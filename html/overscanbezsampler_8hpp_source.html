48 template<typename sampler_def>
49 class OverscanSamplerEdge
50 {
51  typedef typename sampler_def::int_type int_type;
52  typedef typename sampler_def::real_type real_type;
53  typedef typename sampler_def::point_type point_type;
54 
55  static BOOST_CONSTEXPR real_type LargeVal()
56  {
57  return real_services::MinimumMagnitude<real_type, int_type>::val();
58  }
59 
60 public:
61  OverscanSamplerEdge(const point_type& p1, const point_type& p2, bool isAutoCloseSeg) : m_succ(nullptr), m_minMaxResetNeeded(true), m_autoCloseSeg(isAutoCloseSeg), m_isLocalMinima(false)
62  {
63  // make edges in same scanline horizontal
64  if (real_services::Floor(get_y(p1)) == real_services::Floor(get_y(p2)))
65  {
66  if (get_x(p1) < get_x(p2))
67  {
68  get_x(m_bot) = get_x(p1);
69  get_x(m_top) = get_x(p2);
70  }
71  else
72  {
73  get_x(m_top) = get_x(p1);
74  get_x(m_bot) = get_x(p2);
75  }
76  if (get_y(p1) < get_y(p2))
77  {
78  get_y(m_bot) = get_y(p1);
79  get_y(m_top) = get_y(p2);
80  }
81  else
82  {
83  get_y(m_top) = get_y(p1);
84  get_y(m_bot) = get_y(p2);
85  }
86  m_direction = 0;
87  m_delx = 0;
88  }
89  else
90  {
91  if (get_y(p1) < get_y(p2))
92  {
93  m_direction = -1;
94  m_bot = p1;
95  m_top = p2;
96  }
97  else // if (get_y(p1) > get_y(p2))
98  {
99  m_direction = 1;
100  m_bot = p2;
101  m_top = p1;
102  }
103 
104  // if we can't compute the slope make it zero and turn this into a vertical edge
105  real_type delx = get_x(m_bot) - get_x(m_top);
106  real_type dely = get_y(m_bot) - get_y(m_top);
107  m_delx = real_services::DivWillOverflow(delx,dely) ? 0 : delx/dely;
108 
109  // vertical edges may get zero slope without being strictly vertical. adjust them to be strictly vertical
110  // this prevents cases where m_xtop used as endx on vertical/horizontal edges is < begx
111  if (m_delx == 0)
112  get_x(m_top) = get_x(m_bot);
113  }
114 #ifdef BEZSAMPLER_DEBUG
115  if (detail::debugging)
116  cache_logger(TEXT("EDGE %x: b (%.3f,%.3f) t (%.3f,%.3f) w %d\n"),this,
117  real_services::GetFloat(get_x(this->getBot())),real_services::GetFloat(get_y(this->getBot())),
118  real_services::GetFloat(get_x(this->getTop())),real_services::GetFloat(get_y(this->getTop())),
119  this->getDirection());
120 #endif
121  }

123  bool isHorizontalEdge() const
124  {
125  return real_services::Floor(get_y(m_bot)) == real_services::Floor(get_y(m_top));
126  }

128  void extendHorizontalEdge(const point_type& p1, const point_type& p2)
129  {
130  tetraphilia_assert(m_minMaxResetNeeded);
131  real_type v1 = get_x(p1);
132  real_type v2 = get_x(p2);
133  if (v1 > v2)
134  boost::swap(v1,v2);
135  if (v1 < get_x(m_bot))
136  get_x(m_bot) = v1;
137  if (v2 > get_x(m_top))
138  get_x(m_top) = v2;
139 
140  v1 = get_y(p1);
141  v2 = get_y(p2);
142  if (v1 > v2)
143  boost::swap(v1,v2);
144  if (v1 < get_y(m_bot))
145  get_y(m_bot) = v1;
146  if (v2 > get_y(m_top))
147  get_y(m_top) = v2;
148  }
149 
150  // set x values based on Y
151  void CalcXAtY(real_type y)
152  {
153  // minx/maxx calculated in constructor for horizontal/vertical edges
154  if (m_delx != 0)
155  {
156  // watch for roundoff error
157  real_type calcxaty = get_x(m_bot) + m_delx * (y-get_y(m_bot));
158  calcxaty = std::max( -LargeVal(), calcxaty );
159 
160  real_type xatpixelbot = y < get_y(m_bot) ? get_x(m_bot) : calcxaty;
161  real_type xatpixeltop = y+1 < get_y(m_top) ? calcxaty+m_delx : get_x(m_top);
162  // note roundoff error in the calculation can push these values out of expected order
163  if (xatpixelbot < xatpixeltop)
164  {
165  m_minx = xatpixelbot;
166  m_maxx = xatpixeltop;
167  }
168  else
169  {
170  m_maxx = xatpixelbot;
171  m_minx = xatpixeltop;
172  }
173  }
174  else if (m_minMaxResetNeeded)
175  {
176  m_minx = t4::max<real_type>( -LargeVal(), get_x(m_bot) );
177  m_maxx = t4::max<real_type>( -LargeVal(), get_x(m_top) );
178  m_minMaxResetNeeded = false;
179  }
180  tetraphilia_assert(m_minx <= m_maxx);
181  }
182 
183  void IncrementY(real_type y)
184  {
185  // Could call calcxaty - the calcuation here is a bit simpler and faster
186  // CalcXAtY(y);
187  if (m_delx != 0)
188  {
189  tetraphilia_assert(y >= get_y(m_bot));
190  // watch for roundoff error
191  real_type xatpixelbot = get_x(m_bot) + m_delx * (y-get_y(m_bot));
192  real_type xatpixeltop = y+1 < get_y(m_top) ? xatpixelbot+m_delx : get_x(m_top);
193  // note roundoff error in the calculation can push these values out of expected order
194  if (xatpixelbot < xatpixeltop)
195  {
196  m_minx = xatpixelbot;
197  m_maxx = xatpixeltop;
198  }
199  else
200  {
201  m_maxx = xatpixelbot;
202  m_minx = xatpixeltop;
203  }
204  }
205  else if (m_minMaxResetNeeded)
206  {
207  m_minx = get_x(m_bot);
208  m_maxx = get_x(m_top);
209  m_minMaxResetNeeded = false;
210  }
211  }

226  }
227  if (prev->getMaxX() > m_maxx)
228  {
229  m_maxx = prev->getMaxX();
230  m_minMaxResetNeeded = true;
231  }
232  }

376  // single linked list - m_prev is unused
377  void AddEdgeToSampler(Edge *pEdge)
378  {
379  // walk up successor chain?
380 
381  int_type miny = pEdge->getFloorMinY();
382 
383  for (;;)
384  {
385  // TODO: Client should be responsible for these checks
386  // clamp miny to m_rowYMin; discard edges that are out of range
387  if (miny < m_rowYMin)
388  {
389  // discard edges below miny
390  if (pEdge->getFloorMaxY() < m_rowYMin)
391  pEdge = pEdge->getSuccessor();
392  else
393  {
394  miny = m_rowYMin;
395  break; // use pEdge at clamped miny
396  }
397  }
398  else if (t4::add<int_type>(miny, -m_rowYMin) >= t4::convert<int_type>(m_rowHeight)) // discard edges above maxy
399  return;
400  else
401  break; // use pEdge
402 
403  if (!pEdge)
404  return;
405  miny = pEdge->getFloorMinY();
406  }
407 
408  AddEdgeToSampler(pEdge,miny);
409  }
410 
411  // direct call for client to avoid unnecessary edge ymin/ymax filtering
412  void AddEdgeToSampler(Edge *pEdge, int_type miny)
413  {
414  // top of the edge for topleft point scan
415  pEdge->CalcXAtY(real_type(miny));
416  pEdge->m_next = nullptr;
417  while (pEdge->isHorizontalEdge() || pEdge->getFloorMaxY() == miny)
418  {
419  Edge *succ = pEdge->getSuccessor();
420  if (!succ)
421  break;
422  SetEdgeYWithPrev(succ,real_type(miny),pEdge);
423  pEdge = succ;
424  }
425 
426  // adjust miny to be index in m_rows
427  miny -= m_rowYMin;
428 
429  if (static_cast<size_t>(miny) >= m_rowHeight)
430  throw t4::corrupt_data();
431 
432  Edge *first = m_rows[miny];
433 
434  if (!first || !PositionEdgeAfterFirst(first,pEdge))
435  {
436  pEdge->m_next = first;
437  m_rows[miny] = pEdge;
438  }
439  }

649  static Edge *ListRightmost(Edge *list, const Edge *right)
650  {
651  tetraphilia_assert(list != nullptr);
652  tetraphilia_assert(list->getMinX() <= right->getMinX());
653  for (;;)
654  {
655  Edge * next = list->m_next;
656  if (!next || next->getMinX() >= right->getMinX())
657  break;
658  list = next;
659  }
660  return list;

662  static void ListMerge( Edge* left, Edge* right )
663  {
664  while( right )
665  {
666  // j'ai 2 listes, left et right
667  tetraphilia_assert( left != nullptr );
668  tetraphilia_assert( right != nullptr );
669  tetraphilia_assert( left->getMinX() <= right->getMinX() );

812 template<typename transition_list_type, bool evenoddfill>
813 void OverscanBezierSampler<sampler_def>::CalcTransitionTableInternal(transition_list_type& ts, const int_rect_type& constraints)
814 {
815 #ifdef BEZSAMPLER_DEBUG
816  // disable debugging in CalcTransitionTable
817  // SimpleValuePusher<app_traits, bool> pushNoDebug(detail::debugging,false);
818  if (detail::debugging)
819  cache_logger(TEXT("BEGIN CalcTransitionTable %d\n"),this->m_yInt);
820 #endif
821 
822  tetraphilia_assert(ts.size() == 0);
823 
824  Edge *pEdge = this->m_backGroundEdge.m_next;
825  if (pEdge)
826  {
827  // even slots go on - odd slots go off
828  int wind = 0;
829  SegmentID currSeg = kZero;
830 
831  // leave termination in currmax (and not the list) if currSeg == kOne or breaking out of the loop
832  // currmax must always be up to date
833  int_type currmax = limits<int_type>::min();
834 
835  for (;;)
836  {
837  m_yielder.Yield( 100 );
838  // horizontal edges don't change the wind
839  if (evenoddfill)
840  {
841  if (pEdge->getDirection())
842  wind ^= 1;
843  }
844  else
845  wind += pEdge->getDirection();
846 
847  int_type pEdgeMaxX = pEdge->getFloorMaxX();
848  tetraphilia_assert(pEdge->getFloorMinX() <= pEdgeMaxX);
849 
850  if (currSeg == kZero)
851  {
852  tetraphilia_assert((ts.size()&1) == 0);
853  int_type pEdgeMinX = pEdge->getFloorMinX();
854 
855  // either were going to continue the previous seg OR start a new one
856  if (edgeContinuesPreviousSegment(ArgType<real_type>(), pEdgeMinX,currmax,ts))
857  {
858  // extend the previous segment
859  tetraphilia_assert(ts.size());
860  ts.pop_back();
861  tetraphilia_assert(ts.size() != 0 && pEdgeMinX >= ts.back());
862  }
863  else
864  {
865  // start a new segment
866  tetraphilia_assert(ts.size() == 0 || pEdgeMinX > ts.back());
867  ts.push_back(pEdgeMinX);
868  }
869 
870  // update currmax
871  if (pEdgeMaxX > currmax)
872  currmax = pEdgeMaxX;
873 
874  if (wind == 0) // horizontal edge add segment end
875  ts.push_back(std::min(limits<int_type>::max() - 1, currmax) + 1);
876  else
877  currSeg = kOne;
878  }
879  else
880  {
881  // update currmax
882  if (pEdgeMaxX > currmax)
883  currmax = pEdgeMaxX;
884 
885  if (wind == 0)
886  {
887  tetraphilia_assert(!pEdge->getDirection() == false);
888  tetraphilia_assert((ts.size()&1) == 1);
889 
890  // switch off
891  currSeg = kZero;
892  ts.push_back(std::min(limits<int_type>::max() - 1, currmax) + 1);
893  }
894  }
895 
896  // currseg==kone--> ts.size is odd
897  tetraphilia_assert(currSeg != kOne || (ts.size()%2) == 1);
898  // currseg==kzero --> ts.size is even
899  tetraphilia_assert(currSeg != kZero || (ts.size()%2) == 0);
900  if (currmax >= get_xmax(constraints))
901  break; // all done
902 
903  pEdge = pEdge->m_next;
904  if (!pEdge)
905  break;
906  }
907 
908  // and a final on/off -- to infinity and beyond
909  ts.push_back(limits<int_type>::max());
910 
911  // verify the estimate - it should be an absolute maximum
912  tetraphilia_assert(ts.size() <= GetEstimatedMaxTransitions());
913  }
914 
915 #ifdef BEZSAMPLER_DEBUG
916  if (detail::debugging)
917  {
918  cache_logger(TEXT("TransOut y %d:"),this->GetYInt());
919  for (size_t idx2 = 0; idx2 < ts.size(); idx2++)
920  cache_logger(TEXT(" %d"),ts[idx2]);
921  cache_logger(TEXT("\n"));
922  }
923 #endif
