385 TETRAPHILIA_FORCE_INLINE block_type SwapBitsHelper( std::integral_constant<size_t, sz>, const block_type* val )
386 {
387  enum { next_size = sz / 2 };

413 template<typename _surface_traits, typename _signal_traits, typename raster_x_walker_type, typename op_type>
414 class NonArrayLikeSurfaceEncoder : public Encoder<_signal_traits>
415 {
416  typedef _surface_traits surface_traits;
417  typedef typename _signal_traits::app_traits app_traits;
418 
419  typedef _signal_traits signal_traits;
420  typedef typename signal_traits::signal_type source_signal_type;
421  typedef typename app_traits::int_geometry_traits::coord_type int_type;
422 
423 public:
424  NonArrayLikeSurfaceEncoder( const PixelBufferLayout& layout )
425  : m_layout( layout ),
426  m_blockStride( layout.m_pixelStride * surface_traits::pixels_per_block )
427  {
428  m_halftoneMaps = t4::make_transient_array<block_type*>( surface_traits::num_components );
429  MakeHalftoneMapHelper<surface_traits, surface_traits::num_components>::Do( m_halftoneMaps );
430  }
431 
432  typedef typename surface_traits::block_type block_type;
433  static constexpr size_t pixels_per_block = surface_traits::pixels_per_block;
434  static constexpr size_t bits_per_pixel = surface_traits::bits_per_pixel;
435  static constexpr size_t num_components = surface_traits::num_components;
436 
437  static inline block_type GetPixelMask( size_t pixelCount )
438  {
439  return block_type(~0) << static_cast<block_type>( pixelCount * bits_per_pixel );
440  }
441 
442  TETRAPHILIA_FORCE_INLINE block_type LoadPartialBlock( int_type x, int_type y, size_t leftPixToSkip, size_t rightPixToInclude, raster_x_walker_type* walker )
443  {
444  PartialEncodedBlockAccumulator<surface_traits, signal_traits, raster_x_walker_type> accum( MakePhase( x, y ), m_halftoneMaps );
445  return SwapBitsForSurface( AccumulatePixels( rightPixToInclude - leftPixToSkip, walker, block_type(0), accum ) );
446  }
447 
448  TETRAPHILIA_FORCE_INLINE block_type LoadBlock( int_type x, int_type y, raster_x_walker_type* walker )
449  {
450  EncodedBlockAccumulator<surface_traits, signal_traits, raster_x_walker_type> accum( MakePhase( x, y ), m_halftoneMaps );
451  block_type result = AccumulatePixels<pixels_per_block>( walker, block_type(0), accum );
452  return SwapBitsForSurface( result );
453  }
454 
455  TETRAPHILIA_FORCE_INLINE block_type SwapBitsForSurface( block_type val )
456  {
457  using little_endian_type = std::integral_constant<bool, surface_traits::little_endian>;
458  return SwapBits( little_endian_type(), val );
459  }
460 
461  virtual void Encode( int_type xmin, int_type xmax, int_type y, const source_signal_type* src,
462  int_type leftEdgeOfSurfaceX, void* leftEdgeOfSurfacePixels )
463  {
464  tetraphilia_assert( xmin >= leftEdgeOfSurfaceX && xmax >= xmin );
465 
466  raster_x_walker_type xwalker( &m_layout, src );
467  block_type* block = static_cast<block_type*>( leftEdgeOfSurfacePixels );
468  const block_type* lastBlock = block + ( ( xmax - leftEdgeOfSurfaceX ) / pixels_per_block );
469  block += ( xmin - leftEdgeOfSurfaceX ) / pixels_per_block;
470 
471  size_t leftPixToSkip = ( xmin - leftEdgeOfSurfaceX ) % pixels_per_block;
472  block_type leftMask = SwapBitsForSurface( ~GetPixelMask( leftPixToSkip ) );
473 
474  size_t rightPixToInclude = ( xmax - leftEdgeOfSurfaceX ) % pixels_per_block;
475  block_type rightMask = SwapBitsForSurface( GetPixelMask( rightPixToInclude ) );
476 
477  op_type op;
478 
479  if ( block == lastBlock )
480  {
481  *block = static_cast<block_type>( *block & ~( leftMask + rightMask ) )
482  + ( op( LoadPartialBlock( xmin + static_cast<int_type>( leftPixToSkip ), y, leftPixToSkip, rightPixToInclude, &xwalker ), *block ) & ( leftMask + rightMask ) );
483  }
484  else
485  {
486  if ( leftPixToSkip > 0 )
487  {
488  *block = ( *block & ~leftMask ) + ( op( LoadPartialBlock( xmin + static_cast<int_type>( leftPixToSkip ), y, leftPixToSkip, pixels_per_block, &xwalker ), *block ) & leftMask );
489  ++block;
490  xmin += static_cast<int_type>( pixels_per_block - leftPixToSkip );
491  }
492 
493  while ( block != lastBlock )
494  {
495  *block = op( LoadBlock( xmin, y, &xwalker ), *block );
496  ++block;
497  xmin += static_cast<int_type>( pixels_per_block );
498  }
499 
500  if ( rightPixToInclude > 0 )
501  *block = ( *block & ~rightMask ) + ( op( LoadPartialBlock( xmin, y, 0, rightPixToInclude, &xwalker ), *block ) & rightMask );
502  }
503  }
504 
505 private:
506  const PixelBufferLayout& m_layout;
507  size_t m_blockStride;
508  block_type** m_halftoneMaps;
509 };

511 template<typename _surface_traits, typename _signal_traits, typename op_type>
512 class ArrayLikeSurfaceEncoder : public Encoder<_signal_traits>
513 {
514  typedef _surface_traits surface_traits;
515  typedef typename _signal_traits::app_traits app_traits;
516 
517  typedef typename surface_traits::signal_traits dest_signal_traits;
518  typedef typename dest_signal_traits::signal_type dest_signal_type;

523 public:

528  void Encode( int_type xmin, int_type xmax, int_type /*y*/, const source_signal_type* src,
529  int_type leftEdgeOfSurfaceX, void* leftEdgeOfSurfacePixels ) override
530  {
531  dest_signal_type* dst = static_cast<dest_signal_type*>( leftEdgeOfSurfacePixels );
532  dst += static_cast<ptrdiff_t>( surface_traits::pixel_size_in_components ) * ( xmin - leftEdgeOfSurfaceX );
533  dst += ( surface_traits::little_endian ? ( surface_traits::pixel_size_in_components - surface_traits::pad_components - 1 ) : surface_traits::pad_components );
534 
535  while ( xmin != xmax )
536  {
537  const source_signal_type* srcComp = t4::add_bytes( src, m_layout.m_firstSignalOffset );
538  for( size_t i = 0; i < surface_traits::num_components ; ++i )
539  {
540  op_type op;
541  *dst = op( SignalConverter<typename signal_traits::signal_type, typename dest_signal_traits::signal_type>::Do( *srcComp ), *dst );
542  dst += ( surface_traits::little_endian ? -1 : 1 );
543  srcComp = t4::add_bytes( srcComp, m_layout.m_signalStride );
544  }
545 
546  dst += ( surface_traits::little_endian
547  ? ( 2 * surface_traits::pixel_size_in_components - surface_traits::pad_components )
548  : ( surface_traits::pixel_size_in_components - surface_traits::num_components ) );
549  src = t4::add_bytes( src, m_layout.m_pixelStride );
550  ++xmin;
551  }

675 template<typename surface_traits, size_t block_size, size_t pixels_per_block>
676 void ScrollSurfaceFast(
677  const typename surface_traits::app_traits::int_geometry_traits::point_type& delta,
678  const typename surface_traits::app_traits::int_geometry_traits::point_type& surfaceSize,
679  void* baseAddress,
680  size_t scanLineStride )
681 {
682  typedef typename surface_traits::app_traits app_traits;
683  typedef typename app_traits::int_geometry_traits::coord_type int_type;
684  typedef typename app_traits::int_geometry_traits::rect_type int_rect_type;
685 
686  int_rect_type srcRect;
687  MakeRectangle(srcRect,
688  t4::max<int_type>( -get_x( delta ), int_type(0) ),
689  t4::max<int_type>( -get_y( delta ), int_type(0) ),
690  t4::min<int_type>( get_x( surfaceSize ) - get_x( delta ), get_x( surfaceSize ) ),
691  t4::min<int_type>( get_y( surfaceSize ) - get_y( delta ), get_y( surfaceSize ) ) );
692 
693  if ( !RectIsEmpty( srcRect ) )
694  {
695  void* start = t4::add_bytes( baseAddress, get_ymin( srcRect ) * scanLineStride + block_size * get_xmin( srcRect ) / pixels_per_block );
696  void* end = t4::add_bytes( baseAddress, get_ymax( srcRect ) * scanLineStride + block_size * get_xmin( srcRect ) / pixels_per_block );
697  ptrdiff_t lineDelta;
698  if ( get_y( delta ) > 0 )
699  {
700  lineDelta = -( static_cast<ptrdiff_t>( scanLineStride ) );
701  start = t4::add_bytes( start, lineDelta );
702  end = t4::add_bytes( end, lineDelta );
703  boost::swap( start, end );
704  }
705  else
706  lineDelta = static_cast<ptrdiff_t>( scanLineStride );
707 
708  ptrdiff_t memDelta = ( static_cast<ptrdiff_t>( scanLineStride ) ) * get_y( delta ) + block_size * get_x( delta ) / pixels_per_block;
709  size_t bytesPerScanLine = block_size * get_width( srcRect ) / pixels_per_block ;
710  for ( void* addr = start; addr != end; addr = t4::add_bytes( addr, lineDelta ) )
711  {
712  memmove( t4::add_bytes( addr, memDelta ), addr, bytesPerScanLine );
713  }
714  }
715 }

819 template<typename surface_traits, typename signal_traits, typename op_type>
820 void BlitToSurface( const typename signal_traits::app_traits::int_geometry_traits::point_type& clipOrigin,
821  const Constraints& constraints,
822  const typename signal_traits::app_traits::int_geometry_traits::point_type& sourceOrigin,
823  RasterPainter<signal_traits>& source,
824  const typename signal_traits::app_traits::int_geometry_traits::point_type& surfaceSize,
825  void* baseAddress,
826  size_t scanLineStride,
827  op_type )
828 {
829  typedef typename signal_traits::app_traits app_traits;
830  typedef Constraints constraints_type;
831  typedef typename app_traits::int_geometry_traits::coord_type int_type;
832  typedef typename app_traits::int_geometry_traits::rect_type int_rect_type;
833 
834  typedef typename signal_traits::signal_type signal_type;
835  typedef const_GraphicYWalker<signal_traits> segment_type;
836  typedef RasterMachine<signal_traits> machine_type;
837 
838  typedef Encoder<signal_traits>* encoder_ptr_type;
839  t4::transient_vector<encoder_ptr_type> encodeProcs( source.GetNumSegments(), *t4::get_thread_heap() );
840  std::transform( SegmentsBegin( &source ), SegmentsEnd( &source ), encodeProcs.begin(), MakeEncoder<signal_traits, surface_traits, op_type>() );
841 
842  constraints_type surfaceConstraints = TranslateConstraints( constraints, get_x( clipOrigin ), get_y( clipOrigin ) );
843  surfaceConstraints.ConstrainFurther( MakeRectangle<int_rect_type>( 0, 0, get_x( surfaceSize ), get_y( surfaceSize ) ) );
844 
845  int_type clipWalkerXMax = get_xmax( surfaceConstraints.m_interestBounds ) - get_x( clipOrigin );
846 
847  void* dst = baseAddress;
848  dst = t4::add_bytes( dst, t4::multiply( scanLineStride, get_ymin( surfaceConstraints.m_interestBounds ) ) );
849 
850  machine_type machine;
851 
852  SetMachine( &machine, source );
853  SetY( &machine, get_ymin( surfaceConstraints.m_interestBounds ) - get_y( sourceOrigin ) );
854  for ( int_type y = get_ymin( surfaceConstraints.m_interestBounds ); y < get_ymax( surfaceConstraints.m_interestBounds ); ++y )
855  {
856  int_type xstart = get_xmin( surfaceConstraints.m_interestBounds );
857  int_type xstop;
858  while ( xstart < get_xmax( surfaceConstraints.m_interestBounds ) )
859  {
860  xstop = get_xmax( surfaceConstraints.m_interestBounds );
861  tetraphilia_assert( xstop > xstart );
862 
863  xstop = get_x( sourceOrigin ) + SetX( &source, xstart - get_x( sourceOrigin ), xstop - get_x( sourceOrigin ) );
864  SegmentID segID = source.GetCurrentSegmentID();
865  tetraphilia_assert( xstop > xstart );
866  const segment_type* seg = CurrentSegment( &source );
867  const signal_type* color = seg->m_color->m_scanLine;
868  color = t4::add_bytes( color, seg->m_color->m_pixelBuffer.m_layout->m_pixelStride * ( xstart - get_x( sourceOrigin ) - get_xmin( *seg->m_color->m_pixelBuffer.m_bounds ) ) );
869  encodeProcs[segID]->Encode( xstart + get_x( sourceOrigin ), xstop + get_x( sourceOrigin ), y + get_y( sourceOrigin ), color,
870  get_x( sourceOrigin ), dst );
871 
872  xstart = xstop;
873  }
874 
875  dst = t4::add_bytes( dst, scanLineStride );
876  IncrementY( &machine );
877  }
