138 /*
139  If obj is non-null, call the object's dtor and deallocate the pointer using t4::free.
140 */
141 template<typename T>
142 inline void t4_delete( T* obj )
143 {
144  if ( obj )
145  {
146  t4::debug_validate_ptr( obj );
147  obj->~T();
148  t4::free( const_cast<typename std::remove_const<T>::type*>( obj ) );
149  }
150 }

152 /*
153  If arr is non-null, call the dtor on all the objects in the array (in reverse order), and
154  deallocate the pointer using t4::free.
155 */
156 template<typename T>
157 inline void t4_delete_array( T* arr, size_t count )
158 {
159  if ( arr )
160  {
161  debug_validate_ptr( arr );
162  while ( count-- )
163  arr[count]->~T();
164  t4::free( const_cast<typename std::remove_const<T>::type*>( arr ) );
165  }
166 }

452 template<typename T>
453 struct deleter<T[]>
454 {
455  /*
456  t4::unique_ptr (or any unique_ptr with an "interesting" custom deleter)
457  doesn't work for array types. You probably want to use std::array or
458  t4::vector instead.
459 
460  It is in fact impossible (in a portable way, anyway) to do what we're trying
461  to do here, that is, deallocate an array of C++ objects that were allocated
462  using a custom overload of operator new[].
463 
464  We need, in essence, to get the compiler to generate code that invokes all
465  the element's destructors (which means using the compiler-dependent secret
466  code for identifying the number of elements in the array), but follow it
467  up with a call to our custom ::operator delete[]. Ideally, this would be
468  supported by the following syntax:
469 
470  delete[] (t4_new) x;
471 
472  But that's not allowed.
473 
474  http://www.stroustrup.com/bs_faq2.html#placement-delete illuminates some
475  issues similar to this one.
476 
477  That said, if the element type is trivially destructible, then this is a non-issue.
478  */
479  static_assert( std::is_trivially_destructible<typename std::remove_extent<T>::type>::value, "" );
480 
481  deleter( void ) BOOST_NOEXCEPT {}
482 
483  template<typename U>
484  deleter( deleter<U> const&, std::enable_if_t<std::is_convertible<U*, T*>::value>* = 0 ) BOOST_NOEXCEPT {}
485 
486  void operator()( T* p ) const BOOST_NOEXCEPT
487  {
488  // We've already statically asserted that the dtor is trivial
489  t4::free( p );
490  }
491 };

565 /*
566  Create a unique_ptr to an object of type T constructed with args. Additional bytes
567  are allocated after the end of the allocated object; there will be at least extra_bytes_after_result
568  writable bytes at the first memory location after this+1 that is aligned to alignment.
569 */
570 template<typename T, typename... TArgs>
571 unique_ptr<T> make_unique_extra_aligned_bytes( size_t extra_bytes_after_result, size_t alignment, TArgs&&... args )
572 {
573  aligner a;
574  a.push<T>()( extra_bytes_after_result, alignment );
575  size_t explicit_size = a.get();
576  return unique_ptr<T>( new
577  ( t4_new_explicit_size_t( explicit_size ) )
578  T( std::forward<TArgs>( args )... ) );
