332 private:
333  void CopyScanLine( const unsigned char* src, unsigned char* dst )
334  {
335  // src points to the first pixel in the source scan line, dst
336  // points to the edge of the fringe rect. These are not necessarily
337  // the same coordinate.
338  for ( auto x = -m_leftPad; x < 0; ++x )
339  {
340  memcpy( dst, src, m_bpp );
341  dst += m_bpp;
342  }
343 
344  memcpy( dst, src, static_cast<size_t>(m_srcWidth) * m_bpp );
345 
346  src = &m_scanLine[ ( static_cast<size_t>(m_srcWidth) - 1 ) * m_bpp ];
347  dst += static_cast<size_t>(m_srcWidth) * m_bpp;
348  for ( int x = 0; x < m_rightPad; ++x )
349  {
350  memcpy( dst, src, m_bpp );
351  dst += m_bpp;
352  }
353  }
354 
355  const unsigned char* GetNextScanLineImpl( void ) override
356  {
357  if ( m_y <= m_topPad )
358  {}
359  else if ( m_topPad <= ( m_y + 1 ) && m_y < m_topPad + m_srcHeight )
360  {
361  unsigned char* dst = &m_scanLine[0];
362  const unsigned char* src = m_srcPipe->GetNextScanLine();
363  CopyScanLine( src, dst );
364  }
365  else
366  {
367  // Just keep re-using the last scan line
368  }
369 
370  ++m_y;

372  }
