158 template<typename allocator_type, typename T>
159 class allocator_auto_ptr
160 {
161  typedef typename allocator_type::app_traits app_traits;
162  typedef T* ptr_type;
163 
164 public:
165  explicit allocator_auto_ptr( const allocator_type& allocator,
166  ptr_type t = nullptr )
167  : m_allocator(allocator),
168  m_p(t)
169  {}
170 
171  allocator_auto_ptr( allocator_auto_ptr& rhs )
172  : m_allocator( rhs.m_allocator ),
173  m_p( rhs.m_p )
174  {
175  rhs.release();
176  }
177 
178  ~allocator_auto_ptr()
179  {
180  m_allocator.Delete( m_p );
181  }
182 
183  allocator_auto_ptr& operator=(allocator_auto_ptr& rhs)
184  {
185  if( m_p != rhs.m_p )
186  {
187  m_allocator.Delete( m_p );
188  m_p = rhs.m_p;
189  rhs.m_p = 0;
190  }
191  return *this;
192  }
193 
194  friend void swap( allocator_auto_ptr& lhs, allocator_auto_ptr& rhs )
195  {
196  boost::swap( lhs.m_allocator, rhs.m_allocator );
197  boost::swap( lhs.m_p, rhs.m_p );
198  }
199 
200  T& operator*() const { return *m_p; }
201  ptr_type operator->(void) const { return m_p; }
202  ptr_type get(void) const { return m_p; }
203  ptr_type release(void) { ptr_type ret = m_p; m_p = nullptr; return ret; }
204 
205 private:
206  allocator_type m_allocator;
207  ptr_type m_p;
208 };
