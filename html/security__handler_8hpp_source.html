46 /*
47  Enumeration identifying two important contexts in which we deliberately suppress decryption:
48 
49  normal: decrypt everything
50  trailer_top: top level of the trailer dictionary, don't decrypt string values in the /Info keys
51  encrypt_dict: anywhere in the /Encrypt dictionary, don't decrypt any strings
52 */
53 enum class decryption_context
54 {
55  decrypt_all, // everything should be decrypted
56  trailer_container, // about to parse the << of a trailer dictionary
57  trailer_top, // inside the top level of a trailer dictionary
58  encrypt_dict_container, // about to parse the << of the encrypt dictionary
59  id_entry_container, // about to parse the value of the /ID key in a trailer dictionary
60  suppress_decryption // suppress all decryption beyond this point (e.g., inside the encrypt dictionary)
61 };
