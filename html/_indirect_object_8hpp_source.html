87  /*
88  Create a "temporary" IDO. The resulting object is _not_ stored in the IDO cache,
89  does not have a meaningful objNum/generation, cannot ever be the result of a call
90  to ResolveReference, and cannot be reached from the document's root node.
91 
92  A temporary IDO is useful in the creation of placeholder store objects, typically
93  for display in the GUI. The motivating case (but there may be more) is forms
94  editing, where we would like to render a form's appearance without text content
95  behind a text editing widget. We achieve this by creating a temporary appearance
96  object (at the moment only temporary dictionaries may be created) and painting it,
97  instead of the value of the annotation's /AP /N key.
98  */

272 template<typename app_traits>
273 IndirectObject<app_traits>::IndirectObject( typename data_io::BufferedStream& stream,
274  Store<app_traits>* store,
275  const Reference* ref,
276  int_type containerObjNum,
277  bool ignoreRefNums,
278  bool parseObjHeader,
279  t4::pdf::decryption_context dc)
280  : m_heap( kHeapBlockSize, kHeapLargeAllocSize ),
281  m_store(store),
282  m_stack( allocator_type( m_heap ), 2/*chunkCapacity*/ ),
283  m_decrypter(nullptr),
284  m_temporary( false ),
285  m_objNum( 0 ),
286  m_generation( 0 ),
287  m_dirtyState( kNone ),
288  m_isTrailerDict(dc == t4::pdf::decryption_context::trailer_container),
289  m_gettingLoaded(true),
290  m_gettingConstructed(true)
291 {
292  m_stack.SetIDO(this);
293  {
294  if (ref != nullptr)
295  {
296  if ( parseObjHeader )
297  {
298  Parser::SkipCommentsAndWhiteSpace( stream );
299  stream.PeekAssert();
300  Parser::ParseNumber( stream, m_stack );
301  Reference actualRef;
302  m_objNum = actualRef.m_objNum = PopInt( m_stack );
303  if ( !ignoreRefNums && actualRef.m_objNum != ref->m_objNum )
304  throw t4::corrupt_data();
305  Parser::SkipCommentsAndWhiteSpace( stream );
306  stream.PeekAssert();
307  Parser::ParseNumber( stream, m_stack );
308  m_generation = actualRef.m_generation = PopInt( m_stack );
309  Parser::SkipCommentsAndWhiteSpace( stream );
310  Parser::SkipExpectedToken( stream, "obj" );
311 
312  tetraphilia_assert( store );
313  m_decrypter = store->CreateDecrypter( actualRef, m_heap );
314  }
315  else // if it was not populated from the obj header
316  {
317  m_generation = ref->m_generation;
318  m_objNum = ref->m_objNum;
319  }
320  }
321 
322  ObjectStoreParser<object_traits> p( this, true/*stopAfterFirstObject*/, true/*eofIsError*/, dc, m_decrypter );
323  p.ParseWithStack( stream, m_stack, stream.Mark(), containerObjNum );
324  }
325 
326  m_gettingLoaded = false;
327  m_gettingConstructed = false;
328 }

355 template<typename app_traits>
356 void IndirectObject<app_traits>::InitializeWithType( store_type* store, ObjectType type )
357 {
358  m_store = store;
359 
360  m_stack.SetIDO(this);
361 
362  switch ( type )
363  {
364  case store::kObjectTypeInteger:
365  PushInt( m_stack, 0 );
366  break;
367 
368  case store::kObjectTypeReal:
369  PushReal( m_stack, 0 );
370  break;
371 
372  case store::kObjectTypeBool:
373  PushBool( m_stack, false );
374  break;
375 
376  case store::kObjectTypeLargeInteger:
377  PushLargeInt( m_stack, 0 );
378  break;
379 
380  case store::kObjectTypeString:
381  {
382  // @TODO: There is no need to waste a null termination byte here; PDF strings
383  // are not NULL terminated and client code must not assume that.
384  auto strObj = MakeStringObj( m_heap, t4::string(), hex_format::no );
385  PushString( m_stack, strObj );
386  }
387  break;
388 
389  case store::kObjectTypeName:
390  {
391  auto nameObj = MakeStringObj( m_heap, t4::string(), hex_format::no );
392  StringObj* pname(nameObj);
393  PushName( m_stack, pname );
394  }
395  break;
396 
397  case store::kObjectTypeArray:
398  {
399  file_size_type uniqueNum = store->GetUniqueDictOrArrNumber();
400  DictionaryObj<object_traits>* a
401  = TRANSIENT_NEW(DictionaryObj<object_traits>, m_heap, (m_heap, uniqueNum, this, is_dictionary::no) );
402  PushArray<object_traits>( m_stack, a );
403  }
404  break;
405 
406  case store::kObjectTypeDictionary:
407  {
408  file_size_type uniqueNum = store->GetUniqueDictOrArrNumber();
409  DictionaryObj<object_traits>* d
410  = TRANSIENT_NEW(DictionaryObj<object_traits>, m_heap, (m_heap, uniqueNum, this, is_dictionary::yes) );
411  PushDict<object_traits>( m_stack, d );
412  }
413  break;
414 
415  case store::kObjectTypeNull:
416  default: // quiets gcc compiler warning about enumeration types not being handled in a switch statement
417  // push a null ObjectImpl
418  m_stack.Push( ObjectImpl<object_traits>() );
419  break;
420  }
421 }

423 template<typename app_traits>
424 void IndirectObject<app_traits>::InitializeWithCopy( store_type* store, const object_type& objToCopy )
425 {
426  m_store = store;
427 
428  m_stack.SetIDO(this);
429 
430  // We can't copy decrypter objects, so we create a fresh decrypter based on the objnum/gen
431  // of the object being cloned. And because other TIDO's may get created from this one,
432  // we copy the objNum/generation in this IDO so that future decrypters are correctly
433  // constructed.
434  {
435  t4::sptr<IndirectObject<app_traits>> ido = objToCopy.GetPDFObject().GetIDOPtr();
436 
437  // Copy the object and generation number if the object being copied is an indirect object
438  if ( objToCopy.IsIndirectObject() )
439  {
440  m_objNum = ido->GetObjectNum();
441  m_generation = ido->GetGenerationNum();
442  }
443 
444  // If the source IDO doesn't have a decrypter, this means one of two things:
445  // 1) It isn't encrypted
446  // 2) It is the trailer
447  // Under any of these conditions the copied object also does not require a decrypter.
448  if ( ido->m_decrypter )
449  {
450  tetraphilia_assert( false );
451  throw t4::unimplemented();
452 
453  // The code below this line has not been tested! At present there is no need
454  // for CreateTemporaryXXX() in encrypted documents though, so it shouldn't
455  // ever be executed. If we do start caring about TIDOs in encrypted documents
456  // We should delete the above assert/throw code and test this code path carefully.
457  Reference ref( ido->GetObjectNum(), ido->GetGenerationNum() );
458  m_decrypter = store->CreateDecrypter( ref, m_heap );
459  }
460  }
461 
462  switch ( objToCopy.Type() )
463  {
464  case store::kObjectTypeInteger:
465  PushInt( m_stack, objToCopy.IntegerValue_NoTypeCheck() );
466  break;
467 
468  case store::kObjectTypeReal:
469  PushReal( m_stack, objToCopy.RealValue_NoTypeCheck() );
470  break;
471 
472  case store::kObjectTypeBool:
473  PushBool( m_stack, objToCopy.BoolValue_NoTypeCheck() );
474  break;
475 
476  case store::kObjectTypeLargeInteger:
477  PushLargeInt( m_stack, objToCopy.LargeIntegerValue_NoTypeCheck() );
478  break;
479 
480  case store::kObjectTypeString:
481  {
482  string_type str = objToCopy.StringValue_NoTypeCheck();
483  size_t len = static_cast<size_t>( str.end() - str.begin() );
484  auto strObj = MakeStringObj( m_heap, { str.begin(), len }, str.IsHexFormat() );
485  StringObj* pstr(strObj);
486  PushString( m_stack, pstr );
487  }
488  break;
489 
490  case store::kObjectTypeName:
491  {
492  name_type name = objToCopy.NameValue_NoTypeCheck();
493  size_t len;
494  name.get( len );
495  // StringObj is used for both name and string types
496  auto nameObj = MakeStringObj( m_heap, { name.get(), len }, name.IsHexFormat() );
497  PushName( m_stack, nameObj );
498  }
499  break;
500 
501  case store::kObjectTypeArray:
502  {
503  // DictionaryObj is used for both array and dictionary types
504  file_size_type uniqueNum = store->GetUniqueDictOrArrNumber();
505  DictionaryObj<object_traits>* a
506  = TRANSIENT_NEW(DictionaryObj<object_traits>, m_heap, ( *objToCopy.Data().dictAndArrayRep, m_heap, uniqueNum, this, is_dictionary::no) );
507  PushArray<object_traits>( m_stack, a );
508  }
509  break;
510 
511  case store::kObjectTypeDictionary:
512  {
513  file_size_type uniqueNum = store->GetUniqueDictOrArrNumber();
514  DictionaryObj<object_traits>* d
515  = TRANSIENT_NEW(DictionaryObj<object_traits>, m_heap, ( *objToCopy.Data().dictAndArrayRep, m_heap, uniqueNum, this, is_dictionary::yes) );
516  PushDict<object_traits>( m_stack, d );
517  }
518  break;
519 
520  case store::kObjectTypeNull:
521  default: // quiets gcc compiler warning about enumeration types not being handled in a switch statement
522  // push a null ObjectImpl
523  m_stack.Push( ObjectImpl<object_traits>() );
524  break;
525  }
526 }

528 template<typename app_traits>
529 void IndirectObject<app_traits>::SerializeIndirectObject( const t4::sptr<IndirectObject<app_traits>>& ido, typename tetraphilia::data_io::data_io_detail::OutputStream* stream, Store<app_traits> * store ) const
530 {
531  WriteIntType( stream, m_objNum );
532  WriteSpace( stream );
533  WriteIntType( stream, m_generation );
534  WriteSpace( stream );
535  stream->write( "obj\n", 4 );
536 
537  const ObjectImpl<object_traits>& objImpl = GetPDFObject();
538 
539  if(objImpl.type == kObjectTypeDictionary && objImpl.u.dictAndArrayRep->HasStream())
540  {
541  TransientHeap<app_traits> heap( 4096, 1024 ); // a private transient heap
542 
543  Dictionary<StoreObjTraits<app_traits>> dict = MakeStoreObject( store, ido ).DictionaryValue();
544  int_type currentLengthVal = dict.GetInteger("Length", -1);
545  if( currentLengthVal == -1 )
546  throw t4::corrupt_data();
547  auto dbs = GetUnfilteredStream(dict, false);
548 
549  size_t memBufLen = 512;
550  t4::vector<char> memBuf( memBufLen );
551  size_t length = 0;
552  while(1)
553  {
554  t4::data_block db = dbs.get_next_block();
555  if(db.get_size())
556  {
557  auto newLength = t4::add<size_t>( length, db.get_size() );
558  if(newLength > memBufLen)
559  {
560  while(newLength > memBufLen )
561  memBufLen = t4::multiply<size_t>( memBufLen, 2 );
562  memBuf.resize(memBufLen);
563  }
564  ::memcpy( memBuf.data() + length, db.get_bytes(), db.get_size());
565 
566  length+=db.get_size();
567  }
568  else
569  break;
570  }
571 
572  /* For disk streams, the "Length" entry always contains the unfiltered length */
573  if( objImpl.u.dictAndArrayRep->IsDiskStream() && (static_cast<size_t>(currentLengthVal) != length) )
574  {
575  tetraphilia_assert(false);
576  }
577 
578  if ( objImpl.u.dictAndArrayRep->IsUnsavedStream() )
579  dict.PutInteger("Length", t4::convert<int_type>(length));
580 
581  objImpl.SerializeObjectImpl( stream );
582 
583  /* If the stream is in memory then restore the value of "Length" (as it holds the filtered stream length in that case) */
584  if(objImpl.u.dictAndArrayRep->IsUnsavedStream())
585  {
586  dict.PutInteger("Length", currentLengthVal);
587  }
588 
589  stream->write("\nstream\n",8);
590  stream->write(memBuf.data(), length);
591  stream->write("\nendstream",10);
592  }
593  else
594  {
595  objImpl.SerializeObjectImpl( stream );
596  }
597 
598  stream->write( "\nendobj\n", 8 );
599 }
