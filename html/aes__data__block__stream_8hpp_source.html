151 inline t4::data_block get_next_block( AESDataBlockStream& ths )
152 {
153  using namespace aescrypt;
154 
155  /*
156  Callers of this api depend upon dataBlock->len to be zero. In case of aes encrytption, the finalize call
157  returns the last block of data which is mostly non zero. Thus GetNextBlockImpl will thus in such cases will
158  never return block of data len == 0; Hence this check, if you reach end of file (maintained by bool m_EOF)
159  return an empty block.Please mark m_source->AtEOF() == true is valid case for getting the last block of data
160  */
161  if ( ths.m_EOF )
162  return t4::data_block();
163 
164  const char* bytes;
165  size_t thisSize = ths.m_bufferedSource.PeekBytes( bytes, AESCryptor::block_size );
166 
167  // If there are more than zero but fewer than BLOCK_SIZE bytes left, we have an invalid AES stream.
168  if ( 0 < thisSize && thisSize < AESCryptor::block_size )
169  throw t4::corrupt_data();
170 
171  // We need multiples of AESCryptor::block_size
172  AlignSize(thisSize);
173 
174  auto hdbo = t4::make_unique<std::uint8_t[]>( TET_AES_MAX_BLOCK_SIZE );
175  size_t len = 0;
176 
177  size_t numDecryptedBytes = 0;
178  if ( thisSize == 0 )
179  {
180  // No more data to decrypt, so finalize AES
181  // Will get a max of one more AES block of decrypted data
182  len = AESCryptor::block_size;
183  numDecryptedBytes = ths.m_aes->Finish( static_cast<unsigned char *>( hdbo.get() ), len );
184  len = numDecryptedBytes;
185  ths.m_EOF = true;
186  }
187  else
188  {
189  numDecryptedBytes = ths.m_aes->Decrypt( reinterpret_cast<const unsigned char *>( bytes ), thisSize, static_cast<unsigned char *>( hdbo.get() ), TET_AES_MAX_BLOCK_SIZE );
190  ths.m_bufferedSource.Advance( thisSize );
191  len = numDecryptedBytes;
192 
193  /* This code has been added to take care of the scenario when the size of first input block to the AES decryptor is equal to AES Block
194  * Size. In this case, the output block returned to us is of size zero (numDecryptedBytes = 0). This is because the AES decryptor
195  * always retains the last block of output data for handling the padding properly. But if we return the zero sized output from here,
196  * then the zlib code in 'zlib_data_block_stream.hpp' thinks that it has encountered end of the stream which leads to failure in
197  * rendering.
198  *
199  * Fix: We make a recursive call to this method to get more output bytes before returning.
200  */
201 
202  if( 0 == numDecryptedBytes )
203  {
204  tetraphilia_assert(ths.m_aes->block_size == thisSize);
205  return get_next_block( ths );
206  }
207  }
208 
209  return t4::make_data_block( std::move( hdbo ), len );
210 }
