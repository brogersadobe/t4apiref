463  // Get the PDF major and minor versions as an ordered pair of ints.
464  boost::optional<std::pair<int, int>> GetPDFVersion()
465  {
466  // Start with the PDF version from the header string
467  boost::optional<std::pair<int, int>> pdf_version(m_xref->get_pdf_version());
468 
469  // Next check if the Catalog has an overriding Version entry
470  boost::optional<std::pair<int, int>> catalog_version;
471  auto catalog_version_entry = GetRoot().Get("Version");
472  if (catalog_version_entry.Type() == kObjectTypeName)
473  {
474  auto catalog_version_name = catalog_version_entry.NameValue_NoTypeCheck();
475  size_t len = 0;
476  const char *catalog_version_name_string = catalog_version_name.get(len);
477  if (len > 0)
478  {
479  catalog_version = t4::pdf::parse_pdf_version(catalog_version_name_string, len);
480  if (pdf_version < catalog_version)
481  {
482  pdf_version = catalog_version;
483  }
484  }
485  }
486  return pdf_version;
487  }
