65  void ConvertPixel( signal_type * dst, ptrdiff_t dstSignalStride,
66  const signal_type * src, ptrdiff_t /*srcSignalStride*/ ) const
67  {
68  size_t index = signal_traits::signal_to_index( *src );
69  if( index > m_hiVal )

156  IndexedColorConverter( const IndexedColorSpace<app_traits> & srcCS,
157  const color::ColorSpace<app_traits> & dstCS,
158  color::RenderIntent renderIntent,
159  color::ConverterCachingType cachingType,
160  bool linear,
161  hide_this_public_ctor )
162  : m_hiVal( srcCS.GetHiVal() ),
163  m_table( ( m_hiVal + 1 ) * dstCS.GetNumSignals(), t4::transient_allocator<signal_type>( *t4::get_thread_heap() ) ),
164  m_numDstSignals( dstCS.GetNumSignals() ),
165  m_baseConverter( srcCS.GetBaseCS().template CreateConverter<signal_traits>( dstCS, renderIntent, cachingType, linear ) )
166  {
167  // Convert the lookup table to the destination color space:
168  size_t numSrcSignals = srcCS.GetBaseCS().GetNumSignals();
169  size_t numDstSignals = dstCS.GetNumSignals();
170  t4::transient_vector<signal_type> buf( numSrcSignals, t4::transient_allocator<signal_type>( *t4::get_thread_heap() ) );
171  for( size_t tableIndex = 0; tableIndex <= m_hiVal; ++tableIndex )
172  {
173  for( size_t bufIndex = 0; bufIndex < numSrcSignals; ++bufIndex )
174  buf[ bufIndex ] = imaging_model::SignalConverter<unsigned char, signal_type>::Do( srcCS.GetTable()[ tableIndex * numSrcSignals + bufIndex ] );
175  m_baseConverter->Convert( &m_table[ tableIndex * numDstSignals ], sizeof( signal_type ),
176  buf.data(), sizeof( signal_type ) );
177  }
178  }
179 
180 protected:
181  const size_t m_hiVal;
182  t4::transient_vector<signal_type> m_table;
183  const size_t m_numDstSignals;
184  t4::unique_ptr<color::ColorConverter<signal_traits>> m_baseConverter;
185 };
