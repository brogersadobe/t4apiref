81 {

83  return ( static_cast<up_t>( v ) & ( static_cast<up_t>( 1 ) << ( 8 * sizeof( up_t ) - 1 ) ) ) != 0;

265 /*
266  safe_math_op are used to support clients wishing to handle arithmetic failures on their
267  own vs. simply letting exceptions be thrown. For example:
268 
269  auto x = t4::add<boost::optional<int>>( y, z );
270 
271  ...will _never_ throw an exception, where:
272 
273  auto x = t4::add<int>( y, z );
274 
275  ...will throw an exception, if the mathematically correct sum of y and z cannot be represented
276  in an int.
277 
278  safe_math_op takes an "Operation" which is presumed to support an operator() that takes the
279  sequence of arguments "args" and returns a boost::optional<Result>. This result is then returned
280  to the caller in the optional case, or checked (with a possible throw, if uninitialized) in the
281  non-optional case.
282 */
283 template<typename Result, typename Impl, typename... TArgs>
284 BOOST_FORCEINLINE constexpr auto safe_math_op( Impl&& impl, TArgs&&... args )
285 {
286  return safe_math_helper2<Result>()( impl, std::forward<TArgs>( args )... );
287 }
