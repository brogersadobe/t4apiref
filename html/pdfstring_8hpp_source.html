156 template<typename app_traits>
157 void UTF16StringConstIterator<app_traits>::init( const char* begin, const char* end )
158 {
159 
160  // Create another iterator to look for FEFF
161  const char* it = begin;
162  if ( it != end && static_cast<unsigned char>(*it) == 0xFE )
163  {
164  ++it;
165  if ( it != end && static_cast<unsigned char>(*it) == 0xFF )
166  {
167  // If we get here, this string is encoded in Unicode. Skip
168  // past the FEFF.
169  m_isUnicodeUTF16BE = true;
170  ++m_str;
171  ++m_str;
172  }
173  }
174 
175  m_done = ( m_str == m_strEnd );
176  GetNextVal();

179 template<typename app_traits>
180 void UTF16StringConstIterator<app_traits>::GetNextVal( void )
181 {
182  m_done = ( m_str == m_strEnd );
183  if( m_done )
184  return;
185 
186  if ( m_isUnicodeUTF16BE )
187  {
188  m_currVal = GetNextUTF16WordVal();
189  while ( m_currVal == 0x1b ) // language escape sequence
190  {
191  /*
192  The PDF spec is unclear about whether the individual letters in the language and
193  country code values are bytes or unicode characters. Consultation with the PDF
194  group indicates that they are bytes, but we should defend against implementations
195  that use unicode characters. This means that we might have to skip 2, 4 or 8
196  bytes to get to the next U+001b unicode character. The following code also is
197  willing to skip 0 or 6 bytes between the U+001b's, which is malformed PDF, but there
198  is no harm in accepting it.
199  */
200  for( int i = 0; i < 5; ++i )
201  {
202  m_done = ( m_str == m_strEnd );
203  if( m_done )
204  return;
205  m_currVal = GetNextUTF16WordVal();
206  if ( m_currVal == 0x1b )
207  {
208  m_done = ( m_str == m_strEnd );
209  if( m_done )
210  return;
211  m_currVal = GetNextUTF16WordVal();
212  break;
213  }
214  }
215  }
216  }
217  else
218  {
219  m_currVal = gPDFDocToUCS2Map[static_cast<unsigned char>( *m_str )];
220  ++m_str;
221  }
222 }

224 template<typename app_traits>
225 std::uint16_t UTF16StringConstIterator<app_traits>::GetNextUTF16WordVal( void )
226 {
227  tetraphilia_assert( m_isUnicodeUTF16BE );
228  std::uint16_t result = 0;
229  if ( m_str != m_strEnd )
230  {
231  result = static_cast<unsigned char>( *m_str );
232  ++m_str;
233  if ( m_str != m_strEnd )
234  {
235  result = result * 256 + static_cast<unsigned char>( *m_str );
236  ++m_str;
237  }
238  else
239  result = '?'; // string ended in single byte instead of 16-bit value
240  }
241 
242  return result;
243 }

327 template<typename app_traits>
328 void UTF32StringConstIterator<app_traits>::GetNextVal( void )
329 {
330  if( Done() )
331  return;
332 
333  m_currVal = *m_utf16It;
334  ++m_utf16It;
335 
336  if ( 0xD800 <= m_currVal && m_currVal <= 0xDBFF )
337  {
338  if ( !Done() )
339  {
340  std::uint32_t secondValueOfPair = *m_utf16It; // 0xDC00 <= secondValueOfPair <= 0xDFFF or the string is corrupted
341  ++m_utf16It;
342 
343  if ( 0xDC00 <= secondValueOfPair && secondValueOfPair <= 0xDFFF )
344  m_currVal = ((m_currVal - 0xD800) << 10) + (secondValueOfPair - 0xDC00) + 0x10000; // first of pair contains the most significant 10 bits, the second the least significant 10 bits of a 20 bit value. Then 0x10000 must be added to that 20 bit result.
345  else
346  m_currVal = '?'; // corrupted string
347  }
348  else
349  m_currVal = '?'; // corrupted string
350  }

354 // buffer size (including space for the terminator) to hold the entire UTF16 string.
355 // If dstBufSize is not zero: dstBufSize is NOT reset, and a nul-terminated string is copied to the buffer (truncated if necessary).
356 template<typename app_traits>
357 void PDFTextStringToUTF16(std::uint16_t* dstBuf, size_t& dstBufSize, const store::String< store::StoreObjTraits<app_traits> >& srcStr)
358 {
359  typedef typename app_traits::pdf_traits pdf_traits;
360 
361  std::uint16_t* dstIt = dstBuf;
362  std::uint16_t* dstEnd = dstBuf + (dstBufSize - 1);
363 
364  size_t n = 0;
365  UTF16StringConstIterator<app_traits> it( srcStr );
366  while ( !it.Done() && (dstIt < dstEnd || dstBufSize == 0) )
367  {
368  std::uint16_t val = *it;
369 
370  /*
371  PDF_SLANG See Watson bug#1482610
372 
373  It isn't entirely clear what the right answer here is. The PDF Spec fully describes PDFDocEncoding,
374  which is the encoding used for srcStr. In the documentation certain code points are listed as Undefined.
375  For those Undefined code points some have a Unicode code point and some do not. There isn't a clear
376  statement about how one should treat Undefined codes in a PDFDocEncoded string. Desktop Acrobat appears
377  to substitute spaces for any Undefined code, whether it has a Unicode value or not. So in permissive
378  builds we are doing that too.
379  */
380  if ( pdf_traits::permissive && (val <= 0x17 || val == UTF16StringConstIterator<app_traits>::UCS2_UNDEFINED_CHAR) )
381  val = 0x0020;
382 
383  if ( dstBufSize )
384  *dstIt++ = val;
385  else
386  ++n;
387  ++it;
388  }
389 
390  // if unicode string was encoded properly in the PDF file, this should prevent
391  // the first value of a surrogate pair from appearing at the end of the (truncated) string.
392  if ( !it.Done() && dstIt > dstBuf && 0xDC00 <= *it && *it <= 0xDFFF )
393  {
394  tetraphilia_assert( dstBufSize != 0 ); // otherwise !it.Done() would not be true
395  --dstIt;
396  --n;
397  }
398 
399  if ( dstBufSize )
400  *dstIt = 0;
401  else
402  dstBufSize = n + 1;
403 }
404 
405 // If dstBufSize is passed in as zero: No copying is done and dstBufSize is reset to the minimum required
406 // buffer size (including space for the terminator) to hold the entire UTF-32 string.
407 // If dstBufSize is not zero: dstBufSize is NOT reset, and a nul-terminated string is copied to the buffer (truncated if necessary).
408 template<typename app_traits>
409 void PDFTextStringToUTF32(std::uint32_t* dstBuf, size_t& dstBufSize, const store::String< store::StoreObjTraits<app_traits> >& srcStr)
410 {
411  std::uint32_t* dstIt = dstBuf;
412  std::uint32_t* dstEnd = dstBuf + (dstBufSize - 1);
413 
414  size_t n = 0;
415  UTF32StringConstIterator<app_traits> it( srcStr );
416  while ( !it.Done() && (dstIt < dstEnd || dstBufSize == 0) )
417  {
418  std::uint32_t val = *it;
419  if ( dstBufSize )
420  *dstIt++ = val;
421  else
422  ++n;
423  ++it;
424  }
425 
426  if ( dstBufSize )
427  *dstIt = 0;
428  else
429  dstBufSize = n + 1;
430 }
