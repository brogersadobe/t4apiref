29 /*
30  When composing a t4 params-based interface, you need to provide a unique class for each
31  parameter. This class is used as a key when describing and building parameter lists.
32  These classes must descend from parameter_of_type, specialized on the value type of
33  the parameter. For example, a bounding_rect parameter might be defined in this way:
34 
35  struct bounding_rect : parameter_of_type<real_rect> {};
36 
37  These types are called "parameter tags."
38 */
39 template<typename Value>
40 struct parameter_of_type;
41 
42 /*
43  When invoking parameter_description, use required_parameter to indicate that a
44  parameter is required. If you want to provide a default value, use function call
45  syntax. Example:
46 
47  required_parameter<bounding_rect>
48 
49  Indicates that the bounding_rect parameter is required.
50 
51  required_parameter<bounding_rect>( empty<real_rect>() )
52 
53  Indicates that the bounding_rect parameter is required, but if the client does not
54  provide it then the empty rectangle is presumed.
55 */
56 template<typename Parameter>
57 static constexpr auto required_parameter = parameter_detail::required_parameter_t<Parameter>();
58 
59 /*
60  When invoking parameter_description, use optional_parameter to indicate that a
61  parameter is optional. The client can provide a value for this parameter, but
62  doesn't have to. Example:
63 
64  optional_parameter<bounding_rect>
65 
66  Indicates that the bounding_rect parameter is optional.
67 */
68 template<typename Parameter>
69 static auto optional_parameter = parameter_detail::optional_parameter_t<Parameter>();

71 /*
72  Create a "description," or schema, of parameters for a particular API. Each argument
73  must be of one of the following forms:
74 
75  required_parameter<tag>
76 
77  required_parameter<tag>( default_value )
78 
79  optional_parameter<tag>
80 
81  Where "tag" is a parameter tag.
82 
83  Example:
84 
85  struct width : parameter_of_type<int> {};
86  struct height : parameter_of_type<int> {};
87  struct opacity : parameter_of_type<real> {};
88  struct border_color : parameter_of_type<rgb_color> {};
89 
90  auto rectangle_param_desc = parameter_description(
91  required_parameter<width>,
92  required_parameter<height>,
93  required_parameter<opacity>( 1.0 ),
94  optional_parameter<border_color> );
95 
96  The results of two different calls to parameter_description() can be merged using the +
97  operator.
98 
99  Example:
100 
101  struct width : parameter_of_type<int> {};
102  struct height : parameter_of_type<int> {};
103  auto rectangle_desc = parameter_description(
104  required_parameter<width>,
105  required_parameter<height> );
106 
107  struct depth : parameter_of_type<int> {};
108  auto rectangular_prism_desc = rectangle_desc + parameter_description( required_parameter<depth> );
109 
110  Parameter descriptors in the second argument override those in the first.
111 */
112 template<typename... TArgs>
113 constexpr auto parameter_description( TArgs&&... args );
