45 template<typename Device>
46 class fixed_block_size_caching_data_store_base
47 {
48 protected:
49  Device m_device;
50 
51 private:
52  size_t m_block_granularity;
53  cache<data_store_position, data_block> m_block_cache;
54 
55 public:
56  fixed_block_size_caching_data_store_base( Device device, size_t block_granularity )
57  : m_device( std::move( device ) ),
58  m_block_granularity( block_granularity )

60 
61  friend data_block get_block( fixed_block_size_caching_data_store_base const& ths, data_store_position const& pos )
62  {
63  data_store_position block_base = pos / ths.m_block_granularity * ths.m_block_granularity;
64 
65  auto create = [&]( data_store_position const& pos2 )
66  {
67  assert( pos2 % ths.m_block_granularity == 0 );
68  return escape_get_block( ths.m_device, ths.m_block_cache, pos2, ths.m_block_granularity );
69  };
70 
71  data_block result = *ths.m_block_cache.get( block_base, create );
72 
73  // If this assert fires the implementation of create_aligned_block_impl failed to verify that
74  // the block it produced either includes the last byte in the data store or is m_block_granularity
75  // bytes in size.
76  assert( block_base <= pos && pos < block_base + ths.m_block_granularity );
77 
78  size_t ptr_adj = std::min( result.get_size(), static_cast<size_t>( pos - block_base ) );
79  result.truncate_front( ptr_adj );
80  return result;
81  }

86  }

89  {
90  return escape_get_path( ths.m_device );
91  }
92 };

145 template<typename Device>
146 class variable_block_size_caching_data_store
147 {
148  cache<data_store_position, data_block> m_block_cache;
149 
150  Device m_device;
151 
152  friend class variable_block_size_block_stasher;
153 
154 public:
155  variable_block_size_caching_data_store( Device d )
156  : m_device( std::move( d ) )
157  {}
158 
159  friend data_block get_block( variable_block_size_caching_data_store const& ths,
160  data_store_position const& pos )
161  {
162  // Get the map iterator with the largest position less than or equal to the requested position
163  auto q = [&pos]( auto const& map )
164  {
165  auto result = std::upper_bound( map.begin(), map.end(), pos, []( data_store_position const& lhs, cache_entry<data_store_position, data_block> const& rhs )
166  { return lhs < rhs.get_key(); } );
167  if ( result == map.begin() )
168  return map.end();
169  --result;
170  assert( result->get_key() <= pos );
171  if ( result->get_key() + result->get_shared_value()->get_size() <= pos )
172  return map.end();
173  return result;
174  };
175 
176  auto create = [&]( data_store_position const& pos2 )
177  {
178  variable_block_size_block_stasher<Device> stasher( ths );
179  auto result = create_block_containing( ths.m_device, pos2, stasher );
180  return ths.m_block_cache.template make_cache_entry<data_block>( result.first, std::move( result.second ) );
181  };
182 
183  auto e = ths.m_block_cache.query( pos, q, create );
184  auto v = *e->get_shared_value();
185  if ( v.get_size() == 0 )
186  return v;
187  else
188  {
189  assert( e->get_key() <= pos && pos < e->get_key() + v.get_size() );
190  v.truncate_front( static_cast<size_t>( pos - e->get_key() ) );
191  return v;
192  }
193  }
194 
195  friend boost::optional<boost::filesystem::path> get_path( variable_block_size_caching_data_store const& ths )
196  {
197  return escape_get_path( ths );
198  }
