28 /*
29  Template wrapper class used to decorate a parameter or return type as guaranteed to
30  be non-null. Example uses:
31 
32  never_null<shared_ptr<foobar>> get_required_foobar( void )
33  {
34  return make_shared<foobar>();
35  }
36 
37  An instance of never_null exposes operator-> and operator* in a way that forwards to the
38  contained pointer.
39 
40  An instance of never_null<Ptr> may be used wherever a Ptr is expected.
41 
42  Ptr must be a raw pointer type, or a class that implements operator->(), e.g., shared_ptr
43  or unique_ptr.
44 
45  It is impossible to default-construct a never_null, because the resulting object would be
46  null, which sort of undermines the whole point. However it is still possible for a never_null
47  instance to exist with a null pointer inside, after it has been moved. never_null instances
48  in this state must only be assigned to or destroyed.
49 */
50 template<typename Ptr>
51 class never_null : public ptrs_detail::ptr_base<Ptr>::type
52 {
53  typedef typename ptrs_detail::ptr_base<Ptr>::type parent_type;
54  typedef typename ptrs_detail::pointee_type<Ptr>::type pointee_type;
55 
56 public:
57  // Copy ctor
58  never_null( never_null const& rhs )
59  : parent_type( rhs )
60  {}
61 
62  // Move ctor
63  never_null( never_null&& rhs )
64  : parent_type( std::move( rhs ) )
65  {}
66 
67  /*
68  Construct a never_null from an instance of its contained type.
69  */
70  never_null( Ptr p )
71  : parent_type( std::move( p ) )
72  {
73  assert( *this );
74  }
75

77  Construct a never_null from a type convertible to an instance of its contained type.
78  */

80  never_null( Ptr2 p, std::enable_if_t<std::is_convertible<Ptr2, Ptr>::value, int> = 0 )
81  : parent_type( std::move( p ) )

84  }

91  }
