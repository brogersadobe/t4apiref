65 inline real_coord mag_squared( real_point const& scale )
66 {
67  return get_x( scale ) * get_x( scale ) + get_y( scale ) * get_y( scale );
68 }

70 inline real_coord scale_delta( real_point const& target, real_point const& scale )
71 {
72  auto target_m = mag_squared( target );
73  auto scale_m = mag_squared( scale );
74  // If the target zoom > this zoom, we take the square root to introduce a slight bias to magnification over minification
75  return 1 - ( ( target_m > scale_m ) ? ( scale_m / target_m ) : std::sqrt( target_m / scale_m ) );
76 }

78 template<typename TileCacheClient>
79 auto tile_cache<TileCacheClient>::find_best_match(
80  cache_map const& map, int tier, real_point const& scale, int_point const& grid_loc ) const -> boost::optional<tile_match>
81 {
82  tile_key min_key{ tier, grid_loc, make<real_point>( std::numeric_limits<real_coord>::min(), std::numeric_limits<real_coord>::min() ) };
83  tile_key max_key{ tier, grid_loc, make<real_point>( std::numeric_limits<real_coord>::max(), std::numeric_limits<real_coord>::max() ) };
84 
85  auto it = map.upper_bound( min_key );
86  auto end = map.upper_bound( max_key );
87 
88  boost::optional<tile_match> result;
89  real_coord nearest_scale_delta = std::numeric_limits<real_coord>::max();
90  for ( ; it != end; ++it )
91  {
92  auto cp = it->get_shared_value();
93  auto pb = cp->get_progress_and_bitmap();
94 
95  // Skip tiles into which no rendering has occurred (this allows clients to presume
96  // that their bitmaps were all produced by make_bitmap())
97  if ( pb.first == initial_progress<TileCacheClient>() )
98  continue;
99 
100  if ( result )
101  {
102  if ( result->m_progress > pb.first )
103  continue;
104  else
105  {
106  auto sd = scale_delta( scale, it->get_key().m_scale );
107  if ( result->m_progress == pb.first && sd > nearest_scale_delta )
108  continue;
109  nearest_scale_delta = sd;
110  }
111  }
112  else
113  nearest_scale_delta = scale_delta( scale, it->get_key().m_scale );
114 
115  result.emplace( tile_key{ tier, grid_loc, it->get_key().m_scale }, cp, pb.first, pb.second, nearest_scale_delta );
116 
117  // Perfect tile? Exit the loop
118  if ( result->m_progress == complete_progress<TileCacheClient>() && it->get_key().m_scale == scale )
119  break;
120  }
121 
122  return result;
123 }

125 /*
126  Recursive component of the pile_tiles algorithm. This function finds and places the best
127  tiles for a given subgrid at a specified tier. It recursively calls itself to produce this
128  result.
129 
130  Returns: bool indicating whether or not this call completely covered the grid location
131 
132  used_tiles: a vector containing all the tile placements
133  map: the cache
134  tiling: the tiling for the entire tile-piling
135  scale: the actual scale at which we're rendering
136  clip_rect: the aperture in device space
137  tier: the current tier
138  subgrid: the subgrid (in the tiling defined by tier) being piled
139  best_supertile: the most complete tile in a tier < the current tier that covers this
140 */
141 template<typename TileCacheClient>
142 bool tile_cache<TileCacheClient>::pile_tiles_helper(
143  std::vector<tile_placement, no_purge_allocator<tile_placement>>& used_tiles,
144  cache_map const& map,
145  tile_cache_detail::tiling const& tiling,
146  real_point const& scale,
147  int_rect const& clip_rect,
148  int tier,
149  int_rect subgrid,
150  tile_match const* best_supertile )
151 {
152  int ideal_tier = tile_cache_detail::scale_to_tier( scale );
153  if ( tier >= ideal_tier + 4 )
154  return false;
155 
156  auto st = tiling.get_scaled_tiling( scale, tier );
157  int_rect clipped_tile_loc_rect = st.device_rect_to_subgrid( clip_rect );
158  subgrid = intersection( clipped_tile_loc_rect, subgrid );
159 
160  int_point grid_loc;
161  int_rect dst_rect;
162 
163  bool covered = true;
164 
165  // Walk the subgrid
166  get_y( grid_loc ) = get_ymin( subgrid );
167  set_ymin( dst_rect, st.get_row_start( get_y( grid_loc ) ) );
168  for ( ; get_y( grid_loc ) < get_ymax( subgrid ); ++get_y( grid_loc ) )
169  {
170  set_ymax( dst_rect, st.get_row_start( get_y( grid_loc ) + 1 ) );
171  get_x( grid_loc ) = get_xmin( subgrid );
172  set_xmin( dst_rect, st.get_column_start( get_x( grid_loc ) ) );
173  for ( ; get_x( grid_loc ) < get_xmax( subgrid ); ++get_x( grid_loc ) )
174  {
175  set_xmax( dst_rect, st.get_column_start( get_x( grid_loc ) + 1 ) );
176 
177  // Find the best tile in the cache at this exact grid location in this tier.
178  auto best_match = find_best_match( map, tier, scale, grid_loc );
179  tile_match const* best_supertile_this_gridloc = best_supertile;
180 
181  // Figure out the best tile coverage for this cell in the subgrid.
182 
183  // This will be set to true if this cell gets completely covered by tiles,
184  // either at this tier or at higher tiers.
185  bool this_tile_covered = false;
186 
187  // True if we found a tile at this grid location and it's "better" than what
188  // was placed their by the lower tiers.
189  bool this_tile_improves = false;
190  if ( best_match )
191  {
192  // This tile improves on what's already there if:
193  // 1) There isn't anything already there, or
194  // 2) This tile has more progress than what's already there, or
195  // 3) This tile has the same amount of progress but is closer to the ideal zoom
196  this_tile_improves = !best_supertile_this_gridloc
197  || best_match->m_progress > best_supertile_this_gridloc->m_progress
198  || ( best_match->m_progress == best_supertile_this_gridloc->m_progress
199  && best_match->m_scale_delta < best_supertile_this_gridloc->m_scale_delta );
200  }
201 
202  if ( this_tile_improves )
203  best_supertile_this_gridloc = best_match.get_ptr();
204 
205  // We don't recurse more than 3 tiers beyond the ideal_tier (the tier that matches the
206  // scale of the requested piling). We also don't bother if the tile we got from this tier
207  // is already a complete rendering, and we're already at (or past) the ideal tier.
208  if ( tier < ( ideal_tier + 2 )
209  && ( tier <= ideal_tier
210  || !best_match
211  || best_match->m_progress != complete_progress<TileCacheClient>() ) )
212  {
213  int_rect next_tier_subgrid = make_with_size<int_rect>( get_x( grid_loc ) * 2, get_y( grid_loc ) * 2, 2, 2 );
214  this_tile_covered = pile_tiles_helper( used_tiles, map, tiling, scale, clip_rect, tier + 1, next_tier_subgrid, best_supertile_this_gridloc );
215  }
216 
217  if ( !this_tile_covered && this_tile_improves )
218  {
219  this_tile_covered = true;
220  best_match->m_tile->mark_used_now();
221  used_tiles.emplace_back( std::move( *best_match ), dst_rect );
222  }
223 
224  covered = covered && this_tile_covered;
225  set_xmin( dst_rect, get_xmax( dst_rect ) );
226  }
227  set_ymin( dst_rect, get_ymax( dst_rect ) );
228  }
229 
230  return covered;
231 }

239 template<typename TileCacheClient>
240 template<typename F>
241 tile_collection<TileCacheClient> tile_cache<TileCacheClient>::pile_tiles( real_point const& scale, int_rect const& clip_rect, F f )
242 {
243  typedef typename decltype(m_cache)::map_type map_type;
244 
245  // If we haven't even been told any page geometry yet, there can't possibly be
246  // any tiles
247  auto opt_tiling = m_tiling.load();
248  if ( !opt_tiling )
249  return {};
250 
251  auto tiling = *opt_tiling;
252 
253  auto placements = m_cache.query( [this, &scale, &clip_rect, &tiling]( map_type const& map )
254  {
255  std::vector<tile_placement, no_purge_allocator<tile_placement>> result;
256  if ( map.empty() )
257  return result;
258 
259  // tile cache keys are sorted by increasing tier, and we want to start at the lowest one available
260  int start_tier = map.begin()->get_key().m_tier;
261  auto st = tiling.get_scaled_tiling( scale, start_tier );
262  auto subgrid = st.device_rect_to_subgrid( clip_rect );
263  pile_tiles_helper( result, map, tiling, scale, clip_rect, start_tier, subgrid,
264  nullptr );
265  return result;
266  } );
267 
268  tile_collection result;
269  result.m_tiles.reserve( placements.size() );
270  for( auto it = placements.rbegin(); it != placements.rend(); ++it )
271  {
272  auto& tile = it->m_match.m_tile;
273  tile->mark_used_now();
274  result.m_tiles.push_back( tile );
275  f( it->m_match.m_key, it->m_match.m_bitmap, it->m_location );
276  }
277  return result;
278 }
