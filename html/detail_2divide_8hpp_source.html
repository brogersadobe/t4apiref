30 template<typename Result, typename T1, typename T2, typename = void>
31 struct divide_impl_helper
32 {
33  /*
34  This is the fully-general version of divide that works for all combinations of integer types.
35  */
36  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
37  {
38  if ( y == 0 )
39  return {};
40 
41  if ( x == 0 )
42  return 0;
43 
44  // If the result type is unsigned and the result is negative, the only valid case is when the result is 0
45  const bool is_negative_x = is_negative( x );
46  const bool negative_result = is_negative_x != is_negative( y );
47  if ( is_unsigned<Result>::value && negative_result && sm_detail::abs( x ) >= sm_detail::abs( y ) )
48  return {};
49 
50  using up_t1 = typename std::make_unsigned<T1>::type;
51  using up_t2 = typename std::make_unsigned<T2>::type;
52  using up_result = typename std::make_unsigned<Result>::type;
53 
54  // If result is positive we can treat both operands as unsigned
55  if ( !negative_result )
56  return divide_impl_helper<Result, up_t1, up_t2>()( sm_detail::abs( x ), sm_detail::abs( y ) );
57 
58  // In case of negative result we still can piggyback on unsigned version and negate result in the end
59  auto maybe_safe_result = divide_impl_helper<up_result, up_t1, up_t2>()( sm_detail::abs( x ), sm_detail::abs( y ) );
60  if ( maybe_safe_result )
61  {
62  if ( !sm_detail::high_bit_set( *maybe_safe_result ) )
63  return static_cast<Result>( 0 - *maybe_safe_result );
64  // Corner case: the only way to not overflow if high bit set
65  // is if result is negative power of 2
66  else if ( static_cast<up_result>( *maybe_safe_result << 1 ) == 0 )
67  return std::numeric_limits<Result>::min();
68  }
69 
70  return {};
71  }
72 };
