102  T const& operator[]( Key const& key ) const
103  {
104  auto lt = [this]( Key lhs, node_type const& rhs ) { return this->get_cmp()( lhs, rhs.m_stop ); };
105  auto it = std::upper_bound( get_ranges().begin(), get_ranges().end(), key, lt );

110  // Return the first key in [start, end) for which pred( (*this)[key] ) returns true.
111  // Returns end if no such key exists.
112  template<typename Predicate>
113  Key find( Key start, Key end, Predicate const& pred ) const
114  {
115  auto it = find_range_containing( start );
116 
117  while ( it != get_ranges().end() )
118  {
119  if ( pred( it->m_val ) )
120  return std::move( start );
121  start = it->m_stop;
122  if ( start >= end )
123  return end;
124  ++it;
125  }
126 
127  return start;
128  }

130  template<typename Operation>
131  void for_each_range( Key const& start, Key const& stop, Operation const& op ) const
132  {
133  auto it = find_range_containing( start );
134 
135  Key curr = start;
136  while( true )
137  {
138  if ( curr >= stop )
139  break;
140  Key next = std::min( it->m_stop, stop, get_cmp() );
141  op( curr, next, it->m_val );
142  curr = next;
143  }
144  }

148  {

158  void set_range( Key const& start, Key stop, T const& val )
159  {
160  if ( stop <= start )
161  return;
162 
163  auto lt = [this]( node_type const& lhs, Key rhs ) { return this->get_cmp()( lhs.m_stop, rhs ); };
164 
165  // Get the first node whose m_stop >= start
166  auto replacement_start = std::lower_bound( get_ranges().begin(), get_ranges().end(), start, lt );
167  assert( replacement_start != get_ranges().end() ); // can't happen because the terminal node must be max()
168 
169  // Get the first node whose m_stop > stop
170  auto lt2 = [this]( Key lhs, node_type const& rhs ) { return this->get_cmp()( lhs, rhs.m_stop ); };
171  auto replacement_stop = std::upper_bound( replacement_start, get_ranges().end(), stop, lt2 );
172 
173  if ( replacement_start->m_val != val )
174  {
175  if ( replacement_start->m_stop > stop )
176  {
177  // special case: range is splitting an existing range. Two new ranges need to be created.
178  get_ranges().insert( replacement_start, { { start, replacement_start->m_val }, { stop, val } } );
179  return;
180  }
181 
182  replacement_start->m_stop = start;
183  ++replacement_start;
184  }
185 
186  if ( replacement_stop != get_ranges().end() && replacement_stop->m_val == val )
187  {
188  stop = replacement_stop->m_stop;
189  ++replacement_stop;
190  }
191 
192  if ( replacement_start == replacement_stop )
193  get_ranges().insert( replacement_start, { stop, std::move( val ) } );
194  else
195  {
196  *replacement_start++ = { stop, std::move( val ) };
197  get_ranges().erase( replacement_start, replacement_stop );
198  }
199  }
