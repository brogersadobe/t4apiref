42 inline aligner& aligner::push( size_t sz, size_t alignment )
43 {
44  assert( alignment != 0 && align_detail::is_power_of_two( alignment ) );
45  size_t bytes_to_next_aligned_boundary;
46 
47  if ( alignment > m_max_alignment )
48  {
49  bytes_to_next_aligned_boundary = add<size_t>( m_bytes_to_next_max_aligned_boundary, alignment - m_max_alignment );
50  m_bytes_to_next_max_aligned_boundary = bytes_to_next_aligned_boundary;
51  m_max_alignment = alignment;
52  }
53  else

57  m_total = add<size_t>( m_total, adj );

64  unsigned quantity anyway).
65  */
66  m_bytes_to_next_max_aligned_boundary += ( m_max_alignment - adj );
67  m_bytes_to_next_max_aligned_boundary %= m_max_alignment;
