31 /*
32  Class for asserting exclusive access to a resource managed by a cooperative_context.
33 
34  A cooperative_lock can be used to coordinate access to a collection of state which
35  isn't preemptively multithreaded, but _is_ cooperatively multithreaded; i.e., multiple
36  threads can access the state in an interleaved manner. We manage this interleaving by
37  expecting the active threads to explicitly "yield" to each other by calling cooperative_lock::yield().
38 
39  During a call to yield() one or more threads will get a chance to execute for until their
40  next call to yield() (or cooperative_condition_variable::wait, cooperative_mutex::lock, etc.)
41 */
42 class cooperative_lock : boost::noncopyable, public boost::intrusive::list_base_hook<>
43 {
44  cooperative_context& m_context;
45  condition_variable m_condition;
46  boost::intrusive::list<cooperative_detail::yield_hook_base> m_yield_hooks;
47 
48  friend class cooperative_condition_variable;
49  friend class cooperative_context;
50  friend class cooperative_unlock;
51  friend class cooperative_mutex;
52  friend class cooperative_detail::lock_taker;
53  friend class cooperative_detail::lock_waiter;
54  friend class cooperative_detail::lock_yielder;
55  template<typename Pre, typename Post, typename> friend class cooperative_detail::yield_hook;
56 
57 public:
58  /*
59  Construct a cooperative_lock on the provided cooperative_context. Other threads
60  that construct a cooperative_lock on the same cooperative_context object will
61  block until this thread calls one of the yielding APIs on this cooperative_context
62  (e.g., cooperative_lock::yield(), cooperative_condition_variable::wait(), cooperative_mutex::lock())
63  */
64  cooperative_lock( cooperative_context& context );
65 
66  /*
67  Release the lock on the cooperative_context.
68  */
69  ~cooperative_lock( void );
70 
71  /*
72  Yield control to other threads that are sharing this lock's cooperative_context.
73  */
74  void yield( void );
75 
76  /*
77  Add a pre-yield hook that is called before this cooperative_lock yields control to another thread sharing
78  its context. The return type is noncopyable and should live on the stack; the hook will be invoked until the returned
79  object is destroyed.
80 
81  pre's signature should be:
82 
83  <some_type> pre()
84 
85  if some_type is void, post's signature should be:
86 
87  post()
88 
89  otherwise, it should be:
90 
91  post( some_type&& )
92 
93  You can use the return value of pre() to store state local to the thread which might be altered by other threads.
94 
95  The value returned by this function has no public API; as long as it is in scope the yield hooks will be installed.
96  */
97  using yield_hooks = unique_ptr<cooperative_detail::yield_hook_base>;
98 
99  template<typename Pre, typename Post>
100  yield_hooks add_yield_hooks( Pre pre, Post post );
101 
102 private:
103  void lock( void );
104 
105  void unlock( void );

108

129 /*
130  Explicitly release a cooperative_lock's hold on the cooperative_context.
131 */
132 class cooperative_unlock : boost::noncopyable
133 {
134  cooperative_lock& m_lock;
135 
136 public:
137  /*
138  For the lifetime of this cooperative_unlock, lock will not "own" the cooperative_context
139  and other cooperative_lock-holders may access state protected by the cooperative_context.
140 
141  This is useful before calling future::wait and/or thread::join.
142  */
143  cooperative_unlock( cooperative_lock& lock )

159 /*
160  Object that manages exclusive, interleaved access to state.
161 
162  A typical use will involve the creation of a single, global cooperative_context, and context_locks
163  on every thread that intends to use the state governed by that cooperative_context. One could use
164  this as the basis for a system of fibers, although there are more efficient options than this.
165 
166  cooperative_context/cooperative_lock is intended to support T3's legacy cooperative multithreading
167  model.
168 */

174  boost::intrusive::list<cooperative_lock> m_ready_queue;

286 /*
287  Synchronization barrier for cooperative_lock. Expected behavior is that a bunch of threads
288  will all take a context_lock, then call barrier.wait(). They will all wake up as soon as the
289  last thread enters barrier.wait(). The number of threads to wait is passed into cooperative_barrier's
290  ctor.
291 */
292 class cooperative_barrier
293 {
294  size_t m_count;
295  cooperative_condition_variable m_condition;
296 
297 public:
298  /*
299  Create a cooperative_barrier waiting for count threads to hit it
300  */
301  cooperative_barrier( cooperative_context& context, size_t count )
302  : m_count( count ),
303  m_condition( context )
304  {}
305 
306  /*
307  Wait until all other threads have hit the barrier too
308  */
309  void wait( cooperative_lock& lock )
310  {
311  if ( --m_count == 0 )
312  m_condition.notify_all();
313  else
314  m_condition.wait( lock );
315  }
316 };
