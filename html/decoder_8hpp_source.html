30 template<typename signal_traits, typename T>
31 struct DecoderLoop
32 {
33  static void Do( typename signal_traits::signal_type* _dst,
34  const unsigned char* src,
35  size_t numSrcBytes,
36  const void* _table )
37  {
38  T* dst = reinterpret_cast<T*>(_dst);
39  const T* table = static_cast<const T*>(_table);
40  for ( size_t i = 0; i < numSrcBytes; i++, src++, dst+=2 ) {
41  unsigned char s = *src;
42  dst[0] = table[ s >> 4 ];
43  dst[1] = table[ s & 15 ];
44  }
45  }
46 
47 };

62 template<>
63 struct TruncaterLoop<imaging_model::ByteSignalTraits<t4::app_traits>>
64 {
65  static void Do( unsigned char* dst,
66  const unsigned char* src,
67  size_t numSrcBytes,
68  const void* /*table*/ )
69  {
70  tetraphilia_assert( (numSrcBytes & 1) == 0 );
71  for ( size_t i = 0; i < numSrcBytes; i+=2, src+=2, dst++ )
72  dst[0] = src[0];
73  }
74 
75  static void InvDo( unsigned char* dst,
76  const unsigned char* src,
77  size_t numSrcBytes,
78  const void* /*table*/ )
79  {
80  tetraphilia_assert( (numSrcBytes & 1) == 0 );
81  for ( size_t i = 0; i < numSrcBytes; i+=2, src+=2, dst++ )
82  dst[0] = 255 - src[0];
83  }
84 };

99 template<>
100 struct ClipToLowLoop<imaging_model::ByteSignalTraits<t4::app_traits>>
101 {
102  static void Do( unsigned char* dst,
103  const unsigned char* src,
104  size_t numSrcBytes,
105  const void* /*table*/ )
106  {
107  tetraphilia_assert( (numSrcBytes & 1) == 0 );
108  for ( size_t i = 0; i < numSrcBytes; i+=2, src+=2, dst++ )
109  dst[0] = src[0] ? 0xff : src[1];
110  }
111 
112  static void InvDo( unsigned char* dst,
113  const unsigned char* src,
114  size_t numSrcBytes,
115  const void* /*table*/ )
116  {
117  tetraphilia_assert( (numSrcBytes & 1) == 0 );
118  for ( size_t i = 0; i < numSrcBytes; i+=2, src+=2, dst++ )
119  dst[0] = 255 - (src[0] ? static_cast<unsigned char>(0xff) : src[1]);
120  }

132 template<typename signal_traits>
133 struct Decoder
134 {
135  using signal_type = typename signal_traits::signal_type;
136 
137  Decoder( size_t numBitsPerComponent, bool zeroExtend, bool invert )
138  {
139  tetraphilia_assert( !zeroExtend || !invert );
140  switch( numBitsPerComponent )
141  {
142  case 1:
143  m_table = zeroExtend ? g1BitDecodeTableZ.m_table : (invert ? g1BitDecodeTableInv.m_table : g1BitDecodeTableR.m_table);
144  m_proc = DecoderLoop<signal_traits, std::uint32_t>::Do;
145  break;
146  case 2:
147  m_table = zeroExtend ? g2BitDecodeTableZ.m_table : (invert ? g2BitDecodeTableInv.m_table : g2BitDecodeTableR.m_table);
148  m_proc = DecoderLoop<signal_traits, std::uint16_t>::Do;
149  break;
150  case 4:
151  m_table = zeroExtend ? g4BitDecodeTableZ.m_table : (invert ? g4BitDecodeTableInv.m_table : g4BitDecodeTableR.m_table);
152  m_proc = DecoderLoop<signal_traits, std::uint8_t>::Do;
153  break;
154  case 16:
155  /*
156  PDF_SLANG
157 
158  Watson bug#1267284 contains an image that specifies 16 bits per component, and an indexed
159  color table. There is nothing in the PDF spec that makes this illegal, but according to
160  the spec a indexed color space's "hival" (which essentially tells you the maximum legal
161  value of a color index) is required to be less than 256. This implies that, while it might
162  be legal to have 16 bit values in the image data, the high 8 bits of each sample would need
163  to be zero. So there is no advantage to using 16 bit values. A correct implementation of
164  a viewer would clip all values larger than hival to hival. However, the desktop Acrobat
165  viewer does not do this. It seems to read the high 8 bits of each value as if they were
166  the correct value. This is a reasonable thing to do if the color values are 16 bit, but
167  not indexed. But if they are index values this can't be a correct interpretation of the
168  spec. In permissive builds we have the same behavior as desktop Acrobat.
169  */
170  m_table = nullptr;
171  #ifndef TETRAPHILIA_PDF_PERMISSIVE
172  if ( zeroExtend )
173  m_proc = invert ? ClipToLowLoop<signal_traits>::InvDo : ClipToLowLoop<signal_traits>::Do;
174  else
175  #endif
176  m_proc = invert ? TruncaterLoop<signal_traits>::InvDo : TruncaterLoop<signal_traits>::Do;
177  break;
178  default:
179  throw t4::corrupt_data();
180  }
181  }

404 struct ArrayDecoder
405 {
406  ArrayDecoder( const t4::real_coord* decodeValues, size_t numComponents )
407  : m_decodeValues(decodeValues),
408  m_numComponents(numComponents)
409  {
410  m_allSame = true;
411  for ( size_t i = 2; i < numComponents*2; i++ ) {
412  if ( decodeValues[i] != decodeValues[i-2] ) {
413  m_allSame = false;
414  break;
415  }
416  }
417 
418  if ( m_allSame ) {
419  tetraphilia_assert( decodeValues[0] >= 0 && decodeValues[1] >= 0 ); // should be positive
420  t4::real_coord lo = 255 * decodeValues[0];
421  t4::real_coord diff = decodeValues[1] - decodeValues[0];
422  for ( int i = 0; i < 256; i++ ) {
423  // Refer Decode Arrays in the PDF spec:
424  // The output values are linearly interpolated in the range [Dmin Dmax] specified by the decode array.
425  // So, for color range 0 - 255,
426  // y = Dmin + (x * (Dmax - Dmin)/255)
427  // gives the range for y in [0.0 1.0]
428  auto val = real_services::RoundToInt( ArgType<t4::int_coord>(), ( lo + i * diff ));
429 
430  // adjust output values outside the range to the nearest allowed value.
431  if ( val > 255 )
432  val = 255;
433  else if ( val < 0 )
434  val = 0;
435  m_tab[i] = static_cast<unsigned char>( val );
436  }
437  }
438  }
