232  template<typename show_data_type>
233  void HandleTextShow( show_data_type const& showData,
234  pdf::content::TextBBoxInfo & bBoxInfo )
235  {
236  if( this->m_pGState->GetTextFont()->IsUsable() )
237  {
238  auto ctm = this->GetUserToDeviceTransform();
239  if( !IsOrthogonalTransform( ctm ))
240  {
241  // Ignore this text for purposes of underline, etc detection
242  return;
243  }
244 
245  render::ShowInfo showInfo(this->GetGState(),
246  ctm,

249  this->ComputeTextBounds( showInfo, bBoxInfo.m_bounds );
250  auto rotatedBBox = t4::transform( bBoxInfo.m_bounds, m_pageRotation );
251  rotatedBBox = imaging_model::TransformAndBoundRealRect( rotatedBBox, ctm );
252  m_textShows.emplace_back( this->m_curContentPoint, rotatedBBox );
253  }
254  }

490  // Accumulate current figure's bounding box and content range.
491  void ExtendFigure( const t4::real_rect& bBox, bool isAllOrthoLines, t4::real_coord pathLength )
492  {
493  using namespace imaging_model;
494  tetraphilia_assert( m_figureMode != kFigureModeOff );
495  if( !m_figureContentRangeBegun )
496  {
497  m_figureContentRange.m_from = this->m_curContentPoint;
498  m_figureBBox = bBox;
499  m_figureContentRangeBegun = true;
500  m_figureIsOrthoLines = isAllOrthoLines;
501  m_pathLength = pathLength;
502  }
503  else
504  {
505  // If rects are too far apart, treat it as a separate figure
506  if( SeparatedRects( bBox, m_figureBBox ) )
507  {
508  FigureMode curMode = m_figureMode;
509  EndFigure();
510  BeginFigure( curMode );
511  m_figureMarksPage = true;
512  m_figureContentRange.m_from = this->m_curContentPoint;
513  m_figureBBox = bBox;
514  m_figureContentRangeBegun = true;
515  m_figureIsOrthoLines = isAllOrthoLines;
516  m_pathLength = pathLength;
517  }
518  else
519  {
520  m_figureBBox = MakeRectangle<t4::real_rect>(
521  t4::min<t4::real_coord>( get_xmin( m_figureBBox ), get_xmin( bBox ) ),
522  t4::min<t4::real_coord>( get_ymin( m_figureBBox ), get_ymin( bBox ) ),
523  t4::max<t4::real_coord>( get_xmax( m_figureBBox ), get_xmax( bBox ) ),
524  t4::max<t4::real_coord>( get_ymax( m_figureBBox ), get_ymax( bBox ) ) );
525  m_figureIsOrthoLines = m_figureIsOrthoLines && isAllOrthoLines;
526  m_pathLength += pathLength;
527  }
528  }
529  m_figureContentRange.m_to = this->m_curContentPoint;
530  }
531 
532  // RULE: if rects are separated by more than half the max width/height of the two
533  // of them, we call them separated.
534  bool SeparatedRects( const t4::real_rect& bBox1, const t4::real_rect& bBox2 )
535  {
536  if( imaging_model::RectsIntersect( bBox1, bBox2 ) )
537  return false;
538 
539  auto maxY = t4::max<t4::real_coord>( get_height( bBox1 ), get_height( bBox2 ) ) / 2;
540  // 2 above 1
541  if( get_ymin( bBox2 ) - get_ymax( bBox1 ) > maxY )
542  return true;
543  // 1 above 2
544  if( get_ymin( bBox1 ) - get_ymax( bBox2 ) > maxY )
545  return true;
546 
547  auto maxX = t4::max<t4::real_coord>( get_width( bBox1 ), get_width( bBox2 ) ) / 2;
548  // 2 right of 1
549  if( get_xmin( bBox2 ) - get_xmax( bBox1 ) > maxX )
550  return true;
551  // 1 right of 2
552  if( get_xmin( bBox1 ) - get_xmax( bBox2 ) > maxX )
553  return true;
554 
555  return false;
556  }
557 
558  void EndFigure( void )
559  {
560  if ( m_nestedFigureCount > 0 )
561  {
562  --m_nestedFigureCount;
563  return;
564  }
565 
566  if( m_figureContentRangeBegun )
567  {
568  if( m_figureMode == kFigureModeExplicit )
569  {
570  m_figures.emplace_back( m_figureContentRange, m_figureBBox, m_figureIsOrthoLines, m_pathLength );
571  }
572  else if( ( m_figureMode == kFigureModeAutoVector ||
573  m_figureMode == kFigureModeAutoRaster )
574  && m_figureMarksPage )
575  {
576  auto rotatedFigureBBox = t4::transform( m_figureBBox, m_pageRotation );
577  auto width = get_width( rotatedFigureBBox );
578  auto height = get_height( rotatedFigureBBox );
579  if( width > 0 &&
580  height > 0 &&
581  height / width < 20 ) // Don't allow tall, thin auto-figures
582  {
583  m_autoFigures.emplace_back( m_figureContentRange, rotatedFigureBBox, m_figureIsOrthoLines, m_pathLength );
584  }
585  }
586  }
587  m_figureMode = kFigureModeOff;
588  }

590  void HandleImage( const t4::affine_transform& matrix )
591  {
592  using namespace imaging_model;
593  t4::real_rect bBox;
594  MakeRectangle( bBox, 0, 0, 1, 1 );
595  bBox = TransformAndBoundRealRect( bBox, matrix );
596  m_bboxStackImages.push_back( bBox );
597  if( m_figureMode == kFigureModeOff || m_figureMode == kFigureModeAutoVector )
598  {
599  // Treat raster and vector figures as separate figures
600  if( m_figureMode == kFigureModeAutoVector )
601  EndFigure();
602  BeginFigure( kFigureModeAutoRaster );
603  }
604  m_figureMarksPage = true;
605  ExtendFigure( bBox, false, limits<t4::real_coord>::max() );
606  }

608  template<typename show_data_type>
609  void HandleTextShow( show_data_type const& showData,
610  pdf::content::TextBBoxInfo & bBoxInfo )
611  {
612  if( this->m_pGState->GetTextFont()->IsUsable() )
613  {
614  render::ShowInfo showInfo(
615  this->GetGState(),
616  this->GetUserToDeviceTransform(),
617  this->GetTextMatrix() );
618  this->FillShowInfo( showData, showInfo, bBoxInfo );
619  this->ComputeTextBounds( showInfo, bBoxInfo.m_bounds );
620  if( m_figureMode == kFigureModeAutoVector || m_figureMode == kFigureModeAutoRaster )
621  EndFigure();
622  if( m_figureMode == kFigureModeOff )
623  {
624  if( showInfo.GetNumSubstrings() > 0 )
625  {
626  show_info_summary_fmsd_type showInfoSummary( showInfo, this->m_curContentPoint );
627  if( m_priorShowInfoSummary.IsLikelyMultiShow( showInfoSummary ) )
628  m_dropShadows.emplace_back( m_priorTextContentPoint, this->m_curContentPoint );
629  m_priorShowInfoSummary = showInfoSummary;
630  }
631  }
632  else // We're in explicit figure mode
633  {
634  m_figureMarksPage = true;
635  // Assume not all ortho lines and complex figure for now.
636  ExtendFigure( bBoxInfo.m_bounds, false, limits<t4::real_coord>::max() );
637  m_priorShowInfoSummary = show_info_summary_fmsd_type();

643  void HandlePath( pdf::content::DLEntryTreeWalker& walker,
644  bool stroke,
645  const bez_path_iter_type& pathStart,
646  const bez_path_iter_type& pathEnd,
647  pdf::content::FillRule fillRule )
648  {
649  using namespace imaging_model;
650  using namespace t4::literals;
651 
652  if( !stroke && fillRule == content::kNoFill )
653  return;
654  // Documents in the wild sometimes have spurious white rectangles in
655  // the background that we need to ignore:
656  if( ( !stroke || !CurrentColorMarksPage( true /*isStroke*/ ) ) &&
657  ( fillRule == content::kNoFill || !CurrentColorMarksPage( false /*isStroke*/) ) )
658  return;
659  if( m_figureMode == kFigureModeOff || m_figureMode == kFigureModeAutoRaster )
660  {
661  // Treat raster and vector figures as separate figures
662  if( m_figureMode == kFigureModeAutoRaster )
663  EndFigure();
664  BeginFigure( kFigureModeAutoVector );
665  }
666  m_figureMarksPage = true;
667 
668  bool isAllOrthoLines;
669  bool isSingleLine;
670  t4::real_point lineStart;
671  t4::real_point lineEnd;
672  t4::real_coord pathLength;
673  AnalyzePath( pathStart, pathEnd, isAllOrthoLines, isSingleLine, pathLength, lineStart, lineEnd );
674 
675  auto ctm = this->GetUserToDeviceTransform();
676  // We'll only ignore underlines if the current transform has rotation of 0, 90, 180, or 270 degrees.
677  if( isSingleLine && isAllOrthoLines &&
678  TextBBoxConsumer::IsOrthogonalTransform( ctm ) && IgnoreAsUnderline( lineStart, lineEnd ) )
679  return;
680 
681  // Account for stroke width, but only by the minumum amount. The vast majority of paths
682  // acting as glyphs are things like bullets, where if we accounted for the maximum
683  // possible miter limit we'd be putting way too much space around the figure. The
684  // following code was derived from GState::TestUserSpaceBounds:
685  auto lineWidth = this->GetGState()->GetLineWidth() * t4::real_coord(0.707f);
686  if( lineWidth == 0 )
687  lineWidth = .0001_r;
688  t4::real_rect bBox;
689  walker.FetchPathBoundsRect( bBox );
690  bBox = TransformAndBoundRealRect( RectInset( bBox, -lineWidth, -lineWidth ), ctm );
691  ExtendFigure( bBox, isAllOrthoLines, pathLength );
692  }

694  // Determine whether path is all orthogonal lines and whether it's a single line.
695  // If it's a single line, set output vars lineStart and lineEnd to values
696  // in default space.
697  void AnalyzePath( const bez_path_iter_type & pathStart,
698  const bez_path_iter_type & pathEnd,
699  bool & isAllOrthoLines,
700  bool & isSingleLine,
701  t4::real_coord& pathLength,
702  t4::real_point& lineStart,
703  t4::real_point& lineEnd
704  )
705  {
706  using namespace imaging_model;
707 
708  bez_path_iter_type it = pathStart;
709 
710  isAllOrthoLines = true;
711  isSingleLine = true;
712  bool started = false;
713  bool gotLine = false;
714  t4::real_coord prevX = 0;
715  t4::real_coord prevY = 0;
716  pathLength = 0;
717  while( it != pathEnd )
718  {
719  auto move_to = [&]( t4::real_point const& p )
720  {
721  if( started )
722  isSingleLine = false;
723 
724  prevX = get_x( p );
725  prevY = get_y( p );
726  MatrixTransform( this->GetUserToDeviceTransform(), prevX, prevY );
727  set_x( lineStart, prevX );
728  set_y( lineStart, prevY );
729  started = true;
730  };
731 
732  auto line_to = [&]( t4::real_point const& p )
733  {
734  if( gotLine )
735  isSingleLine = false;
736 
737  t4::real_coord x = get_x( p );
738  t4::real_coord y = get_y( p );
739  MatrixTransform( this->GetUserToDeviceTransform(), x, y );
740  set_x( lineEnd, x );
741  set_y( lineEnd, y );
742  if( started )
743  {
744  pathLength += real_services::Sqrt( real_services::Pow( prevX - x, 2 ) + real_services::Pow( prevY - y, 2 ) );
745  if( std::fabs( prevX - x ) > 1 &&
746  std::fabs( prevY - y ) > 1 )
747  isAllOrthoLines = false;
748  }
749  prevX = x;
750  prevY = y;
751  started = true;
752  gotLine = true;
753  };
754 
755  auto curve_to = [&]( t4::real_point const& p0, t4::real_point const& p1, t4::real_point const& p2 )
756  {
757  auto process_pt = [&]( t4::real_point const& p )
758  {
759  t4::real_coord x = get_x( p );
760  t4::real_coord y = get_y( p );
761  MatrixTransform( this->GetUserToDeviceTransform(), x, y );
762  if( started )
763  pathLength += real_services::Sqrt( real_services::Pow( prevX - x, 2 ) + real_services::Pow( prevY - y, 2 ) );
764  isAllOrthoLines = false;
765  isSingleLine = false;
766  prevX = x;
767  prevY = y;
768  started = true;
769  };
770 
771  process_pt( p0 );
772  process_pt( p1 );
773  process_pt( p2 );
774  };
775 
776  auto close_path = [](){};
777 
778  (*it).visit( move_to, line_to, curve_to, close_path );
779  ++it;
780  }
781  }
782 
783  // Ignore lines that are likely representations of text underlines, strikethroughs, and overlines
784  // (subsequently referred to generally as underlines in this comment).
785  // Note this doesn't make exceptions for the case where a line is recoginized as an underline,
786  // but the text is contained within the bounds of another figure. In that case we'd like to keep
787  // the underline (because it will do no harm in text-over-images mode), but we don't.
788  // Note this is pretty naive about efficiency. For every path on the page, all
789  // text runs on the page are examined.
790  // TODO: choose slack margins more intelligently.
791  // TODO: understand and use line orientation compensation.
792  bool IgnoreAsUnderline( const t4::real_point start,
793  const t4::real_point end )
794  {
795  if( m_textShows.size() > 1000 )
796  // Too expensive to calculate
797  return false;
798 
799  bool lineIsHoriz = std::abs( get_x( start ) - get_x( end ) ) > std::abs( get_y( start ) - get_y( end ) );
800  t4::real_coord lineLeft = lineIsHoriz ? t4::min<t4::real_coord>( get_x( start ), get_x( end ) ) : t4::min<t4::real_coord>( get_y( start ), get_y( end ) );
801  t4::real_coord lineRight = lineIsHoriz ? t4::max<t4::real_coord>( get_x( start ), get_x( end ) ) : t4::max<t4::real_coord>( get_y( start ), get_y( end ) );
802  t4::real_coord lineBottom = lineIsHoriz ? t4::min<t4::real_coord>( get_y( start ), get_y( end ) ) : t4::min<t4::real_coord>( get_x( start ), get_x( end ) );
803  t4::real_coord lineTop = lineIsHoriz ? t4::max<t4::real_coord>( get_y( start ), get_y( end ) ) : t4::max<t4::real_coord>( get_x( start ), get_x( end ) );
804 
805  t4::real_coord prevTextTop = 0;
806  t4::real_coord prevTextBottom = 0;
807  bool leftMatch = false;
808  bool rightMatch = false;
809  for( const auto & textShow : m_textShows )
810  {
811  t4::real_rect textBBox = textShow.second;
812  bool textIsHoriz = get_width ( textBBox ) > get_height( textBBox );
813  t4::real_coord textTop = textIsHoriz ? get_ymax( textBBox ) : get_xmax( textBBox );
814  t4::real_coord textBottom = textIsHoriz ? get_ymin( textBBox ) : get_xmin( textBBox );
815  if( textIsHoriz == lineIsHoriz ) // Same direction as line
816  {
817  t4::real_coord margin = ( textTop - textBottom )/5;
818  if( lineBottom <= textTop + margin && lineTop >= textBottom - margin )
819  {
820  // The line may span text shows, so we want to check whether the line's left
821  // and right edges are both within text runs on the same line. We assume that within
822  // a line of text, the DL (as represented by the text bbox stack) is contiguous.
823  // TODO: is there a common heuristic to determine that we're on the same line?
824  // If either top or bottom is a close match with previous line, we consider it the
825  // same line. This allows for superscripts and font size changes.
826  if( std::abs( prevTextTop - textTop ) > margin/2 &&
827  std::abs( prevTextBottom - textBottom ) > margin/2 ) // We're on a new text line.
828  {
829  leftMatch = false;
830  rightMatch = false;
831  }
832  t4::real_coord textLeft = textIsHoriz ? get_xmin( textBBox ) : get_ymin( textBBox );
833  t4::real_coord textRight = textIsHoriz ? get_xmax( textBBox ) : get_ymax( textBBox );
834  if( lineLeft >= textLeft && lineLeft <= textRight )
835  leftMatch = true;
836  if( lineRight >= textLeft && lineRight <= textRight )
837  rightMatch = true;
838  if( leftMatch && rightMatch )
839  return true; // Ignore it as an underline
840  }
841  }
842  prevTextTop = textTop;
843  prevTextBottom = textBottom;
844  }
845  return false;
846  }
