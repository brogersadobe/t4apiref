119 class ContentPointLessThan_StructureOrder
120 {
121 public:
122  ContentPointLessThan_StructureOrder( const Structure& structure )
123  : m_structure( structure )
124  {}
125 
126  bool operator()( const t4::pdf::content_point& lhs, const t4::pdf::content_point& rhs ) const
127  {
128  content::ContentPointLessThan_DocumentOrder order;
129  const StructureNode* root = m_structure.GetRoot();
130  if( !root )
131  return t4::lt( lhs, rhs, order );
132 
133  const StructureNode* lhsPtr =
134  root->SearchContentPointInSubtree( lhs );
135 
136  if( lhsPtr == nullptr )
137  return t4::lt( lhs, rhs, order );
138 
139  const StructureNode* rhsPtr =
140  root->SearchContentPointInSubtree( rhs );
141 
142  if( lhsPtr == rhsPtr ||
143  rhsPtr == nullptr )
144  return t4::lt( lhs, rhs, order );
145  else
146  return ( *lhsPtr < *rhsPtr );
147  }
148 
149 private:
150  const Structure& m_structure;
151 };
