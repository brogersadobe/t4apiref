85 template<typename F>
86 class accessible_content_visitor
87 {
88  const aaa::make_accessible& m_makeAccessible;
89  F const& m_f;
90  t4::vector<structure_element>& m_visited_images;
91 
92  // For internal use only.
93  // Keyed by given tag type, these stacks contain the ancestor nodes currently present
94  // on the stack while performing in-order traversal on the structure tree. This info
95  // can be used to find (and save) the lowest ancestor of a given tag when a node is
96  // reached. This avoids upward traversal to compute the same info.
97  using ancestor_stack_type = t4::vector<t4::pdf::dictionary>;
98  mutable std::array<ancestor_stack_type, size_t(aaa::tag_type::num_tags)> m_ancestor_stacks;
99 
100 public:
101  accessible_content_visitor(const aaa::make_accessible& makeAccessible,
102  F const& f,
103  t4::vector<structure_element>& images)
104  : m_makeAccessible(makeAccessible)
105  , m_f( f )
106  , m_visited_images(images)
107  {}
108 
109  void operator()(boost::variant2::monostate) const
110  {}

112  void operator()(const t4::pdf::structure_tree_root& root) const
113  {
114  // structure_root has no content, so recurse until it finds some content.
115  size_t numChildren = root.get_number_of_children();
116  for (size_t i = 0; i < numChildren; ++i)
117  {
118  try
119  {
120  ( *this )( root.get_child(i) );
121  }
122  catch (const corrupt_data&)
123  {
124  // get_child can throw this exception, but we want to continue traversing the remaining of the tree
125  continue;
126  }
127  }
128  }
129 
130  void operator()(const t4::pdf::structure_element& elem) const
131  {
132  auto tag = get_aaa_tag_type(elem);
133 
134  if (tag)
135  {
136  m_ancestor_stacks[index_for_tag(*tag)].push_back(elem.get_dictionary());
137  }
138 
139  // structure_elements have no content, so recurse until it finds some content.
140  size_t numChildren = elem.get_number_of_children();
141  for (size_t i = 0; i < numChildren; ++i)
142  {
143  try
144  {
145  ( *this )( elem.get_child(i) );
146  }
147  catch (const corrupt_data&)
148  {
149  // get_child can throw this exception, but we want to continue traversing the remaining of the tree
150  continue;
151  }
152  }
153 
154  if (tag)
155  {
156  m_ancestor_stacks[index_for_tag(*tag)].pop_back();
157  }
158  }
159 
160  void operator()(const pdf::structure_content& contentNode) const
161  {
162  for_each_aaa_tag([&](aaa::tag_type tag, size_t index)
163  {
164  if (!m_ancestor_stacks[index_for_tag(tag)].empty())
165  {
166  m_makeAccessible.set_content_ancestor_for_tag_type(contentNode, m_ancestor_stacks[index_for_tag(tag)].back(), tag);
167  }
168 
169  // enumerate all tags
170  return true;
171  });
172 
173  // Absorb all unimplemented feature and corrupt data exceptions so that structure tree traversal continues even if unsupported or corrupt nodes are found.
174  try
175  {
176  auto contentElems = contentNode.get_content(aaa_content_types);
177  for (auto const& i : contentElems)
178  {
179  aaa::accessible acc;
180  content_type type = i.get_content_type();
181  if (type == pdf::content_type::text_run)
182  {
183  acc = get_accessible_text(contentNode, i, m_makeAccessible);
184  if (!is_empty(acc.m_rect)) // Only visit text nodes with non-empty bounding rect
185  m_f(acc);
186  }
187  else
188  {
189  acc = get_accessible_image(contentNode, contentElems, m_makeAccessible, m_visited_images);
190  if (acc.m_role == aaa::role::graphic)
191  {
192  m_f(acc);
193  break;
194  }
195  }
196  }
197  }
198  catch (const t4::unimplemented&)
199  {
200  // Absorbs all unimplemented feature exceptions.
201  assert(false);
202  }
203  catch (const t4::corrupt_data&)
204  {
205  // Absorbs all corrupt data exceptions.
206  }
207  }
208 
209  void operator()(const structure_object_reference& objref) const
210  {
211  t4::string type = objref.get_structure_type();
212  if (type == "Form")
213  {
214  aaa::accessible acc = get_accessible_form_field(objref);
215  m_f(acc);
216  }
217  else if (type == "Annot")
218  {
219  boost::optional<aaa::accessible> acc = get_accessible_link(objref);
220  if (acc)
221  m_f(*acc);
222  }
223  }

231  boost::optional<aaa::tag_type> get_aaa_tag_type(const structure_element& elem) const
232  {
233  boost::optional<aaa::tag_type> elem_tag;
234 
235  for_each_aaa_tag([&](aaa::tag_type tag, size_t index)

238  elem_tag = tag;

242  });
243

245  }
