337  // Cached role

342  t4::string m_spoken_text;
343  // Cached value for this node. E.g., text in a text form field, or heading level for a heading element.
344  t4::string m_value;
345  // Cached document coordinates for this node
346  real_rect m_rect;
347  // Content range for this node
