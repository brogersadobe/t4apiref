112 template<typename _signal_traits>
113 FastPixelBufferRasterPainter<_signal_traits>::FastPixelBufferRasterPainter( const constraints_type& constraints,
114  const raster_source_type& buffers,
115  t4::brief_ptr<BriefPixelStore<signal_traits>> storage )
116  : m_storage( std::move( storage ) )
117 {
118  tetraphilia_assert( !buffers.m_shape || RectContainsRect( *buffers.m_shape->m_bounds, constraints.m_interestBounds ) );
119  tetraphilia_assert( !buffers.m_opacity || RectContainsRect( *buffers.m_opacity->m_bounds, constraints.m_interestBounds ) );
120  tetraphilia_assert( !buffers.m_color || RectContainsRect( *buffers.m_color->m_bounds, constraints.m_interestBounds ) );
121 
122  m_shapeYWalker = MakeWalker( buffers.m_shape );
123  m_opacityYWalker = MakeWalker( buffers.m_opacity );
124  m_colorYWalker = MakeWalker( buffers.m_color );
125

127  m_segment.m_opacity = m_opacityYWalker;
128  m_segment.m_color = m_colorYWalker;
