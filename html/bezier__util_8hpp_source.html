105  Given an input range of bezier path segments, return a view of those same segments transformed through the given
106  affine transform.

108 template<typename BezierPath, typename = std::enable_if_t<is_bezier_path<BezierPath>::value>>
109 auto transform( BezierPath&& path, affine_transform const& xform )
110 {
111  return path | ranges::views::transform( [xform]( auto const& seg )
