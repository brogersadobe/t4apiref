615 // This object maintains a list of all intersections between annotation link
616 // rectangles and content bounding boxes:
617 class ContentToLinkAnnotMaps
618 {
619 public:
620  ContentToLinkAnnotMaps( void ) = default;
621 
622  ContentToLinkAnnotMaps( const ContentToLinkAnnotMaps& rhs ) = default;
623 
624  // Record the fact the content at the specified display list enumeration
625  // index has a bounding box that intersects the specified annotation link
626  // rectangle. Note that the index of the annotation link is not the index
627  // into the annotation dictionary, but rather the zero-based index into
628  // the (conceptual) list of all annotations that are links:
629  void AddLinkAnnot( size_t dlEntryNum, unsigned int annotIndex )
630  {
631  const size_t numAnnots = m_linkAnnots.size();
632  for( size_t index = 0; index < numAnnots; ++index )
633  if( m_linkAnnots[ index ].m_dlEntryNum == dlEntryNum &&
634  m_linkAnnots[ index ].m_linkAnnotIndex == annotIndex )
635  return;
636  m_linkAnnots.push_back( ContentToLinkAnnotMap( dlEntryNum, annotIndex ) );
637  }
638 
639  // Do any annotation link rectangles intersect any of the reflow content?
640  bool HasLinkAnnots( void ) const
641  {
642  return m_linkAnnots.size() > 0;
643  }
644 
645  void Consume( ContentToLinkAnnotMaps & donor )
646  {
647  const size_t numAnnots = donor.m_linkAnnots.size();
648  for( size_t index = 0; index < numAnnots; ++index )
649  AddLinkAnnot( donor.m_linkAnnots[ index ].m_dlEntryNum,
650  donor.m_linkAnnots[ index ].m_linkAnnotIndex );
651  donor.m_linkAnnots.clear();
652  }
653 
654  ContentToLinkAnnotMaps& operator=( const ContentToLinkAnnotMaps& rhs )
655  {
656  m_linkAnnots.clear();
657  const size_t numAnnots = rhs.m_linkAnnots.size();
658  for( size_t index = 0; index < numAnnots; ++index )
659  m_linkAnnots.push_back( rhs.m_linkAnnots[ index ] );
660  return *this;
661  }
662 
663  size_t GetCachedSize() const
664  {
665  return sizeof (*this) + t4::member_size(m_linkAnnots);
666  }
667 
668  t4::vector<ContentToLinkAnnotMap> m_linkAnnots;
669 };
