43 template <typename real_type>
44 real_type VecLen3D(real_type x, real_type y, real_type z)
45 {
46  // this first len is approximate, to keep us from overflowing in the L2 length below.
47  // Note that I would use VectorLength() if this were a 2D distance, but it isn't.
48  size_t overflow = false;
49  real_type taxilen = real_services::Add( real_services::Add( AbsVal(x), AbsVal(y), &overflow ), AbsVal(z), &overflow );
50  if ( overflow )
51  return limits<real_type>::max();
52  x /= taxilen;
53  y /= taxilen;
54  z /= taxilen;
55  return taxilen * real_services::Sqrt(x*x+y*y+z*z);
56 }

88  : m_t0(t0)
89  , m_t1(t1)
90  , m_tmax(t4::max<real_coord_type>(t0, t1))
91  , m_tmin(t4::min<real_coord_type>(t0, t1))
92  , m_numSignals(channelfunc.GetDstColorDimension())
93  , m_rampStorage( t4::multiply<size_t>(t4::add<size_t>(numSteps, 1), channelfunc.GetDstColorDimension()),

103  rampPos += m_numSignals;
104  t += dt;
105  }

108 template <typename _signal_traits>
109 void ShadingChannelRamp<_signal_traits>::GetPixel(signal_type* dst, ptrdiff_t signalStride, real_coord_type t) const
110 {
111  t = t4::pin<real_coord_type>( m_tmin, t, m_tmax );
112  size_t num_steps = m_rampStorage.size() / m_numSignals - 1;
113  ptrdiff_t index = real_services::RoundToInt(ArgType<ptrdiff_t>(), (num_steps * (t-m_t0))/(m_t1-m_t0));
114 
115  size_t buffer_index = static_cast<size_t>(index);
116  // Pin the computed value to the buffer range
117  if (t4::lt(index, 0))
118  buffer_index = 0;
119  else if (t4::gt(index, num_steps))
120  buffer_index = num_steps;
121 
122  const signal_type* ramppos = m_rampStorage.data() + buffer_index * m_numSignals;
123  for (size_t i = 0; i < m_numSignals; i++)
124  {
125  *dst = *ramppos++;
126  dst = t4::add_bytes(dst, signalStride);
127  }
128 }

130 template <typename real_type>
131 class SimplePlanarRegion
132 {
133  enum EType { kWholePlane, kHalfPlane, kCircle };
134  EType m_type;
135  real_type m_x, m_y, m_v1, m_v2;
136 
137  SimplePlanarRegion(real_type x, real_type y, real_type r)
138  : m_type(kCircle)
139  , m_x(x)
140  , m_y(y)
141  , m_v1(r)
142  {}
143 
144  SimplePlanarRegion(real_type x, real_type y, real_type dx, real_type dy)
145  : m_type(kHalfPlane)
146  , m_x(x)
147  , m_y(y)
148  , m_v1(dx)
149  , m_v2(dy)
150  {
151  real_type len = real_services::VectorLength(dx, dy);
152  m_v1 /= len;
153  m_v2 /= len;
154  }
155 
156 public:
157  SimplePlanarRegion()
158  : m_type(kWholePlane)
159  {}
160 
161  bool Contains(real_type* pSafeDistWithoutEdge, real_type x, real_type y) const
162  {
163  switch(m_type)
164  {
165  case kWholePlane:
166  *pSafeDistWithoutEdge = 128; // good enough to speed things up a ton, not dangerous in any real number system
167  break;
168  case kHalfPlane:
169  {
170  real_type sgnonly = real_services::DotForSignOnly(x-m_x, y-m_y, m_v1, m_v2);
171  *pSafeDistWithoutEdge = (x-m_x)*m_v1 + (y-m_y)*m_v2;
172 
173  if (sgnonly < 0)
174  {
175  if (*pSafeDistWithoutEdge > 0 || *pSafeDistWithoutEdge < -128)
176  *pSafeDistWithoutEdge = -128;
177  }
178  else if (sgnonly > 0)
179  {
180  if (*pSafeDistWithoutEdge < 0 || *pSafeDistWithoutEdge > 128)
181  *pSafeDistWithoutEdge = 128;
182  }
183  else // sgnolny == 0
184  {
185  *pSafeDistWithoutEdge = 0;
186  }
187  }
188  break;
189  case kCircle:
190  *pSafeDistWithoutEdge = m_v1 - real_services::VectorLength(x-m_x, y-m_y);
191  break;
192  }
193  if (*pSafeDistWithoutEdge < 0)
194  {
195  *pSafeDistWithoutEdge = -*pSafeDistWithoutEdge;
196  return false;
197  }
198  return true;
199  }

494 template<typename _signal_traits>
495 bool SmoothShadeSampler<_signal_traits>::RepeatOrReflectIntoRange(real_coord_type& t, real_coord_type tmin, RORType rirtypemin, real_coord_type tmax, RORType rirtypemax)
496 {
497  if (t < tmin)
498  {
499  switch (rirtypemin)
500  {
501  case kRORRepeat:
502  t = real_services::Mod(t-tmin, tmax-tmin) + tmin;
503  return false;
504  case kRORReflect:
505  t = real_services::Mod(t-tmin, 2*(tmax-tmin))+tmin;
506  if (t > tmax)
507  {
508  t = 2*tmax - t;
509  return true;
510  }
511  return false;
512  case kRORNone:
513  return false;
514  }
515  }
516 
517  if (t > tmax)
518  {
519  switch (rirtypemax)
520  {
521  case kRORRepeat:
522  t = real_services::Mod(t-tmin, tmax-tmin) + tmin;
523  return false;
524  case kRORReflect:
525  t = real_services::Mod(t-tmin, 2*(tmax-tmin))+tmin;
526  if (t > tmax)
527  {
528  t = 2*tmax - t;
529  return true;
530  }
531  return false;
532  case kRORNone:
533  return false;
534  }
535  }
536

567 template<typename _signal_traits>
568 typename SmoothShadeSampler<_signal_traits>::real_point_type SmoothShadeSampler<_signal_traits>::ComputeShadeSpacePoint(const int_point_type& devicePt) const
569 {
570  // first, find the CENTER of the pixel in question
571  real_point_type p = MakePoint<real_point_type>(real_coord_type(get_x(devicePt))+real_coord_type(0.5), real_coord_type(get_y(devicePt))+real_coord_type(0.5));
572  // ideally, we would mult by inverse(m_transform), but inverse transforms can be bad, particularly in their
573  // translation portions. There is a stabler alternate way, which is what I use here (basically involves translating

987 template<typename _signal_traits>
988 void FunctionShadeSampler<_signal_traits>::ComputeRun( )
989 {
990  tetraphilia_assert(!this->IsInvalid());
991 
992  int_coord_type xmax = get_xmax(this->m_samplerContext.m_constraints.m_interestBounds);
993  int_coord_type startx = get_x(this->m_dstPoint);
994 
995  real_point_type p = this->ComputeShadeSpacePoint(this->m_dstPoint);
996  signal_type* dstFirstSignalPtr = this->ShadeChannelFirstSignalPtr(startx);
997 
998  real_coord_type dpxdx = get_a(this->m_invTransformScalePortionOnly);
999  real_coord_type dpydx = get_b(this->m_invTransformScalePortionOnly);
1000 
1001  for (int_coord_type x = startx; x < xmax; x++)
1002  {
1003  signal_type alpha;
1004 
1005  if (get_x(p) < m_xmin || get_x(p) > m_xmax || get_y(p) < m_ymin || get_y(p) > m_ymax)
1006  {
1007  if (this->m_hasBackground)
1008  {
1009  this->m_channelFunc.GetBackgroundSignal(dstFirstSignalPtr, this->ShadeChannelSignalStride());
1010  alpha = signal_traits::one();
1011  }
1012  else
1013  alpha = signal_traits::zero();
1014  }
1015  else
1016  {
1017  real_coord_type ps[2] = { get_x( p ), get_y( p ) };
1018  this->m_channelFunc.GetPixel(dstFirstSignalPtr, this->ShadeChannelSignalStride(), ps);
1019  alpha = signal_traits::one();
1020  }
1021 
1022  if (x == startx)
1023  this->SetActiveSegmentFromAlphaValue(alpha);
1024  else if (this->GetActiveSegment() != this->GetSegmentFromAlphaValue(alpha))
1025  {
1026  this->m_xStop = x;
1027  return;
1028  }
1029 
1030  dstFirstSignalPtr = t4::add_bytes(dstFirstSignalPtr, this->ShadeChannelPixelStride());
1031  get_x(p) += dpxdx;
1032  get_y(p) += dpydx;
1033 
1034  this->m_samplerContext.Yield(10000);
1035  }
1036  this->m_xStop = xmax;
1037 }

1367  // has its axis through the origin along the t axis. It starts at t=-1, and ends at t=1. The

1370

1372  // following shape:
1373  //

1375  // [ 0 1/R 0 ] * [ 0 1 0 ]
1376  // [ 0 0 1 ] [ x0-x1 y0-y1 2 ]

1405  }

1515  // direction v with the normalized cone (and trimming to between t in [-1,1]
1516  // qe_a is a constant, calculated in ctor.

1523  // when the starting and ending circles touch at an edge

1527  }

1538  if (useSmallerT)
