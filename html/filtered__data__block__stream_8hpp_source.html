64  // normal behavior of decompression filters is to stop attempting to decompress data
65  // once we have run out of source data. But some filters, such as CCITT, are highly
66  // tolerant of damanged source data, and have their own mechanisms to determine when
67  // they have done enough work to no longer bother attempting to "decompress" more data.
68  // So we have a virtual method that controls this behavior. Default is to stop.

71  return true;

79  virtual data_block get_next_block_impl( void )
80  {
81  if( m_encountered_eod_marker && stop_at_eod() )
82  return data_block();
83 
84  static const size_t filtered_stream_block_size = 4096;
85  auto hdbo = t4::make_unique<std::uint8_t[]>( filtered_stream_block_size );
86  size_t len = 0;
87 
88  while( true )
89  {
90  if (m_char_beyond == -1)
91  len = process_data(static_cast<unsigned char *>(hdbo.get()), filtered_stream_block_size);
92  else
93  {
94  unsigned char *buffer = static_cast<unsigned char*>( hdbo.get() );
95  assert( 0 <= m_char_beyond && m_char_beyond <= 255 );
96  *buffer = static_cast<unsigned char>(m_char_beyond);
97  m_char_beyond = -1;
98  len = process_data( buffer + 1, filtered_stream_block_size - 1 ) + 1;
99  }
100 
101  if( m_encountered_eod_marker || len > 1 )
102  break;
103 
104  // save any length 1 block in m_char_beyond (which is often already equal to this char value)
105  if( len == 1 )
106  {
107  m_char_beyond = static_cast<unsigned char const*>( hdbo.get() )[0];
108  }
109  }
110 
111  auto block = t4::make_data_block( std::move( hdbo ), len );
112 
113  // Ensures we don't return the stream's last byte unless we've parsed the EOD.
114  assert( !( !m_encountered_eod_marker && block.get_size() <= 1 ) ); // len >= 2 unless EOD has been reached
115  if( !m_encountered_eod_marker )
116  {
117  block.truncate_back( 1 );
118  m_char_beyond = static_cast<unsigned char const*>( block.get_bytes() )[block.get_size()];
119  }
120  assert( m_encountered_eod_marker || ( m_char_beyond >= 0 && m_char_beyond <= 255 && block.get_size() > 0 ) );
121 
122  return block;
123  }
