383  friend t4::data_block get_next_block( CMapDecompressedDataBlockStream& ths )
384  {
385  size_t len = 0;
386  const char *buf = nullptr;
387  if (!ths.m_atEOF)
388  {
389  buf = ths.m_decompressor->PullChars(&len);
390  if (len == 0)
391  ths.m_atEOF = true;
392  }
393 
394  return t4::make_data_block( buf, len );
395  }
