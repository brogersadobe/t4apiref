170  CSFamily srcFamily = umSrcCS.GetFamily();
171  if( dstFamily == kDeviceRGB )

178  result = t4::make_unique<color_detail::DeviceRGBFromDeviceCMYK<signal_traits>>( linear );
179  else if( srcFamily == kLab )
180  result = t4::make_unique<color_detail::DeviceRGBFromLab<signal_traits>>( umSrcCS, linear );
181  }
182  else if( dstFamily == kDeviceCMYK )
183  {
184  if( srcFamily == kDeviceCMYK )
185  result = t4::make_unique<color_detail::IdentityConverter<signal_traits>>( 4u, linear );
186  }

292  const signal_type * src, ptrdiff_t srcSignalStride )
293  {

295 #ifdef TETRAPHILIA_CUSTOM_CMYK_TO_RGB
296  // TODO_READER_MOBILE<jpitteng>: Investigate whether this color conversion could be accelerated by
297  // calling ACEOptimizedCMYKtoRGB on an array of pixels rather than one at a time, if the
298  // signal traits line up just-so.
299 
300  BRVUns32 aceRGB;
301 
302  // This assumes ace-mobile is using a white0-based conversion; if it's black0 instead, then
303  // the signal_traits::complement() must be called prior to the signal conversion.
304  unsigned char currSignal = imaging_model::SignalConverter<signal_type, unsigned char>::Do(color_helper::Pin(*src));
305  BRVUns32 aceCMYK = PUTBYTE0(currSignal);
306  src = t4::add_bytes( src, srcSignalStride );
307  currSignal = imaging_model::SignalConverter<signal_type, unsigned char>::Do(color_helper::Pin(*src));
308  aceCMYK |= PUTBYTE1(currSignal);
309  src = t4::add_bytes( src, srcSignalStride );
310  currSignal = imaging_model::SignalConverter<signal_type, unsigned char>::Do(color_helper::Pin(*src));
311  aceCMYK |= PUTBYTE2(currSignal);
312  src = t4::add_bytes( src, srcSignalStride );
313  currSignal = imaging_model::SignalConverter<signal_type, unsigned char>::Do(color_helper::Pin(*src));
314  aceCMYK |= PUTBYTE3(currSignal);
315 
316  #ifdef NDEBUG
317  BOOST_ATTRIBUTE_UNUSED
318  #endif
319  ACE_Error error = ACEOptimizedCMYKtoRGB (reinterpret_cast<const ACE_CMYK8*>(&aceCMYK),
320  reinterpret_cast<ACE_pRGB8*>(&aceRGB),
321  1);
322  tetraphilia_assert(error == ACE_Error_None);
323 
324  // The initial byte of the result is always zero
325  tetraphilia_assert(GETBYTE0(aceRGB) == 0);
326  currSignal = GETBYTE1(aceRGB);
327  *dst = imaging_model::SignalConverter<unsigned char, signal_type>::Do(currSignal);
328  dst = t4::add_bytes( dst, dstSignalStride );
329  currSignal = GETBYTE2(aceRGB);
330  *dst = imaging_model::SignalConverter<unsigned char, signal_type>::Do(currSignal);
331  dst = t4::add_bytes( dst, dstSignalStride );
332  currSignal = GETBYTE3(aceRGB);
333  *dst = imaging_model::SignalConverter<unsigned char, signal_type>::Do(currSignal);
334 #else
335  signal_type black = color_helper::Pin( *t4::add_bytes( src, 3 * srcSignalStride ) );
336  signal_type maxVal = signal_traits::complement( black );
337  signal_type tmp;
338 
339  tmp = color_helper::Pin( *src );
340  tmp = tmp > maxVal ? signal_traits::one() : tmp + black;
341  *dst = signal_traits::complement( tmp );
342  if( this->m_linear )
343  *dst = color_helper::Pin( color_helper::Linearize( *dst ) );
344  dst = t4::add_bytes( dst, dstSignalStride );
345  src = t4::add_bytes( src, srcSignalStride );
346 
347  tmp = color_helper::Pin( *src );
348  tmp = tmp > maxVal ? signal_traits::one() : tmp + black;
349  *dst = signal_traits::complement( tmp );
350  if( this->m_linear )
351  *dst = color_helper::Pin( color_helper::Linearize( *dst ) );
352  dst = t4::add_bytes( dst, dstSignalStride );
353  src = t4::add_bytes( src, srcSignalStride );
354 
355  tmp = color_helper::Pin( *src );
356  tmp = tmp > maxVal ? signal_traits::one() : tmp + black;
357  *dst = signal_traits::complement( tmp );

441  void ConvertPixel( signal_type * dst, ptrdiff_t dstSignalStride,
442  const signal_type * src, ptrdiff_t srcSignalStride )
443  {
444  typedef color::ColorHelper<signal_traits> color_helper;
445  color_conv_type color[ 3 ];
446  m_cache.NewSearch();
447 
448  m_cache.SetKey( 0, *src );
449  color[ 0 ] = color_helper::PinAndScale( *src, 0, 100 );
450  src = t4::add_bytes( src, srcSignalStride );
451 
452  m_cache.SetKey( 1, *src );
453  color[ 1 ] = color_helper::PinAndScale( *src, m_aMin, m_aMax );
454  src = t4::add_bytes( src, srcSignalStride );
455 
456  m_cache.SetKey( 2, *src );
457  color[ 2 ] = color_helper::PinAndScale( *src, m_bMin, m_bMax );
458 
459  if( m_cache.Found() )
460  {
461  for( unsigned int index = 0; index < 3; ++index )
462  {
463  *dst = m_cache.GetResult( index );
464  dst = t4::add_bytes( dst, dstSignalStride );
465  }
466  return;
467  }
468 
469  color::color_detail::ConvertLabToXYZ( color );
470 
471  // Compensate based on Observer = 2(degrees) and Illuminant = D65:
472  color[ 0 ] *= color_conv_type( 0.95047 );
473  color[ 2 ] *= color_conv_type( 1.08883 );
474 
475  // Use the matrix for converting to sRGB based on reference Illuminant D65:
476  color_conv_type R = color[ 0 ] * color_conv_type( 3.24071 )
477  + color[ 1 ] * color_conv_type( -1.53726 )
478  + color[ 2 ] * color_conv_type( -0.498571 );
479  color_conv_type G = color[ 0 ] * color_conv_type( -0.969258 )
480  + color[ 1 ] * color_conv_type( 1.87599 )
481  + color[ 2 ] * color_conv_type( 0.0415557 );
482  color_conv_type B = color[ 0 ] * color_conv_type( 0.0556352 )
483  + color[ 1 ] * color_conv_type( -0.203996 )
484  + color[ 2 ] * color_conv_type( 1.05707 );
485 
486  if ( R > color_conv_type( 0.0031308 ) )
487  R = color_conv_type( 1.055 ) * real_services::Pow( R, color_conv_type( 1.0 / 2.4 ) ) - color_conv_type( 0.055 );
488  else
489  R *= color_conv_type( 12.92 );
490 
491  if ( G > color_conv_type( 0.0031308 ) )
492  G = color_conv_type( 1.055 ) * real_services::Pow( G, color_conv_type( 1.0 / 2.4 ) ) - color_conv_type( 0.055 );
493  else
494  G *= color_conv_type( 12.92 );
495 
496  if ( B > color_conv_type( 0.0031308 ) )
497  B = color_conv_type( 1.055 ) * real_services::Pow( B, color_conv_type( 1.0 / 2.4 ) ) - color_conv_type( 0.055 );
498  else
499  B *= color_conv_type( 12.92 );
500

502 
503  m_cache.NewResult();

505  if( this->m_linear )
506  *dst = color_helper::Pin( color_helper::Linearize( *dst ) );
507  m_cache.SetResult( 0, *dst );
508  dst = t4::add_bytes( dst, dstSignalStride );

510  if( this->m_linear )
511  *dst = color_helper::Pin( color_helper::Linearize( *dst ) );
512  m_cache.SetResult( 1, *dst );

516  *dst = color_helper::Pin( color_helper::Linearize( *dst ) );

518  }
