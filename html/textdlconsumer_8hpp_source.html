84  template<typename show_data_type>
85  void HandleTextShow( show_data_type const& showData,
86  pdf::content::TextBBoxInfo & bBoxInfo )
87  {
88  auto pFont = this->m_pGState->GetTextFont();
89  if( !pFont->IsUsable() )
90  return;
91 
92  auto& dc = m_teContext.GetDocumentCaches();
93  ptr_to_UCM_type pUCM = pFont->GetToUnicodeCharCodeMap( text::GetToUnicodeCache(dc), text::GetFontCache(dc) );
94  tetraphilia_assert(pUCM);
95 
96  show_info_type showInfo( this->GetGState(),
97  this->GetUserToDeviceTransform(),
98  this->GetTextMatrix() );
99  this->FillShowInfo( showData, showInfo, bBoxInfo );
100  if( showInfo.GetTotalNumGlyphs() == 0 )
101  return;
102 
103  if( this->InReflowMode() &&
104  this->m_reflowLayoutIter != this->m_reflowLayoutEnd &&
105  pUCM->IsValid() )
106  showInfo.AdjustForReflow( this->m_reflowLayoutIter,
107  this->m_reflowLayoutEnd,
108  this->m_curContentPoint );
109 
110  tetraphilia_assert( !m_pShowInfo );
111  m_pShowInfo = &showInfo;
112 
113  const bool isVisible = IsVisibleOnPage( showInfo );
114  if( HandleShow_Begin( pUCM->IsValid(), isVisible ) )
115  {
116  try
117  {
118  HandleTextShowHelper( showData,
119  showInfo,
120  pUCM,
121  isVisible );
122  }
123  catch ( StopEnumerateContentException const& )
124  {
125  HandleShow_End( true );
126  throw;
127  }
128  }
129 
130  HandleShow_End( false );
131  m_pShowInfo = nullptr;
132  }
