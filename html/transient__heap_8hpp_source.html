63 /*
64  Allocate and construct an object in a transient_heap with extra bytes
65  "just past the end."
66 
67  This behaves just like make_transient, except that the object is allocated
68  with an "extra" number of bytes after the memory occupied by the structure.
69  This can be used to reduce allocations and co-locate objects alongside
70  variable-length data that they own. For example, one could imagine writing
71  a memory-efficient string class like this:
72 
73  class my_string : boost::noncopyable
74  {
75  size_t m_len;
76 
77  // A private sentinel class that allows us to have a "public" ctor
78  // that can't be used by any client. This is important because my_string
79  // instances _must_ be allocated out of a transient_heap, so we can't
80  // allow the client to construct one wherever they like. But the ctor
81  // still needs to be public, so that make_transient_extra_bytes can invoke
82  // it.
83  struct privatize_ctor {};
84 
85  public:
86  my_string( size_t len, privatize_ctor ) : m_len( len ) {}
87 
88  size_t length( void ) const { return m_len; }
89 
90  // Get a pointer to the bytes in the string.
91  char const* get( void ) const { return reinterpret_cast<char const*>( this + 1 ); }
92 
93  // Create a my_string with the specified value in the heap
94  static my_string* create( transient_heap& heap, char const* str )
95  {
96  auto len = strlen( str );
97 
98  // allocate a my_string object out of the transient_heap, with enough space
99  // for the string and its null termination byte
100  auto result = make_transient_extra_bytes<my_string>( heap, strlen( str ) + 1, privatize_ctor() );
101 
102  // Copy the string just past the end of the allocated object
103  strcpy( reinterpret_cast<char *>( result + 1 ), str );
104 
105  return result;
106  }
107  };
108 
109  Some caveats:
110 
111  1) If you use this technique, _every_ instance you create must be created using this
112  technique. Classes using this technique cannot be regular.
113  2) The instance will be valid for the lifetime of the transient_heap's current
114  transient_snapshot, if there is any; if there isn't, then the object will
115  be valid for the lifetime of the transient_heap.
116  3) If you reflexively loathe reinterpret_cast, you can instead write:
117 
118  char const* get( void ) const { return static_cast<char const*>( static_cast<void const*>( this + 1 ) ); }
119 
120  But that doesn't exactly help readability.
121 */
122 template<typename T, typename... TArgs>
123 std::enable_if_t<!std::is_trivially_destructible<T>::value,
124 T*> make_transient_extra_bytes( transient_heap& heap, size_t extra_bytes, TArgs&&... args );
125 
126 template<typename T, typename... TArgs>
127 std::enable_if_t<std::is_trivially_destructible<T>::value,
128 T*> make_transient_extra_bytes( transient_heap& heap, size_t extra_bytes, TArgs&&... args );

151 /*
152  Container class for heterogeneous collections of objects that the user expects to create
153  at various times, but dispose of in batches (typically all at once).
154 
155  Objects are created in a transient_heap using the make_transient function, e.g.:
156 
157  {
158  transient_heap my_heap;
159  foo* my_foo = t4::make_transient<foo>( my_heap, 12, "foo" ); // 12 and "foo" are forwarded to foo's ctor
160  bar* my_bar = t4::make_transient<bar>( my_heap ); // invokes bar's default ctor
161 
162  // my_foo and my_bar are destroyed when my_heap goes out of scope, here
163  }
164 
165  transient_heap is ideal for collections of objects that are created over time by an algorithm
166  that builds a complex object whose exact structure isn't known initially, where memory co-location
167  of those objects might be beneficial. It is extremely useful for heterogeneous collections of small
168  objects with trivial dtors. For example, the display list uses a transient_heap to store almost all
169  its data.
170 
171  Internally, the transient_heap allocates large blocks of memory and suballocates those blocks when
172  creating objects inside make_transient. The suballocation is not sophisticated--it simply allocates
173  memory sequentially, and does not record where individual objects start and stop. If there isn't
174  enough space in the current block for a new object it allocates another block. If an object has a
175  non-trivial dtor, it is added to an internal linked list of objects to be destroyed when the
176  heap goes out of scope. If its dtor is trivial, no overhead is incurred beyond the bytes required
177  to store the object.
178 
179  "Large" blocks (by default 4K or larger, but that's client-configurable) are not suballocated, but
180  rather allocated directly out of the global memory heap. But they are still automatically reclaimed
181  the same way small blocks are, so this is an internal implementation detail as far as the client is
182  concerned.
183 */

400 /*
401  Instances of transient_snapshot are useful for one particular application of transient_heaps,
402  which is the stack-like allocation of large blocks of memory. Example usage:
403 
404  {
405  transient_heap my_heap;
406 
407  foo* my_foo = make_transient<foo>( my_heap, 12, "foo" );
408 
409  {
410  transient_snapshot ss( my_heap );
411 
412  bar* my_bar = make_transient<bar>( my_heap );
413 
414  // my_bar is destroyed as soon as ss goes out of scope, right here
415  }
416 
417  // my_foo is still pointing to valid memory
418  assert( my_foo->get_value() == 12 );
419 
420  // but it is destroyed as soon as my_heap goes out of scope, right here
421  }
422 */
423 class transient_snapshot
424 {
425  /*
426  When a snapshot is constructed, it is associated with an instance of transient_heap
427  and the snapshot records enough state in the heap to restore the heap to its original
428  state at destruction time.
429 
430  Specifically, this means that any objects allocated out of the transient_heap during
431  the lifetime of this snapshot are destroyed. Completely empty blocks are released
432  back to the global heap, except for the last two. All large blocks allocated during
433  the snapshot are released as well.
434  */
435  transient_heap& m_heap;
436 
437 #if !T4_HARDENED_TRANSIENT_HEAP
438  using transient_block = transient_heap_detail::transient_block;
439  using block_iterator = boost::intrusive::list<transient_block>::iterator;
440  using dtor_record = transient_heap_detail::dtor_record;
441 
442  // This is the top of the dtor stack at the time the snapshot is created. Any objects
443  // linked into this stack after this point are destroyed from the snapshot's dtor.
444  dtor_record const* m_dtor_top;
445 
446  // This is the current block at the time the snapshot is created, i.e., the correct block
447  // from which to attempt the next request for memory. This isn't necessarily the last
448  // block in the heap's list though! But it is the last nonempty one.
449  block_iterator m_current_block;
450 
451  // This is the current large block at the time the snapshot is created. Unlike m_current_block,
452  // there is no guarantee that there are any large blocks, so this is wrapped up in an optional.
453  boost::optional<block_iterator> m_current_large_block;
454 
455  // This is the number of bytes allocated out of the block pointed to by m_current_block.
456  size_t m_current_block_top;
457 #else
458  using dispose_record = transient_heap_detail::dispose_record;
459 
460  t4::stable_chunked_vector<dispose_record>::iterator m_deallocations_top;
461  t4::stable_chunked_vector<dispose_record>::iterator m_dtors_top;
462 #endif /* T4_HARDENED_TRANSIENT_HEAP */
463 
464 #if T4_TRANSIENT_HEAP_CHECK_SNAPSHOT_USE
465  transient_snapshot* m_prev_snapshot;
466 #endif
467 
468 #if T4_TRANSIENT_HEAP_BUFFER_GUARDS
469  size_t m_canary_count;
470 #endif
