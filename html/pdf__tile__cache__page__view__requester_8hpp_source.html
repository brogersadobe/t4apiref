33 /*
34  Implementation of the pdf_tile_cache_requester interface which keeps track of a
35  "visible subrectangle map" for an entire pdf document. This map maps page numbers
36  to a data structure which describes what zoom level at which to paint the page, and
37  the subset of the page that is visible at that zoom level. Pages start out in the
38  invisible state, and there is no data associated with invisible pages.
39 */
40 template<typename PDFTCClient>
41 class pdf_tile_cache_page_view_requester : public pdf_tile_cache_requester<PDFTCClient>
42 {
43  using paint_request = paint_request<PDFTCClient>;
44  using clock = std::chrono::steady_clock;
45 
46  clock::duration m_update_interval;
47  synchronized<map<page_num, shared_ptr<paint_request>>> m_requests;
48 
49 public:
50  template<typename Duration = std::chrono::milliseconds>
51  pdf_tile_cache_page_view_requester( Duration update_interval = std::chrono::milliseconds( 500 ) )
52  : m_update_interval( std::chrono::duration_cast<std::chrono::milliseconds>( update_interval ) )
53  {}
54 
55  ~pdf_tile_cache_page_view_requester( void ) noexcept override {}
56 
57  /*
58  Tell the pdf_tile_cache that the specified page is visible and being displayed with the provided
59  draw_params. clip_rect describes the page view's aperture in device page space (the coordinate
60  space in which the origin is the top left corner of the page, y grows down, and 1 unit = 1 pixel).
61 
62  If the pdf_tile_cache is already working on that page under different draw_params, it will abort
63  that rendering and restart it.
64 
65  Rendering will occur inside the intersection of clip_rect and the page bounds, if nonempty.
66 
67  It is expected that multiple pages can be visible at one time; the client may call this function
68  multiple times with different pages. The pdf_tile_cache will attempt to render all pages that
69  are visible.
70 
71  See the comments in pdf_tile_cache_requester.hpp for an explanation of the update_timing parameter.
72  */
73  void set_page_visible( page_num page, draw_params const& draw_params, int_rect const& clip_rect,
74  update_timing timing = update_timing::right_now );
75 
76  /*
77  Tell the pdf_tile_cache that the specified page no longer intersects the aperture. If a rendering
78  for this page is in progress it will be aborted.
79 
80  See the comments in pdf_tile_cache_requester.hpp for an explanation of the update_timing parameter.
81  */
82  void set_page_invisible( page_num page, update_timing timing = update_timing::right_now );
83 
84 private:
85  shared_ptr<paint_request> get_next_request_impl( void ) override;
86 };
