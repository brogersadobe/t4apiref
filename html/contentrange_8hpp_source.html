167 inline bool ContentPoint::operator== (const ContentPoint& rhs) const
168 {
169  if ( m_pageID != rhs.m_pageID ||
170  m_dlEntryNum != rhs.m_dlEntryNum ||
171  m_stringNum != rhs.m_stringNum ||
172  m_glyphNum != rhs.m_glyphNum )
173  {
174  return false;
175  }
176 
177  // If m_totalNumUnicode == 0, we're dealing with a glyph boundary, which is
178  // equivalent to either the leading or trailing edge of the glyph.
179  if ( m_totalNumUnicode == 0 )
180  {
181  // this is a glyph, not a unicode character boundary
182  if ( !m_isAfter )
183  {
184  // This corresponds to the leading edge of the glyph, i.e., the
185  // ContentPoint before the first unicode char.
186  return rhs.m_unicodeNum == 0 && !rhs.m_isAfter;
187  }
188  else
189  return rhs.m_isAfter && ( rhs.m_totalNumUnicode == 0 || rhs.m_unicodeNum == rhs.m_totalNumUnicode - 1 );
190  }
191  else if ( rhs.m_totalNumUnicode == 0 )
192  {
193  if ( !rhs.m_isAfter )
194  return m_unicodeNum == 0 && !m_isAfter;
195  else
196  return m_isAfter && ( m_unicodeNum == m_totalNumUnicode - 1 );
197  }
198  else
199  {
200  tetraphilia_assert(m_totalNumUnicode == rhs.m_totalNumUnicode);
201 
202  if ( m_unicodeNum != rhs.m_unicodeNum )
203  return false;
204  return m_isAfter == rhs.m_isAfter;
205  }
206 }

208 inline bool ContentPointLessThan_SamePage( const ContentPoint& lhs, const ContentPoint& rhs )
209 {
210  // You should only call this on ContentPoints that are on the same page
211  tetraphilia_assert( lhs.m_pageID == rhs.m_pageID );
212 
213  if ( lhs.m_dlEntryNum < rhs.m_dlEntryNum )
214  return true;
215  if ( lhs.m_dlEntryNum > rhs.m_dlEntryNum )
216  return false;
217 
218  if ( lhs.m_stringNum < rhs.m_stringNum )
219  return true;
220  if ( lhs.m_stringNum > rhs.m_stringNum )
221  return false;
222 
223  if ( lhs.m_glyphNum < rhs.m_glyphNum )
224  return true;
225  if ( lhs.m_glyphNum > rhs.m_glyphNum )
226  return false;
227 
228  // If m_totalNumUnicode == 0, we're dealing with a glyph boundary, which is
229  // equivalent to either the leading or trailing edge of the glyph.
230  if ( lhs.m_totalNumUnicode == 0 )
231  {
232  // this is a glyph, not a unicode character boundary
233  if ( !lhs.m_isAfter )
234  {
235  // This corresponds to the leading edge of the glyph, i.e., the
236  // ContentPoint before the first unicode char. This point is less
237  // than all other points in the glyph except the point before the
238  // first unicode character.
239  return rhs.m_unicodeNum != 0 || rhs.m_isAfter;
240  }
241  else
242  {
243  // This corresponds to the right edge of the glyph, i.e., the
244  // ContentPoint after the last unicode char. This is less than
245  // no other point in the glyph.
246  return false;
247  }
248  }
249  else if ( rhs.m_totalNumUnicode == 0 )
250  {
251  if ( !rhs.m_isAfter )
252  {
253  // Leading edge of rhs' glyph
254  return false;
255  }
256  else
257  {
258  // Trailing edge of rhs' glyph, only the trailing edge of the same glyph is not less than
259  return !(lhs.m_isAfter && lhs.m_unicodeNum == lhs.m_totalNumUnicode);
260  }
261  }
262  else
263  {
264  tetraphilia_assert(lhs.m_totalNumUnicode == rhs.m_totalNumUnicode);
265 
266  if ( lhs.m_unicodeNum < rhs.m_unicodeNum )
267  return true;
268  if ( lhs.m_unicodeNum > rhs.m_unicodeNum )
269  return false;
270 
271  return !lhs.m_isAfter && rhs.m_isAfter;
272  }
273 }

275 // This comparator describes ContentPoint ordering in document order. This is
276 // subject to change across page reorderings, so it should not be used as a key
277 // comparator in maps that are supposed to be useful across page reorderings.
278 struct ContentPointLessThan_DocumentOrder
279 {
280  bool operator()( const ContentPoint& lhs, const ContentPoint& rhs ) const
281  {
282  lhs.Validate();
283  rhs.Validate();
284  if ( lhs.m_pageID == rhs.m_pageID )
285  return ContentPointLessThan_SamePage( lhs, rhs );
286  auto lhsPageIndex = t3::pdf::document::get_page_num_from_page_id( lhs.m_pageID );
287  auto rhsPageIndex = t3::pdf::document::get_page_num_from_page_id( rhs.m_pageID );
288  tetraphilia_assert( lhsPageIndex != rhsPageIndex );
289  return lhsPageIndex < rhsPageIndex;
290  }
291 };

293 // This comparator describes ContentPoint ordering in document order, but it is a
294 // client error to invoke this comparator on two content points from different pages.
295 struct ContentPointLessThan_SamePageDocumentOrder
296 {
297  ContentPointLessThan_SamePageDocumentOrder( void )
298  {}
299 
300  bool operator()( const ContentPoint& lhs, const ContentPoint& rhs ) const
301  {
302  assert( lhs.m_pageID == rhs.m_pageID );
303  lhs.Validate();
304  rhs.Validate();
305  return ContentPointLessThan_SamePage( lhs, rhs );
306  }
307 };

309 // This comparator describes ContentPoint ordering using PageID ordering. This
310 // isn't a useful ordering when describing content traversal order, but it does
311 // allow the user to use ContentPoints as keys in maps that will survive across
312 // page reorderings.
313 struct ContentPointLessThan_PageIDOrder
314 {
315  bool operator()( const ContentPoint& lhs, const ContentPoint& rhs ) const
316  {
317  lhs.Validate();
318  rhs.Validate();
319  if ( lhs.m_pageID == rhs.m_pageID )
320  return ContentPointLessThan_SamePage( lhs, rhs );
321  return lhs.m_pageID < rhs.m_pageID;
322  }
323 };

517 // Returns true if the ranged bounded by the provided ContentPoints inside the content range,
518 // in document order.
519 inline bool ContentRange::Contains(const ContentPoint& from, const ContentPoint& to) const
520 {
