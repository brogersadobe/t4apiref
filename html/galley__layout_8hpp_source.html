33 /*
34  "Galley space" is the coordinate space in which one or more pages of PDF are
35  laid out. Typically this layout will comprise a single page, or a single
36  column of all the pages in the document, but other layouts are supported,
37  such as two-up or a large n x m display of thumbnails.
38 
39  Coordinates, points and rectangles in galley space are expressed using
40  galley_coord, galley_point and galley_rect, which model the Coordinate,
41  Point and Rectangle concept, respectively.
42 */
43 typedef t4::real_coord galley_coord;
44 typedef point<galley_coord> galley_point;
45 typedef rectangle<galley_coord> galley_rect;
46 
47 /*
48  To compute a galley_layout one must provide an instance of galley_layout_params.
49  This structure describes various aspects of the layout, specifically:
50 
51  - Which pages are visible and the number of rows and columns in which they appear
52  - The size (in pixels) of the various margins between the pages and other pages
53  or the galley's edges
54 
55  Each page in the galley lives inside a "cell" which is at least large enough
56  to hold the page, and possibly larger (if larger pages exist in the same row
57  or column). Cells are arranged in rows and columns.
58 
59  Cells may be empty. For example, the leftmost cell in a two-up display will
60  typically be empty when viewing the first page.
61 */
62 struct galley_layout_params
63 {
64  /*
65  The page number corresponding to the top left cell. This doesn't have to
66  be a valid page number. For example, in the two-up case mentioned above,
67  the page number should be -1 (so that the first page, page 0, appears in
68  the right column).
69  */
70  pdf::page_num m_top_left_page_num;
71 
72  /*
73  The total number of pages in the document.
74  */
75  pdf::page_num m_num_pages;
76 
77  /*
78  The size of the grid. The horizontal component is the number of columns;
79  the vertical is the number of rows. For example, in a 200 page document
80  in continuous mode, this would be set to (1, 200).
81  */
82  int_point m_grid_size;
83 
84  /*
85  The minimum number of pixels between the left edge of the galley and the left edge
86  of the pages in the leftmost column.
87  */
88  galley_coord m_galley_page_margin_left;
89 
90  /*
91  The minimum number of pixels between the top edge of the galley and the top edge
92  of the pages in the topmost row.
93  */
94  galley_coord m_galley_page_margin_top;
95 
96  /*
97  The minimum number of pixels between the right edge of the galley and the right edge
98  of the pages in the rightmost column.
99  */
100  galley_coord m_galley_page_margin_right;
101 
102  /*
103  The minimum number of pixels between the bottom edge of the galley and the bottom edge
104  of the pages in the bottommost column.
105  */
106  galley_coord m_galley_page_margin_bottom;
107 
108  /*
109  The minimum number of pixels between the right and left edges of pages in the same
110  row and adjacent columns.
111  */
112  galley_coord m_interpage_margin_horizontal;
113 
114  /*
115  The minimum number of pixels between the bottom and top edges of pages in the same
116  columns and adjacent rows.
117  */
118  galley_coord m_interpage_margin_vertical;
119 };

125 {

152 {

156 inline boost::optional<int_point> page_to_grid_loc( galley_layout_params const& glp, pdf::page_num page )
157 {
158  if ( includes_page( glp, page ) )
159  return make<int_point>( static_cast<int_coord>( page - glp.m_top_left_page_num ) % get_x( glp.m_grid_size ), static_cast<int_coord>( page - glp.m_top_left_page_num ) / get_x( glp.m_grid_size ) );
160  return {};
161 }

163 inline pdf::page_num grid_loc_to_page( galley_layout_params const& glp, int_point const& grid_loc )
164 {
165  return glp.m_top_left_page_num + static_cast<pdf::page_num>( get_x( glp.m_grid_size ) * get_y( grid_loc ) + get_x( grid_loc ) );

258  template<typename F>
259  void for_each_cell( galley_rect const& aperture, F const& f ) const
260  {
261  int_rect subgrid = get_spanning_subgrid( aperture );
262  galley_rect cell_rect;
263  int_point grid_loc;
264 
265  set_y( grid_loc, get_ymin( subgrid ) );
266  set_ymin( cell_rect, get_row_start( get_y( grid_loc ) ) );
267  for ( ; get_y( grid_loc ) < get_ymax( subgrid ); ++get_y( grid_loc ) )
268  {
269  set_ymax( cell_rect, get_row_start( get_y( grid_loc ) + 1 ) );
270  set_x( grid_loc, get_xmin( subgrid ) );
271  set_xmin( cell_rect, get_column_start( get_x( grid_loc ) ) );
272 
273  for ( ; get_x( grid_loc ) < get_xmax( subgrid ); ++get_x( grid_loc ) )
274  {
275  set_xmax( cell_rect, get_column_start( get_x( grid_loc ) + 1 ) );
276  pdf::page_num page = grid_loc_to_page( m_galley_layout_params, grid_loc );
277  if ( includes_page( m_galley_layout_params, page ) )
278  {
279  auto opi = get_page_info( page );
280  f( page, opi, cell_rect, grid_loc );
281  }
282  set_xmin( cell_rect, get_xmax( cell_rect ) );
283  }
284 
285  set_ymin( cell_rect, get_ymax( cell_rect ) );
286  }
287  }
