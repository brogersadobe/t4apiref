157  static void writeParagraphHTML(std::string &pHtml, GeneratorState& pState)
158  {
159  std::stringstream lTextRunHtmlStream;
160  std::string lAlignmentStr;
161 
162  if (pState.m_alignment & Paragraph<app_traits>::LEFT_ALIGNED)
163  lAlignmentStr = "left";
164  else if (pState.m_alignment & Paragraph<app_traits>::RIGHT_ALIGNED)
165  lAlignmentStr = "right";
166  else if (pState.m_alignment & Paragraph<app_traits>::CENTER_ALIGNED)
167  lAlignmentStr = "center";
168  else
169  lAlignmentStr = "left"; // default
170 
171  std::vector<TextRunInfo> lTextRunsInfo;
172 
173  PreProcessParagraph(pState, lTextRunsInfo);
174 
175  pState.m_lineHeight = std::ceil(pState.m_lineHeight);
176 
177  size_t numTextRuns = pState.m_textRunIndexes.size();
178  assert(numTextRuns == lTextRunsInfo.size());
179 
180  for (size_t i = 0; i < numTextRuns; ++i)
181  {
182  writeTextRunHTML(lTextRunHtmlStream, pState.m_textRunSink.GetAt(pState.m_textRunIndexes[i]), lTextRunsInfo[i], pState);
183  }
184 
185  std::stringstream lParagraphHtmlStream;
186 
187  lParagraphHtmlStream
188  << "<div id=\"container_wrapper\"><div id=\"container\" "
189  << "containerLineHeight=\"" << pState.m_lineHeight << "\" style=\""
190  << "font-size:"<< pState.m_lineHeight <<"px; line-height:1;\">";
191 
192  // Webkit places baseline of the first line arbitrarily and positions remaining baselines by adding the user-provided lineheight.
193  // In javascript, these are calculated by approximating first line's baseline and adding lineheight to get subsequent baselines.
194  // To do that, we insert a dummy text as the first line. It is always hidden and has pre-defined attributes which are used to allow
195  // us to make the approximation accurately.
196  lParagraphHtmlStream
197  << "<div id=\"dummySpan\" style=\"background:red;\">"
198  << "<div id=\"dummy_baseline_marker\" style=\"background:green;width:0px;height:0px;display:inline-block;\"></div>"
199  << "</div>";
200 
201  lParagraphHtmlStream << "<div id=\"container_span\" autocomplete=\"off\" autocorrect=\"off\" paragraphAlignment=\"" << lAlignmentStr << "\" contenteditable=\"true\">" << lTextRunHtmlStream.str() << "</div></div></div></body></html>";
202 
203  pHtml = lParagraphHtmlStream.str();
204  }

223  static void PreProcessParagraph(GeneratorState& pState, std::vector<TextRunInfo>& pTextRunsInfo)
224  {
225  const auto& fontMap = pState.m_fontMap;
226  const real_type& zoomLevel = pState.m_zoomLevel;
227  const real_type& xMin = get_xmin(pState.m_bbox);
228  const real_type& yMax = get_ymax(pState.m_bbox);
229  real_type& lineHeight = pState.m_lineHeight;
230 
231  real_type prevBaseline = 0.0f;
232 
233  for (size_t i = 0; i < pState.m_textRunIndexes.size(); ++i)
234  {
235  size_t lIndex = pState.m_textRunIndexes[i];
236 
237  const TextRun<app_traits>& lTextRun = pState.m_textRunSink.GetAt(lIndex);
238 
239  const matrix_type& lTextMatrix = lTextRun.m_textMatrix;
240  const matrix_type& lDerivedTextMatrix = lTextRun.m_derivedTextMatrix;
241 
242  matrix_type lTextTransform = lTextMatrix * lTextRun.m_userSpaceToDefaultUserSpaceMatrix; // Text space to default user space
243  matrix_type lDerivedTextTransform = lDerivedTextMatrix * lTextRun.m_userSpaceToDefaultUserSpaceMatrix;
244 
245  // calculate BBox vertices transformed from normalized PDF user space (bottom-left origin) to device space (top-left origin)
246  // relative to the top-left vertice.
247  const real_rect_type& lTextRunBbox = lTextRun.GetNormalizedSpaceBbox();
248  real_type x1 = (lTextRun.m_baselineXMin - xMin) * zoomLevel;
249  real_type x2 = (lTextRun.m_baselineXMax - xMin) * zoomLevel;
250  real_type y1 = (yMax - get_ymax(lTextRunBbox)) * zoomLevel;
251  real_type y2 = (yMax - get_ymin(lTextRunBbox)) * zoomLevel;
252 
253  real_type ascent = 0;
254  real_type descent = 0;
255 
256  auto lFontIter = fontMap.find(lTextRun.m_fontKey);
257  if(lFontIter != fontMap.end())
258  {
259  ascent = std::abs(lFontIter->second.m_ascent);
260  descent = std::abs(lFontIter->second.m_descent);
261  }
262 
263  // Although Ascent/Descent are required entries in font descriptor, sometimes,
264  // they contain redundant value of 0. Hence, using default values.
265  if (ascent == 0 && descent == 0)
266  {
267  ascent = 940;
268  descent = 288;
269  }
270 
271  real_type baseline = y1 + ((y2 - y1) * ascent / (ascent + descent));
272 
273  if (prevBaseline > 0.0f)
274  lineHeight = std::max(lineHeight, std::abs(baseline - prevBaseline));
275 
276  lineHeight = std::max(lineHeight, y2 - y1);
277  prevBaseline = baseline;
278 
279  pTextRunsInfo.push_back( TextRunInfo(lTextTransform, lDerivedTextTransform, zoomLevel, baseline, x1, y1, x2, y2));
280  }
281  }

283  static void writeTextRunHTML(std::stringstream& htmlStream, const TextRun<app_traits>& pTextRun, const TextRunInfo& pTextRunInfo, GeneratorState& pState)
284  {
285  const auto& fontMap = pState.m_fontMap;
286  real_type& prevX1 = pState.m_prevX1;
287  real_type& prevX2 = pState.m_prevX2;
288  real_type& prevY1 = pState.m_prevY1;
289  real_type& prevY2 = pState.m_prevY2;
290  real_type& lineHeight = pState.m_lineHeight;
291  real_type& currentYMax = pState.m_currentYMax;
292  real_type& prevBaseline = pState.m_prevBaseline;
293  int& lineNumber = pState.m_lineNumber;
294  int alignment = pState.m_alignment;
295  size_t dlEntryNum = pTextRun.m_dlEntryNum;
296 
297  real_type baseline = pTextRunInfo.m_baseline;
298  real_type lScaleX = pTextRunInfo.m_textScaleX;
299  real_type lDerivedScaleX = pTextRunInfo.m_derivedTextScaleX;
300 
301  real_type x1 = pTextRunInfo.m_x1;
302  real_type x2 = pTextRunInfo.m_x2;
303  real_type y1 = pTextRunInfo.m_y1;
304  real_type y2 = pTextRunInfo.m_y2;;
305 
306 #if defined(_DEBUG) && DISPLAY_TEXT_METRICS
307  // display PDF paragraph BBoxes for reference
308  htmlStream << "<div class='coloredBG' style = 'position:absolute; height:" << y2 - y1 << "px; width:" << x2 - x1 << "px; left : " << x1 << "px; top:" << y1 << "px; background:rgba(0,0,0,0.2); z-index : -100;'></div>";
309 #endif
310 
311  int r = pTextRun.m_color[0];
312  int g = pTextRun.m_color[1];
313  int b = pTextRun.m_color[2];
314  real_type lFontSize = lDerivedScaleX;
315  real_type lCharSpaceX = pTextRun.m_charSpace * lScaleX;
316  real_type lWordSpaceX = pTextRun.m_wordSpace * lScaleX;
317  const auto& lTextSubStrVector = pTextRun.m_subStrVector;
318 
319  // Do not read back the zero-width space character (unicode 8203) which we saved into the PDF as workaround
320  // to the webkit bug (Apple Bug Reporter bug # 15029844) because :
321  //
322  // 1. When webkit generates the content stream for the given character, it places it in a separate Tj.
323  // The bounding box for this single character textrun should be calculated to have zero-width but
324  // it is non-zero (possibly because it's width is not written to the widths array in font dictionary)
325  // disturbing the rendering in HTML.
326  //
327  // 2. Reading this character back would accumulate it unnecessarily upon repeated edits on the same paragraph.
328  if (lTextSubStrVector.size() == 1 && lTextSubStrVector[0].first.size() == 1 && lTextSubStrVector[0].first[0] == 8203)
329  return;
330 
331  t4::string lFontName;
332  real_type ascent = 0;
333  real_type descent = 0;
334 
335  auto lFontIter = fontMap.find(pTextRun.m_fontKey);
336  if(lFontIter != fontMap.end())
337  {
338  real_type lFontWeight = lFontIter->second.m_weight;
339  real_type lItalicAngle = lFontIter->second.m_italicAngle;
340  ascent = lFontIter->second.m_ascent;
341  descent = lFontIter->second.m_descent;
342 
343  auto& fontNameMap = pState.m_store.GetFontDictToFallbackSystemFontMap();
344  auto ret = fontNameMap.emplace(lFontIter->second.m_fontKey, "");
345 
346  if (ret.second)
347  {
348  const auto& fontPSName = lFontIter->second.m_fontPSName;
349  if(!fontPSName.empty())
350  {
351  ret.first->second = t4::font::get_closest_match_system_font_name(fontPSName, pState.m_systemFontNamesMap, lFontIter->second.m_glyphWidths, lFontIter->second.m_fontDescriptorFlags, lFontWeight, lItalicAngle);
352  }
353  }
354 
355  lFontName = ret.first->second;
356  }
357 
358  if (lFontName.empty())
359  lFontName = DEFAULT_FALLBACK_SYSTEM_FONT;
360 
361  // Although Ascent/Descent are required entries in font descriptor, sometimes,
362  // they contain redundant value of 0. Hence, using default values.
363  if (ascent == 0 && descent == 0)
364  {
365  ascent = 940;
366  descent = 288;
367  }
368 
369  real_type lHorizontalShift = 0.0f;
370  bool startInNewline = false;
371 
372  if ((x1 < prevX2/2.0 && y1 > prevY1 && y2 > prevY2) || (x1 >= prevX2/2.0 && y1 > currentYMax))
373  {
374  startInNewline = true;
375  lineNumber++;
376  lHorizontalShift = x1;
377  }
378  else
379  lHorizontalShift = x1 - prevX2;
380 
381  // insert line break (<br> tag) if new line
382  if (startInNewline && lineNumber > 1)
383  {
384  real_type verticalShift = prevBaseline - (lineNumber - 1) * lineHeight;
385  htmlStream << "<br class='as' style=\"top:" << SET_FIXED_POINT_PRECISION << verticalShift << "px;\">";
386  }
387 
388  real_type verticalShift = baseline - lineNumber * lineHeight;
389 
390  // insert horizontal shift between text-runs
391  if (fabs(lHorizontalShift) >= WEBKIT_MARGIN_MAGNITUDE_THRESHOLD && !(startInNewline && ((alignment & Paragraph<app_traits>::RIGHT_ALIGNED) || (alignment & Paragraph<app_traits>::CENTER_ALIGNED))))
392  {
393  htmlStream << "<span class=\"hs\" style=\"top:" << SET_FIXED_POINT_PRECISION << verticalShift << "px; margin-left:" << SET_FIXED_POINT_PRECISION << lHorizontalShift << "px;\"></span>";
394  }
395 
396  // generate HTML for text-run
397  htmlStream << "<span class=\"as atr\" style=\"top:" << SET_FIXED_POINT_PRECISION << verticalShift << "px;";
398 
399  // avoid negative char/word-spacings to diverge due to webkit's inability to handle fractional values
400  if (lCharSpaceX < 0.0f)
401  lCharSpaceX = std::ceil(lCharSpaceX);
402 
403  if (lWordSpaceX < 0.0f)
404  lWordSpaceX = std::ceil(lWordSpaceX);
405 
406  if (fabs(lCharSpaceX) > 0.0f)
407  htmlStream << "letter-spacing:" << SET_FIXED_POINT_PRECISION << lCharSpaceX << "px;";
408 
409  if (fabs(lWordSpaceX) > 0.0f)
410  htmlStream << "word-spacing:" << SET_FIXED_POINT_PRECISION << lWordSpaceX << "px;";
411 
412  htmlStream << "font-family:" << lFontName << ";" <<
413  "font-size:" << lFontSize << "px;" <<
414  "color:rgba(" << r << "," << g << "," << b << ",1.0);" <<
415  "\">";
416 
417  bool isWordSpacingSet = fabs(lWordSpaceX) > 0.0f;
418 
419  bool stringContainsSelectionStart = false;
420  bool stringContainsSelectionEnd = false;
421 
422  if (pState.m_handleTextSelection)
423  {
424  if (dlEntryNum == pState.m_selectionStartContentPoint.m_dlEntryNum)
425  stringContainsSelectionStart = true;
426 
427  if (dlEntryNum == pState.m_selectionEndContentPoint.m_dlEntryNum)
428  stringContainsSelectionEnd = true;
429  }
430 
431  size_t subStrVectorSize = lTextSubStrVector.size();
432 
433  for (size_t i = 0; i < subStrVectorSize; ++i)
434  {
435  const text_string_type& text = lTextSubStrVector[i].first;
436  real_type inlineAdjustmentX = -1 * lTextSubStrVector[i].second * lDerivedScaleX;
437 
438  if (fabs(inlineAdjustmentX) >= WEBKIT_MARGIN_MAGNITUDE_THRESHOLD)
439  {
440  htmlStream << "<span class='ia' style=\"margin-left:" << SET_FIXED_POINT_PRECISION << inlineAdjustmentX << "px;\">" << ZERO_WIDTH_NON_BREAKING_HTML_ENTITY << "</span>";
441  }
442 
443  writeSubStringUsingHtmlEscapes(htmlStream, text, isWordSpacingSet,
444  (stringContainsSelectionStart && i == pState.m_selectionStartContentPoint.m_stringNum),
445  (stringContainsSelectionEnd && i == pState.m_selectionEndContentPoint.m_stringNum),
446  pState,
447  lFontName);
448  }
449 
450  htmlStream << "</span>";
451 
452 #if defined(_DEBUG) && DISPLAY_TEXT_METRICS
453  // display PDF baselines for reference
454  htmlStream << "<div class=\"baseline\" style=\"" <<
455  "background:black;"<<
456  "position:absolute;" <<
457  "white-space:nowrap;" <<
458  "left : " << x1 <<"px;" <<
459  "top : " << baseline <<"px;" <<
460  "width : " << x2 - x1 <<"px;" <<
461  "height : " << 1 <<"px;\"></div>";
462 #endif
463  prevX1 = x1;
464  prevX2 = x2;
465  prevY1 = y1;
466  prevY2 = y2;
467  prevBaseline = baseline;
468  currentYMax = std::max(currentYMax, std::max(y1, y2));
469  }

471  static void writeSubStringUsingHtmlEscapes(std::stringstream& htmlStream, const text_string_type& str, bool isWordSpacingSet,
472  bool substringContainsSelectionStart, bool substringContainsSelectionEnd, GeneratorState& pState,
473  const t4::string& fontName)
474  {
475  size_t strSize = str.size();
476 
477  for (size_t i = 0; i < strSize; ++i)
478  {
479  auto c = str[i];
480 
481  // In Unicode, Private Use Areas (codepoints 57344 to 63743) are intentionally not assigned any characters
482  // so that third parties can define their own characters without conflicting with the Unicode Consortium
483  // assignments. Souce : http://en.wikipedia.org/wiki/Private_Use_Areas
484  //
485  // Example of such usage is a PDF file with a ligature (Th) which has no assignment in unicode. The returned
486  // unicode for that ligature falls in the Private Use Area but iOS uses the same area for Emojis/Emoticons.
487  // Hence, upon "Copy"/"Paste" or during editing in our case, the emoticon appears instead of the glyph for which
488  // no unicode exists. Hence. we skip over these characters for editing for now.
489  //
490  // TODO_READER_MOBILE<rosinha> : should we cancel editing in presence of such invalid characters ?
491  // Another solution is to find a solution which preserves the glyph bitmap from the PDF font file during editing.
492  if (c >= 57344 && c <= 63743)
493  continue;
494 
495  // [Bug# 3592753]: There seem to be files out there with following unicode patterns :
496  // 9 13 32 160
497  // 9 13 32 non-160
498  // where 9 = tab, 13 = carriage return, 32 = space, 160 = non-breaking space
499  //
500  // Since carriage return implies a line break, we cannot use it in the textrun.
501  // Ignoring their logical meaning, in order to have these unicodes match text's
502  // rendering, it appears that carriage return should eat up the preceding tab.
503  // Also, non-breaking space should eat up the preceding space. Hence, the above
504  // patterns are used as follows :
505  // 9 13 32 160 -> 160
506  // 9 13 32 non-160 -> 32 non-160
507 
508  if (c == 9 && i+2 < strSize && str[i+1] == 13 && str[i+2] == 32)
509  {
510  if (i+3 < strSize && str[i+3] == 160)
511  i+=2; // found unicodes (9 13 32 160)
512  else
513  i++; // found unicodes (9 13 32 non-160)
514 
515  continue;
516  }
517 
518  // disallow NULL, newline and carriage return characters within a span
519  if (c == 0 || c == 10 || c == 13)
520  continue;
521 
522  bool textSelectionBoundaryChar = false;
523 
524  // If we have successfully matched the ContentPoint corresponding to text selection begin/end by matching the following :
525  //
526  // 1. DL entry number (corresponding to Tj operator)
527  // 2. substring index within the matched Tj operator
528  // 3. glyph index within the matched substring
529  //
530  // we are ready to mark the begin/end characters within HTML which will be used in javascript to initialize the same text selection
531  // upon display of webview.
532  if (substringContainsSelectionStart)
533  {
534  size_t effectiveGlyphNum = pState.m_selectionStartContentPoint.m_glyphNum;
535 
536  if (pState.m_selectionStartContentPoint.m_isAfter != 0)
537  ++effectiveGlyphNum;
538
