117 public:
118  GrayToAlphaRasterPainter( const constraints_type&,
119  t4::brief_ptr<raster_painter_type> src )
120  : m_src( std::move( src ) ),
121  m_segments( m_src->GetNumSegments(), *t4::get_thread_heap() ),
122  m_identityWalker{}
123  {
124  auto resultIt = m_segments.begin();
125  const segment_type* srcIt = SegmentsBegin( t4::get_ptr( m_src ) );
126  const segment_type* srcEnd = SegmentsEnd( t4::get_ptr( m_src ) );
127  while ( srcIt != srcEnd )
128  {
129  resultIt->m_shape = MakeIdentityWalker();
130  resultIt->m_opacity = srcIt->m_color;
131  resultIt->m_color = {};
132  ++srcIt;
133  ++resultIt;
134  }
135

137  // RasterPainter never needs SetY calls. SetX is only needed if src needs SetX.
138  imaging_model::RasterPainterProperties props = imaging_model::kSetY_Unneeded;
139  if ( m_src->GetProperties() & imaging_model::kSetX_Unneeded )
140  props |= imaging_model::kSetX_Unneeded;
141  this->Initialize( m_src->GetNumSegments(), m_segments.data(), m_src->GetNonZeroBounds(), props );
142 
143  #ifdef _DEBUG
144  this->SetName( "GrayToAlpha" );
145  #endif
146  }

344 template<typename signal_traits>
345 typename SoftMask<signal_traits>::smart_ptr_type MakeSoftMask(
346  document::DocumentCaches<typename signal_traits::app_traits>& documentCaches,
347  const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& curCS,
348  typename signal_traits::app_traits::int_geometry_traits::coord_type greekThreshold,
349  GStateConsumerContext& gsContext,
350  display_event_stream const& eventStream,
351  const store::Dictionary<store::StoreObjTraits<typename signal_traits::app_traits> >* pageResources,
352  const imaging_model::Constraints& constraints,
353  const RenderGState<signal_traits>& gState,
354  const store::Object<store::StoreObjTraits<typename signal_traits::app_traits> >& softMaskObj )
355 {
356  using namespace imaging_model;
357 
358  typedef typename signal_traits::app_traits app_traits;
359  typedef typename SoftMask<signal_traits>::smart_ptr_type result_type;
360  typedef store::StoreObjTraits<app_traits> obj_traits_type;
361  typedef store::Dictionary<obj_traits_type> store_dict_type;
362  typedef store::Name<obj_traits_type> store_name_type;
363  typedef store::Object<obj_traits_type> store_object_type;
364  typedef typename app_traits::real_geometry_traits::matrix_type matrix_type;
365  typedef TransparencyGroup<signal_traits> group_type;
366  typedef GState gstate_type;
367  typedef gstate_type* gstate_ptr_type;
368  typedef typename color::ColorSpace<app_traits>::smart_ptr_type color_space_ptr_type;
369  typedef RasterPainter<signal_traits> raster_painter_type;
370  typedef t4::brief_ptr<raster_painter_type> raster_painter_ptr_type;
371  typedef typename app_traits::pdf_traits::display_traits display_traits;
372 
373  auto& heap = *t4::get_thread_heap();
374 
375  if ( softMaskObj.Type() == store::kObjectTypeName && softMaskObj.NameValue_NoTypeCheck() == "None" )
376  return result_type();
377 
378  store_dict_type softMaskDict = softMaskObj.DictionaryValue();
379  store_dict_type formDict = softMaskDict.GetRequiredDictionary( "G" );
380 
381  matrix_type mat;
382  store_object_type obj = formDict.Get( "Matrix" );
383  if ( obj.Type() != store::kObjectTypeNull )
384  mat = store::GetMatrix<matrix_type>( obj.ArrayValue() );
385  else
386  mat = IdentityMatrix<matrix_type>();
387 
388  boost::optional<store_dict_type> groupDict = formDict.GetDictionary( "Group" );
389 
390  GraphicLayoutDescriptor desc = MakeGraphicLayoutDescriptor( PixelBufferLayout::kReplicatedSignal,
391  GraphicLayoutDescriptor::kShapeImplicitlyOne,
392  GraphicLayoutDescriptor::kOpacityReplicated,
393  GraphicLayoutDescriptor::kColorIgnored );
394 
395  typedef GraphicStore<signal_traits> gstore_type;
396  typedef typename app_traits::color_traits color_traits;
397 
398  // This GraphicStore stores the shape/opacity for the result. If this is an /Alpha type soft mask, we just render directly into
399  // this. For /Luminosity soft masks we need to render and convert into this buffer.
400  auto gStore = t4::make_unique<gstore_type>( constraints.m_interestBounds, desc, false );
401 
402  {
403  t4::transient_snapshot ss( heap );
404  auto contentRec = t4::make_unique<content::FormXObjectContentRecord>( pageResources, formDict );
405 
406  store_name_type softMaskType = softMaskDict.GetRequiredName( "S" );
407  bool isAlphaGroup = true;
408 
409  group_type* group = nullptr;
410 
411  if ( softMaskType == "Alpha" )
412  {
413  TransparencyBackdrop<signal_traits>* backdrop = t4::make_transient<IsolatedBackdrop<signal_traits>>( heap, curCS );
414  group = t4::make_transient<group_type>( heap, constraints, gStore->GetWritableBuffers(), curCS, desc, backdrop, true /*actually irrelevant*/,
415  groupDict ? groupDict->GetBoolean( "K", false ) : false );
416  }
417  else if ( softMaskType == "Luminosity" )
418  {
419  isAlphaGroup = false;
420  color_space_ptr_type cs = GetSoftMaskColorSpace( pdfcolor::GetColorSpaceCache(documentCaches), formDict, contentRec->GetResources().get_ptr() );
421  if ( !cs )
422  throw t4::corrupt_data();
423  if ( cs && !display_traits::kRespectTransparencyGroupColorSpaces )
424  cs = curCS;
425  group = MakeSoftMaskLuminosityTransparencyGroup<signal_traits>( softMaskDict, constraints, pdfcolor::GetColorSpaceCache(documentCaches), contentRec->GetResources().get_ptr(), cs );
426  }
427  else
428  throw t4::unimplemented();
429 
430  typedef HardClip<signal_traits> hard_clip_type;
431  hard_clip_type hardClip( constraints.m_interestBounds );
432 
433  RenderContext<signal_traits>* rc = t4::make_transient<RenderContext<signal_traits>>( heap,
434  *group,
435  hardClip,
436  gState.GetUnreflowedDefaultUserSpaceToDeviceSpaceTransform(),
437  documentCaches,
438  greekThreshold );
439 
440  content::DisplayList* dl = t4::make_transient<content::DisplayList>( heap, std::move(contentRec), documentCaches );
441 
442  gstate_ptr_type clonedGState = gsContext.CloneGState( &gState );
443  clonedGState->ResetTransparencyGState();
444  clonedGState->SetContentRecord(&dl->GetContentRecord());
445  clonedGState->Concat( mat );
446 
447  RenderConsumer<signal_traits> cons( *rc, document::PageID() /*pageID*/, &dl->GetContentRecord(), eventStream, nullptr /*reflowInfoPtr*/, nullptr /* ocState */, nullptr /*croppedMediaBox*/, nullptr /*colorServerOverride*/ );
448  cons.SetGState( static_cast<RenderGState<signal_traits>&>( *clonedGState ) );
449  dl->EnumerateContent( cons );
450 
451  if ( !isAlphaGroup )
452  {
453  // If this is an alpha-based group, the group painted the correct alpha values directly into the
454  // GraphicStore, and we don't need to do anything. For luminosity-based groups we need to convert
455  // to XYZ, and use the luminosity as alpha.
456  raster_painter_ptr_type rp = ConvertLuminosityToAlpha( std::integral_constant<bool, color_traits::supportICC>(),
457  constraints, group->GetColorSpace(), group->GetResult( constraints, group->GetColorSpace() ) );
458  Blit( constraints, &gStore->GetWritableBuffers(), std::move( rp ) );
459  }
460  }
461 
462  content::ConvertBuffer( gStore->GetWritableBuffers().m_opacity, constraints, softMaskDict.Get( "TR" ) );
463  return t4::make_shared<SoftMask<signal_traits>>( std::move( gStore ) );
464 }
