152 template<typename stroker_traits>
153 void ContourSegment<stroker_traits>::Reverse()
154 {
155  ReversePointList(m_point, m_numPts);
156  coord_type tmp = m_startT;
157  m_startT = 1 - m_endT;
158  m_endT = 1 - tmp;
159  flags_type newFlags = m_flags & ~(kStartIsUE | kEndIsUE);
160  if (m_flags & kStartIsUE)
161  newFlags |= kEndIsUE;
162  if (m_flags & kEndIsUE)
163  newFlags |= kStartIsUE;
164  m_flags = newFlags;
165 }

169 {
170  point_type* pA = pts;
171  point_type* pB = pts+(numPts-1);
172  while (pA < pB)
173  boost::swap(*pA++, *pB--);

250 template<typename stroker_traits>
251 typename ContourSegment<stroker_traits>::coord_type ContourSegment<stroker_traits>::computeTrapezoidWidthForArea(coord_type heightA, coord_type heightB, coord_type width, coord_type desiredArea)
252 {
253  // We now need to find a width such that the "first part of the given
254  // trapezoid up to newWidth" has the requested area.
255  coord_type avgHeight = real_services::Average(heightA, heightB);
256 
257  // sanity check our inputs. We want positive heights, and a trpaezoid whose area is bigger then
258  // the desired area (return width is between zero and width).
259  tetraphilia_assert(width > 0 && heightA >= 0 && heightB >= 0);
260  tetraphilia_assert(width*avgHeight >= desiredArea);
261 
262  // To keep precision up, we scale to a coordinate system where the trapezoids width is 1 and
263  // average height is one. We then wnat to know for what "x" the area is desiredArea/(width*avgHeight).
264  coord_type A = desiredArea / (width*avgHeight);
265  coord_type d = (heightB-heightA)/(heightB+heightA);
266  coord_type newWidth = 0;
267  if (AbsVal(d) < coord_type(0.01))
268  {
269  // quadratic formula wont work very well, but the trapezoid is nearly
270  // a rectangle so linear interp is good anyhow
271  newWidth = desiredArea/avgHeight;
272  }
273  else
274  {
275  coord_type x = ((d-1) + real_services::Sqrt((1-d)*(1-d) + 4*d*A))/(2*d);
276  newWidth = x * width;
277  }
278 
279  if (newWidth < 0)
280  return 0;
281  if (newWidth > width)
282  return width;
283  return newWidth;
284 }

286 template<typename stroker_traits>
287 typename ContourSegment<stroker_traits>::coord_type ContourSegment<stroker_traits>::ComputeEndForArclen(coord_type arclen, coord_type bezInterval_dt, coord_type startT, coord_type* pEndT ) const
288 {
289  if (IsBezier())
290  {
291  coord_type remaining = arclen;
292 
293  // special handling of fairly common zero length dash case
294  if (remaining == 0)
295  {
296  *pEndT = startT;
297  return 0;
298  }
299 
300  // Now compute the derivative function, first compute the cubic poly,
301  // then derivative.
302 
303  point_type dcoeffs[4];
304  point_type p21_3 = (m_point[2] - m_point[1]) * 3;
305  dcoeffs[0] = m_point[0];
306  dcoeffs[3] = m_point[3] - p21_3 - m_point[0];
307  dcoeffs[1] = (m_point[1] - m_point[0]) * 3;
308  dcoeffs[2] = p21_3 - dcoeffs[1];
309 
310 #ifdef _DEBUG
311  cubic_bernstein_poly_storage<coord_type> cX;
312  cX.Set(get_x(m_point[0]), get_x(m_point[1]), get_x(m_point[2]), get_x(m_point[3]));
313  tetraphilia_assert(cX.c[0] == dcoeffs[3].x);
314  tetraphilia_assert(cX.c[1] == dcoeffs[2].x);
315  tetraphilia_assert(cX.c[2] == dcoeffs[1].x);
316 #endif
317 
318  dcoeffs[0] = dcoeffs[1];
319  dcoeffs[1] = 2*dcoeffs[2];
320  dcoeffs[2] = 3*dcoeffs[3];
321  // dcoeffs[3] = 0; not used
322 
323  // now we numerically integrate t=[arcBeginT, ??] speed(t) until
324  // we eat up the requested length. This integration is done using
325  // the standard trapezoidal numerical integration method.
326  coord_type intervalBeginT = startT;
327  coord_type intervalBeginSpeed = (dcoeffs[0] + intervalBeginT*(dcoeffs[1] + intervalBeginT*dcoeffs[2])).Mag();
328  while (intervalBeginT < m_endT)
329  {
330  coord_type intervalEndT = intervalBeginT + bezInterval_dt;
331 
332  // see if we past the end of of the curve, if so we will
333  // decrease dt to account for it
334  if (intervalEndT > m_endT)
335  {
336  bezInterval_dt -= intervalEndT-m_endT; // we can just modify bezInterval_dt because we know we are done no matter what
337  intervalEndT = m_endT;
338  }
339 
340  // this calculates the arclen ~= trapezoidal area of the interval, i.e.
341  // one numerical integration step using the trapezoidal method.
342  coord_type intervalEndSpeed = (dcoeffs[0] + intervalEndT*(dcoeffs[1] + intervalEndT*dcoeffs[2])).Mag();
343  coord_type intervalArclen = bezInterval_dt*(intervalBeginSpeed+intervalEndSpeed)/2;
344 
345  // see if we overshot....
346  if (remaining <= intervalArclen)
347  {
348  // We're done! The answer lies in this section.
349  // Find a new interpolated dt;
350  coord_type interpolated_dt = computeTrapezoidWidthForArea(intervalBeginSpeed, intervalEndSpeed, bezInterval_dt, remaining);
351 
352  // BIG NOTE: once Chip decided to see whether adding a refinement on here would
353  // help precision a lot. The idea was that if there were no "whole dt steps" so far,
354  // we would take the dt given by the computeTrapezoidWidthForArea() call and feed
355  // that back into the integration step (once) to refine the numbers.
356  // It did improve accuracy, but VERY little. It appears that for monotone beziers
357  // the dt number 1/16 is such that it truly captures the curvature in t=[0,1], so the
358  // refinement is simply unnecessary.
359  *pEndT = intervalBeginT + interpolated_dt;
360 
361  // note that precision problems can cause the above to produce a number larger than
362  // and and that will trip up some of the logic later on, so here we pin the value
363  if ( *pEndT > 1 )
364  *pEndT = 1;
365  return 0;
366  }
367 
368  // OK, we can just eat this segment up and move onto the next interval
369  remaining -= intervalArclen;
370  intervalBeginT = intervalEndT;
371  intervalBeginSpeed = intervalEndSpeed;
372  }
373 
374  // oh well, ran out of curve.
375  *pEndT = m_endT;
376  return remaining;
377  }
378  else
379  {
380  tetraphilia_assert(IsLine());
381  coord_type remaining = arclen;
382 
383  // special handling of fairly common zero length dash case
384  if (remaining == 0)
385  {
386  *pEndT = startT;
387  return remaining;
388  }
389 
390  coord_type lineLen = (m_point[1] - m_point[0]).Mag();
391 
392  // see if we should eat up the whole line. Note that this
393  // also removes the case where lineLen == 0, so we can divide by
394  // it later
395  coord_type maxLen = lineLen*(m_endT-startT);
396  if (maxLen <= remaining)
397  {
398  *pEndT = m_endT;
399  return remaining - maxLen;
400  }
401 
402  // figure out what endT would yield a length of *pLengthRemaining
403  *pEndT = remaining/lineLen + startT;
404  if (*pEndT > m_endT)
405  *pEndT = m_endT; // make sure we dont overflow because of imprecison, should always be less than m_endT
406  return 0;
407  }
408 }

410 template<typename stroker_traits>
411 void ContourSegment<stroker_traits>::AddToPath(output_path_type& path, bool includeMoveTo) const
412 {
413  if (IsBezier())
414  {
415  if (m_startT == 0 && m_endT == 1)
416  {
417  if (includeMoveTo)
418  path.MoveTo(m_point[0]);
419  path.CurveTo(m_point[1], m_point[2], m_point[3]);
420  }
421  else
422  {
423  point_type q0, q1, q2, q3;
424  ComputeBezSection(m_point[0], m_point[1], m_point[2], m_point[3], m_startT, m_endT, q0, q1, q2, q3);
425  if (includeMoveTo)
426  path.MoveTo(q0);
427  path.CurveTo(q1, q2, q3);
428  }
429  }
430  else
431  {
432  tetraphilia_assert(IsLine());
433  if (includeMoveTo)
434  path.MoveTo(GetStartPoint());
435  path.LineTo(GetEndPoint());
436  }
437 }

600 template <typename stroker_traits>
601 void Contour<stroker_traits>::AppendUserBezierSegment( const point_type& p0, const point_type& p1, const point_type& p2, const point_type& p3 )
602 {
603  // First, see if this is one of the Illustrator lines masquerading as a true bezier,
604  // if so just commit it as a line.
605  if (p0 == p1 && p2 == p3)
606  {
607  AppendUserLineSegment(p0, p3);
608  return;
609  }
610 
611  if (m_params.ShouldOptimizeOrigPathOutsideOfVisibleRect())
612  {
613  if (m_params.InOneOctantOutsideOfVisible(p0, p1, p2, p3))
614  return; // can skip this entirely and save computation
615  else
616  connectLastSegToPtAsNeeded(p0);
617  }
618 
619  // we split bezier curves where either the x or y derivative vanishes; this
620  // turns out to dramatically improve the performance of dashing and offset
621  // curve computations since it "removes" crimps/cusps and makes us able
622  // to assume that the beziers are monotonic in x and y, reducing some other
623  // complexitites.
624  point_type monotonePts[16];
625  size_t numSections = BezGetMonotoneSections(monotonePts, p0, p1, p2, p3);
626  tetraphilia_assert(numSections <= 5);
627 
628  point_type* curPos = monotonePts;
629  for ( size_t section = 0; section < numSections; section++ )
630  {
631  typename contour_seg_type::flags_type flags = 0;

641 template <typename stroker_traits>
642 void Contour<stroker_traits>::AppendFlattenedBezierSegment( point_type p0, point_type p1, point_type p2, point_type p3, typename contour_seg_type::flags_type flags, size_t depth )
643 {
644  tetraphilia_assert(depth <= m_params.GetMaxFlatnessRecursionDepth());
645 
646  // we do these "per split" because often big parts of the curve are out the clip when zoomed in, we do not
647  // want high precision computations outside of the drawn region.
648  if (m_params.ShouldOptimizeOrigPathOutsideOfVisibleRect())
649  {
650  if (m_params.InOneOctantOutsideOfVisible(p0, p1, p2, p3))
651  return; // can skip this entirely and save computation
652  else
653  connectLastSegToPtAsNeeded(p0);
654  }
655 
656  // we do a dot product "distance" check
657  point_type v; // perp
658  v.x = get_y(p3)-get_y(p0);
659  v.y = get_x(p0)-get_x(p3);
660  coord_type vlen = v.TaxicabLength();
661 
662  // we know the control points are within the box defined by the
663  // endpoints (outset by m_flatness), and the math starts to go bad
664  // if the two endpoints are too close together, so the PS code calls it
665  // close enough if the len is small (compared to m_flatness)
666 
667  coord_type tolerance = m_params.GetUserSpaceFlatness();
668 
669  if (depth < m_params.GetMaxFlatnessRecursionDepth() && vlen > real_services::DivideBy4( tolerance ))
670  {
671  // watch for overflow and split
672  size_t overflow = false;
673 
674  // multiply tolerance by vlen for comparison
675  tolerance = real_services::Multiply(tolerance,vlen,&overflow);
676 
677  if (AbsVal(Dot(v, (p1-p0), &overflow)) > tolerance || AbsVal(Dot(v, (p2-p0), &overflow)) > tolerance || overflow)
678  {
679  // We need to split and recurse
680  point_type r0, r1, r2, r3; // "right part" of curve
681  r3 = p3;
682  r2 = (p2+p3)/static_cast<int_type>(2);
683  p3 = (p1+p2)/static_cast<int_type>(2);
684  p1 = (p0+p1)/static_cast<int_type>(2);
685  p2 = (p1+p3)/static_cast<int_type>(2);
686  r1 = (p3+r2)/static_cast<int_type>(2);
687  r0 = (p2+r1)/static_cast<int_type>(2);
688  p3 = r0;
689  AppendFlattenedBezierSegment(p0, p1, p2, p3,
690  static_cast<typename contour_seg_type::flags_type>(flags & ~(contour_seg_type::kEndIsUE)), // typecast required to quiet VC.NET Level 4 (/W4) warning
691  depth+1);
692  AppendFlattenedBezierSegment(r0, r1, r2, r3,
693  static_cast<typename contour_seg_type::flags_type>(flags & ~(contour_seg_type::kStartIsUE)), // typecast required to quiet VC.NET Level 4 (/W4) warning
694  depth+1);
695  return;
696  }
697  }
698 
699  // flat enough call it a line
700  AppendLineSegment(p0, p3, flags);
701 }

703 template <typename stroker_traits>
704 void Contour<stroker_traits>::Close(const point_type& curpt, const point_type& startpt)
705 {
706  // The original path has a closepath, but the "closing segment" may be clipped out. Therefore the contour we are
707  // stroking may or may not actually be closed.
708  if (m_params.ShouldOptimizeOrigPathOutsideOfVisibleRect() && m_params.InOneOctantOutsideOfVisible(curpt, startpt))
709  return; // no need to close, the closing segment is entirely outside the clip.
710 
711  m_isClosed = true;
712 
713  // see if the curve basically is already closed, if so just set the is closed flag and don't generate a
714  // countour segment.
715  if (GetNumSegs() > 0 && point_type::EssentiallyEqual(curpt, startpt))
716  {
717  tetraphilia_assert(m_segmentStack.Top().GetEndPoint() == curpt);
718  m_segmentStack.Top().SnapEndToPoint(startpt);
719  return;
720  }
721 
722  // OK, looks like we need a segment to express the close operation.
723  AppendUserLineSegment(curpt, startpt);
724 }

737 template <typename stroker_traits>
738 void Contour<stroker_traits>::AddToPath(output_path_type& path, bool includeMoveTo) const
739 {
740  typename contour_seg_stack_type::const_iterator i = m_segmentStack.begin();
741  typename contour_seg_stack_type::const_iterator end = m_segmentStack.end();
742  if (i == end)
743  {
744  tetraphilia_assert(!m_isClosed);
745  return;
746  }
747 
748  i->AddToPath(path, includeMoveTo);
749  while (1)
750  {
751  ++i;
752  if (i != end)
753  i->AddToPath(path, false);
754  else
755  break;
756  }
757 
758  if (m_isClosed)
759  path.ClosePath();
760 }
761 
762 // computes an endT val for a portion of seg, starting at arcBeginT, ending at the
763 // endT val, having length AT MOST arclen. The return value is the amount of arclength
764 // NOT consumed; it will be 0 if the whole arclen
765 // is consumed.
766 template <typename stroker_traits>
767 typename Contour<stroker_traits>::coord_type Contour<stroker_traits>::ComputeEndForArclen( coord_type arclen, const contour_pos_type& start, contour_pos_type* pEnd, bool* pHitEnd) const
768 {
769  coord_type remaining = arclen;
770  *pEnd = start;
771  while (1)
772  {
773  // advances pEnd possibly up to the end of its current segment
774  remaining = m_segmentStack[pEnd->m_segIndex].ComputeEndForArclen(remaining, m_params.GetDasherArcLenIntegrationStep(), pEnd->m_segT, &pEnd->m_segT);
775 
776  if (remaining == 0 || pEnd->m_segIndex == m_segmentStack.size()-1)
777  {
778  // the PS interpreter does not "finish" a line or bezier
779  // segment until it moves PAST the end of that segment. Therefore
780  // we cannot be done with the contour until we have gone past
781  // the end of the contour, i.e. there is some remaining.
782  // Think of a line of length 100, dashed with [0, 100]. The question
783  // is whether the is a dash at the beginning and end, or just at the beginning.
784  *pHitEnd = ((remaining != 0) && *pEnd == GetEndPosition());
785  return remaining;
786  }
787 
788  // advance to next segment
789  pEnd->m_segIndex++;
790  pEnd->m_segT = m_segmentStack[pEnd->m_segIndex].GetStartT();
791  }
792 }

808  AppendContourSegmentSubsection(contour[i], 0, 1);
809  AppendContourSegmentSubsection(contour[end.m_segIndex], 0, end.m_segT);
810  }
