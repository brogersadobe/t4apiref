186 /* IMPORTANT NOTE<vmarda>: Recently, via CL#129675 in Prithvi branch, I have replaced all structure order comparisons of content
187  ranges by document order comparisons. After this change, it's not correct to use RestartableTextDLConsumer for text-extraction
188  on a single content range spanning multiple structure nodes. That might lead to incorrect results. After this change, the client
189  of RestartableTextDLConsumer needs to ensure that document order comparisons are okay for the content range on which the consumer
190  is working. For example, the StructureEnumerator uses this consumer for content range belonging to a single structure node at a
191  time. Hence structure comparisons are not required for that use-case, which happens to be the only usage of this consumer right
192  now.
193  */

242  template<typename show_data_type>
243  void HandleTextShow( show_data_type const& showData,
244  pdf::content::TextBBoxInfo & bBoxInfo )
245  {
246  m_pROContentRange = m_consumerClient.GetRange();
247  tetraphilia_assert( m_pROContentRange );
248 
249  bool isInside = false;
250  if( m_pROContentRange->m_contentRange.Contains( this->m_curContentPoint ) )
251  isInside = true;
252  else
253  {
254  content::ContentPointLessThan_DocumentOrder order;
255 
256  // Although m_curContentPoint is outside the current range,
257  // roContentRange.m_contentRange might start somewhere within the current
258  // DLEntry, but not with the very first glyph; should that be the case, we
259  // first advance m_curContentPoint until it matches the start of the range,
260  // then we call DoTextShow to handle the portion of the current DLEntry
261  // that's inside the range. If we are in reflow mode we need to also
262  // advance the reflow iterator so that it points to the next
263  // ContentRangeReflowLayout. Normally, this iterator advances once per
264  // show, but this is a special case in which the show spans multiple
265  // paragraphs:
266  if( this->m_curContentPoint.m_dlEntryNum ==
267  m_pROContentRange->m_contentRange.m_from.m_dlEntryNum )
268  {
269  while( this->m_curContentPoint.m_stringNum <
270  m_pROContentRange->m_contentRange.m_from.m_stringNum )
271  this->m_curContentPoint.AdvanceStringNum();
272 
273  while( this->m_curContentPoint.m_glyphNum <
274  m_pROContentRange->m_contentRange.m_from.m_glyphNum )
275  this->m_curContentPoint.AdvanceGlyphNum();
276 
277  tetraphilia_assert( this->m_curContentPoint.m_stringNum ==
278  m_pROContentRange->m_contentRange.m_from.m_stringNum &&
279  this->m_curContentPoint.m_glyphNum ==
280  m_pROContentRange->m_contentRange.m_from.m_glyphNum );
281 
282  if( this->InReflowMode() )
283  while( this->m_reflowLayoutIter != this->m_reflowLayoutEnd
284  && t4::lt( (*this->m_reflowLayoutIter)->GetContentRange().m_to, this->m_curContentPoint, order ) )
285  {
286  ++this->m_reflowLayoutIter;
287  }
288 
289  isInside = true;
290  }
291  }
292 
293  if( isInside )
294  TextDLConsumer::DoTextShowImpl( showData, bBoxInfo );
295  }

384 template<typename consumer_client_type>
385 bool RestartableTextDLConsumer<consumer_client_type>::HandleShow_Begin(
386  bool hasValidUCM,
387  bool isVisible )
388 {
389  m_pROContentRange = m_consumerClient.GetRange();
390  m_attrib = m_consumerClient.GetAttributes();
391  tetraphilia_assert( m_pROContentRange && m_attrib );
392  mark_handler_type& markHandler =
393  m_consumerClient.GetMarkHandler();
394 
395  bool goDeeper = false;
396  if( !m_attrib->IsDropShadow() || markHandler.Supports( kDropShadows ) )
397  goDeeper = markHandler.HandleShow_Begin(
398  *this->m_pGState,
399  this->m_curContentPoint,
400  *m_pROContentRange,
401  m_attrib,
402  *this->m_pShowInfo,
403  hasValidUCM,
404  isVisible && m_attrib->IsVisible() );
405 
406  if( markHandler.IsStopping() )
407  this->StopConsumption();
408 
409  m_checkRangeWhenDone = true;
410  m_checkContentPoint = this->m_curContentPoint;
411  if( !goDeeper )
412  {
413  tetraphilia_assert( this->m_pShowInfo );
414  m_checkContentPoint.m_stringNum = this->m_pShowInfo->GetNumSubstrings() - 1;
415  m_checkContentPoint.m_glyphNum = this->m_pShowInfo->GetNumGlyphsInSubstring( m_checkContentPoint.m_stringNum ) - 1;
416  m_checkContentPoint.m_totalNumUnicode = 0;
417  m_checkContentPoint.m_isAfter = true;
418  }
419 
420  return goDeeper;
421 }

997 template<typename app_traits>
998 void StructureEnumerator<app_traits>::EnumerateCurrentRange( void )
999 {
1000  content::ContentPointLessThan_DocumentOrder order;
1001 
1002  tetraphilia_assert( m_curStructureNode );
1003 
1004  const ro_content_range_type* roContentRange = m_curStructureNode->GetRange();
1005  tetraphilia_assert( roContentRange );
1006  if( !roContentRange )
1007  throw t4::unknown();
1008 
1009 #ifndef T4_SINGLE_THREADED
1010  if( !IsValid() ||
1011  t4::gt( this->m_curContentPoint, roContentRange->m_contentRange.m_to, order ) )
1012  {
1013  JoinThread();
1014 
1015  /*
1016  Explanation:
1017 
1018  Whenever any exception is thrown in the 'worker' thread that thread stashes the error object
1019  in its 'm_error' member, signals m_rangeDoneEvent, and then rethrows the error. This includes
1020  thread termination errors. The reason for this is that in most cases we want to propagate any
1021  error that is thrown in that thread into this thread. But in the line above when we reset
1022  m_thread, it can be because this new content range is back before the one that m_thread handled
1023  last time, so we need to restart that thread at the beginning. If that happens that thread is
1024  killed by injecting a thread termination error, which will cause it to set m_error and signal
1025  m_rangeDoneEvent. We don't want to consider either of those in this case where we have intentionally
1026  killed the thread.
1027 
1028  */
1029  m_newRangeEvent.Reset();
1030  m_rangeDoneEvent.Reset();
1031  m_error = {};
1032 
1033  // create a new thread and get it running.
1034  m_thread = t4::thread( [ this ] {
1035  try
1036  {
1037  FPUControl<float> fpuControl;
1038  t4::establish_thread_heap eth;
1039  boost::optional<ContextLock> lock;
1040  if ( m_contextLocked )
1041  lock.emplace();
1042 #endif // T4_SINGLE_THREADED
1043  StructureEnumeratorFunctor<app_traits>( GetGlobalContext(), *this, m_reflowLayoutPtr )();
1044 #ifndef T4_SINGLE_THREADED
1045  }
1046  catch(...)
1047  {
1048  boost::optional<ContextLock> lock;
1049  if ( m_contextLocked )
1050  lock.emplace();
1051  m_error = t4::current_exception();
1052  m_rangeDoneEvent.Signal();
1053  }
1054  } );
1055  }
1056 
1057  try
1058  {
1059  m_newRangeEvent.Signal();
1060  m_rangeDoneEvent.Wait();
1061  }
1062  catch( ... )
1063  {
1064  if ( !m_error )
1065  m_error = t4::current_exception();
1066  }
1067 
1068  if ( m_error )
1069  {
1070  tetraphilia_assert( m_thread != t4::thread() );
1071 
1072  /*
1073  We're not in a catch clause here, but we really don't want JoinThread() to throw--we're
1074  about to propagate an exception and it's vital that the child thread be completely finished
1075  and unwound before we proceed.
1076  */
1077  boost::this_thread::disable_interruption di;
1078  JoinThread();
1079  t4::rethrow_exception( m_error );
1080  }
1081 #endif // T4_SINGLE_THREADED
1082 }
