31 /*
32  The global console-directed logging class. The implementation uses different APIs on different
33  platforms:
34 
35  iOS/OSX: pipes to stdout
36  Android: __android_log_print
37  Windows: OutputDebugStringA
38 
39  Example:
40 
41  // Writes "Hello world!\n" to the console
42  t4::log << "Hello world" << std::endl;
43 
44  Multithreading guarantees:
45 
46  All calls to operator<< between sequence points (i.e., semicolons) will appear uninterrupted
47  on the console. So you can be sure that the following statement:
48 
49  t4::log << "this " << "is " << "a " << "sentence.";
50 
51  Will write "this is a sentence." to the console, regardless of what other threads may be doing
52  with t4::log at the same time.
53 
54  The arguments to operator<< are not guaranteed to be _evaluated_ with any synchronization. Under the
55  hood all elements in a chain of << operators are evaluated and stored in a buffer; only when the
56  complete sequence is complete is the accumulated output piped to the console atomically. So if
57  you have the following code:
58 
59  static std::atomic<int> counter;
60 
61  auto a = std::async( [&](){ t4::log << "A: " << ++counter << " " << ++counter << " " << ++counter << std::endl; } );
62  auto b = std::async( [&](){ t4::log << "B: " << ++counter << " " << ++counter << " " << ++counter << std::endl; } );
63 
64  a.wait(); b.wait();
65 
66  You might get:
67 
68  A: 1 4 5
69  B: 2 3 6
70 
71  or:
72 
73  A: 3 5 6
74  B: 1 2 4
75 
76  or even:
77 
78  B: 4 5 6
79  A: 1 2 3
80 
81  But you will _not_ get weird interleaved output like:
82 
83  A: B 2 13 4
84  5 6
85 
86  It is important to note that t4::log is a debugging facility. It is not desirable to ship code
87  with active logging statements; indeed it is preferred that users not even submit enabled logging
88  statements to source control.
89 
90  t4::log is enabled in release builds, though, because of the possibility that it would be
91  useful in debugging issues that only appear in release builds. Nor is there a macro (like PRE_RC_ONLY)
92  to disable logging "just before we ship." Logging statements that are important to you are most
93  likely distracting noise to other engineers, so as a matter of policy please do not submit active
94  uses of t4::log to source control.
95 */

98 /*
99  Abstract base class for logging that supports C++ iostream-style streaming in a
100  multithreaded environment.
101 
102  Most of the time clients do not need to write their own implementations of log_impl
103  and can simply write their output to t4::log (see above). But if you have some private
104  logging target for which you would like to support C++ iostream-style streaming and
105  the atomicity guarantees noted above, it's easy to do by simply subclassing t4::log_impl.
106 
107  For example:
108 
109  class my_log_impl : public t4::log_impl
110  {
111  t4::ostringstream m_stm;
112 
113  public:
114  // Retrieve all logging info so far, and flush the contents. Use while_locked() to
115  // ensure that we don't flush in the middle of a call to write_c_str() from another
116  // thread.
117  t4::string get_and_flush( void )
118  {
119  return while_locked( [this]()
120  {
121  auto result = m_stm.str();
122  result.str( "" );
123  return result;
124  } );
125  }
126 
127  private:
128  void write_c_str( char const* str )
129  {
130  m_stm << str;
131  }
132  };
133 
134  Implementers do not need to do their own synchronization in write_c_str--you may presume that
135  your implementation of write_c_str will never be simultaneously entered by different threads.
136  You might need to use while_locked to insure that some other method doesn't overlap with a call
137  to write_c_str though.
138 */
139 class log_impl
140 {
141  friend class log_detail::log_buffer;
142  t4::mutex m_mutex;
143 
144 public:
145  /*
146  Various operator<< overloads to support standard C++ iostream-style streaming
147  */
148  template<typename T>
149  friend log_detail::log_buffer operator<<( log_impl& ths, T&& val );
150 
151  friend log_detail::log_buffer operator<<( log_impl& ths, std::ios& (*func)( std::ios& ) );
152 
153  friend log_detail::log_buffer operator<<( log_impl& ths, std::ostream& (*func)( std::ostream& ) );
154 
155  virtual ~log_impl( void ) {}
156 
157 protected:
158  /*
159  Implementers can use this method to perform operations that they don't want to collide with
160  a call to write_c_str from another thread.
161  */
162  template<typename F>
163  auto while_locked( F const& f ) -> decltype(f());
164 
165 private:
166  /*
167  Clients must implement this. A standard null-terminated C string in the current locale is
168  provided.
169  */
170  virtual void write_c_str( char const* str ) = 0;
171 
172  void write( std::ostringstream& stm );
173 };
