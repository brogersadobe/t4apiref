109 /*
110  Metafunction that identifies types that model the Point concept.
111 
112  A Point identifies a location in the two-dimensional coordinate plane. If
113  p is an instance of a model of the Point concept, then the following expressions:
114 
115  get_x( p )
116  get_y( p )
117 
118  retrieve the corresponding values in the point (typically using ADL). These expressions
119  must all return values of the same type. That type is available through
120  the convenience metafunction "coord_type." If point_type is a model of
121  the Point concept, then:
122 
123  coord_type<point_type>::type
124 
125  identifies the type of values returned by get_x and get_y. Implementers do not
126  need to specialize this metafunction explicitly, its implementation is derived
127  from the implementation of get_x.
128 
129  NOTE: Currently get_x/get_y are (when operating on mutable instances) must return
130  references to their internal values, so support old code. This usage is deprecated
131  and set_x/set_y is preferred.
132 
133  If p is an instance of a model of the Point concept, then the following statments:
134 
135  set_x( p, new_x );
136  set_y( p, new_y );
137 
138  Should modify the corresponding values of the point.
139 
140  The following operators must work on Points and provide expected behavior:
141 
142  operator==
143  operator!=
144  operator+
145  operator+=
146  operator-
147  operator-=
148  operator<<( std::ostream& )
149 
150  If point_type is a model of the Point concept, then the "is_point_type"
151  template metafunction should be specialized to return true_type. Only types
152  for which this metafunction returns true are available for the various point
153  algebra functions like contains, scale, etc.
154 */
155 template<typename Point>
156 struct is_point_type : public std::false_type {};

158 /*
159  Metafunction that identifies types that model the Rectangle concept.
160 
161  A Rectangle is an upright region in the coordinate plane identified by
162  xmin, ymin, xmax and ymax values. If r is an instance of a model of the
163  Rectangle concept, then the following expressions:
164 
165  get_xmin( r )
166  get_ymin( r )
167  get_xmax( r )
168  get_ymax( r )
169 
170  retrieve the corresponding values in the rectangle (typically using ADL). These expressions
171  must all return values of the same type. That type is available through
172  the convenience metafunction "coord_type." If rect_type is a model of
173  the Rectangle concept, then:
174 
175  coord_type<rect_type>::type
176 
177  identifies the type of values returned by get_xmin et al. Implementers do not
178  need to specialize this metafunction explicitly, its implementation is derived
179  from the implementation of get_xmin.
180 
181  The convenience functions:
182 
183  get_width( r )
184  get_height( r )
185 
186  must also exist and return get_xmax( r ) - get_xmin( r ) and get_ymax( r ) - get_ymin( r ),
187  respectively. If the type natively stores width and height instead of xmax/ymax, you
188  should overload these functions to return the value directly.
189 
190  If r is an instance of a model of the Rectangle concept, then the following statments:
191 
192  set_xmin( r, new_xmin );
193  set_ymin( r, new_ymin );
194  set_xmax( r, new_xmax );
195  set_ymax( r, new_ymax );
196  set_width( r, new_width );
197  set_height( r, new_height );
198  set_all( r, new_xmin, new_ymin, new_xmax, new_ymax );
199 
200  Should modify the corresponding values of the rectangle. By convention, set_width
201  and set_height should modify the xmax or ymax value, and leave the xmin and ymin unchanged.
202 
203  The following operators must work on Rectangles and provide expected behavior:
204 
205  operator==
206  operator!=
207  operator<<( std::ostream& )
208 
209  If rect_type is a model of the Rectangle concept, then the "is_rectangle_type"
210  template metafunction should be specialized to return true_type. Only types
211  for which this metafunction returns true are available for the various rectangle
212  algebra functions like is_empty, intersect, union, etc.
213 */
214 template<typename Rectangle>
215 struct is_rectangle_type : public std::false_type {};

624 bool matrix_invertible_partial( Coord a, Coord b, Coord c, Coord d )
625 {
626  auto det = tetraphilia::real_services::ComputeMatrixDeterminantOrInf( a, b, c, d );
627  det = std::abs( det );

632  // we know that 0 <= mx < numeric_limits<Coord>::max. In particular,

637  && (std::abs( c ) < mx) && (std::abs( d ) < mx);

647 template<typename AffineTransform>
648 std::enable_if_t<is_affine_transform_type<AffineTransform>::value,
649 boost::optional<AffineTransform>> invert_if_invertible( AffineTransform const& m )
650 {
651  using coord = typename coord_type<AffineTransform>::type;
652 
653  coord a = get_a( m );
654  coord b = get_b( m );
655  coord c = get_c( m );
656  coord d = get_d( m );
657  coord e = get_e( m );
658  coord f = get_f( m );
659  coord scale;
660  coord epsilon = std::numeric_limits<coord>::epsilon();
661 
662  // Can't invert values of no or subnormal magnitude
663  if ( (std::abs(a) < epsilon) && (std::abs(b) < epsilon) && (std::abs(c) < epsilon) && (std::abs(d) < epsilon) )
664  return {};
665  tetraphilia::real_services::ScaleMatrixToAvoidDeterminantOverflow( &a, &b, &c, &d, &scale );
666 
667  if ( !matrix_invertible_partial( a, b, c, d ) )
668  return {};
669 
670  coord ia;
671  coord ib;
672  coord ic;
673  coord id_;
674  coord ie;
675  coord _if;
676 
677  // In floating point builds, we compute the inverse of the determininant and
678  // multiply the four values by that. This helps keep the compiler from generating
679  // slight computational differences in debug vs. release builds.
680  auto det = 1 / determinant( m );
681  ia = d * det;
682  ib = -b * det;
683  ic = -c * det;
684  id_ = a * det;
685  ie = -ia * e - ic * f;
686  _if = -ib * e - id_ * f;
687 
688  return make<AffineTransform>( ia, ib, ic, id_, ie, _if );
689 }

1459 /*

1461  tolerance, or both empty.

1464  xmin >= xmax OR ymin >= ymax) are equal, regardless of their actual coordinate values. But "almost"
1465  empty rectangles (per the tolerance) have a real spatial location and thus should not be considered to be
1466  "almost" equal unless their coordinates are close.

1476

1784  { get_xmax( r ), get_ymin( r ) },
