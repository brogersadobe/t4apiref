53 template<typename Client>
54 auto pdf_tile_cache_thumbnail_requester<Client>::make_thumbnail_paint_request( page_num page ) -> shared_ptr<paint_request>
55 {
56  // Target number of tiles in a thumbnail. @todo gdow: make this configurable
57  static size_t default_thumbnail_size_in_tiles = 6;
58 
59  auto ptc = this->get_pdf_tile_cache();
60 
61  auto pg = get_cached_page_geometry( *ptc->get_document(), page );
62 
63  // Ideally we'd like thumbnails to consume at most default_thumbnail_size tiles, where
64  // each tile is the tile cache's minimum length on a side.
65  auto aspect_ratio = get_width( pg.m_cropped_media_box ) / get_height( pg.m_cropped_media_box );
66  auto mts = static_cast<size_t>( ptc->get_minimum_tile_size() );
67  auto target_tile_size = mts * mts * default_thumbnail_size_in_tiles;
68  auto target_tile_width = std::sqrt( aspect_ratio * target_tile_size );
69  auto device_scale = target_tile_width / get_width( pg.m_cropped_media_box );
70 
71  draw_params dp( make<real_point>( device_scale, device_scale ) );
72  auto clip_rect = enclose<int_rect>( scale( pg.m_cropped_media_box, device_scale ) );
73  return make_shared<paint_request>( paint_job_params{ page, dp, clip_rect, std::chrono::steady_clock::duration::max() }, 20 );
74 }

91 template<typename Client>
92 void pdf_tile_cache_thumbnail_requester<Client>::set_important_page_sync( page_num page, size_t num_to_cache_ahead )
93 {
94  auto ptc = this->get_pdf_tile_cache();
95  if ( !ptc )
96  return;
97 
98  page_num page_count = ptc->get_document()->get_num_pages();
99 
100  m_requests.while_locked( [ this, page, page_count, &num_to_cache_ahead ]( request_vector& requests )
101  {
102  if ( requests.size() == num_to_cache_ahead + 1 && requests[0]->get_paint_job_params().m_page == page )
103  return;
104 
105  request_vector new_requests;
106 
107  new_requests.reserve( std::min( static_cast<size_t>( page_count ), num_to_cache_ahead ) );
108  new_requests.push_back( get_request_for( requests, page ) );
109 
110  page_num above = page - 1;
111  page_num below = page + 1;
112 
113  while ( 0 <= above || below < page_count )
114  {
115  if ( num_to_cache_ahead == 0 )
116  break;
117 
118  if ( below < page_count )
119  {
120  new_requests.push_back( get_request_for( requests, below++ ) );
121  --num_to_cache_ahead;
122  }
123 
124  if ( num_to_cache_ahead == 0 )
125  break;
126  if ( above >= 0 )
127  {
128  new_requests.push_back( get_request_for( requests, above-- ) );
129  --num_to_cache_ahead;
130  }
131 
132  if ( num_to_cache_ahead == 0 )
133  break;
134  if ( below < page_count )
135  {
136  new_requests.push_back( get_request_for( requests, below++ ) );
137  --num_to_cache_ahead;
138  }
139  }
140 
141  requests = std::move( new_requests );
142  } );
143 
144  ptc->poke();
145 }
