31 /*
32  Constexpr-friendly template class supporting a fixed-size array of bits.
33 
34  This is mostly a workalike to std::bitset, with the following differences:
35 
36  - Everything that can be constexpr is
37  - Support construction from an initializer_list<bool>
38  - No support for to/from ulong or ullong
39  - No support for to/from strings of sequences of chars (like "00101100")
40 */
41 template<size_t N, typename Word = size_t>
42 class bitset : private bitset_detail::bitset_impl<N, Word>,
43  private t4::totally_ordered<bitset<N, Word>,
44  t4::bitwise<bitset<N, Word>,
45  t4::shiftable<bitset<N, Word>, size_t>>>
46 {
47  using impl = bitset_detail::bitset_impl<N, Word>;
48 
49 public:
50  /*
51  Type returned by mutable operator[] which models a reference to a single bit in memory.
52  */
53  using reference = typename impl::reference;
54 
55  /*
56  Default ctor
57  */
58  constexpr bitset( void ) noexcept = default;
59 
60  /*
61  Construct a bitset from an initializer_list of booleans, e.g.:
62 
63  bitset<3> x{{ true, false, true }};
64  */
65  constexpr bitset( std::initializer_list<bool> list ) noexcept
66  : impl( list )
67  {}

70  Construct a bitset from the given integer value. If N is larger than the number of bits
71  in UInteger, the corresponding bits are initialized to zero.
72  */
73  template<typename UInteger, std::enable_if_t<std::is_unsigned<UInteger>::value, int> = 0>
74  constexpr bitset( UInteger bits ) noexcept
75  : impl( bits )
76  {}
