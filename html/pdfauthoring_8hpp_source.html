84 struct parallelogram
85 {
86  typedef t4::affine_transform matrix_type;
87  typedef t4::real_point real_point_type;
88  typedef t4::real_rect real_rect_type;
89 
90  real_point_type m_topLeft, m_topRight, m_bottomLeft, m_bottomRight;
91 
92  parallelogram(const real_rect_type& pNormalizedSpaceBbox, t4::real_coord pRotation)
93  {
94  namespace im = imaging_model;
95  const matrix_type& lRotationMatrix = GetTransformationMatrix(pRotation);
96 
97  m_topLeft = im::MatrixTransform( lRotationMatrix, im::GetTopLeft<real_point_type>( pNormalizedSpaceBbox ) );
98  m_topRight = im::MatrixTransform( lRotationMatrix, im::GetTopRight<real_point_type>( pNormalizedSpaceBbox ) );
99  m_bottomLeft = im::MatrixTransform( lRotationMatrix, im::GetBottomLeft<real_point_type>( pNormalizedSpaceBbox ) );
100  m_bottomRight = im::MatrixTransform( lRotationMatrix, im::GetBottomRight<real_point_type>( pNormalizedSpaceBbox ) );
101  }
102 
103  real_rect_type GetBbox() const
104  {
105  real_rect_type lBbox;
106 
107  set_xmin( lBbox, std::min( { get_x( m_topLeft ), get_x( m_topRight ), get_x( m_bottomLeft ), get_x( m_bottomRight ) } ) );
108  set_ymin( lBbox, std::min( { get_y( m_topLeft ), get_y( m_topRight ), get_y( m_bottomLeft ), get_y( m_bottomRight ) } ) );
109  set_xmax( lBbox, std::max( { get_x( m_topLeft ), get_x( m_topRight ), get_x( m_bottomLeft ), get_x( m_bottomRight ) } ) );
110  set_ymax( lBbox, std::max( { get_y( m_topLeft ), get_y( m_topRight ), get_y( m_bottomLeft ), get_y( m_bottomRight ) } ) );
111 
112  return lBbox;
113  }
114 
115  matrix_type GetTransformationMatrix(t4::real_coord pRotation) const
116  {
117  t4::real_coord lSinVal = std::sin(pRotation);
118  t4::real_coord lCosVal = std::cos(pRotation);
119 
120  matrix_type lRotationMatrix;
121  imaging_model::MakeMatrix(lRotationMatrix, lCosVal, lSinVal, -lSinVal, lCosVal, 0.0, 0.0);
122 
123  return lRotationMatrix;
124  }
125 
126  bool ContainsDefaultUserSpacePoint(const real_rect_type& pNormalizedSpaceBbox, const t4::real_point& pt, t4::real_coord pRotation) const
127  {
128  namespace im = tetraphilia::imaging_model;
129 
130  const matrix_type& lRotationMatrix = GetTransformationMatrix(pRotation);
131 
132  matrix_type lInverseRotationMatrix;
133  if(!im::MatrixInvert(lInverseRotationMatrix, lRotationMatrix))
134  {
135  pdfAssert(0);
136  return false;
137  }
138 
139  im::MatrixTransform( lInverseRotationMatrix, pt ); // changes pt
140  return (get_xmin(pNormalizedSpaceBbox) <= get_x(pt) && get_x(pt) <= get_xmax(pNormalizedSpaceBbox)
141  && get_ymin(pNormalizedSpaceBbox) <= get_y(pt) && get_y(pt) <= get_ymax(pNormalizedSpaceBbox));
142  }
143 
144  t4::real_coord GetArea() const
145  {
146  const real_rect_type& lBbox = GetBbox();
147  return get_width(lBbox) * get_height(lBbox);
148  }

151 template<typename app_traits>
152 class bbox_utility
153 {
154  typedef typename app_traits::real_geometry_traits::rect_type real_rect_type;
155  typedef typename app_traits::real_geometry_traits::coord_type real_coord_type;
156 
157 public:
158  static bool HorizontallyOverlaps(const real_rect_type& pBbox1, const real_rect_type& pBbox2)
159  {
160  real_coord_type lXMin1 = get_xmin(pBbox1);
161  real_coord_type lXMax1 = get_xmax(pBbox1);
162  real_coord_type lXMin2 = get_xmin(pBbox2);
163  real_coord_type lXMax2 = get_xmax(pBbox2);
164 
165  return !(lXMax1 < lXMin2 || lXMin1 > lXMax2);
166  }
167 
168  static bool VerticallyOverlaps(const real_rect_type& pBbox1, const real_rect_type& pBbox2)
169  {
170  real_coord_type lYMin1 = get_ymin(pBbox1);
171  real_coord_type lYMax1 = get_ymax(pBbox1);
172  real_coord_type lYMin2 = get_ymin(pBbox2);
173  real_coord_type lYMax2 = get_ymax(pBbox2);
174 
175  bool lLies1 = (lYMin1 <= lYMin2 && lYMin2 <= lYMax1);
176  bool lLies2 = (lYMin1 <= lYMax2 && lYMax2 <= lYMax1);
177  bool lLies3 = (lYMin2 <= lYMin1 && lYMin1 <= lYMax2);
178  bool lLies4 = (lYMin2 <= lYMax1 && lYMax1 <= lYMax2);
179 
180  if(lLies1 || lLies2 || lLies3 || lLies4)
181  {
182  if((lLies1 && lLies2) || (lLies3 && lLies4))
183  return true;
184 
185  real_coord_type lYSpan = std::max(lYMax1 - lYMin1, lYMax2 - lYMin2);
186 
187  real_coord_type lYIntermediate1 = (lYMin1 < lYMin2) ? lYMin2 : lYMin1;
188  real_coord_type lYIntermediate2 = (lYMax1 > lYMax2) ? lYMax2 : lYMax1;
189 
190  real_coord_type lPercentOverlap = (AbsVal(lYIntermediate1 - lYIntermediate2) / lYSpan) * 100;
191  return (lPercentOverlap >= 30);
192  }
193 
194  return false;
195  }

468  int DetermineAlignment(const real_rect_type& pParagraphBbox, const real_rect_type& pLineBbox, real_type pMaxWidthPerChar)
469  {
470  bool lLeftAligned = (AbsVal(get_xmin(pParagraphBbox) - get_xmin(pLineBbox)) < pMaxWidthPerChar);
471  bool lRightAligned = (AbsVal(get_xmax(pParagraphBbox) - get_xmax(pLineBbox)) < pMaxWidthPerChar);
472  bool lCenterAligned = (AbsVal((get_xmin(pParagraphBbox) + get_xmax(pParagraphBbox))/2.0 - (get_xmin(pLineBbox) + get_xmax(pLineBbox))/2.0) < pMaxWidthPerChar);
473 
474  int lAlignment = UNKNOWN_ALIGNMENT;
475  if(lLeftAligned)
476  lAlignment |= LEFT_ALIGNED;
477 
478  if(lRightAligned)
479  lAlignment |= RIGHT_ALIGNED;
480 
481  if(lCenterAligned)
482  lAlignment |= CENTER_ALIGNED;
483

487  void DetermineAlignment(const real_rect_type& pPreMergeBbox, const real_rect_type& pPostMergeBbox, const Line<app_traits>& pLine, const real_rect_type& pLineBbox)
488  {
489  size_t lCount = this->GetCount(); // including current line as it has already been merged
490  pdfAssert(lCount >= 2);
491 
492  if(lCount > 2 && m_alignment == UNKNOWN_ALIGNMENT)
493  return;
494 
495  int lCurrentAlignment = DetermineAlignment(pPostMergeBbox, pLineBbox, pLine.m_maxWidthPerChar);
496  int lModifiedAlignment = DetermineAlignment(pPostMergeBbox, pPreMergeBbox, pLine.m_maxWidthPerChar);
497 
498  if(lCount == 2)
499  m_alignment = lModifiedAlignment;
500  else if(m_alignment != lModifiedAlignment)
501  m_alignment &= lModifiedAlignment;
502 
503  if(m_alignment != UNKNOWN_ALIGNMENT && m_alignment != lCurrentAlignment)
504  {
505  if(lCurrentAlignment == UNKNOWN_ALIGNMENT)
506  m_alignment = UNKNOWN_ALIGNMENT;
507  else
508  m_alignment &= lCurrentAlignment;
509  }
510  }

738  bool CanCombine(const real_rect_type& pLastBbox, const real_rect_type& pCurrBbox, real_type pLineMaxCharWidth, real_type pTextRunMaxCharWidth)
739  {
740  real_coord_type lYSpan1 = get_ymax(pLastBbox) - get_ymin(pLastBbox);
741  real_coord_type lYSpan2 = get_ymax(pCurrBbox) - get_ymin(pCurrBbox);
742 
743  if(lYSpan2 >= 5.0 * lYSpan1 || lYSpan1 >= 5.0 * lYSpan2)
744  return false;
745 
746  if(bbox_utility<app_traits>::HorizontallyOverlaps(pLastBbox, pCurrBbox))
747  return true;
748 
749  real_coord_type lDist = get_xmin(pCurrBbox) - get_xmax(pLastBbox);
750  if(lDist < 0)
751  lDist = AbsVal(get_xmax(pCurrBbox) - get_xmin(pLastBbox));
752 
753  real_type lMaxCharWidth = std::max(pLineMaxCharWidth, pTextRunMaxCharWidth);
754  if(lMaxCharWidth != 0)
755  return (lDist < 5 * lMaxCharWidth);
756 
757  return (lDist < 2.0 * (std::max(get_xmax(pLastBbox) - get_xmin(pLastBbox), get_xmax(pCurrBbox) - get_xmin(pCurrBbox))));
758  }

763  {

765  if((get_ymax(lBbox) - get_ymin(lBbox)) > 2.0 * (get_ymax(pCurrBbox) - get_ymin(pCurrBbox)))
766  return true;
767  }
768 
769  return false;
770  }

779  }
780  else

794  && (lLastLine.GetRotation() == pTextRun.GetRotation()))
795  {
796  lMerged = true;
797  lLastLine.Merge(pTextRunIndex, lBbox2, lActualBbox2);

799  lLastLine.m_uneditable |= pTextRun.m_uneditable;
800  lLastLine.m_hasValidNonWhitespaceChar |= pTextRun.m_hasValidNonWhitespaceChar;

802 
803  if(!lMerged)

805  }

826  bool TwoLookaheadsFormCloserParagraph(real_coord_type pParagraphSpacing, real_coord_type pPrevLineSpacing, const real_rect_type& pCurrentLineBbox, const real_rect_type& pNextLineBbox, const real_rect_type& pNextNextLineBbox)

835  bool OneLookaheadFormsCloserParagraph(real_coord_type pParagraphSpacing, real_coord_type pPrevLineSpacing, const real_rect_type& pPrevLineBbox, const real_rect_type& pCurrentLineBbox, const real_rect_type& pNextLineBbox)
836  {
837  real_coord_type lDist = AbsVal(get_ymin(pNextLineBbox) - get_ymin(pCurrentLineBbox));
838  real_coord_type lSpacing = ( pParagraphSpacing != 0 ) ? pParagraphSpacing : pPrevLineSpacing;

842  real_coord_type lNextYSpan = get_ymax(pNextLineBbox) - get_ymin(pNextLineBbox);
843 
844  return (!IsLineSpacingOkay(lSpacing, lDist) && (lDist < lSpacing) && (AbsVal(lNextYSpan - lCurrentYSpan) <= AbsVal(lPrevYSpan - lCurrentYSpan)));
845  }

855  return true;
856

863  bool IsLineSeparationTooHigh(const real_rect_type& pPrevLineBbox, const real_rect_type& pCurrLineBbox)
864  {
865  real_coord_type lYMin1 = get_ymin(pPrevLineBbox);
866  real_coord_type lYMax1 = get_ymax(pPrevLineBbox);
867  real_coord_type lYMin2 = get_ymin(pCurrLineBbox);
868  real_coord_type lYMax2 = get_ymax(pCurrLineBbox);
869 
870  real_coord_type lYSpan = 2 * std::max(std::abs(lYMax1 - lYMin1), std::abs(lYMax2 - lYMin2));
871  return (std::abs(lYMin2 - lYMin1) > lYSpan);
872  }
873 
874  bool LineIndentationProhibitsMerging(const Paragraph<app_traits>& pParagraph, const Line<app_traits>& pLastLine, const Line<app_traits>& pLine, const Line<app_traits>* pLookaheadLine1)
875  {
876  if(pParagraph.m_alignment == Paragraph<app_traits>::CENTER_ALIGNED || pParagraph.m_alignment == Paragraph<app_traits>::RIGHT_ALIGNED)
877  return false;
878 
879  if(pParagraph.GetCount() >= 2) // No need to split paragraphs with one line only
880  {
881  const real_rect_type& lParagraphBbox = pParagraph.GetNormalizedSpaceBbox();
882  const real_rect_type& lBbox1 = pLastLine.GetNormalizedSpaceBbox();
883  const real_rect_type& lBbox2 = pLine.GetNormalizedSpaceBbox();
884 
885  if(pLookaheadLine1)
886  {
887  const real_rect_type& lBbox3 = pLookaheadLine1->GetNormalizedSpaceBbox();
888 
889  if(AbsVal(get_xmin(lBbox3) - get_xmin(lParagraphBbox)) < pLine.m_maxWidthPerChar)
890  {
891  bool lPrevIndent = (get_xmin(lBbox1) <= get_xmin(lBbox2) - pLine.m_maxWidthPerChar * 2);
892  bool lNextIndent = (get_xmin(lBbox3) <= get_xmin(lBbox2) - pLine.m_maxWidthPerChar * 2);
893 
894  if(lPrevIndent && lNextIndent)
895  {
896  bool lPrematurePrevTermination = (get_xmax(lBbox1) >= get_xmax(lBbox2) + pLine.m_maxWidthPerChar * 2);
897  bool lPrematureNextTermination = (get_xmax(lBbox3) >= get_xmax(lBbox2) + pLine.m_maxWidthPerChar * 2);
898 
899  return (!lPrematurePrevTermination || !lPrematureNextTermination);
900  }
901 
902  bool lPrematurePrevLineEnd = (get_xmax(lParagraphBbox) >= get_xmax(lBbox1) - pLine.m_maxWidthPerChar * 4);
903  bool lMinorPrevIndent = (get_xmin(lBbox1) <= get_xmin(lBbox2) - pLine.m_maxWidthPerChar);
904  bool lMinorNextIndent = (get_xmin(lBbox3) <= get_xmin(lBbox2) - pLine.m_maxWidthPerChar);
905 
906  return (lMinorPrevIndent && lMinorNextIndent && lPrematurePrevLineEnd);
907  }
908  }
909  else
910  {
911  return (get_xmin(lParagraphBbox) <= get_xmin(lBbox2) - pLine.m_maxWidthPerChar * 4); // check if there is more than a tab of leading space
912  }
913  }
914 
915  return false;
916  }

918  void operator() (const Line<app_traits>& pLine, size_t pLineIndex, const Line<app_traits>* pLookaheadLine1, const Line<app_traits>* pLookaheadLine2)
919  {
920  if(m_sink.IsEmpty())
921  {
922  m_sink.Accept(Paragraph<app_traits>(pLineIndex, pLine.GetNormalizedSpaceBbox(), pLine.GetNormalizedSpaceActualBbox(), pLine.GetRotation(), pLine.m_uneditable, pLine.m_hasValidNonWhitespaceChar));
923  }
924  else
925  {
926  Paragraph<app_traits>& lLastParagraph = m_sink.GetLast();
927  const Line<app_traits>& lLastLine = (m_authoringStructure.GetLastLineSink().GetAt(lLastParagraph.GetAt(lLastParagraph.GetCount() - 1)));
928 
929  bool lMerged = false;
930 
931  const real_rect_type& lParagraphBbox = lLastParagraph.GetNormalizedSpaceBbox();
932  const real_rect_type& lBbox1 = lLastLine.GetNormalizedSpaceBbox();
933  const real_rect_type& lBbox2 = pLine.GetNormalizedSpaceBbox();
934  const real_rect_type& lActualBbox2 = pLine.GetNormalizedSpaceActualBbox();
935 
936  if(!lLastLine.m_dropcap && !pLine.m_dropcap && bbox_utility<app_traits>::HorizontallyOverlaps(lParagraphBbox, lBbox2) && (pLine.GetRotation() == lLastParagraph.GetRotation()))
937  {
938  real_coord_type lDist1 = AbsVal(get_ymin(lBbox2) - get_ymin(lBbox1));
939 
940  bool lShouldMerge = !(lLastParagraph.m_lineSpacing != 0 && !IsLineSpacingOkay(lLastParagraph.m_lineSpacing, lDist1, lLastLine, pLine));
941  if(lShouldMerge)
942  {
943  const Line<app_traits>* lLookaheadLine1 = pLookaheadLine1;
944  const Line<app_traits>* lLookaheadLine2 = pLookaheadLine2;
945 
946  if(lLookaheadLine1 && (lLookaheadLine1->m_dropcap || pLine.GetRotation() != lLookaheadLine1->GetRotation()))
947  lLookaheadLine1 = lLookaheadLine2 = NULL;
948 
949  if(lLookaheadLine2 && (lLookaheadLine2->m_dropcap || pLine.GetRotation() != lLookaheadLine2->GetRotation()))
950  lLookaheadLine2 = NULL;
951 
952  const real_rect_type* lBbox3 = lLookaheadLine1 ? &lLookaheadLine1->GetNormalizedSpaceBbox() : NULL;
953  const real_rect_type* lBbox4 = lLookaheadLine2 ? &lLookaheadLine2->GetNormalizedSpaceBbox() : NULL;
954 
955  if(lBbox3 && bbox_utility<app_traits>::HorizontallyOverlaps(lBbox2, *lBbox3))
956  {
957  if(lBbox4 && bbox_utility<app_traits>::HorizontallyOverlaps(lBbox2, *lBbox4)) // Two lookaheads available
958  {
959  if(TwoLookaheadsFormCloserParagraph(lLastParagraph.m_lineSpacing, lDist1, lBbox2, *lBbox3, *lBbox4))
960  lShouldMerge = false;
961  }
962 
963  if(lShouldMerge)
964  {
965  // Only one lookahead available
966  if(OneLookaheadFormsCloserParagraph(lLastParagraph.m_lineSpacing, lDist1, lBbox1, lBbox2, *lBbox3))
967  lShouldMerge = false;
968  }
969  }
970 
971  if(lShouldMerge)
972  {
973  // No lookahead available
974  if(lLastParagraph.m_lineSpacing != 0 && IsLineSeparationTooHigh(lBbox1, lBbox2))
975  lShouldMerge = false;
976  }
977

979  {
980  lLastParagraph.MergeLine(pLineIndex, pLine, lBbox2, lActualBbox2);
981  lLastParagraph.m_uneditable |= pLine.m_uneditable;
982  lLastParagraph.m_hasValidNonWhitespaceChar |= pLine.m_hasValidNonWhitespaceChar;
983 
984  if(lLastParagraph.m_lineSpacing == 0)
985  lLastParagraph.m_lineSpacing = lDist1;
986 
987  lMerged = true;
988  }
989  }
990  }
991 
992  if(!lMerged)
993  m_sink.Accept(Paragraph<app_traits>(pLineIndex, lBbox2, lActualBbox2, pLine.GetRotation(), pLine.m_uneditable, pLine.m_hasValidNonWhitespaceChar));
994  }

2095  bool handleCJKGlyphWithMultipleUnicodes(text_string_type& pTextStr, text::CharCodeLookup lookup, bool& pValidNonWhitespaceChar)

2098  * associated with it.
2099  *
2100  * The Chinese, Japanese and Korean (CJK) scripts share a common background.
2101  * In the process called Han unification, the common (shared) characters were
2102  * identified and named "CJK Unified Ideographs". In addition, there are
2103  * separate unicode blocks which contain similar glyphs with minor variations
2104  * corresponding to various CJK characters.
2105  *
2106  * Some system fonts (Heiti SC, AlNile etc...) when embedded into the PDF by
2107  * iOS contain multiple unicodes for the same CJK glyph with each unicode's
2108  * actual glyph almost same (with minor variations) but belonging to different
2109  * blocks.
2110  *
2111  * For e.g. given "STHeitiSC-Light" font, for Kangxi radicals (a list of 214 Chinese
2112  * characters), upto 3 unicodes are encoded for each glyph, each of which belongs
2113  * to one of the following blocks :
2114  *
2115  * 1. 0x2F00 - 0x2FDF : Kangxi Radicals
2116  * 2. 0x2E80 - 0x2EFF : CJK Radicals Supplement (alternative forms of the Kangxi radicals)
2117  * 3. 0x4E00 - 0x9FFF : CJK Unified Ideographs
2118  * (More info : http://www.unicode.org/Public/UNIDATA/CJKRadicals.txt)
2119  *
2120  * Hence, since we use unicodes to display associated glyph during editing on
2121  * platform, for now, we must choose a single unicode else risk displaying
2122  * multiple glyphs on platform for a single CJK glyph in PDF.
2123  *
2124  * TODO_READER_MOBILE<rosinha>: Investigate if extracting a single unicode
2125  * for multi-unicode CJK glyphs breaks any behaviour.
2126  */

2130

2132  {

2136  }

2140  }
