60 /*
61  The PDF spec says that the function entry in a shading dictionary must be "A 2-in, n-out function or
62  an array of n 2-in, 1-out functions (where n is the number of colour components in the shading
63  dictionaryâ€™s colour space)." If the function is a single function dictionary, that function must
64  produce a value with the same number of components as the shading dictionary's color space. Otherwise
65  it must be an array with that number of elements, and each element must be a function producing
66  exactly one component.
67 */
68 template <typename _app_traits>
69 PDFSmoothShadeFunction<_app_traits>::PDFSmoothShadeFunction( const obj_type& funcDefObj )
70  : m_funcConvStack( TransientAllocator<app_traits>(*t4::get_thread_heap()) )
71 {
72  if (funcDefObj.Type() == store::kObjectTypeDictionary)
73  {
74  m_funcConvStack.Push( content::FunctionFactory<app_traits>::Create( funcDefObj.DictionaryValue_NoTypeCheck() ));
75  }
76  else if (funcDefObj.Type() == store::kObjectTypeArray)
77  {
78  array_type a = funcDefObj.ArrayValue_NoTypeCheck();
79  for (typename array_type::const_iterator_type i = a.begin(); i != a.end(); ++i)
80  {
81  m_funcConvStack.Push( content::FunctionFactory<app_traits>::Create( i->DictionaryValue() ) );
82  /*
83  This code verifies that, if the entry is an array, all the member
84  functions produce one signal of output. See comments at the top of
85  this function definition.
86  */
87  if ( m_funcConvStack.Top().GetRangeDimension() != 1 )
88  throw t4::corrupt_data();
89  }
90  }
91  else
92  throw t4::corrupt_data();
93 }

95 template <typename _app_traits>
96 void PDFSmoothShadeFunction<_app_traits>::Eval(real_type* dst, ranges::span<const real_type> src)
97 {
98  if (m_funcConvStack.size() == 1)
99  {
100  m_funcConvStack.Top().ConvertPixel( dst, sizeof(real_type), src.data(), sizeof(real_type) );
101  }
102  else
103  {
104  for (typename func_conv_stack_type::iterator i = m_funcConvStack.begin(); i!= m_funcConvStack.end(); ++i)
105  {
106  i->ConvertPixel( dst, sizeof(real_type), src.data(), sizeof(real_type) );
107  dst++;
108  }
109  }
110 }

113 size_t PDFSmoothShadeFunction<_app_traits>::GetRangeDimension(void)
114 {
115  if (m_funcConvStack.size() == 1 )

121  return m_funcConvStack.size();

401 template <typename _app_traits>
402 typename NumericBitStream<_app_traits>::real_type NumericBitStream<_app_traits>::ReadNBitsAsReal(size_t n, real_type min, real_type max)
403 {
404  unsigned int data = m_pBitStream->template GetNextNBitsAsInteger<unsigned int>(n);
405  real_type frac;
406  if (!real_services::RealTypeHelper<real_type>::isFloatingPointType && n > 15)
407  frac = (data >> (n - 15)) / real_type(32767);
408  else
409  {
410  if (n < 31)
411  frac = data / real_type((std::uint32_t(1) << n)-1);
412  else if (n == 32)
413  frac = data / real_type(0xFFFFFFFF);
414  else
415  frac = data / (real_services::Pow(real_type(2), real_type(n)) - 1);
416  }
417 
418  return min*(1-frac) + max*frac;
419 }

558 template <typename _app_traits>
559 std::uint8_t LatticeFormGouraudVertexStream<_app_traits>::Next(vertex_type& v, size_t numSignals)
560 {
561  // Security Fix - QNX PR 94288 [See constructor]
562  if(m_verticesPerRow == 0)
563  return 0xFF;
564 
565  if (m_curVertex == 2*m_verticesPerRow)
566  {
567  if (this->m_numBitStream.AtEOF())
568  return 0xFF;
569 
570  // hit the end of this row, better be another since not EOF
571  boost::swap(m_topRow, m_bottomRow);
572  for (int_type i = 0; i < m_verticesPerRow; i++)
573  this->ReadVertex(m_bottomRow[i], numSignals);
574  m_curVertex = 0;
575  }
576 
577  if (m_curVertex & 1)
