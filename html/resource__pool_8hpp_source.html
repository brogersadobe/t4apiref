71  resource_weight_t sized = resource_weight_t( std::max( 1_z, m_sz.load() ) );
72  std::tuple<size_t, cache_duration> cache_info = m_control.get_cache_info();
73  resource_weight_t aged = resource_weight_t( 1 + now_tick - std::get<0>( cache_info ) );

75  m_weight = ttcd / sized / std::log( aged );

94  std::atomic<size_t> m_amount_in_use{ 0 };
95 
96  // If a large number of allocations are made, the resource_pool can get in a state where
97  // nothing is purgeable and yet it's still above m_purge_threshold. A naive implementation
98  // (i.e., the one before this addition here) would then attempt a purge on every single
99  // allocation (and fail to purge anything!), bringing the application to a crawl.
100  //
101  // m_amount_in_use_after_last_purge stores the amount of memory (according to amount_in_use())
102  // that was available at the completion of the last purge. Even if we're above m_purge_threshold,
103  // we do not attempt to purge unless amount_in_use is greater than this value _plus_ m_purge_amount.
104  std::atomic<size_t> m_amount_in_use_after_last_purge{ 0 };
