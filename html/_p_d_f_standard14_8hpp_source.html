307 template<typename app_traits>
308 FauxFont<app_traits>::FauxFont( typename app_traits::context_type& context,
309  const FontDesc& desc,
310  const std::uint16_t* widths /*[kNumSubstitutionGlyphs]*/ )
311  : parsers::CFF<app_traits>( context, GetDataStore(context,desc) ),
312  m_desc( desc ),
313  m_baseDesc((desc.m_Flags & kSerif)?
314  MMDesc::ASerifMMDesc : MMDesc::ASansMMDesc)
315 
316 {
317  memcpy( m_widths, widths, sizeof(m_widths) );
318 
319  // This block of code corrects for unreasonable values in the font descriptor. It seems like this
320  // should be PDF_SLANG code, but the PDFRM claims that values like XHeight are optional and the default
321  // value is 0. This clearly indicates that at least for this value you can't consider these corrections
322  // to be slang-only corrections. Because of this, we go ahead and correct all values, even in non-slang builds
323  if ( m_desc.m_StemV == 0 )
324  m_desc.m_StemV = m_baseDesc.defaultMetrics.m_StemV;
325  else
326  m_desc.m_StemV = pin( m_desc.m_StemV, 20, 300 );
327  if ( m_desc.m_Flags & kSerif ) {
328  m_desc.m_CapHeight = 662;
329  if ( m_desc.m_XHeight == 0 )
330  m_desc.m_XHeight = 480;
331  else
332  m_desc.m_XHeight = pin( m_desc.m_XHeight, 300, 480 );
333  } else {
334  m_desc.m_CapHeight = 718;
335  if ( m_desc.m_XHeight == 0 )
336  m_desc.m_XHeight = 532;
337  else
338  m_desc.m_XHeight = pin( m_desc.m_XHeight, 300, 532 );
339  }
340  m_desc.m_ItalicAngle = pin( m_desc.m_ItalicAngle, -45, 45 );
341 
342  m_fRec.slant = tangent(m_desc.m_ItalicAngle);
343  FauxFD<app_traits>* lcMetrics = &m_fRec.FDArray[0];
344  FauxFD<app_traits>* ucMetrics = &m_fRec.FDArray[1];
345 
346  CalcFauxScales(desc,
347  lcMetrics->scale, m_desc.m_XHeight, static_cast<std::uint16_t>( m_baseDesc.XHeight ),
348  ucMetrics->scale, m_desc.m_CapHeight, static_cast<std::uint16_t>( m_baseDesc.CapHeight ) );
349 
350  ScaleMetrics(lcMetrics->scale,
351  lcMetrics->StdVW, m_baseDesc.StemSnapV[0]);
352  ScaleMetrics(ucMetrics->scale,
353  ucMetrics->StdVW, m_baseDesc.StemSnapV[0]);
354 
355  // this->m_maxGlyphBounds has been set in our parent's ctor based on parsing the cff_FontBBox
356  // opcode from the font data. This is the union of the font bbox's of all 4 of the master fonts
357  // in the mm font. We expand that based on some heuristics about the widths of each glyph.
358  int widestWidth = 0;
359  for ( size_t i = 0; i < kNumSubstitutionGlyphs; i++ ) {
360  if ( m_widths[i] > widestWidth )
361  widestWidth = m_widths[i];
362  }
363  real_type widest = real_services::DivideBy1000( real_type(widestWidth) );
364  if ( widest > get_xmax(this->m_maxGlyphBounds) ) {
365  real_type xscale = real_type(widest) / real_type(get_xmax(this->m_maxGlyphBounds));
366  set_xmax(this->m_maxGlyphBounds, widest );
367  set_xmin(this->m_maxGlyphBounds, get_xmin(this->m_maxGlyphBounds) * xscale );
368  }
369  if ( m_fRec.slant != 0 ) {
370  if ( m_fRec.slant > 0 ) {
371  set_xmax(this->m_maxGlyphBounds, get_xmax(this->m_maxGlyphBounds) + get_ymax(this->m_maxGlyphBounds) * m_fRec.slant );
372  set_xmin(this->m_maxGlyphBounds, get_xmin(this->m_maxGlyphBounds) + get_ymin(this->m_maxGlyphBounds) * m_fRec.slant );
373  } else {
374  set_xmax(this->m_maxGlyphBounds, get_xmax(this->m_maxGlyphBounds) + get_ymin(this->m_maxGlyphBounds) * m_fRec.slant );
375  set_xmin(this->m_maxGlyphBounds, get_xmin(this->m_maxGlyphBounds) + get_ymax(this->m_maxGlyphBounds) * m_fRec.slant );
376  }
377  }

381 template<typename app_traits>
382 void FauxFont<app_traits>::ParseHintedGlyphOutlineConst( t4::transient_bezier_path& path, matrix_type& outMatrix, size_t CID_or_GID, const matrix_type& inMatrix, const real_type* weightVector, real_type * /*hShift*/ ) const
383 {
384  using namespace imaging_model;
385 
386  static const unsigned char ToUpper[] =
387  {
388  #include "tetraphilia/fonts/data/BaseToUpper.h"
389  };
390 
391  size_t gid = CID_or_GID;
392  tetraphilia_assert( gid < kNumSubstitutionGlyphs );
393 
394  std::uint16_t targetWidth = m_widths[gid];
395 
396  if ( targetWidth == 0 ) {
397  /* Probably trying to faux a TrueType "nonmarkingreturn" glyph; return
398  without adding any path data */
399  return;
400  }
401 
402  bool isUpper = gid == ToUpper[gid];
403  if (m_desc.m_Flags & (kAllCap|kSmallCap))
404  /* Map lowercase to uppercase */
405  gid = ToUpper[gid];
406 
407  const FauxFD<app_traits>* metrics = &m_fRec.FDArray[isUpper ? 1 : 0];
408 
409  real_type rm[4];
410  ScaleMetrics(metrics->scale, rm, m_baseDesc.widths[gid]);
411 
412  real_type wv[4];
413  int fontFitResult = FontFit(wv,
414  m_desc.m_StemV, metrics->StdVW,
415  targetWidth, rm,
416  m_baseDesc.extrap == nullptr ? nullptr:
417  (*m_baseDesc.extrap)[gid] );
418 
419  typedef typename app_traits::real_geometry_traits::matrix_type matrix_type;
420  matrix_type mat;
421  real_type hShift = 0; // character space horizontal shift
422 
423  if ( fontFitResult == kFontFit_LargeWidth ) {
424  /* Width too big: resort to matrix distortion */
425  real_type d = metrics->scale;
426  real_type c = metrics->scale * m_fRec.slant;
427  real_type z = real_type(0);
428  real_type a = d * real_type(targetWidth) / (wv[0]*rm[0] + wv[1]*rm[1] + wv[2]*rm[2] + wv[3]*rm[3]);
429  MakeMatrix( mat, a, z, c, d, z, z );
430  } else {
431  if ( fontFitResult == kFontFit_SmallWidth ) {
432  // shift by half the width error
433  hShift = ( real_type(targetWidth) - (wv[0]*rm[0] + wv[1]*rm[1] + wv[2]*rm[2] + wv[3]*rm[3]) ) / 2;
434  }
435  MakeMatrix( mat, metrics->scale, 0, metrics->scale * m_fRec.slant, metrics->scale, 0, 0 );
436  }
437 
438  parsers::CFF<app_traits>::ParseHintedGlyphOutlineConst( path, outMatrix, gid, mat * inMatrix, &wv[0], &hShift );
439 }

441 // Calculate height adjustment scaling factors.
442 template<typename app_traits>
443 void FauxFont<app_traits>::CalcFauxScales( const FontDesc& fauxDesc,
444  real_type& lcScale, std::uint16_t lcTargetHeight, std::uint16_t lcBaseHeight,
445  real_type& ucScale, std::uint16_t ucTargetHeight, std::uint16_t ucBaseHeight)
446 {
447  ucScale = real_type(ucTargetHeight) / ucBaseHeight;
448 
449  if (fauxDesc.m_Flags & kSmallCap)
450  lcScale = real_type(lcTargetHeight) / ucBaseHeight;
451  else if (fauxDesc.m_Flags & kAllCap)
452  lcScale = ucScale;
453  else
454  lcScale = real_type(lcTargetHeight) / lcBaseHeight;
455 
456  // Don't scale above 1.0 to avoid line collision
457  if (ucScale > 1)
458  ucScale = 1;
459  if (lcScale > 1)
460  lcScale = 1;
461 }

539  /* Compute tangent from angle in degrees. Reduces angle argument to range -45
540  to +45 degrees (negative values clockwise from vertical). Values outside
541  this range map to extreams. (This is safe enough since the most sloped font
542  in the Adobe library is -42 degrees.) */

544 typename FauxFont<app_traits>::real_type FauxFont<app_traits>::tangent(int angle)
545 {
546  int index = angle;
547  if (index < 0)
548  index = -index;
549  if (static_cast<size_t>(index) >= (sizeof(TangentTable<real_type>::table)/sizeof(TangentTable<real_type>::table[0])))
550  index = sizeof(TangentTable<real_type>::table)/sizeof(TangentTable<real_type>::table[0]) - 1;
551  real_type ret = TangentTable<real_type>::table[index];
552  return (angle < 0) ? ret : -ret;
553 }

557 {

564 template<typename app_traits>
565 bool FauxFont<app_traits>::inLimits(real_type wv[4],
566  const RealCoordPoint& r, const std::uint16_t* limits)
567 {
568  if (limits == nullptr) {
569  // No limits; test against nominal design space
570  if (r.x < 0 || r.x > 1 || r.y < 0 || r.y > 1)
571  return false;
572  } else {
573  // Determine region and test against extrapolation limits.
574  // Test for degenerate triangular sections and return false
575  /* Bug #1362970: using the expression (1 - y0)*(r.x - x1) - (x1 - 1)*(r.y - 1) < 0
576  * and others like it (e.g., hand-optimized) in the 4 triangle tests below
577  * produced bad code with /O2 optimization in both VC7 and VC8.
578  * Workaround is to use variables for the constants 0 and 1.
579  */
580  real_type x0, x1, y0, y1;
581  if (r.x < 0) {
582  if (r.y < 0) { // lower left triangle
583  x0 = 0;
584  y0 = real_services::DivideBy1024( real_type(-limits[4]) );
585  x1 = real_services::DivideBy1024( real_type(-limits[0]) );
586  y1 = 0;
587  if (y0 == 0 || x1 == 0 || (y1 - y0)*(r.x - x1) - (x1 - x0)*(r.y - y1) < 0)
588  return false;
589  } else if (r.y > 1) { // upper left triangle
590  x0 = real_services::DivideBy1024( real_type(-limits[1]) );
591  y0 = 1;
592  x1 = 0;
593  y1 = 1 + real_services::DivideBy1024( real_type(limits[7]) );
594  if (x0 == 0 || y1 == 1 || (y1 - y0)*(r.x - x1) - (x1 - x0)*(r.y - y1) < 0)
595  return false;
596  } else { // left quad
597  x0 = real_services::DivideBy1024( real_type(-limits[0]) );
598  // y0 = 0
599  x1 = real_services::DivideBy1024( real_type(-limits[1]) );
600  // y1 = 1
601  if ((r.x < x0 || r.x < x1) &&
602  (r.x - x1) - (x1 - x0)*(r.y - 1) < 0)
603  return false;
604  }
605  }
606  else if (r.x > 1) {
607  if (r.y < 0) { // lower right triangle
608  x0 = 1 + real_services::DivideBy1024( real_type(limits[3]) );
609  y0 = 0;
610  x1 = 1;
611  y1 = real_services::DivideBy1024( real_type(-limits[5]) );
612  if (x0 == 1 || y1 == 0 || (y1 - y0)*(r.x - x1) - (x1 - x0)*(r.y - y1) < 0)
613  return false;
614  } else if (r.y > 1) { // upper right triangle
615  x0 = 1;
616  y0 = 1 + real_services::DivideBy1024( real_type(limits[6]) );
617  x1 = 1 + real_services::DivideBy1024( real_type(limits[2]) );
618  y1 = 1;
619  if (y0 == 1 || x1 == 1 || (y1 - y0)*(r.x - x1) - (x1 - x0)*(r.y - y1) < 0)
620  return false;
621  } else { // right quad
622  x0 = 1 + real_services::DivideBy1024( real_type(limits[2]) );
623  // y0 = 1
624  x1 = 1 + real_services::DivideBy1024( real_type(limits[3]) );
625  // y1 = 0
626  if ((r.x > x0 || r.x > x1) && (x1 - r.x) - (x1 - x0)*r.y < 0)
627  return false;
628  }
629  }
630  else {
631  if (r.y < 0) { // bottom quad
632  // x0 = 1
633  y0 = real_services::DivideBy1024( real_type(-limits[5]) );
634  // x1 = 0
635  y1 = real_services::DivideBy1024( real_type(-limits[4]) );
636  if ((r.y < y0 || r.y < y1) && (y1 - y0)*r.x + (r.y - y1) < 0)
637  return false;
638  } else if (r.y > 1) { // top quad
639  // x0 = 0
640  y0 = 1 + real_services::DivideBy1024( real_type(limits[7]) );
641  // x1 = 1
642  y1 = 1 + real_services::DivideBy1024( real_type(limits[6]) );
643  if ((r.y > y0 || r.y > y1) &&
644  (y1 - y0)*(r.x - 1) - (r.y - y1) < 0)
645  return false;
646  }
647  }
648  }
649 
650  NDV2WV(wv, r);
651  return true;

654 template<typename app_traits>
655 void FauxFont<app_traits>::isectHypWithLine(
656  int& n, RealCoordPoint (&r)[16],
657  const RealCoordPoint& p0, const RealCoordPoint& p1,
658  real_type u0, real_type u1, real_type u2, real_type u3
659  )
660 {
661  // Integer part of u0-3 is about 10 bits and the line end points are about
662  // 3 bits. Reducing u0-3 by 3 bits keeps the calculations from overflowing
663  // when calculating values a, b, and c. The calculation in solveQuad
664  // cancels this range reduction.
665  u0 = real_services::DivideBy8(u0);
666  u1 = real_services::DivideBy8(u1);
667  u2 = real_services::DivideBy8(u2);
668  u3 = real_services::DivideBy8(u3);
669 
670  real_type dx = p1.x - p0.x;
671  real_type dy = p1.y - p0.y;
672  real_type a = dx*dy*u3;
673  real_type b = dx*u2 - dy*u1 + (p0.x*dy + p0.y*dx)*u3;
674  real_type c = p0.x*u2 - u0 - p0.y*u1 + p0.x*p0.y*u3;
675  real_type t0;
676  real_type t1;
677 
678  switch (solveQuad(a, b, c, t0, t1)) {
679  case 2:
680  if ( t1 >= 0 && t1 <= 1) {
681  r[n].x = p0.x + t1*dx;
682  r[n].y = p0.y + t1*dy;
683  ++n;
684  }
685  // fall through
686  TETRAPHILIA_SWITCH_FALLTHROUGH;
687 
688  case 1:
689  if (t0 >= 0 && t0 <= 1) {
690  r[n].x = p0.x + t0*dx;
691  r[n].y = p0.y + t0*dy;
692  ++n;
693  }
694  // fall through
695  TETRAPHILIA_SWITCH_FALLTHROUGH;
696 
697  case 0:
698  break;
699  }
700 }

702 template<typename app_traits>
703 int FauxFont<app_traits>::findClosest(int n, const RealCoordPoint r[8],
704  real_type x, const real_type xm[4],
705  real_type y, const real_type ym[4])
706 {
707  real_type d = 0;
708  int i = -1;
709  while (n--) {
710  real_type xErr = x - (xm[0] + r[n].x*(xm[1] - xm[0]) +
711  r[n].y*(xm[2] - xm[0]) +
712  r[n].x*r[n].y*(xm[0] - xm[1] - xm[2] + xm[3]));
713  if (xErr < 0)
714  xErr = -xErr;
715 
716  real_type yErr = 0;
717  if ( ym != nullptr){
718  yErr = y - (ym[0] + r[n].x*(ym[1] - ym[0]) +
719  r[n].y*(ym[2] - ym[0]) +
720  r[n].x*r[n].y*(ym[0] - ym[1] - ym[2] + ym[3]));
721  if (yErr < 0)
722  yErr = -yErr;
723  }
724 
725  // second target (width) is weighted less than the first
726  real_type err = 5 * xErr + yErr;
727 
728  // search for first minimum error
729  if (i == -1 || err < d) {
730  i = n;
731  d = err;
732  }
733  }
734  return i;
735 }

737 template<typename app_traits>
738 int FauxFont<app_traits>::solveQuad(
739  real_type a,
740  real_type b,
741  real_type c,
742  real_type& x0,
743  real_type& x1
744 )
745 {
746  if (a == 0) {
747  if (b != 0) {
748  x0 = -c/b;
749  return 1;
750  }
751  } else {
752  b /= a;
753  c /= a;
754  real_type d = b*b - c*4;
755  if (d > 0) {
756  d = real_services::Sqrt(d);
757  x0 = real_services::DivideBy2(-b - d);
758  x1 = real_services::DivideBy2(-b + d);
759  return 2;
760  } else if (d == 0) {
761  x0 = -real_services::DivideBy2(b);
762  return 1;
763  }
764  }
765  return 0;
766 }

769 template<typename app_traits>
770 int FauxFont<app_traits>::FontFit(real_type wv[4],
771  std::uint16_t xt, const real_type xm[4],
772  std::uint16_t yt, const real_type ym[4],
773  const std::uint16_t *limits )
774 {
775  real_type u0 = xt - xm[0];
776  real_type u1 = xm[0] - xm[2];
777  real_type u2 = xm[1] - xm[0];
778  real_type u3 = xm[3] - xm[1] + u1;
779  real_type v0 = static_cast<int>(yt) - ym[0];
780  real_type v1 = ym[0] - ym[2];
781  real_type v2 = ym[1] - ym[0];
782  real_type v3 = ym[3] - ym[1] + v1;
783 
784  real_type p0 = real_services::DivideBy4( u0 );
785  real_type p1 = real_services::DivideBy4( u1 );
786  real_type p2 = real_services::DivideBy4( u2 );
787  real_type p3 = real_services::DivideBy4( u3 );
788  real_type q0 = real_services::DivideBy16( v0 );
789  real_type q1 = real_services::DivideBy16( v1 );
790  real_type q2 = real_services::DivideBy16( v2 );
791  real_type q3 = real_services::DivideBy16( v3 );
792 
793  real_type a = p3*q1 - p1*q3;
794  real_type b = p2*q1 + p3*q0 - p0*q3 - p1*q2;
795  real_type c = p2*q0 - p0*q2;
796 
797  RealCoordPoint r[16];
798 
799  switch (solveQuad(a, b, c, r[0].y, r[1].y)) {
800  case 2:
801  r[1].x = v2 + r[1].y*v3;
802  if (r[1].x != 0) {
803  r[1].x = (v0 + r[1].y*v1)/r[1].x;
804  if (inLimits(wv, r[1], limits))
805  return kFontFit_Success;
806  }
807  // fall through
808  TETRAPHILIA_SWITCH_FALLTHROUGH;
809 
810  case 1:
811  r[0].x = v2 + r[0].y*v3;
812  if (r[0].x != 0) {
813  r[0].x = (v0 + r[0].y*v1)/r[0].x;
814  if (inLimits(wv, r[0], limits))
815  return kFontFit_Success;
816  }
817  // fall through
818  TETRAPHILIA_SWITCH_FALLTHROUGH;
819 
820  case 0:
821  break;
822  }
823 
824  // no roots
825  int n = 0;
826  if (limits == nullptr) {
827  // Compute weight match on edge of design space
828  if (u1 != 0) {
829  r[n].x = 0;
830  r[n].y = u0/-u1;
831  if (r[n].y >= 0 && r[n].y <= 1)
832  ++n;
833  }
834  if (u2 != 0) {
835  r[n].x = u0/u2;
836  r[n].y = 0;
837  if (r[n].x >= 0 && r[n].x <= 1)
838  ++n;
839  }
840  a = xm[3] - xm[1];
841  if (a != 0) {
842  r[n].x = 1;
843  r[n].y = (static_cast<int>(xt) - xm[1])/a;
844  if (r[n].y >= 0 && r[n].y <= 1)
845  ++n;
846  }
847  a = xm[3] - xm[2];
848  if (a != 0) {
849  r[n].x = (static_cast<int>(xt) - xm[2])/a;
850  r[n].y = 1;
851  if (r[n].x >= 0 && r[n].x <= 1)
852  ++n;
853  }
854 
855  // Choose nearest width
856  n = findClosest(n, r, real_type(yt), ym);
857  if (n != -1)
858  NDV2WV(wv, r[n]);
859  else {
860  // Find "best" match of weight and width on corner of design space
861  r[0].x = 0; r[0].y = 0;
862  r[1].x = 1; r[1].y = 0;
863  r[2].x = 0; r[2].y = 1;
864  r[3].x = 1; r[3].y = 1;
865  n = findClosest(4, r, real_type(xt), xm, real_type(yt), ym);
866  NDV2WV(wv, r[n]);
867  }
868  } else {
869  // Compute weight match on limits polygon
870  RealCoordPoint p[8];
871  p[0].x = real_services::DivideBy1024( real_type(-limits[0]) );
872  p[0].y = 0;
873  p[1].x = real_services::DivideBy1024( real_type(-limits[1]) );
874  p[1].y = 1;
875  p[2].x = 0;
876  p[2].y = 1 + real_services::DivideBy1024( real_type(limits[7]) );
877  p[3].x = 1;
878  p[3].y = 1 + real_services::DivideBy1024( real_type(limits[6]) );
879  p[4].x = 1 + real_services::DivideBy1024( real_type(limits[2]) );
880  p[4].y = 1;
881  p[5].x = 1 + real_services::DivideBy1024( real_type(limits[3]) );
882  p[5].y = 0;
883  p[6].x = 1;
884  p[6].y = real_services::DivideBy1024( real_type(-limits[5]) );
885  p[7].x = 0;
886  p[7].y = real_services::DivideBy1024( real_type(-limits[4]) );
887 
888  // Intersect edges of extrapolation limits polygon with weight curve
889  isectHypWithLine(n, r, p[0], p[1], u0, u1, u2, u3);
890  isectHypWithLine(n, r, p[1], p[2], u0, u1, u2, u3);
891  isectHypWithLine(n, r, p[2], p[3], u0, u1, u2, u3);
892  isectHypWithLine(n, r, p[3], p[4], u0, u1, u2, u3);
893  isectHypWithLine(n, r, p[4], p[5], u0, u1, u2, u3);
894  isectHypWithLine(n, r, p[5], p[6], u0, u1, u2, u3);
895  isectHypWithLine(n, r, p[6], p[7], u0, u1, u2, u3);
896  isectHypWithLine(n, r, p[7], p[0], u0, u1, u2, u3);
897 
898  if (n > 0) {
899  // Got weight match; find nearest width
900  n = findClosest(n, r, real_type(yt), ym);
901  NDV2WV(wv, r[n]);
902  } else {
903  // No weight match; find "best" match of weight and width on corners of design space
904  n = findClosest(8, p, real_type(xt), xm, real_type(yt), ym);
905  NDV2WV(wv, p[n]);
906  }
907  }
908 
909  // Width mismatch; return whether too big or too small
910  real_type width = wv[0]*ym[0] + wv[1]*ym[1] + wv[2]*ym[2] + wv[3]*ym[3];
911  return (width < yt)? kFontFit_SmallWidth: kFontFit_LargeWidth;
912 }
