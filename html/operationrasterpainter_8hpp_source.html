78  template<typename x_walker_cluster_type>
79  TETRAPHILIA_FORCE_INLINE void operator()( const x_walker_cluster_type* walkers )
80  {
81  size_t count = NumStoredSignals( walkers, Shape() );
82  for ( size_t i = 0; i < count; ++i )
83  GetDestSignal( walkers, Shape(), i ) = m_op( walkers, Shape(), i );
84 
85  count = NumStoredSignals( walkers, Opacity() );
86  for ( size_t i = 0; i < count; ++i )
87  GetDestSignal( walkers, Opacity(), i ) = m_op( walkers, Opacity(), i );
88 
89  count = NumStoredSignals( walkers, Color() );
90  for ( size_t i = 0; i < count; ++i )
91  GetDestSignal( walkers, Color(), i ) = m_op( walkers, Color(), i );
92  }
93 
94  separable_operation_type m_op;
95 
96  #ifdef _DEBUG
97  const char* GetName( void )
98  {
99  return m_op.GetName();
100  }
101  #endif
