80 // a bitmap cache key combines gid and subpixel positions (phases)
81 // phase is a combination of quantized (e.g., to 1/4 pixel) x and y subpixel positions
82 // the only requirement is that m_phases be unique for each quantized subpixel position

134  bool operator==( const BitmapCacheKey& rhs ) const
135  {
136  return (this->m_gid == rhs.m_gid) && (this->m_phases == rhs.m_phases) && (this->m_clipRect == rhs.m_clipRect);
137  }
138 
139  bool operator!=( const BitmapCacheKey& rhs ) const
140  {
141  return !operator==( rhs );
142  }
143 
144  bool operator<( const BitmapCacheKey& rhs ) const
145  {
146  if ( this->m_gid != rhs.m_gid )
147  return this->m_gid < rhs.m_gid;
148  if ( this->m_phases != rhs.m_phases )
149  return this->m_phases < rhs.m_phases;
150  if ( !this->m_clipRect )
151  return bool( rhs.m_clipRect );
152  else if ( !rhs.m_clipRect )
153  return false;
154  else
155  {
156  const int_rect_type& l = *this->m_clipRect;
157  const int_rect_type& r = *rhs.m_clipRect;
158 
159  if ( get_xmin( l ) != get_xmin( r ) )
160  return get_xmin( l ) < get_xmin( r );
161  if ( get_ymin( l ) != get_ymin( r ) )
162  return get_ymin( l ) < get_ymin( r );
163  if ( get_xmax( l ) != get_xmax( r ) )
164  return get_xmax( l ) < get_xmax( r );
165  return get_ymax( l ) < get_ymax( r );
166  }
167  }

174  if ( key.m_clipRect )

177  }

194 // SubpixelPositionPolicy class encapsulates a policy that converts x,y positions into a
195 // controlled number of distinct renderings, based on quantized subpixel x and y positions
196 // This parent class sets a default policy which may be overridden by font types that
197 // subclass this.
198 // Subpixel positions returned are in device space.
199 template<typename app_traits>
200 class SubpixelPositionPolicy
201 {
202  typedef typename app_traits::real_geometry_traits::matrix_type matrix_type;
203  typedef typename app_traits::real_geometry_traits::coord_type real_type;
204  typedef typename app_traits::real_geometry_traits::point_type real_point_type;
205  typedef typename app_traits::int_geometry_traits::coord_type int_type;
206  typedef typename app_traits::int_geometry_traits::point_type int_point_type;
207  typedef typename app_traits::int_geometry_traits::rect_type int_rect_type;
208 
209 protected:
210  unsigned int m_xPositionsPerPixel;
211  unsigned int m_yPositionsPerPixel;
212 
213 public:
214  // default policy is to disable subpixel positioning
215  SubpixelPositionPolicy( const matrix_type& /*mat*/ ) :
216  m_xPositionsPerPixel( 1 ),
217  m_yPositionsPerPixel( 1 )
218  {}
219 
220  // helper functions compute subpixel phase and key for a bitmap in this BitmapCache
221  // for example, with 4 positions per pixel,
222  // .000 <= x < .125 is phase 0,
223  // .125 <= x < .375 is phase 1, etc.
224  // .875 <= x < 1.00 is phase 0 but with carry
225  void GetIntCarries( real_type xOrigin, real_type yOrigin, unsigned int& xCarry, unsigned int& yCarry ) const
226  {
227  xCarry = XPhasePlus( xOrigin ) / m_xPositionsPerPixel;
228  yCarry = YPhasePlus( yOrigin ) / m_yPositionsPerPixel;
229  }
230  void GetFracPositions( real_type xOrigin, real_type yOrigin, real_type& xPosition, real_type& yPosition ) const
231  {
232  xPosition = real_type(XPhase( xOrigin )) / real_type(m_xPositionsPerPixel);
233  yPosition = real_type(YPhase( yOrigin )) / real_type(m_yPositionsPerPixel);
234  }
235  void GetPositions( real_type xOrigin, real_type yOrigin, real_type& xPosition, real_type& yPosition ) const
236  {
237  xPosition = real_services::Floor(xOrigin) + real_type(XPhasePlus( xOrigin )) / real_type(m_xPositionsPerPixel);
238  yPosition = real_services::Floor(yOrigin) + real_type(YPhasePlus( yOrigin )) / real_type(m_yPositionsPerPixel);
239  }
240 
241  // build a key for a glyph cache; x and y phases are encoded into a single int value
242  BitmapCacheKey MakeKey( size_t gid, real_type xOrigin, real_type yOrigin, const int_rect_type* clipRect ) const
243  {
244  tetraphilia_assert( m_xPositionsPerPixel < 256 && m_yPositionsPerPixel < 256 );
245  return BitmapCacheKey( gid, ( YPhase(yOrigin) << 8 ) + XPhase(xOrigin), clipRect );
246  }
247 
248  // compute fractional positions from a decoded key
249  void GetFracPositions( const BitmapCacheKey& key, real_type& xPosition, real_type& yPosition ) const
250  {
251  xPosition = real_type( key.GetPhases() & 0xff ) / real_type(m_xPositionsPerPixel);
252  yPosition = real_type( key.GetPhases() >> 8 ) / real_type(m_yPositionsPerPixel);
253  }
254 
255  int_point_type ComputeDeviceLoc( const real_point_type& p )
256  {
257  real_type xfrac = real_type( 1 ) / ( 2 * m_xPositionsPerPixel );
258  real_type yfrac = real_type( 1 ) / ( 2 * m_yPositionsPerPixel );
259 
260  return imaging_model::MakePoint<int_point_type>(
261  real_services::FloorToInt( ArgType<int_type>(), get_x( p ) + xfrac ),
262  real_services::FloorToInt( ArgType<int_type>(), get_y( p ) + yfrac ) );
263  }
264 
265 private:
266  // return integer phase, including possible carry
267  // E.g., for 4 positions, 0 and 4 may both be returned for phase 0
268  unsigned int XPhasePlus( real_type xOrigin ) const
269  {
270  return real_services::RoundToInt( ArgType<unsigned int>(),
271  real_services::Fraction( xOrigin ) * m_xPositionsPerPixel);
272  }
273  unsigned int YPhasePlus( real_type yOrigin ) const
274  {
275  return real_services::RoundToInt( ArgType<unsigned int>(),
276  real_services::Fraction( yOrigin ) * m_yPositionsPerPixel);
277  }
278 
279  // return integer phase exclusive of carry
280  unsigned int XPhase( real_type xOrigin ) const
281  {
282  return XPhasePlus( xOrigin ) % m_xPositionsPerPixel;
283  }
284  unsigned int YPhase( real_type yOrigin ) const
285  {
286  return YPhasePlus( yOrigin ) % m_yPositionsPerPixel;
287  }
288 };

1002  // completely empty (see bug 2684474). Rather than fail these files completely, treat
1003  // them as if they were unembedded (like the desktop reader does)
1004  if ( bds->AtEOF() )
1005  return kUnSupported;
1006  }
1007 
1008  unsigned int first4Bytes = data_io::BytesToInteger<unsigned int>( *bds, 4 );
1009  switch ( first4Bytes ) {
1010  case sfr_true_tag:
1011  return kTrueType;
1012  case sfr_v1_0_tag:
1013  case sfr_OTTO_tag:
1014  case sfr_typ1_tag: {
1015  unsigned int numTables = data_io::BytesToInteger<unsigned int>( *bds, 2 );
1016  bds->Advance( 6 );
1017  unsigned int i;
1018  for ( i = 0; i < numTables; i++ ) {
1019  unsigned int tag = data_io::BytesToInteger<unsigned int>( *bds, 4 );
1020  if ( tag == CTL_TAG<'C', 'F', 'F', ' '>::val )
1021  return kCFF;
1022  else
1023  bds->Advance( 12 );
1024  }
1025  return kTrueType; // TODO get TrueType, etc...
1026  };
1027 
1028  default:
1029  bds->Seek( 0 );
1030  }
1031 
1032  if ( data_io::BytesToInteger<size_t>( *bds, 1 ) == 1 )
1033  return kCFF;
1034 
1035  // check for "!PS-AdobeFont"
1036  const char* bytes;
1037  bds->Seek( 0 );
1038  size_t len = bds->PeekBytes( bytes, 14 );
1039  if ( len >= 14 && ((memcmp( bytes, "%!PS-AdobeFont", 14 ) == 0)
1040  || (memcmp( bytes, "%!FontType1", 11 ) == 0)
1041  || (memcmp( bytes, "%!FontType 1", 12) == 0)) ) // TODO make sure this is the right test
1042  return kType1;
1043 
1044  // check for PFB. This is not permitted by the spec, but some PDF producers embed PFB files into their font dictionaries.
1045  // A PFB font is just a Type1 font with a six byte prefix. The first two bytes are 0x8001, the next four are a length indicator.
1046  // PFB also sticks 0x8002 followed by another length immediately before the binary section of the data (after the eexec).
1047  if ( app_traits::pdf_traits::permissive )
1048  {
1049  bds->Seek( 0 );

1052  if ( len >= 2 && memcmp( bytes, kPFBHeader, 2 ) == 0 )

1054  }
