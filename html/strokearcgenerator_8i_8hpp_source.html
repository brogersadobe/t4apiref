25 // This class constructs a clockwise circular-ish arc of beziers, starting at startOffset,
26 // perpendicular to startOffset-startCenter, ending at endOffset, perpendicular to
27 // endOffset - endCenter. Note that the algorithm is designed to work "perfectly" in the
28 // case where the startCenter and endPath points are identical, and should gracefully adapt
29 // as they move apart. So long as the distance between the two path points is much smaller than
30 // the radius, I believe this will be very well behaved and produce pleasing output.
31 // For extreme cases (which should never arise), I do not have a good idea of what will come out....
32 //
33 // The output is 3*n points to be interpreted as CurveTo's from the current point,
34 // which should be identical to startOffset.
35 template <typename stroker_traits>
36 class ArcGenerator
37 {
38 public:
39  TETRAPHILIA_STROKER_IMPL_TYPEDEF_BLOCK
40 
41  ArcGenerator( const point_type& startCenter,
42  const point_type& startOffset,
43  const point_type& endCenter,
44  const point_type& endOffset,
45  coord_type radius );
46 
47  size_t GetNumCurveToPoints() const { return m_numPoints; }
48  const point_type* GetCurveToPoints() const { return m_points; }
49  void AppendToPath( output_path_type& outPath) const;
50 
51  static constexpr t4::int_coord kLambdaTableScale = 128;
52 
53 private:
54  static unsigned int getCWQuadrant(coord_type x, coord_type y);
55  static unsigned int getNumCWSpannedQuandrants(const point_type& start, const point_type& end);
56  point_type FindArcMidpoint(const point_type& a, const point_type& b) const;
57 
58  // the startCenter and endCenter are used to guarantee perpendicularity to the real
59  // edges, while the virtual center is used for the math and quadrants
60  void addCWArc( const point_type& startOrigin,
61  const point_type& start,
62  const point_type& endOrigin,
63  const point_type& end );

65  enum { kBufferSize = 15 }; // max of five arcs
66 
67  const point_type fCenter;
68  coord_type fRadius;
69  size_t m_numPoints;
70  point_type m_points[kBufferSize];
71 };

226 template <typename stroker_traits>
227 ArcGenerator<stroker_traits>::ArcGenerator(const point_type& startCenter, const point_type& startOffset, const point_type& endCenter, const point_type& endOffset, coord_type radius)
228 : fCenter(AveragePt(startCenter, endCenter)), fRadius(radius), m_numPoints(0)
229 {
230  if (fRadius > 0)
231  {
232  point_type start = startOffset - fCenter;
233  point_type end = endOffset - fCenter;
234  point_type startOrigin = startCenter - fCenter;
235  point_type endOrigin = endCenter - fCenter;
236 
237  unsigned int quadrantSpan = getNumCWSpannedQuandrants(start, end);
238  if (quadrantSpan == 1)
239  {
240  // one <90deg arc is good enough
241  addCWArc(startOrigin, start, endOrigin, end);
242  }
243  else if (quadrantSpan == 2)
244  {
245  point_type midpoint;
246  midpoint = FindArcMidpoint(start, end);
247  addCWArc(startOrigin, start, point_type(0,0), midpoint);
248  addCWArc(point_type(0,0), midpoint, endOrigin, end);
249  }
250  else
251  {
252  point_type onequarterpoint, midpoint, threequarterpoint;
253  midpoint = FindArcMidpoint(start, end);
254  onequarterpoint = FindArcMidpoint(start, midpoint);
255  threequarterpoint = FindArcMidpoint(midpoint, end);
256  addCWArc(startOrigin, start, point_type(0,0), onequarterpoint);
257  addCWArc(point_type(0,0), onequarterpoint, point_type(0,0), midpoint);
258  addCWArc(point_type(0,0), midpoint, point_type(0,0), threequarterpoint);
259  addCWArc(point_type(0,0), threequarterpoint, endOrigin, end);
260  }
261  // recenter
262  for (size_t i = 0; i < m_numPoints; i++)
263  m_points[i] += fCenter;
264  }
265 }

267 template <typename stroker_traits>
268 unsigned int ArcGenerator<stroker_traits>::getCWQuadrant(coord_type x, coord_type y)
269 {
270  // computes which quadrant a point is in, with:
271  // [3o'clock, 6o'clock) in quadrant 1
272  // [6o'clock, 9o'clock) in quadrant 2
273  // [9o'clock, 12o'clock) in quadrant 3
274  // [12o'clock, 3o'clock) in quadrant 4
275  // origin is arbitrarily in quadrant 1
276 
277  if (x > 0 && y <= 0)
278  return 1;
279  if (x <= 0 && y < 0)
280  return 2;
281  if (x < 0 && y >= 0)
282  return 3;
283  if (x >= 0 && y > 0)
284  return 4;
285  return 1; // x == y == 0
286 }

288 template <typename stroker_traits>
289 unsigned int ArcGenerator<stroker_traits>::getNumCWSpannedQuandrants(const point_type& start, const point_type& end)
290 {
291  // This basically converts to a coordinate system where start is on the positive x axis,
292  // and start.Rotate90CCW() is on the positive y axis.
293  return getCWQuadrant(DotForSignOnly(start, end), CrossForSignOnly(start, end));
294 }

296 template <typename stroker_traits>
297 typename ArcGenerator<stroker_traits>::point_type ArcGenerator<stroker_traits>::FindArcMidpoint(const point_type& a, const point_type& b) const
298 {
299  // this basically find losslessly computed points "near" the expected answer, then
300  // finds the midpoint of those; it then scales out to the radius.
301  point_type mp;
302  switch(getNumCWSpannedQuandrants(a, b))
303  {
304  case 1:
305  mp = (a+b)/static_cast<int_type>(2);
306  break;
307  case 2:
308  case 3:
309  mp = (a.Rotate90CW()+b.Rotate90CCW())/static_cast<int_type>(2);
310  break;
311  case 4:
312  mp = -(a+b)/static_cast<int_type>(2);
313  break;
314  default:
315  tetraphilia_assert_always();
316  }
317  return mp * (fRadius/mp.Mag());
318 }

320 // the startCenter and endCenter are used to guarantee perpendicularity to the real
321 // edges, while the virtual center is used for the math
322 template <typename stroker_traits>
323 void ArcGenerator<stroker_traits>::addCWArc(const point_type& startOrigin, const point_type& start, const point_type& endOrigin, const point_type& end)
324 {
325  tetraphilia_assert(getNumCWSpannedQuandrants(start, end) < 3); // should be 1, but might be barely two
326  tetraphilia_assert(m_numPoints+3 <= kBufferSize); // room for 3 more points?
327 
328  coord_type costheta = Dot(start,end)/(fRadius*fRadius); // needs len(start) == len(end) == radius, which it is
329  std::int32_t tableIndex = real_services::RoundToInt(ArgType<std::int32_t>(), costheta*coord_type(kLambdaTableScale));
330 
331  // because we are not really dividing by the mags of the vectors, but an approx,
332  // this could actually produce cosines slightly greater than 1.0 or slightly less
333  // than 0. (Despite the name, t4::pin does not require integral operands)
334  tableIndex = t4::pin<std::int32_t>( 0, tableIndex, std::int32_t( kLambdaTableScale ) );
