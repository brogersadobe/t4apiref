178 template<typename _signal_traits>
179 template<typename glyph_iterator_type>
180 TextRasterPainter<_signal_traits>::TextRasterPainter( const constraints_type& constraints,
181  const typename fonts::Font<app_traits>::ptr_type& font,
182  const matrix_type& ctm,
183  const matrix_type& fontMatrix,
184  const glyph_iterator_type& start,
185  const glyph_iterator_type& end )
186  : m_glyphs( *t4::get_thread_heap() ),
187  m_cache(font->GetCache( fontMatrix)),
188  m_nonZeroBounds( EmptyRect<int_rect_type>() ),
189  m_pixelStore( *t4::get_thread_heap() ),
190  m_zeroWalker( t4::make_brief<pixel_walker_type>( IdentityPixelBuffers<signal_traits>::ZerosBuffer() ) ),
191  m_constraints( constraints ),
192  m_font( font )
193 {
194  m_glyphs.reserve( 16 );
195  auto& heap = *t4::get_thread_heap();
196  const int_rect_type* clipRect = m_cache->CanCacheBitmap() ? nullptr : &constraints.m_interestBounds;
197  auto bitmapsHolder = CacheAllGlyphs( start, end, ctm, clipRect );
198 
199  // Fill the glyph vector
200  glyph_iterator_type it = start;
201 
202  while ( it != end )
203  {
204  m_glyphs.push_back( glyph_data_type() );
205  glyph_data_type* glyph = &m_glyphs[m_glyphs.size() - 1];
206 
207  glyph->m_glyphStartPt = MatrixTransform( ctm, it->glyphStartPt );
208  glyph->m_key = m_cache->GetBitmapCacheKey( it->gid, get_x( glyph->m_glyphStartPt ), get_y( glyph->m_glyphStartPt ), clipRect );
209  fonts::Bitmap* b = m_cache->GetBitmap_NoCreate( glyph->m_key );
210 
211  /* In case of web rendering, the bitmap may be returned null in first pass of rendering for unembedded fonts and we don't want to cache
212  the null bitmaps because those would be set with proper value in the second pass of rendering.
213  */
214  if(!b)
215  break;
216 
217  int_rect_type glyphBounds;
218  glyphBounds = b->GetBounds();
219 
220  // adjust integer origin for subpixel rounding
221  int_point_type deviceLoc = m_cache->ComputeDeviceLoc( glyph->m_glyphStartPt );
222  MakeRectangle( glyph->m_bounds, get_xmin(glyphBounds) + get_x( deviceLoc ), get_ymin(glyphBounds) + get_y( deviceLoc ), get_xmax(glyphBounds) + get_x( deviceLoc ), get_ymax(glyphBounds) + get_y( deviceLoc ) );
223 
224  if ( RectsIntersect( glyph->m_bounds, constraints.m_interestBounds ) )
225  {
226  glyph->m_bitmap = b;
227  // Aggregate bounds
228  m_nonZeroBounds = RectUnion( m_nonZeroBounds, glyph->m_bounds );
229  }
230  ++it;
231  }
232 
233  m_pixelStore.Init( m_nonZeroBounds,
234  PixelBufferLayoutDescriptor(0, PixelBufferLayout::kReplicatedSignal, 1,
235  true/*littleEndian*/, false/*planar*/, false/*tileX*/, true/*tileY*/, true/*replicateSignals*/ ), false, heap );
236 
237  m_pixelWalker = t4::make_brief<pixel_walker_type>( m_pixelStore.GetConstPixelBuffer() );
238 
239  m_segments[0] = segment_type( m_zeroWalker, {}, {} );
240  m_segments[1] = segment_type( m_pixelWalker, {}, {} );
241 
242  this->Initialize( 2, m_segments, m_nonZeroBounds, kEverythingNeeded);
243 
244  #ifdef _DEBUG
245  this->SetName( "Text" );
246  #endif
247 }

249 template<typename signal_traits>
250 template<typename glyph_iterator_type>
251 auto TextRasterPainter<signal_traits>::CacheAllGlyphs( const glyph_iterator_type& start,
252  const glyph_iterator_type& end,
253  const matrix_type& ctm,
254  const int_rect_type* clipRect ) -> t4::vector<bitmap_entry>
255 {
256  using namespace imaging_model;
257  typedef decltype(*start) glyph_type;
258 
259  t4::vector<bitmap_entry> bitmapsHolder;
260  bitmapsHolder.reserve( static_cast<size_t>( end - start ) );
261  std::transform( start, end, std::back_inserter( bitmapsHolder ), [this,&ctm,clipRect]( const glyph_type& info )
262  {
263  real_point_type pt = MatrixTransform( ctm, info.glyphStartPt );
264  return std::make_pair( this->m_cache->GetBitmapCacheKey( info.gid, get_x( pt ), get_y( pt ), clipRect ), t4::nullptr_to<fonts::Bitmap> );
265  } );
266 
267  // Erase duplicates
268  std::sort( bitmapsHolder.begin(), bitmapsHolder.end(), []( const bitmap_entry& e1, const bitmap_entry& e2 ) { return e1.first < e2.first; } );
269  auto bhend = bitmapsHolder.end();
270 
271  // remove duplicates
272  bhend = std::unique( bitmapsHolder.begin(), bhend,
273  []( const bitmap_entry& e1, const bitmap_entry& e2 ) { return e1.first == e2.first; } );
274 
275  // remove entries that are already created
276  bhend = std::remove_if( bitmapsHolder.begin(), bhend, [=]( const bitmap_entry& e )
277  {
278  auto b = this->m_cache->GetBitmap_NoCreate( e.first );
279  return b != nullptr;
280  } );
281 
282  // erase removed entries
283  bitmapsHolder.erase( bhend, bitmapsHolder.end() );
284 
285 #ifndef T4_SINGLE_THREADED_GLYPH_RENDERING
286  size_t bSize = bitmapsHolder.size();
287  /* disable multiThreading (i.e. parallel tasks) for very low number of glyphs */
288  const bool multiThreading = m_font->SupportsMT() && (bSize > 3);
289  t4::thread_pool* tp = multiThreading ? &GetThreadPool() : nullptr;
290 #endif /* T4_SINGLE_THREADED_GLYPH_RENDERING */
291  t4::synchronized<t4::exception_ptr> exc;
292 
293  {
294  boost::optional<t4::share_heap> sh;
295  auto futures = t4::make_transient_vector<t4::future<void>>( *t4::get_thread_heap() );
296 
297 #ifndef T4_SINGLE_THREADED_GLYPH_RENDERING
298  // share the bitmap cache's transient heap; this puts it in a mode where
299  // multiple threads can allocate from it.
300  if ( multiThreading )
301  sh.emplace( m_cache->GetCacheHeap() );
302 #endif /* T4_SINGLE_THREADED_GLYPH_RENDERING */
303  // Queue all bitmap creation tasks
304  for ( auto& e : bitmapsHolder )
305  {
306  fonts::Bitmap** dst = &e.second;
307  auto task = [this, e, &exc, dst]()
308  {
309  try
310  {
311  t4::fpu_control fc;
312  t4::establish_thread_heap eth;
313 
314  //TODO_READER_MOBILE<vingupta>: Handle out of memory due to high peak value (due to multiple threads)
315  *dst = m_cache->GetBitmap( e.first, this->m_font, this->m_cache->GetCacheHeap() );
316  tetraphilia_assert( *dst );
317  }
318  catch( ... )
319  {
320  // If multiple glyphs fail, we might overwrite this, but it doesn't really matter
321  // as long as _something_ is thrown to signal failure
322  *exc = t4::current_exception();
323  }
324  };
325 
326 #ifndef T4_SINGLE_THREADED_GLYPH_RENDERING
327  if ( multiThreading )
328  futures.push_back( tp->schedule( std::forward<decltype(task)>( task ) ) );
329  else
330 #endif /* T4_SINGLE_THREADED_GLYPH_RENDERING */
331  task();
332  }
333 
334  for ( auto& f : futures )
335  f.wait();
336  }
337 
338  if ( *exc != t4::exception_ptr() )
339  t4::rethrow_exception( *exc );
340 
341  #ifdef _DEBUG
342  for ( auto& e : bitmapsHolder )
343  tetraphilia_assert( e.second );
344  #endif
345 
346  return bitmapsHolder;
347 }

349 template<typename _signal_traits>
350 void TextRasterPainter<_signal_traits>::SetYImpl( int_type y )
351 {
352  typedef typename app_traits::int_geometry_traits::coord_type int_type;
353 
354  m_y = y;
355  m_insideNZBounds = ( y >= get_ymin(m_nonZeroBounds) && y < get_ymax(m_nonZeroBounds) );
356  if ( m_insideNZBounds ) {
357 
358  signal_type* buff = m_pixelStore.GetPixelBuffer().GetPixels();
359 
360  memset( buff, 0, static_cast<size_t>( get_width(m_nonZeroBounds) ) );
361 
362  for ( auto it = m_glyphs.begin(); it != m_glyphs.end(); ++it )
363  {
364  if ( get_ymin(it->m_bounds) <= y && get_ymax(it->m_bounds) > y )
365  {
366  if ( it->m_bitmap )
367  {
368  const unsigned char* src = it->m_bitmap->GetBits();
369  int_type scanStride = get_xmax(it->m_bounds) - get_xmin(it->m_bounds);
370  src += (y - get_ymin(it->m_bounds)) * scanStride;
371  unsigned char* pp = buff + get_xmin(it->m_bounds) - get_xmin(m_nonZeroBounds);
372  for ( int_type x = 0; x < scanStride; x++, pp++, src++ )
373  *pp |= *src;
374  }
375  }
376 
377  }
378 
379  }

382 template<typename app_traits, typename glyph_iterator_type>
383 void GetTextOutlines( const typename fonts::Font<app_traits>::ptr_type& font,
384  const typename app_traits::real_geometry_traits::matrix_type& ctm,
385  const typename app_traits::real_geometry_traits::matrix_type& fontMatrix,
386  const glyph_iterator_type& start,
387  const glyph_iterator_type& end,
388  t4::stable_transient_bezier_path& resultPath)
389 {
390  typedef typename app_traits::real_geometry_traits::matrix_type matrix_type;
391  typedef typename app_traits::real_geometry_traits::point_type real_point_type;
392 
393  matrix_type scaledFontMatrix = font->GetFontMatrix() * fontMatrix; // (fontMatrix * charMatrix * y-flip * ctm) without translation
394 
395  glyph_iterator_type it = start;
396  while ( it != end ) {
397 
398  real_point_type pt = MatrixTransform( ctm, it->glyphStartPt );
399 
400  matrix_type outMatrix;
401  t4::transient_bezier_path tmpPath( *t4::get_thread_heap() );
402  font->ParseHintedGlyphOutline( tmpPath, outMatrix, it->gid, scaledFontMatrix );
403 
404  fonts::SubpixelPositionPolicy<app_traits> pos = font->NewSubpixelPositionPolicy( scaledFontMatrix );
405 
406  // adjust subpixel position of the path via matrix translation elements
407  pos.GetPositions( get_x(pt), get_y(pt), get_e(outMatrix), get_f(outMatrix) );
408 
409  for ( auto r : tmpPath )
410  {
411  auto moveto = [&]( t4::real_point const& p )
412  {
413  resultPath.moveto( t4::transform( p, outMatrix ) );
414  };
415 
416  auto lineto = [&]( t4::real_point const& p )
417  {
418  resultPath.lineto( t4::transform( p, outMatrix ) );
419  };
420 
421  auto curveto = [&]( t4::real_point const& p0, t4::real_point const& p1, t4::real_point const& p2 )
422  {
423  resultPath.curveto( t4::transform( p0, outMatrix ), t4::transform( p1, outMatrix ), t4::transform( p2, outMatrix ) );
424  };
425 
426  auto closepath = [&](){ resultPath.closepath(); };
427 
428  r.visit( moveto, lineto, curveto, closepath );
429  }
430 
431  ++it;
432  }
433 
434 }
