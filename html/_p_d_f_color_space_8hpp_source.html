98  smart_ptr_type CreateCalCS(
99  color::CSFamily family,
100  const t4::pdf::dictionary& dict,
101  bool halveRangeValues = false )
102  {
103  tetraphilia_assert( family == color::kCalGray || family == color::kCalRGB || family == color::kLab );
104 
105  if( family == color::kCalGray )
106  family = color::kDeviceGray;
107  else if( family == color::kCalRGB )
108  family = color::kDeviceRGB;
109  smart_ptr_type result( get_ColorContext( GetGlobalContext() ).GetDeviceColorSpace( family ) );
110  if( family == color::kLab )
111  {
112  boost::optional<store::Array<store::StoreObjTraits<app_traits> > > rangeArray = dict.GetArray( "Range" );
113  if( rangeArray )
114  {
115  unsigned int index;
116  color_conv_type ranges[ 6 ];
117  ranges[ 0 ] = 0;
118  ranges[ 1 ] = 100;
119  for( index = 2; index < 6; ++index )
120  ranges[ index ] = rangeArray->Get( index - 2 ).RealValue();
121  if( halveRangeValues )
122  for( index = 2; index < 6; ++index )
123  ranges[ index ] /= 2;
124  result->SetRanges( ranges, 6 );
125  }
126  }
127  return result;
128  }

157 inline color::RenderIntent RenderIntentFromName( const char* intent )
158 {
159  if( strcmp( intent, "AbsoluteColorimetric" ) == 0 )
160  return color::kAbsColorimetric;
161  if ( strcmp( intent, "Saturation" ) == 0 )
162  return color::kSaturation;
163  if( strcmp( intent, "Perceptual" ) == 0 )
164  return color::kPerceptual;
165  // Per the PDF spec, treat the rendering intent as RelativeColorimetric in
166  // the case where the rendering intent name is unrecognized:
167  return color::kRelColorimetric;
168 }
