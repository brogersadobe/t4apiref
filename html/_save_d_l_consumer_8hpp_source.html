77  std::vector<Reference> CommitToStore()
78  {
79  auto& store = m_documentCaches.GetStore();
80  dict_type pageDict = m_contentRecord.GetContentDictionary();
81 
82  std::vector<Reference> refVector;
83  Object<object_traits> contentsObj = pageDict.Get_NoResolve("Contents");
84 
85  ObjectType type = contentsObj.Type();
86  tetraphilia_assert(type == kObjectTypeNull || type == kObjectTypeReference || type == kObjectTypeArray);
87 
88  if(type == kObjectTypeReference)
89  {
90  Reference ref = contentsObj.ReferenceValue_NoTypeCheck();
91  refVector.emplace_back(ref.m_objNum, ref.m_generation);
92  }
93  else if(type == kObjectTypeArray)
94  {
95  typedef tetraphilia::pdf::store::ArrayIterator<object_traits, false> const_iterator_type;
96 
97  Array<object_traits> array = contentsObj.ArrayValue_NoTypeCheck();
98 
99  for(const_iterator_type iter = array.begin_noresolve(); iter != array.end_noresolve(); ++iter)
100  {
101  tetraphilia_assert(iter->Type() == kObjectTypeReference || iter->Type() == kObjectTypeNull);
102  if(iter->Type() == kObjectTypeReference)
103  {
104  Reference ref = iter->ReferenceValue_NoTypeCheck();
105  refVector.emplace_back(ref.m_objNum, ref.m_generation);
106  }
107  }
108  }
109 
110  /* Always create a new contents dictionary. Never modify the existing one because we don't know if it's been used by some other
111  * dictionary or not. For example, see Bug#3682032. */
112  ref_type newContentsDictRef = store.CreateNewStreamDictionary();
113  pageDict.DeleteKey( "Contents" );
114  pageDict.PutReference( "Contents", newContentsDictRef ); // Need to explicitly delete first to avoid triggering type mismatch exception.
115 
116  dict_type newContentsDict = store.ResolveReference( newContentsDictRef ).DictionaryValue();
117  size_t length = 0;
118  const char* buffer = m_bufStream.getBuffer(length);
119  newContentsDict.SetStream(buffer, length);
120  //Set Filter Name to Flate
121  newContentsDict.PutName("Filter", "FlateDecode");
122 
126  return refVector;
127  }

129  t4::string CommitToStoreAsFormXObject(dict_type& resourcesDict, const dict_type& srcFormXDict)
130  {
131  auto& store = m_documentCaches.GetStore();
132 
133  // create new indirect stream dictionary
134  ref_type formDictRef = store.CreateNewStreamDictionary();
135  dict_type formDict = store.ResolveReference(formDictRef).DictionaryValue();
136 
137  // iterate through source formXObject dictionary and copy all it's entries
138  // into new formXObject
139  typename dict_type::const_iterator_noresolve_type iter = srcFormXDict.begin_noresolve();
140  while (iter != srcFormXDict.end_noresolve())
141  {
142  const char *key = iter->GetKey();
143 
144  // "Length" key for a stream dictionary represents length of the contained
145  // stream. However, since we set newly created formXObject stream dict's stream
146  // via. dict.SetStream() API which sets the "Length" entry correctly in the
147  // store dictionary, we don't need to copy it from source dict.
148  if (strcmp(key, "Length") != 0)
149  {
150  const typename Store<app_traits>::obj_type& value = iter->GetValue();
151 
152  switch ( value.Type() )
153  {
154  case store::kObjectTypeInteger:
155  formDict.PutInteger(key, value.IntegerValue_NoTypeCheck());
156  break;
157 
158  case store::kObjectTypeReal:
159  formDict.PutReal(key, value.RealValue_NoTypeCheck());
160  break;
161 
162  case store::kObjectTypeBool:
163  formDict.PutBoolean(key, value.BoolValue_NoTypeCheck());
164  break;
165 
166  case store::kObjectTypeLargeInteger:
167  formDict.PutLargeInteger(key, value.LargeIntegerValue_NoTypeCheck());
168  break;
169 
170  case store::kObjectTypeString:
171  {
172  const auto& str = value.StringValue_NoTypeCheck();
173  size_t len = static_cast<size_t>( str.end() - str.begin() );
174  formDict.PutString(key, { str.begin(), len } );
175  }
176  break;
177 
178  case store::kObjectTypeName:
179  formDict.PutName(key, value.NameValue_NoTypeCheck().get());
180  break;
181 
182  case store::kObjectTypeArray:
183  formDict.CreateArrayFrom(key, value.ArrayValue_NoTypeCheck());
184  break;
185 
186  case store::kObjectTypeDictionary:
187  formDict.CreateDictionaryFrom(key, value.DictionaryValue_NoTypeCheck());
188  break;
189 
190  case store::kObjectTypeReference:
191  formDict.PutReference(key, value.ReferenceValue_NoTypeCheck());
192  break;
193 
194  case store::kObjectTypeNull:
195  // do nothing
196  break;
197  }
198  }
199 
200  ++iter;
201  }
202 
203  // set stream
204  size_t length = 0;
205  const char* buffer = m_bufStream.getBuffer(length);
206  formDict.SetStream(buffer, length);
207 
208  // write new formX dictionary to "XObject" dict with a unique key
209  dict_type xobjDict = resourcesDict.GetRequiredDictionary("XObject");
210  auto uniqueKey = store::GenerateUniqueKey(xobjDict, "Fm"); // "Fm" is a common prefix for FormXObject dict names
211  xobjDict.PutReference(uniqueKey.c_str(), formDictRef);
212 
213  // return new key
214  return uniqueKey;
215  }

519  void DoPath( content::DLEntryTreeWalker& walker,
520  bool insideRenderBounds, const bez_path_iter_type& begPath, const bez_path_iter_type& endPath,
521  bool stroke, pdf::content::FillRule fillRule, pdf::content::FillRule clipRule) override
522  {
523  // Path object construction operators
524  t4::real_coord lastMoveTo_x = 0, lastMoveTo_y = 0;
525 
526  auto m = [&]( t4::real_point const& p )
527  {
528  WriteRealType(&m_bufStream, get_x(p));
529  WriteSpace(&m_bufStream);
530  WriteRealType(&m_bufStream, get_y(p));
531  WriteSpace(&m_bufStream);

539  {
540  WriteRealType(&m_bufStream, get_x(p));
541  WriteSpace(&m_bufStream);
542  WriteRealType(&m_bufStream, get_y(p));
543  WriteSpace(&m_bufStream);
544  m_bufStream.write("l\n", 2);
545  };

548  {
549  WriteRealType(&m_bufStream, get_x(p0));
550  WriteSpace(&m_bufStream);

561  m_bufStream.write("c\n", 2);
562  };

564  auto h = [&]()
565  {
566  m_bufStream.write("h\n", 2);
567  };
568 
569  if ( !clean( ranges::make_subrange( begPath, endPath ), t4::make_segment_visitor( std::move( m ), std::move( l ), std::move( c ), std::move( h ) ) ) )
570  {
571  // Means path is incorrect, so don't emit anything
572  return;
573  }
574 
575  //Clipping path operators
576  switch (clipRule)
577  {
578  case content::kWinding:
579  m_bufStream.write("W\n", 2);
580  break;
581 
582  case content::kEvenOdd:
583  m_bufStream.write("W*\n", 3);
584  break;
585 
586  case content::kNoFill:
587  break;
588  }
589 
590  //Path painting operators
591  switch (fillRule)
592  {
593  case content::kWinding:
594  if (stroke)
595  m_bufStream.write("B\n", 2); // stroke and fill using nonzero winding number rule
596  else
597  m_bufStream.write("f\n", 2); // fill using nonzero winding number rule
598  break;
599 
600  case content::kEvenOdd:
601  if (stroke)
602  m_bufStream.write("B*\n", 3); // stroke and fill using even-odd rule
603  else
604  m_bufStream.write("f*\n", 3); // fill using even-odd rule
605  break;
606 
607  case content::kNoFill:
608  if (stroke)
609  m_bufStream.write("S\n", 2); // Stroke
610  else
611  m_bufStream.write("n\n", 2); // End path without filling or stroking
612  break;
613  }

616  void DoForm( const char* name, const matrix_type& mat, const real_rect_type& bBox) override
617  {
618  // A FormXObject will 'always' have a parent DL from which this particular instance has been invoked
619  assert(this->GetFormXObjectDLEntryRangesPtr() && this->GetDeletedEntryNumIter() && this->GetDeletedEntryNumEndIter());
620 
621  // Resources->XObject dictionary is required because there must be a FormX dict corresponding to refer to
622  dict_type formDict = content::GetRequiredResourceDictionary( m_contentRecord.GetResources().get_ptr(), "XObject", name );
623 
624  // Increment the FormXObjectDisplayListID counter to obtain the correct DLEntryNum range
625  this->AdvanceFormXObjectDisplayListID();
626  size_t formXDisplayListID = this->GetFormXObjectDisplayListID();
627 
628  // Find the DLEntryNum range for the temporary DL about to be created and check if it contains any deleted DL entries.
629  auto dlEntryRanges = *(this->GetFormXObjectDLEntryRangesPtr());
630  std::pair<size_t, size_t> dlEntryRange = dlEntryRanges[formXDisplayListID];
631 
632  auto deletedEntryNumIter = *(this->GetDeletedEntryNumIter());
633  auto deletedEntryNumEndIter = *(this->GetDeletedEntryNumEndIter());
634 
635  auto iter = std::lower_bound(deletedEntryNumIter, deletedEntryNumEndIter, dlEntryRange.first);
636  bool formxIsDirty = (iter != deletedEntryNumEndIter && *iter <= dlEntryRange.second);
637 
638  // If current reference to FormXObject was edited, a new edited FormXObject needs to be created
639  // (not containing the deleted DL entries) for save to PDF else a simple reference to the original
640  // FormXObject suffices
641  if ( !formxIsDirty || !content::FormIsVisible( formDict, m_ocStack, m_contentRecord.GetResources().get_ptr(), m_ocInfo ) )
642  {
643  WriteNameType(&m_bufStream, name);
644  m_bufStream.write(" Do\n", 4);
645  }
646  else
647  {
648  try
649  {
650  // Create a new consumer to generate content stream for the edited FormXObject
651  auto contentRec = t4::make_unique<content::FormXObjectContentRecord>( m_contentRecord.GetPageResources().get_ptr(), formDict );
652  store::SaveDLConsumer<app_traits> saveConsumer( *contentRec, m_documentCaches );
653 
654  saveConsumer.setDeletedEntryNumIters(this->GetDeletedEntryNumIter(), this->GetDeletedEntryNumEndIter());
655  saveConsumer.setFormXObjectDLEntryRangesPtr(this->GetFormXObjectDLEntryRangesPtr());
656  saveConsumer.SetFormXObjectDisplayListID(formXDisplayListID);
657 
658  auto dl = t4::make_unique<content::DisplayList>(std::move(contentRec), m_documentCaches);
659 
660  // Set new consumer's dlEntry counter to a start-value as calculated during earlier runs.
661  // Without this, DeletedDLEntryNums won't make sense
662  saveConsumer.SetDLEntryNum(dlEntryRange.first - 1);
663 
664  dl->EnumerateContent( saveConsumer );
665 
666  boost::optional<dict_type> pageResources = m_contentRecord.GetResources();
667  tetraphilia_assert(pageResources);
