30 /*
31  Utility class which can be used to make easily manipulated bool-like objects that
32  cannot be converted to unrelated types.
33 
34  We've all made this mistake at some point:
35 
36  void search( bool whole_words, bool ignore_case );
37 
38  ...
39 
40  search( flags & ignore_case_bit, flags & whole_words_bit );
41 
42  The traditional way of preventing this problem is to use enums:
43 
44  enum class whole_words_flag { yes, no };
45  enum class ignore_case_flag { yes, no };
46 
47  And redeclare search this way:
48 
49  void search( whole_words_flag whole_words, ignore_case_flag ignore_case );
50 
51  Now if you mix up the parameter ordering you'll get a compiler error:
52 
53  search( ignore_case_flag::yes, whole_words_flag::no ); // fails to compile, and that's good
54 
55  But there are some downsides to this approach. Being able to do boolean
56  operations on these types was nice, but now the compiler can no longer
57  negate them, test them, or combine them in boolean algebra. search's implementation
58  used to look like this:
59 
60  search( bool whole_words, bool ignore_case )
61  {
62  auto text = get_text();
63  if ( ignore_case )
64  text = case_normalize( text );
65  ...
66  }
67 
68  But now it has to be:
69 
70  search( whole_words ww, ignore_case ic )
71  {
72  auto text = get_text();
73  if ( ic == ignore_case_flag::yes )
74  text = case_normalize( text );
75  ...
76  }
77 
78  And the calling code is even worse:
79 
80  search( ( flags & whole_words_bit ) ? whole_words_flags::yes : whole_words_flags::no,
81  ( flags & ignore_case_bit ) ? ignore_case_flags::yes : ignore_case_flags::no );
82 
83  The goal here was never to make boolean algebra difficult with these types. The
84  goal was simply to make it impossible to make a bunch of enums that are proxies for
85  booleans but not implicitly convertible from bool (or other unrelated boolean proxies).
86 
87  typed_bool is a CRTP-based template which implements support for boolean logic on its
88  derived type without permitting implicit conversion from bool (conversion _to_ bool is
89  permitted and frequently useful). Clients typically won't even see this mechanism--just
90  use the T4_TYPED_BOOL macro to create a new "typed boolean":
91 
92  T4_TYPED_BOOL( whole_words );
93  T4_TYPED_BOOL( ignore_case );
94 
95  search( whole_words ww, ignore_case ic )
96  {
97  auto text = get_text();
98  if ( ic )
99  text = case_normalize( text );
100  ...
101  }
102 
103  And the client code:
104 
105  search( whole_words( flags & whole_words_bit ), ignore_case( flags & ignore_case_bit ) );
106 
107  And most importantly, none of the following lines of code will compile:
108 
109  search( true, false );
110  search( ignore_case::yes, whole_words::no );
111  search( ignore_case( flags & ignore_case_bit ), whole_words( flags & whole_words_bit ) );
112 
113 */
114 template<typename Derived>
115 class typed_bool : public t4::totally_ordered<typed_bool<Derived>>
116 {
117  bool m_state;
118 
119 protected:
120  constexpr typed_bool( void ) {}
121  constexpr explicit typed_bool( bool val ) : m_state( val ) {}
122 
123 public:
124  static Derived const no;
125  static Derived const yes;
126 
127  constexpr operator bool( void ) const { return m_state; }
128  constexpr bool operator==( typed_bool rhs ) const { return m_state == rhs.m_state; }
129  constexpr bool operator<( typed_bool rhs ) const { return m_state < rhs.m_state; }
130  friend std::ostream& operator<<( std::ostream& stm, typed_bool ths )
131  {
132  return stm << ( ths ? "yes" : "no" );
133  }
134 
135  /*
136  Boolean algebra on typed_bools works, but the returned type depends on
137  the inputs. If the input types are the same typed_bool, then a typed_bool
138  is returned. Otherwise the logic gracefully degrades to bool.
139 
140  This allows us to use expressions like:
141 
142  search( ignore_case( ica ) || ignore_case( icb ), whole_words::yes );
143 
144  but not:
145 
146  search( ignore_case( ica ) || whole_words( wwb ), whole_words::yes );
147 
148  Examples:
149 
150  auto a = ignore_case::yes;
151  auto b = ignore_case::no;
152  auto c = whole_words::yes;
153  bool d = true;
154 
155  auto aorb = x || y; // result is ignore_case::yes (type = ignore_case)
156  auto bandc = b && c; // result is false (type = bool)
157  auto candd = c && d; // result is true (type = bool)
158  auto notc = !c; // result is whole_words::no (type = whole_words)
159  */
160  constexpr Derived operator!( void ) const { return Derived( !m_state ); }
161  constexpr Derived operator&&( Derived rhs ) const { return Derived( m_state && rhs.m_state ); }
162  constexpr Derived operator||( Derived rhs ) const { return Derived( m_state || rhs.m_state ); }
163 };

199 /*
200  Define a "typed boolean" with the provided name. After this definition
201  (which takes place in whatever namespace is currently active) "name"
202  will identify a regular type that is implicitly convertible to, but
203  only explicitly from, bool.
204 
205  For example:
206 
207  T4_TYPED_BOOL( is_cold );
208 
209  bool x = is_cold::yes; // x <- true
210  is_cold ic = is_cold::no; // ic <- false
211  bool z = x || ic; // z <- true
212 
213  is_cold ic2 = false; // does not compile
214  is_cold ic3 = ic || ic2; // ic3 <- ic || ic2 (compiles because they're all the same type)
215  bool ic4 = ic || false; // ic4 <- false
216 */
217 #define T4_TYPED_BOOL( name ) class name : public ::t4::typed_bool<name> \
218  { \
219  public: \
220  constexpr name( void ) {} \
221  constexpr explicit name( bool val ) : typed_bool<name>( val ) {} \
222  }
