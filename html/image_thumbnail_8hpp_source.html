56 template<typename signal_traits>
57 void DoFull1DBoxFilter( typename signal_traits::signal_type* dst,
58  const typename signal_traits::signal_type* src,
59  size_t numBlocks,
60  size_t numSignalsPerBlock,
61  size_t numSamplesPerSignal,
62  int sampleCountPower,
63  size_t sampleStride,
64  size_t blockStride
65  )
66 {
67  typedef typename signal_traits::signal_type signal_type;
68 
69  for ( size_t blockCount = 0; blockCount < numBlocks; blockCount++ ) {
70  for ( size_t sigCount = 0; sigCount < numSignalsPerBlock; sigCount++ ) {
71  typename signal_traits::intermediate_type sum = 0;
72  const signal_type* sampleSrc = src;
73  for ( size_t sampleCount = 0; sampleCount < numSamplesPerSignal; sampleCount++ ) {
74  sum += *sampleSrc;
75  sampleSrc = t4::add_bytes( sampleSrc, sampleStride );
76  }
77  *dst++ = static_cast<signal_type>( DivByPowerOf2<typename signal_traits::intermediate_type>::Do( sum, sampleCountPower ) );
78  ++src;
79  }
80  src = t4::add_bytes( src, blockStride - numSignalsPerBlock*sizeof(signal_type) );
81  }
82 }

84 template<typename signal_traits>
85 void DoPartial1DBoxFilter( typename signal_traits::signal_type* dst,
86  const typename signal_traits::signal_type* src,
87  size_t numDstSignals,
88  size_t numSamplesPerSignal,
89  size_t numSamplesAvail,
90  int sampleCountPower,
91  size_t sampleStride
92  )
93 {
94  typedef typename signal_traits::signal_type signal_type;
95 
96  for ( size_t sigCount = 0; sigCount < numDstSignals; sigCount++ ) {
97  typename signal_traits::intermediate_type sum = 0;
98  const signal_type* sampleSrc = src;
99  for ( size_t sampleCount = 0; sampleCount < numSamplesPerSignal; sampleCount++ ) {
100  sum += *sampleSrc;
101  if ( sampleCount < numSamplesAvail - 1 )
102  sampleSrc = t4::add_bytes( sampleSrc, sampleStride );
103  }
104  *dst++ = static_cast<signal_type>( DivByPowerOf2<typename signal_traits::intermediate_type>::Do( sum, sampleCountPower ) );
105  ++src;
106  }
107 }

167 template<typename signal_traits>
168 HorizThumbnailImagePipe<signal_traits>::HorizThumbnailImagePipe (
169  size_t numSignals,
170  ImagePipe* pipe,
171  int scaleDownPower )
172  : m_pipe( pipe ),
173  m_bounds( pipe->GetBounds() ),
174  m_bitsPerPixel( 8 * sizeof( signal_type ) * numSignals ),
175  m_scaleDownPower( scaleDownPower ),
176  m_scaleDownFactor( 1 << scaleDownPower ),
177  m_numSignalsPerPixel( numSignals )
178 {
179  tetraphilia_assert( scaleDownPower > 0 && scaleDownPower < 32 );
180  tetraphilia_assert( get_xmin(m_bounds) % static_cast<int_coord>(m_scaleDownFactor) == 0 );
181 
182  int_coord fullBlockWidth = get_width( m_bounds ) & static_cast<int_coord>(~(m_scaleDownFactor - 1)); // width, in pixels, that we will process as "full" blocks
183  m_numRemainderPixels = static_cast<size_t>( get_width( m_bounds ) - fullBlockWidth );
184  m_dstFullPixelWidth = static_cast<size_t>( fullBlockWidth >> scaleDownPower );
185 
186  int_coord imageWidth = (get_width(m_bounds) + static_cast<int_coord>(m_scaleDownFactor) - 1) >> scaleDownPower; // width, in pixels, of our scaled down buffer
187  set_xmin( m_bounds, get_xmin(m_bounds) >> scaleDownPower );
188  set_xmax( m_bounds, get_xmin(m_bounds) + imageWidth );
189 
190  m_bytesPerScanLine = t4::multiply<size_t>( numSignals, sizeof(signal_type) );
191  m_bytesPerScanLine = t4::multiply<size_t>( m_bytesPerScanLine, imageWidth );
192 
193  auto& heap = *t4::get_thread_heap();
194  m_buff = t4::make_transient_array<signal_type>( heap, t4::multiply<size_t>( numSignals, imageWidth ) );
195 
196  m_indexOfLastDstPixel = static_cast<size_t>(imageWidth-1)*m_numSignalsPerPixel;
197  m_indexOfCorrespondingSourcePixel = m_indexOfLastDstPixel << m_scaleDownPower;
198 }

200 template<typename signal_traits>
201 const unsigned char* HorizThumbnailImagePipe<signal_traits>::GetNextScanLine()
202 {
203  const signal_type* src = static_cast<const signal_type*>(this->m_pipe->GetNextScanLine());
204 
205  DoFull1DBoxFilter<signal_traits>( m_buff, src,
206  m_dstFullPixelWidth,
207  m_numSignalsPerPixel,
208  m_scaleDownFactor,
209  m_scaleDownPower,
210  sizeof(signal_type)*m_numSignalsPerPixel,
211  sizeof(signal_type)*m_numSignalsPerPixel<<m_scaleDownPower);
212 
213  if ( m_numRemainderPixels )
214  DoPartial1DBoxFilter<signal_traits>(
215  &m_buff[m_indexOfLastDstPixel],
216  &src[m_indexOfCorrespondingSourcePixel],
217 
218  m_numSignalsPerPixel, // numDstSignals -- only doing the last (single) pixel
219  m_scaleDownFactor, // numSamplesPerSignal
220  m_numRemainderPixels, // numSamplesAvail
221  m_scaleDownPower,
222  sizeof(signal_type)*m_numSignalsPerPixel );
223 
224  return reinterpret_cast<const unsigned char*>( m_buff );
225 }

302 template<typename signal_traits>
303 const unsigned char* VertThumbnailImagePipe<signal_traits>::GetNextScanLine()
304 {
305  size_t numScanLines;
306  char* dst = reinterpret_cast<char*>( m_buff );
307  int_coord yMax = get_ymax( m_pipe->GetBounds() );
308  for ( numScanLines = 0; numScanLines < m_scaleDownFactor; numScanLines++ ) {
309  if ( m_curSrcY == yMax )
310  break;
311  ++m_curSrcY;
312  memcpy( dst, m_pipe->GetNextScanLine(), m_bytesPerScanLine );
313  dst += m_bytesPerScanLine;
314  }
315 
316  if ( numScanLines == m_scaleDownFactor ) {
317 
318  DoFull1DBoxFilter<signal_traits>( m_buff, m_buff,
319  1, // process the entire scan line as a single "block"
320  m_numSignalsPerScanLine, // get_width(srcBounds)*numSignalsPerPixel
321  m_scaleDownFactor,
322  m_scaleDownPower,
323  m_bytesPerScanLine,
324  0 // don't care about this since we are specifying only one "block"
325  );
326  } else {
327  DoPartial1DBoxFilter<signal_traits>( m_buff, m_buff,
328  m_numSignalsPerScanLine, // get_width(srcBounds)*numSignalsPerPixel
329  m_scaleDownFactor, // numSamplesPerSignal
330  numScanLines, // numSamplesAvail
331  m_scaleDownPower,
332  m_bytesPerScanLine );
333  }
334 
335  return reinterpret_cast<unsigned char*>( m_buff );
336 }
