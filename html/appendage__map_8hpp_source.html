37 /*
38  Multithreaded container that maps types to singleton objects.
39 
40  An appendage_map is a threadsafe associative map from types to lazily-created instances,
41  usually of that same type. Example usage:
42 
43  appendage_map map;
44 
45  // Demonstrates that mapped types are singletons (within the appendage_map)
46  struct foo {};
47  foo& f = map.get<foo>();
48  foo& f2 = map.get<foo>();
49  assert( &f == &f2 );
50 
51  // Any parameters are forwarded to the instance type's ctor
52  struct bar
53  {
54  int m_x;
55  bar( int x ) : m_x( x ) {}
56  };
57 
58  bar& b = map.get<bar>( 23 );
59  assert( b.m_x == 23 );
60  bar& b2 = map.get<bar>( 42 );
61  assert( b.m_x == 42 ); // ctor parameters are ignored if the object has already been created!
62 
63  // The key type doesn't have to be the same as the mapped type. Override the appendage_type
64  // template metafunction to use this feature.
65  namespace t4 {
66 
67  struct foo_key1;
68  struct foo_key2;
69  template<> appendage_type<foo_key1> { typedef foo type; };
70  template<> appendage_type<foo_key2> { typedef foo type; };
71 
72  } // end namespace t4
73 
74  // These calls all produce different objects of type foo.
75  foo& f = map.get<foo>();
76  foo& f1 = map.get<foo_key1>();
77  foo& f2 = map.get<foo_key2>();
78  assert( &f != &f1 && &f1 != &f2 && &f2 != &f );
79 */
