116  // Returns a floating-point value for BSpline function
117  // Filter Routines called from FilterScale
118  // input "x" is distance from CENTER to the pixel
119  // box (*) box (*) box (*) box
120  // This kernel is sometime called Parzen window.
121  // Note that this kernel is strictly positive.
122  static void BSplineFilter(real_coord xbr, real_coord * /*filterParam*/, real_coord* result)
123  {
124  using namespace t4::literals;
125  real_coord x = xbr;
126  real_coord xSqr;
127 
128  if(x < 0)
129  {
130  x = -x;
131  }
132  if(x < 1)
133  {
134  xSqr = SQR<real_coord>(x);
135  *result = xSqr * x / 2 - xSqr + 2_r / 3_r;
136  }
137  else if(x < 2)
138  {
139  x = 2 - x;
140  real_coord t1 = 1_r / 6_r;
141  *result = t1 * x * x * x;
142  }
143  else
144  *result = 0;
145  }

153  static void MitchellFilter(real_coord xbr, real_coord *filterParam, real_coord* result)
154  {
155  real_coord x = xbr;
156  real_coord xSqr = (x*x);
157  real_coord m_b1 = filterParam[0];
158  real_coord m_c1 = filterParam[1];
159 
160  if(x < 0)
161  {
162  x = -x;
163  }
164  if(x < 1)
165  {
166  real_coord t1 = 6 * m_c1;
167  x = (((12 - 9 * m_b1 - t1) * (x * xSqr))
168  + ((-18 + 12 * m_b1 + t1) * xSqr)
169  + (6 - 2 * m_b1));
170 
171  *result = static_cast<real_coord>(x / 6);
172  }
173  else if(x < 2)
174  {
175  real_coord t1 = 6 * m_c1;
176  x = (((-1 * m_b1 - t1) * (x * xSqr))
177  + ((6 * m_b1 + 30 * m_c1) * xSqr)
178  + ((-12 * m_b1 - 48 * m_c1) * x)
179  + (8 * m_b1 + 24 * m_c1));
180 
181  *result = static_cast<real_coord>(x / 6);
182  }

185  *result = 0;
