416 inline RedBlackNodeBase* RedBlackTreeBase::InsertNodeReference( RedBlackNodeBase& n, const void* key )
417 {
418  search_state state;
419 
420  if ( SetSearchState( state, key ) ) {
421  RedBlackNodeBase *removed = *state.atn;
422  *state.atn = &n;
423  if ( removed->is_red() )
424  n.make_red();
425  else
426  n.make_black();
427  n.m_parent = removed->m_parent;
428  n.m_left = removed->m_left;
429  if ( n.m_left )
430  n.m_left->m_parent = &n;
431  n.m_right = removed->m_right;
432  if ( n.m_right )
433  n.m_right->m_parent = &n;
434  if ( m_del )
435  m_del( removed );
436  } else
437  InsertNodeReference( n, state );
438 
439  return &n;
440 }

442 inline bool RedBlackTreeBase::SetSearchState( search_state& state, const void* key )
443 {
444  RedBlackNodeBase **next;
445 
446  state.atn = &m_root;
447  state.atp = state.atg = nullptr;
448 
449  while ( *state.atn ) {
450 
451  if ( (*state.atn)->is_three_node() )
452  SplitNode(state);
453 
454  int c = m_comp(key, *state.atn);
455  if ( c < 0 )
456  next = &(*state.atn)->m_left;
457  else if ( c > 0 )
458  next = &(*state.atn)->m_right;
459  else
460  break;
461 
462  state.atg = state.atp;
463  state.atp = state.atn;
464  state.atn = next;
465  }
466 
467  return *state.atn != nullptr;
468 }

481 inline RedBlackNodeBase** RedBlackTreeBase::RotateNode(RedBlackNodeBase** atNode, RedBlackNodeBase** atParent)
482 {
483  RedBlackNodeBase* parent = *atParent;
484  RedBlackNodeBase* node = *atNode;
485  RedBlackNodeBase** atChild = (node == parent->m_left) ? &node->m_right : &node->m_left;
486  *atParent = node;
487  node->m_parent = node->m_parent->m_parent;
488  *atNode = *atChild;
489  if ( *atChild )
490  (*atChild)->m_parent = parent;
491  *atChild = parent;
492  parent->m_parent = node;
493  return atParent;

496 inline void RedBlackTreeBase::SplitNode( search_state& state )
497 {
498  RedBlackNodeBase *n = *state.atn;
499  n->make_red();
500  if ( n->m_left )
501  n->m_left->make_black();
502  if ( n->m_right )
503  n->m_right->make_black();
504 
505  if ( state.atg && (*state.atp)->is_red() ) {
506  RedBlackNodeBase *g = *state.atg;
507  g->make_red();
508  if ((g->m_left == *state.atp) != ((*state.atp)->m_left == n))
509  state.atp = RotateNode(state.atn, state.atp);
510  state.atn = RotateNode(state.atp, state.atg);
511  (*state.atn)->make_black();
512  }
513 }

525 inline RedBlackNodeBase* RedBlackTreeBase::UnlinkNode(RedBlackNodeBase** node)
526 {
527  RedBlackNodeBase* unlinkNode = *node;
528  if ( unlinkNode->m_left ) {
529  if ( unlinkNode->m_right ) {
530  *node = UnlinkNode( FindMinNode( &unlinkNode->m_right ) );
531  (*node)->m_parent = unlinkNode->m_parent;
532  (*node)->m_left = unlinkNode->m_left;
533  unlinkNode->m_left->m_parent = *node;
534  (*node)->m_right = unlinkNode->m_right;
535  if ( unlinkNode->m_right )
536  unlinkNode->m_right->m_parent = *node;
537  } else {
538  unlinkNode->m_left->m_parent = unlinkNode->m_parent;
539  *node = unlinkNode->m_left;
540  }
541  } else {
542  if ( unlinkNode->m_right )
543  unlinkNode->m_right->m_parent = unlinkNode->m_parent;
544  *node = unlinkNode->m_right;
545  }
546  if ( *node )
547  (*node)->copy_color( *unlinkNode );
548  return unlinkNode;
549 }
