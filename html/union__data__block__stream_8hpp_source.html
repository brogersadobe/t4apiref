49 template<typename object_traits>
50 class UnionDataBlockStream
51 {
52  typedef typename object_traits::app_traits app_traits;
53 
54 public:
55  UnionDataBlockStream( bool intersperseWhiteSpaceBlocks = false )
56  : m_intersperseWhiteSpaceBlocks(intersperseWhiteSpaceBlocks),
57  m_needWhiteSpaceBlock(true)
58  {
59  }
60 
61  UnionDataBlockStream( UnionDataBlockStream&& rhs ) = default;
62 
63  void Append( Dictionary<object_traits>& dict, bool isInlineImage = false)
64  {
65  assert( !m_cur );
66  m_elems.emplace_back( dict, isInlineImage );
67  }
68 
69  friend t4::data_block get_next_block( UnionDataBlockStream& ths )
70  {
71  if ( ths.m_elems.size() == 0 || ( ths.m_cur && *ths.m_cur == ths.m_elems.end() && ths.m_stream.at_eof() ) )
72  return t4::data_block();
73  if ( !ths.m_cur )
74  {
75  ths.m_cur = ths.m_elems.begin();
76  tetraphilia_assert(!ths.m_stream); // no open stream
77  ths.m_stream = GetFilteredStream( (*ths.m_cur)->m_dict, (*ths.m_cur)->m_isInlineImage );
78  }
79 
80  t4::data_block result = ths.m_stream.get_next_block();
81 
82  while ( result.get_size() == 0 ) // this is 'while' because there can be empty streams in the middle of an array, and we want to skip those
83  {
84  if ( ths.m_intersperseWhiteSpaceBlocks && ths.m_needWhiteSpaceBlock )
85  {
86  ths.m_needWhiteSpaceBlock = false;
87  static char const whitespace = ' ';
88  return t4::data_block( &whitespace, 1, t4::make_shared<t4::data_block_owner>() );
89  }
90  ths.m_needWhiteSpaceBlock = true;
91  ++(*ths.m_cur);
92  if ( *ths.m_cur == ths.m_elems.end() )
93  break; // Propagate the empty block; we're really at EOF
94  ths.m_stream = {}; // reset first so that only one stream is open
95  ths.m_stream = GetFilteredStream( (*ths.m_cur)->m_dict, (*ths.m_cur)->m_isInlineImage );
96  result = ths.m_stream.get_next_block();
97  }
98 
99  return result;
100  }
101 
102  friend void rewind( UnionDataBlockStream& ths )
103  {
104  ths.m_cur.reset();
105  ths.m_stream = {};
106  }
107 
108  friend bool at_eof( UnionDataBlockStream& ths )
109  {
110  return ths.m_cur && *ths.m_cur == ths.m_elems.end() && ths.m_stream.at_eof();
111  }
112 
113 private:
114  size_t GetCachedSizeImpl() const
115  {
116  return sizeof( *this ) + t4::member_size( m_elems );
117  }
