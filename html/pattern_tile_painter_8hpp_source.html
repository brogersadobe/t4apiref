125 template<typename _signal_traits>
126 PDFPatternTilePainter<_signal_traits>::PDFPatternTilePainter(
127  display_list_type& displayList,
128  const matrix_type& defaultParentContentStreamSpaceToCurrentDeviceSpaceMatrix, // mapping from default parent content stream space to ACTUAL CURRENT DEVICE (not necessarily final page device for we could be drawing into an offscreen)
129  const store_dict_type& patternDict,
130  document::DocumentCaches<app_traits>& docCaches,
131  display_event_stream eventStream,
132  ColorServer<signal_traits>* colorServer )
133  : m_context( GetGlobalContext() )
134  , m_patternDict( patternDict )
135  , m_displayList( displayList )
136  , m_documentCaches( docCaches )
137  , m_encounteredNonNormalBlendMode( false )
138  , m_eventStream( MakePatternEventStream( std::move( eventStream ) ) )
139  , m_colorServer( colorServer )
140 {
141  using namespace imaging_model;
142 
143  // get pattern matrix
144  optional_array_type matrixArray = patternDict.GetArray( "Matrix" );
145  matrix_type patternMatrix = matrixArray ? store::GetMatrix<matrix_type>( *matrixArray ) : IdentityMatrix<matrix_type>();
146 
147  int_type paintType = patternDict.GetRequiredInteger( "PaintType" );
148  int_type tilingType = patternDict.GetRequiredInteger( "TilingType" ); // we check that TilingType is valid but don't currently use it in any way
149  if ( paintType < 1 || paintType > 2 || tilingType < 1 || tilingType > 3 )
150  throw t4::corrupt_data();
151 
152  get_x( m_tilingPatternInfo.tileStepping ) = patternDict.GetRequiredReal( "XStep" );
153  get_y( m_tilingPatternInfo.tileStepping ) = patternDict.GetRequiredReal( "YStep" );
154  if ( get_x( m_tilingPatternInfo.tileStepping ) == 0 || get_y( m_tilingPatternInfo.tileStepping ) == 0 )
155  throw t4::corrupt_data();
156 
157  /*
158  The issue here is that sometimes the pattern matrix has a very large e and/or f
159  component. In a fixed point build that can cause overflow problems when rendering
160  at higher zoom levels. Since the cells of a pattern are tiled using the xstep
161  and ystep values, any matrix that has the same "phase" as the given one can be
162  used. The goal of this code is to find the matrix that has the smallest e and
163  f values such that it has the same phase as the specified matrix. To find this
164  we compute 'm' and 'n' values such that:
165 
166  patternMatrix' = [1 0 0 1 m*xStep n*yStep] * patternMatrix
167 
168  produces a patternMatrix' that has e == 0 and f == 0. That is, we find the number
169  of tiling steps in both x and y that are needed to get to the origin. This produces
170  real (non-integer) values for m and n. We then round these values to the closest
171  integer. This gives an offset that preserves the phase of the original matrix.
172  */
173  real_type xs = get_x( m_tilingPatternInfo.tileStepping );
174  real_type ys = get_y( m_tilingPatternInfo.tileStepping );
175  real_type a = get_a(patternMatrix);
176  real_type b = get_b(patternMatrix);
177  real_type c = get_c(patternMatrix);
178  real_type d = get_d(patternMatrix);
179  real_type e = get_e(patternMatrix);
180  real_type f = get_f(patternMatrix);
181  real_type det = a*d - c*b;
182 
183  // There exist PDF files (PDFRM 1.7, page 1149) that contain pattern objects with extremely
184  // large step values--32767 in the aforementioned example. This is done to prevent any possibility
185  // of actual tiling taking place; i.e., the client really just wants to render some artwork
186  // within some context where only a pattern can work. This would cause the denominator in the
187  // computation of m and n to overflow, leading to completely wrong values for e and f in the
188  // pattern matrix. This code detects the overflow, and when it occurs sets the resulting value
189  // to 0, which it should be if the denominator is arbitrarily large.
190  size_t overflow = false;
191  real_type m = real_services::Multiply( det, xs, &overflow );
192  m = overflow ? 0 : (c*f - d*e) / m;
193 
194  overflow = false;
195  real_type n = real_services::Multiply( det, ys, &overflow );
196  n = overflow ? 0 : (b*e - a*f) / n;
197 
198  // modulo precision problems, these two expressions should be exactly 0.
199  // The expressions to compute 'm' and 'n' above come from solving these two
200  // equations when set to 0 (not 1).
201 #if 0
202  // this turns out to be a somewhat impractical thing to assert
203  // because precision issues can cause this result to be arbitrarily
204  // far from 0, particularly in FixedPoint builds. The expressions
205  // that compute m and n really do imply that the below expressions
206  // would be exactly 0. So it also isn't all that interesting to
207  // continue to assert that it is close to zero other than during
208  // development of the code. Perhaps another solution would be to
209  // only make these assertions in floating point builds where it is
210  // considerably harder to get the precision problems to effect the
211  // result so much.
212  tetraphilia_assert( AbsVal(e + a*m*xs + c*n*ys) < 1 );
213  tetraphilia_assert( AbsVal(f + b*m*xs + d*n*ys) < 1 );
214 #endif
215 
216  int im = real_services::RoundToInt( ArgType<int>(), m );
217  int in = real_services::RoundToInt( ArgType<int>(), n );
218  get_e( patternMatrix ) = e + a * xs * im + c * ys * in;
219  get_f( patternMatrix ) = f + b * xs * im + d * ys * in;
220 
221  m_tilingPatternInfo.patternSpaceToCurrentDeviceSpaceMatrix = patternMatrix * defaultParentContentStreamSpaceToCurrentDeviceSpaceMatrix;
222  m_tilingPatternInfo.tileCellBBox = store::GetRectangle<real_rect_type>( patternDict.GetRequiredArray( "BBox" ) );
223 
224  m_tilingPatternInfo.isUncolored = (paintType == 2);
225 }

227 template<typename _signal_traits>
228 bool PDFPatternTilePainter<_signal_traits>::PaintPatternTile( transparency_group_type& transparency_group, const matrix_type& transform ) const
229 {
230  using namespace imaging_model;
231  // Technically for COLORED patterns, according to PDF 1.5 section 4.6.2
232  // the color is supposed to be inherited from the INITIAL color of the
233  // parent (current) content stream (which we currently don't save).
234  // In practice we expect colored patterns to always supply their own
235  // color so this should not be a major issue. And for uncolored patterns
236  // we only capture shape so the current color doesn't matter.
237 
238  TransientHeap<app_traits>& heap = get_TransientHeap(m_context);
239  // create a snapshot for each tile we paint
240  TransientSnapShot<app_traits> ss(heap);
241 
242  typedef HardClip<signal_traits> hard_clip_type;
243  int_rect_type hardClipBounds = CastToIntCoordRect<int_rect_type>( TransformAndBoundRealRect( m_tilingPatternInfo.tileCellBBox, transform ) );
244  hard_clip_type hardclip( hardClipBounds );
245 
246  RenderContext<signal_traits> rc( transparency_group,
247  hardclip,
248  transform,
249  m_documentCaches,
250  0 );
251 
252  RenderConsumer<signal_traits> cons( rc, document::PageID() /*pageID*/, &m_displayList.GetContentRecord(), m_eventStream, nullptr /*reflowInfoPtr*/, nullptr /*ocState*/, nullptr/*croppedMediaBox*/, m_colorServer );
253  m_displayList.EnumerateContent(cons);
254  return !m_encounteredNonNormalBlendMode;
255 }
