64  type the same size as the (signed) source, but in the general case the source may or may not be the same size.
65  */

79  constexpr boost::optional<Result> operator()( T1 const& x ) const
80  {
81  // Compute the number of signficant bits needed to represent the negation (this is the sbc of the positive value minus one, with zero
82  // as a special case)
83  size_t x_sbc = ( x == 0 ) ? 0 : sm_detail::significant_bit_count( x - 1 );
84  if ( x_sbc <= significant_bit_count_in_type<Result> )
85  {
86  // Create an _unsigned_ int as large as the larger of the two types
87  using unsigned_promotion_type = std::make_unsigned_t<std::conditional_t<( sizeof( T1 ) > sizeof( Result ) ), T1, Result>>;
88  using signed_promotion_type = std::make_signed_t<unsigned_promotion_type>;

90  // Negating an unsigned type is okay, and produces a correct result mod 2
