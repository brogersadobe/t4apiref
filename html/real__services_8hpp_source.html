263 template <typename _real_type>
264 bool QuickQuadraticHasRootsInZeroToOne(_real_type a, _real_type b, _real_type c)
265 {
266  if (!RealTypeHelper<_real_type>::isFloatingPointType)
267  {
268  _real_type mx = AbsVal(a);
269  _real_type tmp = AbsVal(b);
270  if ( tmp > mx )
271  mx = tmp;
272  tmp = AbsVal(c);
273  if ( tmp > mx )
274  mx = tmp;
275  while ( mx > 64 ) {
276  a /= 2;
277  b /= 2;
278  c /= 2;
279  mx /= 2;
280  }
281  }
282 
283  // NOTE: the a == 0 case is rare, I don't check for it because I don't have to. The
284  // math below yields correct results even for a = 0.
285 
286  // See whether the quadratic equation would yield any real roots at all, if not
287  // then bail. For "flattish" offset curves (the common case) this will quickly yield
288  // "no roots"
289  _real_type discriminant = b*b - a*c*4;
290  if (discriminant < 0)
291  return false;
292 
293  // Note that the quadratic evaluated at x=0 and x=1 is simple, so we can see if the
294  // curve is negative on one side and positive on the other with a mimimum amount of
295  // math.
296  if (c*(a+b+c) <= 0)
297  return true;
298 
299  // normalize a bit, makes the ineqalities below easier
300  if (a < 0)
301  {
302  a = -a;
303  b = -b;
304  c = -c;
305  }
306 
307  // OK, we now know that there is not just one root between zero and one
308  // (above check would have been true) so we know there are either zero or two roots.
309  // This is really important: WE KNOW: ZERO OR TWO ROOTS
310 
311  // Now we know that the equation ax*x + bx + c == 0 can be rewritten as x*x + bx/a + c/a == 0
312  // and also as (x - r1)(x - r2) == 0 where r1 and r2 are the roots.
313  // Expand and set the coefficients equal yields
314  //
315  // r1*r2 = c/a
316  // r1+r2 = -b/a
317  //
318  // Therefore in order for r1 and r2 to both be in [0,1] we must have:
319  //
320  // 0 <= c/a <= 1 implies 0 <= c <= a (since a >= 0)
321  // 0 <= -b/a <= 2 implies -2a <= b <= 0 (since a >= 0)
322  //
323  // OK, so we know there are either zero or two roots in [0.1], and in order for
324  // there to be two the above ineqalities must hold, so we can throw out a lot of
325  // cases by simple compares.
326 
327  if (c < 0 || c > a || b > 0 || b < -a*2 )
328  return false;
329 
330  // So now we need to work. We want to see whether 0 <= (-b +- sqrt(discriminant))/(2a) <= 1
331  // but this is an expensive check. Since a >= 0 we can rearrange to yield:
332  //
333  // b <= +-sqrt(discriminant) <= 2a + b
334  //
335  // Now again, we know that there are either zero or two roots, so either BOTH the
336  // +sqrt(discriminant) and the -sqrt(discriminant) must be between the two ends.
337  // Note that we always know -sqrt(x) <= sqrt(x) as sqrt always yields the positive value.
338  //
339  // b <= -sqrt(discriminant) <= +sqrt(discriminant) <= 2a + b
340  //
341  // Since -sqrt(x) <= sqrt(x) we can rewrite as:
342  //
343  // b <= -sqrt(discriminant) and
344  // sqrt(discriminant) <= 2a+b
345  //
346  // so
347  //
348  // sqrt(discriminant) <= -b and
349  // sqrt(discriminant) <= 2a+b
350  //
351  // so
352  //
353  // sqrt(discriminant) <= min(-b, 2a+b)
354  //
355  // Note that both -b and 2a+b are guranteed positive by sanity checks above, so
356  // squaring preserves the inequality, so:
357  //
358  // discriminant <= min(-b, 2a+b)^2
359  //
360  // The whole point then is that there are two roots in [0,1] iff the above inequality
361  // holds. Since we know there are either two or zero, this inequality tells us also
362  // if there are ANY roots. QED
363 
364  // I originally had code like this here, but then John Nash pointed out that it is unnecessary.
365  // The inequalities above not returning out imply the result of the code below to be true.
366  // Why? I leave that to you, when I read his proof, it was clearly (and surprisingly) true.
367  // are false.force this:
368  // BRVRealCoord k = 2*a+b;
369  // if (-b < k)
370  // k = -b;
371  // return discriminant <= k*k;
372  return true;
373 }
