30 /*
31  Given an implementation object, create an event_stream which can consume the events
32  enumerated in the Events template parameter pack.
33 
34  For each type E in the Events parameter pack:
35 
36  If e is a const instance of E, and impl a const instance of Impl, the following expression
37  must by supported:
38 
39  impl << e;
40 
41  The return type can be whatever you like, but it might as well be void.
42 
43  The returned object is a runtime-polymorphic entity which forwards invocations of operator<<
44  to a contained copy of the Impl for each of the types in the parameter pack.
45 
46  Example:
47 
48  struct my_stream
49  {
50  void operator<<( int x ) const
51  {
52  std::cout << "encountered integer " << x << "." << std::endl;
53  }
54 
55  void operator<<( char const*( ) const
56  {
57  std::cout << "encountered string " << s << "." << std::endl;
58  }
59  };
60 
61  auto es = make<event_stream<int, std::string>>( my_stream() );
62 
63  es << 3;
64  es << "hello";
65 
66  Outputs:
67 
68  encountered integer 3.
69  encountered string hello.
70 */
71 template<typename EventStream, typename Impl>
72 std::enable_if_t<is_instantiation_of<EventStream, event_stream>::value,
73 EventStream> make( Impl impl );

75 /*
76  Template class for objects returned by make<event_stream<>>.
77 
78  Instances of event_stream are regular and have efficient move semantics.
79  Default-constructed event_streams are "empty" and convert to false in boolean
80  contexts.
81 
82  Any object of a type specified in the Events template parameter pack can be
83  reported to an event_stream, e.g.:
84 
85  event_stream<int> es = make<event_stream<int>>( some_impl() );
86  es << 5;
87 
88  Expected Usage:
89 
90  Users of event_stream will typically be providing a service API of some sort
91  in which they wish to notify clients of particular "events" that might occur
92  during an algorithm's execution. Suppose we were implementing an algorithm "foo" that
93  traversed a PDF object hierarhcy and in which the following sorts of "events" might
94  be of particular interest to clients:
95 
96  1) An object with a /Subtype but no /Type was encountered.
97  2) An array with no elements is encountered.
98 
99  We might represent these "events" descriptively with the following two classes
100  (written as structs here for simplicity):
101 
102  struct subtype_without_type
103  {
104  t4::string m_subtype;
105  };
106 
107  struct array_with_no_elements {};
108 
109  We use these types, and any more we might come up with later that might be of
110  interest to a client, to create a particular instantiation of event_stream:
111 
112  using foo_event_stream = t4::event_stream<subtype_without_type, array_with_no_elements>;
113 
114  and a helper creation function:
115 
116  template<typename Impl>
117  foo_event_stream make_foo_event_stream( Impl impl );
118 
119  Then you'd provide a top-level API to the "foo" algorithm that looked like this:
120 
121  foo_result foo( ... parameters ... , foo_event_stream const& stm );
122 
123  Your implementation of foo will include occasional calls to stm's operator<<, like this:
124 
125  stm << subtype_without_type( subtype );
126 
127  or:
128 
129  stm << array_with_no_elements();
130 
131  Clients, then, would implement simple classes that consume some subset of these events:
132 
133  struct my_foo_event_stream_impl
134  {
135  friend void operator<<( my_foo_event_stream_impl const& impl, subtype_without_type const& event )
136  {
137  std::cout << "missing type for /Subtype = " << event.m_subtype;
138  }
139  };
140 
141  (Note that clients don't need to process _all_ event types; they just process the ones they care about).
142 
143  And create an event_stream like this:
144 
145  auto my_foo_event_stream = make_foo_event_stream( my_foo_event_stream_impl() );
146 
147  This object can be passed into foo():
148 
149  auto result = foo( ... parameters ... , my_foo_event_stream );
150 */
151 template<typename... Events>
152 class event_stream
153 {
154  std::shared_ptr<event_detail::event_stream_impl_base<Events...>> m_impl;
155 
156 public:
157  event_stream( void ) = default;
158  event_stream( event_stream const& ) = default;
159  event_stream( event_stream&& ) = default;
160  event_stream& operator=( event_stream const& ) = default;
161  event_stream& operator=( event_stream&& ) = default;
162 
163  explicit operator bool( void ) const { return bool( m_impl ); }
164 
165  template<typename Event>
166  event_stream const& operator<<( Event const& e ) const
167  {
168  /*
169  If you trigger this static assertion trying to stream an event of type E into
170  an event_stream of types E1, E2, E3, then one of the following is happening:
171 
172  1) E is not convertible to _any_ of E1, E2, E3
173 
174  for example:
175 
176  auto es = make<event_stream<int, std::string>>( my_impl );
177  es << std::make_pair( true, 23 );
178 
179  2) E is convertible to more than one of E1, E2, E3, but is not an exact match
180  for any.
181 
182  for example:
183 
184  auto es = make<event_stream<int, t4::string, std::string>>( my_impl );
185  es << t4::string( "hello" ) // unambiguous, no error
186  es << "hello"; // error, can't decide between t4::string and std::string
187  */
188  static_assert( event_detail::no_more_than_one_match_exists<Event, Events...>, "" );
189  if ( m_impl )
190  *m_impl << e;
191  return *this;
192  }
193 
194 private:
195  template<typename EventStream, typename Impl>
196  friend struct event_detail::make_event_stream_impl;
197 
198  event_stream( std::shared_ptr<event_detail::event_stream_impl_base<Events...>> impl )
199  : m_impl( std::move( impl ) )
200  {}
201 };

203 /*
204  A log of events that can be played into another listener. Duplicates are discarded.
205  During playback, events will be reordered.
206 
207  If you use event_set, there are some stronger requirements on the types of Events: each
208  type must be totally ordered (providing operator== and operator<).
209 
210  event_set is _not_ an event_stream! To populate one, create it and pass it to make_event_stream_recorder().
211 
212  Example:
213 
214  event_set<int, t4::string> events;
215  auto estm = make_event_stream_recorder( events );
216  events << 3 << 2 << "world" << 1 << "hello" << 2;
217 
218  struct reporter
219  {
220  template<typename Event> operator<<( Event const& e ) const { std::cout << e; }
221  };
222 
223  estm.replay( reporter() ); // outputs "123helloworld"
224 */
