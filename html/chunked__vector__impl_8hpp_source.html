85 template<typename T, bool StableEndIterator>
86 void chunked_vector_base<T, StableEndIterator>::validate( void )
87 {
88  chunk_iterator_type it = this->m_chunks.begin();
89  size_t count = 0;
90 
91  int zone = 0;
92 
93  while ( it != this->m_chunks.end() )
94  {
95  if ( it->empty() )
96  zone = 2;
97  else if ( it->full() )
98  {
99  assert( zone == 0 );
100  }
101  else
102  {
103  assert( zone == 0 );
104  zone = 1;
105  }
106 
107  count += it->size();
108  ++it;
109  }
110 
111  if ( StableEndIterator )
112  {
113  assert( !this->m_chunks.empty() );
114  assert( !this->m_end_chunk->full() );
115  }
116  else
117  assert( this->m_size == 0 || !this->m_end_chunk->empty() );
118 
119  assert( count == this->m_size );
120 }

276 template<typename T, typename Allocator, typename Traits>
277 void chunked_vector<T, Allocator, Traits>::shrink_to_fit( void )
278 {
279  this->validate();
280  if ( !allocator_free_is_trivial<Allocator>::value )
281  {
282  if ( has_stable_end_iterator )
283  {
284  assert( !this->m_chunks.empty() );
285  auto it = this->m_chunks.iterator_to( this->m_chunks.back() );
286  while ( this->m_chunks.size() > 1 && it->empty() )
287  {
288  auto prev = std::prev( it );
289  if ( prev->full() )
290  break;
291  set_capacity( capacity() - it->capacity() );
292  pop_back_chunk();
293  it = prev;
294  }
295  }
296  else
297  while ( !this->m_chunks.empty() && this->m_chunks.back().empty() )
298  {
299  set_capacity( capacity() - this->m_chunks.back().capacity() );
300  pop_back_chunk();
301  }
302  }
303  this->validate();
304 }

351 template<typename T, typename Allocator, typename Traits>
352 auto chunked_vector<T, Allocator, Traits>::make_space_for_push( void ) -> chunk_iterator_type
353 {
354  chunk_iterator_type nrv;
355 
356  if ( has_stable_end_iterator )
357  {
358  // If this fires you are attempting to mutate a chunked_vector in the moved-from state.
359  assert( !this->m_chunks.empty() );
360  nrv = this->m_end_chunk;
361  assert( !nrv->full() );
362  if ( nrv->first_unused() + 1 == nrv->storage_end() )
363  {
364  // Since this is a stable chunked_vector, make sure m_end_chunk points to the next chunk with any available space
365  auto next = std::next( nrv );
366  if ( next == this->m_chunks.end() )
367  {
368  // We do _not_ update nrv here--it has space for the element, in the last slot. We're
369  // just making sure that the next slot already exists so we have a stable end iterator.
370  next = add_chunk();
371  assert( next->empty() );
372  }
373 
374  this->m_end_chunk = next;
375  }
376 
377  return nrv;
378  }
379  else
380  {
381  if ( this->m_size == 0 )
382  nrv = this->m_chunks.size() == 0 ? add_chunk() : this->m_chunks.begin();
383  else
384  {
385  nrv = this->m_end_chunk;
386  assert( nrv != this->m_chunks.end() );
387  if ( nrv->full() )
388  {
389  ++nrv;
390  if ( nrv == this->m_chunks.end() )
391  nrv = add_chunk();
392  assert( nrv->empty() );
393  }
394  }
395  }
396 
397  return nrv;
398 }

416 void chunked_vector<T, Allocator, Traits>::pop_back( void )
417 {
418  this->validate();

424  --this->m_end_chunk;

433  --this->m_end_chunk;
434  }
435  this->validate();
436 }

447 template<typename T, typename Allocator, typename Traits>
448 auto chunked_vector<T, Allocator, Traits>::insert( const_iterator position, size_type n, value_type const& val ) -> iterator
449 {
450  this->validate();
451  ptrdiff_t distance_to_end = this->end() - position;
452  ptrdiff_t count = static_cast<ptrdiff_t>( n );
453 
454  while ( n-- )
455  push_back( val );
456 
457  iterator dstEnd = this->end() - distance_to_end;
458  iterator dst = dstEnd - count;
459  iterator src = this->end() - count;
460  while ( dst != dstEnd )
461  boost::swap( *src++, *dst++ );
462 
463  iterator result = this->end() - ( count + distance_to_end );
464  if ( 0 < distance_to_end && 0 < count )
465  {
466  ptrdiff_t hop_count = count - ( count - 1 ) / distance_to_end * distance_to_end;
467  std::rotate( this->end() - distance_to_end, this->end() - hop_count, this->end() );
468  }
469  this->validate();
470  return result;
471 }

504 auto chunked_vector<T, Allocator, Traits>::insert( const_iterator position, value_type&& val ) -> iterator
505 {
506  this->validate();
507  bool was_end = ( position == this->cend() );
508 
509  append( 1, std::move( val ) );
510 
511  // because of chunked_vector's implementation, position will stay valid unless it's the
512  // end, but if it was the end we don't need to do anything

514  {

517  this->validate();
518  return this->const_iterator_cast( position );
519  }
520  else
521  {
522  this->validate();
523  return std::prev( this->end() );
524  }

609 auto chunked_vector<T, Allocator, Traits>::emplace( const_iterator position, Targs&&... args ) -> iterator
610 {

612 
613  // because of chunked_vector's implementation, position will stay valid unless it's the
614  // end _and_ the end iterator is not stable, but in that one special case the rotate call won't
615  // do anything.

624  this->validate();
625  return this->const_iterator_cast( position );
626  }
627  else
628  {
629  this->validate();

701 template<typename T, typename Allocator, typename Traits>
702 auto chunked_vector<T, Allocator, Traits>::add_chunk( void ) -> chunk_iterator_type
703 {
704  using chunked_vector_detail::chunk;
705 
706  using chunk_allocator = typename Allocator::template rebind<chunk<T, has_stable_end_iterator>>::other;
707  chunk_allocator ca( get_allocator() );
708  auto new_chunk_size = this->m_chunks.empty() ? first_chunk_size<Traits>::value : next_chunk_size<Traits>::value;
709 
710  /*
711  What we really want here is a block of memory in which to store a chunk<T> followed by
712  an array of T, respecting alignment requirements of both types. This means that there
713  may be a gap between the chunk<T> and the first T (and that's accounted for by the call to
714  align_up in chunk::get_start()).
715 
716  There doesn't exist a way to request a block of memory with these alignment requirements
717  from a standard C++ allocator, so instead we compute the number of chunk<T>'s required to
718  fill the requested space (accounting for the possible need for a gap between the chunk<T>
719  and the first T). This can lead to a small number of wasted bytes, but there's really no
720  way around this problem without using an allocator that is somehow able to figure out how
721  to allocate an object followed by an array of different objects with potentially different
722  alignments.
723  */
724  size_t count = sizeof( chunk<T, has_stable_end_iterator> );
725  count = align_up( count, alignof( T ) );
726  count += new_chunk_size * sizeof( T );
727  count = ( count + sizeof( chunk<T, has_stable_end_iterator> ) - 1 ) / sizeof( chunk<T, has_stable_end_iterator> );
728 
729  allocator_ptr<chunk_allocator> p( ca, ca.allocate( count ), count );
730  auto ch = ::new ( p.get() ) chunk<T, has_stable_end_iterator>( this, this->m_chunks.size(), new_chunk_size );
731 
732  this->m_chunks.push_back( *ch );
733  p.release();
734 
735  chunk<T, has_stable_end_iterator> const& back = this->m_chunks.back();
736  set_capacity( capacity() + back.capacity() );
737  return std::prev( this->m_chunks.end() );
738 }
