278 {
279 public:

281  : m_unreflowedRect( unreflowedRect )
282  {}

285 
286  void AddReflowRect( const t4::real_rect& reflowedRect,
287  size_t lineBreakIndex )
288  {
289  if( t4::is_empty( reflowedRect ) )
290  return;
291  m_reflowedRects.push_back( reflowedRect );
292  m_lineBreakIndices.push_back( lineBreakIndex );
293  }
294 
295  void AdjustReflow( t4::real_coord deltaY,
296  size_t lineBreakIndex )
297  {
298  for( size_t index = 0; index < m_reflowedRects.size(); ++index )
299  if( m_lineBreakIndices[ index ] == lineBreakIndex )
300  m_reflowedRects[ index ] = t4::translate( m_reflowedRects[ index ], 0, deltaY );
301  }
302 
303  void Finalize( void )
304  {
305  m_lineBreakIndices.clear();
306  if( m_reflowedRects.size() <= 1 )
307  return;
308 
309  // Merge adjacent rectangles if possible:
310  auto rectIter = m_reflowedRects.begin();
311  auto nextRectIter = std::next( rectIter );
312  while( nextRectIter != m_reflowedRects.end() )
313  {
314  auto ptTLprev = t4::get_top_left( *rectIter );
315  auto ptBRprev = t4::get_bottom_right( *rectIter );
316  auto ptTLcur = t4::get_top_left( *nextRectIter );
317  auto ptBRcur = t4::get_bottom_right( *nextRectIter );
318  if( ( CloseEnough( get_x( ptBRprev ), get_x( ptTLcur ) ) &&
319  get_y( ptTLprev ) == get_y( ptTLcur ) &&
320  get_y( ptBRprev ) == get_y( ptBRcur ) ) || // Horizontal reflow
321  ( CloseEnough( get_y( ptBRprev ), get_y( ptTLcur ) ) &&
322  get_x( ptTLprev ) == get_x( ptTLcur ) &&
323  get_x( ptBRprev ) == get_x( ptBRcur ) ) ) // Vertical reflow
324  {
325  *nextRectIter = t4::make<t4::real_rect>( ptTLprev, ptBRcur );
326  auto iterIndex = rectIter - m_reflowedRects.begin();
327  m_reflowedRects.erase( rectIter, nextRectIter );
328  rectIter = m_reflowedRects.begin() + iterIndex;
329  nextRectIter = std::next( rectIter );
330  }
331  else
332  {
333  ++rectIter;
334  ++nextRectIter;
335  }
336  }
337  }
338 
339  auto begin( void ) const
340  {
341  return m_reflowedRects.begin();
342  }
343 
344  auto end( void ) const
345  {
346  return m_reflowedRects.end();
347  }
348 
349  size_t GetCachedSize() const
350  {
351  return sizeof( *this )
352  + t4::member_size( m_reflowedRects )
353  + t4::member_size( m_lineBreakIndices );
354  }
355 
356 private:
357  static bool CloseEnough( t4::real_coord a, t4::real_coord b )
358  {
359  return std::abs( a - b ) < 0x40000 * real_services::Epsilon( b );
360  }
361 
362  t4::vector<t4::real_rect> m_reflowedRects;
363  t4::vector<size_t> m_lineBreakIndices;
364 }; // class AnnotReflowLayout

371 struct ReflowLayoutKey

373  ReflowLayoutKey( const t4::pdf::page_id& pageID,
374  t4::real_coord width,
375  const t4::orthogonal_rotation screenTransform,
376  const t4::orthogonal_rotation basePageTransform,
377  t4::real_coord subpageHeight = 0,
378  t4::real_coord subpagePhase = 0,
379  bool reverseLayout = false )
380  : m_pageID( pageID )
381  , m_width( width )
382  , m_screenTransform( screenTransform )
383  , m_basePageTransform( basePageTransform )
384  , m_subpageHeight( reverseLayout ? -subpageHeight : subpageHeight )
385  , m_subpagePhase( reverseLayout ? -subpagePhase : subpagePhase )
386  {
387  if( m_subpageHeight > 0 )
388  tetraphilia_assert( m_subpagePhase >= 0 );
389  else if( m_subpageHeight < 0 )
390  tetraphilia_assert( m_subpagePhase <= 0 );
391  }
392 
393  ReflowLayoutKey() = default;
394 
395  bool operator==( const ReflowLayoutKey& rhs ) const
396  {
397  return std::tie( m_pageID, m_width, m_subpageHeight, m_subpagePhase, m_screenTransform, m_basePageTransform )
398  == std::tie( rhs.m_pageID, rhs.m_width, rhs.m_subpageHeight, rhs.m_subpagePhase, rhs.m_screenTransform, rhs.m_basePageTransform );
399  }
400 
401  bool operator!=( const ReflowLayoutKey& rhs ) const
402  {
403  return !operator==( rhs );
404  }
405 
406  bool operator<( const ReflowLayoutKey& rhs ) const
407  {
408  return std::tie( m_pageID, m_width, m_subpageHeight, m_subpagePhase, m_screenTransform, m_basePageTransform )
409  < std::tie( rhs.m_pageID, rhs.m_width, rhs.m_subpageHeight, rhs.m_subpagePhase, rhs.m_screenTransform, rhs.m_basePageTransform );
410  }
411 
412  t4::pdf::page_id m_pageID;

414  t4::orthogonal_rotation m_screenTransform;
415  t4::orthogonal_rotation m_basePageTransform;
416 
417  // Only interesting in pagination mode. Positive values for
418  // m_subpageHeight and m_subpagePhase indicate that we are in forward
419  // pagination mode, negative values indicate that we are in reverse
420  // pagination mode, and m_subpageHeight of zero indicates that we are in
421  // regular reflow mode (in which case m_subpagePhase should be ignored):
422  t4::real_coord m_subpageHeight;
423  t4::real_coord m_subpagePhase;
