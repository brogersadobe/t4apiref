206 /*
207  Utility class for updating the size and time-to-create cost of cacheables.
208 
209  If set_update_required() is called during the lifetime of this object, the
210  dtor will increase the time-to-create for this object by the amount of time
211  passed since the call to set_update_required(). It will also mark the object's
212  size as dirty, forcing a size recompute at the time of the next purge.
213 */
214 class cache_updater
215 {
216  t4::cacheable const& m_cacheable;
217  bool m_update_required;
218  cache_time_point m_start_time;
219 
220 public:
221  cache_updater( t4::cacheable const& cacheable )
222  : m_cacheable( cacheable ),
223  m_update_required( false )
224  {}
225 
226  void set_update_required( void )
227  {
228  if ( !m_update_required )
229  {
230  m_update_required = true;
231  m_start_time = current_cache_time();
232  }
233  }
234 
235  ~cache_updater()
236  {
237  if ( m_update_required )
238  {
239  cache_time_point now = current_cache_time();
240  m_cacheable.increase_time_to_create( now - m_start_time );
241  m_cacheable.update_sizes();
242  }
243  }
244 };
