41 /*
42  Class that lazily creates a contained object based on a functor and provided arguments.
43 
44  Instances of make_once expose their contained object via the * and -> operators, much like
45  boost::optional. When one of those operators is invoked for the first time, the functor
46  provided to make_once's constructor is invoked and the contained value is constructed
47  with the result.
48 
49  make_once is threadsafe. Multiple threads may call operator* or operator-> on it safely
50  without any external synchronization.
51 
52  Examples:
53 
54  using namespace t4;
55 
56  // These functions are presumed to be very expensive to call.
57 
58  // Computes all the prime factors of x.
59  extern vector<int> compute_prime_factors( int x );
60 
61  // Computes the smallest prime greater than x
62  extern int compute_next_larger_prime( int x );
63 
64  // Simple class that lazily computes interesting properties of a member integer
65  class integer_factoids
66  {
67  int m_x;
68 
69  make_once<vector<int>> m_prime_factors{ [this](){ return compute_prime_factors( this->m_x ); } };
70  make_once<int> m_next_larger_prime{ [this](){ return compute_next_larger_prime( this->m_x ); } };
71 
72  public:
73  integer_factoids( int x ) : m_x( x ) {}
74 
75  // Lazily compute the prime factors of m_x. This method is threadsafe! Multiple threads
76  // may call it. One of them will wind up doing the actual computation; the others will
77  // wait until the computation is complete and get the same result.
78  //
79  // The result reference is valid for the lifetime of the instance of the m_prime_factors member.
80  vector<int> const& get_prime_factors( void ) const
81  {
82  return *m_prime_factors;
83  }
84 
85  // Same, but for the next larger prime
86  int get_next_larger_prime( void ) const
87  {
88  return *m_next_larger_prime;
89  }
90  };
91 */
92 template<typename T>
93 class make_once : boost::noncopyable
94 {
95  mutable std::once_flag m_flag;
96  mutable boost::optional<T> m_val;
97  t4::function<void()> m_initializer;
98 
99 public:
100  template<typename F>
101  make_once( F&& callable )
102  {
103  init( std::forward<F>( callable ) );
104  }
105 
106  /*
107  Construct an invalid make_once instance. The instance is not usable until
108  init is called.
109  */
110  make_once( void ) {}
111 
112  /*
113  Assign an initializer to this. It is not allowed to call this after the first call
114  to operator-> or operator*.
115  */
116  template<typename F>
117  void init( F&& callable )
118  {
119  assert( !m_initializer );
120  m_initializer = [this, callable](){ this->m_val = callable(); };
121  }
122 
123  /*
124  Threadsafe pointer operator.
125  */
126  T* operator->( void )
127  {
128  create();
129  return m_val.operator->();
130  }
131 
132  /*
133  Threadsafe dereference operator.
134  */
135  T& operator*( void )
136  {
137  create();
138  return *m_val;
139  }
140 
141 private:
142  void create( void ) const
143  {
144  std::call_once( m_flag, m_initializer );
145  }
146 };
