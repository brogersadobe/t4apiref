199  bool IsDirectionCompatible( const DirectedRectRegion& rhs, bool disallowSkepticism = false ) const
200  {
201  if( disallowSkepticism )
202  return GetCompatibleRotation( rhs ) == drr_detail::kRotate_0;
203  else
204  {
205  if( SkepticalOfDirection() )
206  return ( GetCompatibleRotation(rhs) & drr_detail::kRotate_Orthogonal ) ? true : false;
207  return ( rhs.GetCompatibleRotation(*this) & drr_detail::kRotate_Orthogonal ) ? true : false;
208  }

211  DirectedRectRegion Union( const DirectedRectRegion& rhs ) const
212  {
213  if( !IsDirectionCompatible( rhs ) )
214  return DirectedRectRegion(); // invalid
215  if( rhs.SkepticalOfDirection() )
216  {
217  const DirectedRectRegion rhsCompat = rhs.CalcCompatibleRotation( *this );
218  // Due to precision limitations, CalcCompatibleRotation can fail:
219  if( !IsDirectionCompatible( rhsCompat ) )
220  return DirectedRectRegion(); // invalid
221  else
222  return RawUnion( rhsCompat );
223  }
224  if( SkepticalOfDirection() )
225  {
226  const DirectedRectRegion thisCompat = CalcCompatibleRotation( rhs );
227  // Due to precision limitations, CalcCompatibleRotation can fail:
228  if( !thisCompat.IsDirectionCompatible( rhs ) )
229  return DirectedRectRegion(); // invalid
230  else
231  return thisCompat.RawUnion( rhs );
232  }
233  return RawUnion( rhs );
234  }

236  static DirectedRectRegion DirectedUnion( const DirectedRectRegion & start,

242  // this guy is the same as union, unless the directed regions have skeptical directions. In that
243  // case we need to be careful to deduce the directions from how the regions live relative to one another.
244  // Normally for unions, the major direction is deduced to be from the "start" towards the "end".
245  // This function indicates that the union here is in the "CW" direction, not the main direction, so
246  // we need to be careful.
247  static DirectedRectRegion DirectedPerpUnion( const DirectedRectRegion& start, const DirectedRectRegion& end )
248  {
249  return DirectedUnion( start.CalcRotation( drr_detail::kRotate_90 ),
250  end.CalcRotation( drr_detail::kRotate_90 ) ).CalcRotation( drr_detail::kRotate_270 );

253  DirectedRectRegion Intersection( const DirectedRectRegion& rhs ) const
254  {
255  if( !IsDirectionCompatible( rhs ) )
256  return DirectedRectRegion(); // invalid
257  if( rhs.SkepticalOfDirection() )
258  return RawIntersection( rhs.CalcCompatibleRotation( *this ) );
259  if( SkepticalOfDirection() )
260  return rhs.RawIntersection( CalcCompatibleRotation( rhs ) );
261  return RawIntersection(rhs);
262  }

264  template <typename context_type>
265  bool Intersects( context_type & context,
266  const DirectedRectRegion & rhs ) const
267  {
268  using namespace imaging_model;
269  if( IsInvalid() || rhs.IsInvalid() )
270  return false;
271  if( ( SkepticalOfDirection() &&
272  ( GetCompatibleRotation(rhs) == drr_detail::kRotate_NonOrthogonal ) ) ||
273  ( rhs.GetCompatibleRotation(*this) == drr_detail::kRotate_NonOrthogonal ))
274  {
275  // We have to do it the hard way:
276  typedef typename context_type::app_traits app_traits;
277  typedef typename app_traits::real_geometry_traits::rect_type real_rect;
278  typedef typename app_traits::real_geometry_traits::matrix_type matrix_type;
279  real_rect thisRect;
280  MakeRectangle( thisRect, m_minInDir, m_minInCWDir,
281  m_maxInDir, m_maxInCWDir );
282  real_rect rhsRect;
283  MakeRectangle( rhsRect, rhs.m_minInDir, rhs.m_minInCWDir,
284  rhs.m_maxInDir, rhs.m_maxInCWDir );
285  bool status = false;
286  try
287  {
288  matrix_type rhsToNorm;
289  MakeMatrix( rhsToNorm, m_dirX, -m_dirY,
290  m_dirY, m_dirX,
291  0, 0 );
292  if ( !MatrixInvert( rhsToNorm, rhsToNorm ) )
293  throw t4::corrupt_data();
294  matrix_type normToThis;
295  MakeMatrix( normToThis,
296  m_dirX, -m_dirY,
297  m_dirY, m_dirX,
298  0, 0 );
299  status = DoNonAxisAlignedRectsIntersect(
300  thisRect, rhsRect, rhsToNorm * normToThis );
301  }
302  catch( ... )
303  {
304  // Fall through; status remains false
305  }
306  return status;
307  }
308  else
309  {
310  DirectedRectRegion intersection = rhs.RawIntersection( CalcCompatibleRotation( rhs ) );
311  return !intersection.IsInvalid() &&
312  intersection.Length() != 0 &&
313  intersection.Width() != 0;
314  }
315  }

323 template<typename real_coord>
324 DirectedRectRegion<real_coord> DirectedRectRegion<real_coord>::CalcRotation( drr_detail::RotationType r ) const
325 {
326  if( r == drr_detail::kRotate_0 )
327  return *this;
328 
329  DirectedRectRegion res;
330  if( r == drr_detail::kRotate_Impossible )
331  return res; // leave it invalid
332 
333  switch( r )
334  {
335  case drr_detail::kRotate_90:
336  res.m_dirX = -m_dirY;
337  res.m_dirY = m_dirX;
338  res.m_minInDir = m_minInCWDir;
339  res.m_maxInDir = m_maxInCWDir;
340  res.m_minInCWDir = -m_maxInDir;
341  res.m_maxInCWDir = -m_minInDir;
342  break;
343  case drr_detail::kRotate_180:
344  res.m_dirX = -m_dirX;
345  res.m_dirY = -m_dirY;
346  res.m_minInDir = -m_maxInDir;
347  res.m_maxInDir = -m_minInDir;
348  res.m_minInCWDir = -m_maxInCWDir;
349  res.m_maxInCWDir = -m_minInCWDir;
350  break;
351  case drr_detail::kRotate_270:
352  res.m_dirX = m_dirY;
353  res.m_dirY = -m_dirX;
354  res.m_minInDir = -m_maxInCWDir;
355  res.m_maxInDir = -m_minInCWDir;
356  res.m_minInCWDir = m_minInDir;
357  res.m_maxInCWDir = m_maxInDir;
358  break;
359 
360  default:
361  // Commenting out the assert because we expect to see pages
362  // where adjacent text is non-orthogonal. We don't yet have
363  // a way to deal with it gracefully, but at this point if we
364  // do nothing we'll return an invalid DRR which should result
365  // in the read order consumer putting the text in a separate
366  // paragraph which is probably good enough for now.
367  //tetraphilia_assert_always();
368  break;
369  }
370  res.m_flags = m_flags;
371  return res;
372 }

374 template<typename real_coord>
375 drr_detail::RotationType DirectedRectRegion<real_coord>::GetCompatibleRotation( const DirectedRectRegion& rhs ) const
376 {
377  if( IsInvalid() || rhs.IsInvalid() )
378  return drr_detail::kRotate_Impossible;
379 
380  assertok();
381  rhs.assertok();
382 
383  if( m_dirX == rhs.m_dirX && m_dirY == rhs.m_dirY )
384  return drr_detail::kRotate_0;
385 
386  real_coord d = dotdir(rhs);
387 
388  if( d > real_coord( kDRRCompatibleThreshold ) )
389  return drr_detail::kRotate_0;
390 
391  if( !SkepticalOfDirection() )
392  return drr_detail::kRotate_Impossible;
393 
394  if( d < real_coord( -kDRRCompatibleThreshold ) )
395  return drr_detail::kRotate_180;
396 
397  real_coord c = crossdir(rhs);
398 
399  if( c > real_coord( kDRRCompatibleThreshold ) )
400  return drr_detail::kRotate_90;
401 
402  if( c < real_coord( -kDRRCompatibleThreshold ) )
403  return drr_detail::kRotate_270;
404 
405  return drr_detail::kRotate_NonOrthogonal;
406 }

408 template<typename real_coord>
409 DirectedRectRegion<real_coord>
410 DirectedRectRegion<real_coord>::DirectedUnion( const DirectedRectRegion<real_coord> & start,
411  const DirectedRectRegion<real_coord> & end )
412 {
413  DirectedRectRegion u = start.Union( end );
414  if( u.IsInvalid() )
415  return u;
416 
417  if( !start.SkepticalOfDirection() )
418  {
419  tetraphilia_assert( !u.SkepticalOfDirection() && u.IsDirectionCompatible( start ) );
420  if( u.m_minInDir + end.Length()/10 < start.m_minInDir )
421  return DirectedRectRegion(); // invalid, at least 10% of the new guy was before the start (growing the wrong way)
422  return u;
423  }
424 
425  if( !end.SkepticalOfDirection() )
426  {
427  DirectedRectRegion cstart = start.CalcCompatibleRotation( u );
428  if( u.m_minInDir + end.Length()/10 < cstart.m_minInDir )
429  return DirectedRectRegion(); // invalid, at least 10% of the new guy was before the start (growing the wrong way)
430  return u;
431  }
432 
433  // now we were skeptical of the directions of both, so we need to tease out the "biggest growth"
434  // direction and make the union point that way. First get everybody in the same coordinate system
435  DirectedRectRegion cstart = start.CalcCompatibleRotation( u );
436  if( ( u.Length() - cstart.Length() ) < ( u.Width() - cstart.Width() ) )
437  {
438  // it grew more wide than long, this is wrong, rotate
439  u = u.CalcRotation( drr_detail::kRotate_90 );
440  cstart = cstart.CalcRotation( drr_detail::kRotate_90 );
441  }
442  tetraphilia_assert( u.IsDirectionCompatible( cstart ) && ( u.Length() - cstart.Length( )) >= ( u.Width() - cstart.Width() ) );
443 
444  // Now we look to see whether the "start" rect's min in dir equals the union's, if so it grew the right direction
445  if( u.m_minInDir == cstart.m_minInDir )
446  {
447  // right direction, did we actually grow? If so we are confident...
448  if( u.m_maxInDir > cstart.m_maxInDir )
449  u.m_flags &= static_cast<unsigned int>( ~drr_detail::kFlag_SkepticalOfDirection );
450  return u;
451  }
452 
453  return DirectedRectRegion(); // invalid
454 }

461 template<typename real_coord>
462 DirectedRectRegion<real_coord>
463 DirectedRectRegion<real_coord>::DirectedUnion( const DirectedRectRegion<real_coord> & start,
464  DirectedRectRegion<real_coord> & end,
465  bool weakenEndIfHelpful )
466 {
467  DirectedRectRegion u = DirectedUnion( start, end );
468  if( u.IsInvalid() && weakenEndIfHelpful )
469  {
470  // So far we have detemined that the two rectangles do not mesh; this is usually due
471  // to both knowing their directions, and the directions being different. There is
472  // one case where this may yet be OK, which is the hack here. There is vertical
473  // Japanese typography where the half-width western chars and numbers are displayed
474  // in small "horizontal" blocks, this is called TCY (I think). We have decided that
475  // an established DRR can absorb any DRR which "is contained in its extension to
476  // infinity". What this means is that if the later DRR is covered by where we
477  // expect the established DRR to grow, then it is accepted regardless of
478  // orientation. --chip 11/06
479 
480  // First weaken the direction of the end DRR
481  DirectedRectRegion weakend = end;
482  weakend.m_flags |= drr_detail::kFlag_SkepticalOfDirection;
483 
484  // now union the weak end with the original start
485  u = DirectedUnion( start, weakend );
486  if( u.IsInvalid() )
487  return u; // not even unionable in a weak sense. Means the coords are incompatible.
488 
489  // OK, so it unioned, but we need to make sure it didn't get *too* fat.
490  // After discussions with Maki, decided to allow at most 100% growth.
491  if( u.Width() > 2*start.Width() )
492  return DirectedRectRegion(); // invalid, end was not well contained in the starts CW direction bounds
493 
494  // the weaker version passed, so we commit the change and return.
495  end = weakend.CalcCompatibleRotation(u);
496  }
497  return u;
498 }

500 template<typename real_coord>
501 DirectedRectRegion<real_coord> DirectedRectRegion<real_coord>::RawUnion( const DirectedRectRegion<real_coord>& rhs ) const
502 {
503  tetraphilia_assert( IsDirectionCompatible( rhs, true ) );
504  DirectedRectRegion res;
505  res.m_flags = ( SkepticalOfDirection() && rhs.SkepticalOfDirection() ) ? drr_detail::kFlag_SkepticalOfDirection : 0;
506  res.m_dirX = m_dirX;
507  res.m_dirY = m_dirY;
508  res.m_minInDir = t4::min<real_coord>( m_minInDir, rhs.m_minInDir );
509  res.m_maxInDir = t4::max<real_coord>( m_maxInDir, rhs.m_maxInDir );
510  res.m_minInCWDir = t4::min<real_coord>( m_minInCWDir, rhs.m_minInCWDir );
511  res.m_maxInCWDir = t4::max<real_coord>( m_maxInCWDir, rhs.m_maxInCWDir );
512  return res;
513 }

517 {

519  DirectedRectRegion res;

521  res.m_dirX = m_dirX;
522  res.m_dirY = m_dirY;
523  res.m_minInDir = t4::max<real_coord>( m_minInDir, rhs.m_minInDir );
524  res.m_maxInDir = t4::min<real_coord>( m_maxInDir, rhs.m_maxInDir );
525  res.m_minInCWDir = t4::max<real_coord>( m_minInCWDir, rhs.m_minInCWDir );
526  res.m_maxInCWDir = t4::min<real_coord>( m_maxInCWDir, rhs.m_maxInCWDir );
527  if( res.m_minInDir > res.m_maxInDir || res.m_minInCWDir > res.m_maxInCWDir )
528  res.m_flags = drr_detail::kFlag_IsInvalid;
529  return res;
