219 class DLDataStackEntry
220 {
221 public:
222  using int_type = boost::int_t<8 * sizeof( t4::real )>::exact;
223  using uint_type = boost::uint_t<8 * sizeof( t4::real )>::exact;
224  static constexpr size_t unit_size = sizeof( int_type );
225 
226 private:
227  union
228  {
229  t4::real m_real;
230  int_type m_int;
231  uint_type m_uint;
232  };
233 
234 public:
235  /*
236  Create a DLDataStackEntry for a signed integer. The integer must not be larger than an int_type.
237  (A compiler error will ensue).
238  */
239  template<typename T>
240  explicit DLDataStackEntry( T val, std::enable_if_t<std::is_signed<T>::value && t4::is_integral<T>::value>* = {} )
241  : m_int( static_cast<int_type>( val ) )
242  {
243  // DLDataStackEntry can't handle integers larger than a t4::real
244  static_assert( sizeof( T ) <= sizeof( int_type ), "" );
245  }
246 
247  /*
248  Create a DLDataStackEntry for an unsigned integer. The integer must not be larger than an uint_type.
249  (A compiler error will ensue).
250  */
251  template<typename T>
252  explicit DLDataStackEntry( T val, std::enable_if_t<!std::is_signed<T>::value && t4::is_integral<T>::value>* = {} )
253  : m_uint( static_cast<uint_type>( val ) )
254  {
255  // DLDataStackEntry can't handle integers larger than a t4::real
256  static_assert( sizeof( T ) <= sizeof( uint_type ), "" );
257  }
258 
259  template<typename T>
260  explicit DLDataStackEntry(T val, std::enable_if_t<t4::is_floating_point<T>::value>* = {} )
261  : m_real( val )
262  {}
263 
264  template<typename T>
265  std::enable_if_t<std::is_signed<T>::value && t4::is_integral<T>::value,
266  T> Get( void ) const
267  {
268  // DLDataStackEntry can't handle integers larger than a t4::real
269  static_assert( sizeof( T ) <= sizeof( int_type ), "" );
270  return static_cast<T>( m_int );
271  }
272 
273  template<typename T>
274  std::enable_if_t<!std::is_signed<T>::value && t4::is_integral<T>::value,
275  T> Get( void ) const
276  {
277  // DLDataStackEntry can't handle integers larger than a t4::real
278  static_assert( sizeof( T ) <= sizeof( uint_type ), "" );
279  return static_cast<T>( m_uint );
280  }
281 
282  template<typename T>
283  std::enable_if_t<t4::is_floating_point<T>::value,
284  T> Get( void ) const
285  {
286  return static_cast<T>( m_real );
287  }
288 
289  friend bool operator==( const DLDataStackEntry& lhs, const DLDataStackEntry& rhs )
290  {
291  return lhs.m_int == rhs.m_int;
292  }

295  {
296  return !operator==( lhs, rhs );

1908 /*
1909  This function allows two clips to be "sequential" even if they have marked content operators
1910  between them. It currently does not permit any other operators between them (while still
1911  considering them to be sequential)
1912 */
1913 template<typename accumulator_type>
1914 void AccumulateSequentialClips( DisplayList& dl, DLConsumer* consumer, pdf::content::DLEntryTreeWalker& walker, accumulator_type& accum )
1915 {
1916  typedef DisplayList::bez_path_iter_type bez_path_iter_type;
1917  typedef DLEntryTreeWalker::entry_stack_iter_type entry_stack_iter_type;
1918  typedef DLEntryTreeWalker::data_stack_iter_type data_stack_iter_type;
1919 
1920  entry_stack_iter_type opIt = walker.GetOPCodeIterator();
1921  data_stack_iter_type dataIt = walker.GetDataIterator();
1922 
1923  while( dl.DLHasMoreEntries(opIt) )
1924  {
1925  DLOPCode op = *opIt;
1926  // if this is anything other than a path (which includes clips) then stop looping
1927  switch ( op )
1928  {
1929  case DLOPCode::kOPExecuteEntry:
1930  {
1931  // Peek at the top item; consume it if it's marked content
1932  auto const& e = walker.GetTopPtr();
1933  if( e->GetEntryType() == DLOPCode::kOPBeginMarkedContent )
1934  {
1935  walker.FetchPtr(); // advance it
1936  walker.FetchOPCode();
1937  consumer->AdvanceDLEntryNum();
1938  e->Do( *consumer, walker);
1939  }
1940  }
1941  break;
1942 
1943  case DLOPCode::kOPEndMarkedContent:
1944  walker.FetchOPCode();
1945  consumer->AdvanceDLEntryNum();
1946  consumer->DoEndMarkedContent();
1947  break;
1948 
1949  case DLOPCode::kOPPath:
1950  {
1951  // high 8 bits holds the flags - low 24 bits holds the path length
1952  int pathLengthAndFlags = dataIt->template Get<int>(); ++dataIt;
1953  unsigned char flags = static_cast<unsigned char>(pathLengthAndFlags >> kFlagsShift);
1954 
1955  size_t pathLength = pathLengthAndFlags & kMaxNonFlagsInt;
1956  if (flags&kPathSpecifiesBounds)
1957  dataIt += 4;
1958 
1959  // unpack stroke, fillRule and clipRule
1960  bool stroke = (flags&0x10) ? true : false;
1961  FillRule fillRule = static_cast<FillRule>((flags>>kFillShift) & kFillRuleMask);
1962  FillRule clipRule = static_cast<FillRule>((flags>>kClipShift) & kFillRuleMask);
1963 
1964  // if this is anything other than a clip of some kind, then stop looping
1965  if ( stroke || fillRule != pdf::content::kNoFill || clipRule == pdf::content::kNoFill )
1966  return;
1967 
1968  // we need to increment these
1969  /*pathOpCode =*/ walker.FetchOPCode();
1970  /*pathLength =*/ walker.template FetchInt<int>();
1971  consumer->AdvanceDLEntryNum();
1972  if ( flags&kPathSpecifiesBounds )
1973  walker.UpdatePathBoundsRectLocation();
1974 
1975  if (op == DLOPCode::kOPPath)
1976  {
