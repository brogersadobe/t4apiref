233 template<typename app_traits>
234 Type0Function<app_traits>::Type0Function( const array_type& domain, const obj_type& rangeObj,
235  const dict_type& funcDict )
236  : Function<app_traits>( domain, rangeObj, 0 /*type*/ )
237 {
238  if( this->m_dimRange == 0 )
239  throw t4::corrupt_data();
240 
241  array_type size = funcDict.Get( "Size" ).ArrayValue();
242  size_t i;
243  auto bitsPerSample = t4::convert<size_t>( funcDict.GetRequiredInteger( "BitsPerSample" ) );
244  size_t totBits = bitsPerSample * this->m_dimRange;
245  for ( i = 0; i < this->m_dimDomain; i++ )
246  totBits = t4::multiply<size_t>( totBits, size.Get(i).IntegerValue() );
247  size_t required = t4::add<size_t>( totBits, 7 ) / 8;
248 
249  t4::data_block_stream str =
250  GetFilteredStream( funcDict, false /*isInlineImage*/ );
251  data_io::BufferedDataBlockStream bdbs( std::move( str ) );
252  const char* samples;
253  size_t avail = bdbs.PeekBytes( samples, required );
254  if ( avail < required )
255  throw t4::corrupt_data();
256 
257  m_size = t4::make_transient_array<int_coord>( this->m_heap, this->m_dimDomain );
258  int_coord* sizePtr = m_size;
259  for ( i = 0; i < this->m_dimDomain; i++ )
260  {
261  sizePtr[i] = size.Get(i).IntegerValue();
262  if ( sizePtr[i] <= 0 )
263  throw t4::corrupt_data();
264  }
265 
266  size_t dimRangeTimes2 = t4::multiply<size_t>( this->m_dimRange, 2 );
267  m_decode = t4::make_transient_array<real_coord>( this->m_heap, dimRangeTimes2 );
268  m_normalizedDecode = t4::make_transient_array<real_coord>( this->m_heap, dimRangeTimes2 );
269 
270  obj_type decodeObj = funcDict.Get( "Decode" );
271  if ( decodeObj.Type() != store::kObjectTypeNull )
272  {
273  array_type decode = decodeObj.ArrayValue();
274  real_coord* decodePtr = m_decode;
275  for ( i = 0; i < this->m_dimRange; i++ )
276  {
277  decodePtr[i*2] = decode.Get(i*2).RealValue();
278  decodePtr[i*2+1] = decode.Get(i*2+1).RealValue();
279  }
280  }
281  else {
282  std::copy( this->m_range, this->m_range + dimRangeTimes2, m_decode );
283  }
284 
285  /*
286  Note that although the PDF spec says that the default value for the Decode
287  is that it is the same as the Range this is problematic for the way we treat
288  Function objects combined with the way PDF treats image data values in an Lab color space.
289  An Lab color space will typically state that its Range is [0 100 -128 127 -128 127]
290  even though 8 bit per pixel Lab image data is encoded using all 8 bits of
291  each channel. PDF viewers just use an implicit mapping for Lab image color values.
292  For image data, Tetraphilia's Function objects always produce values in the range of 0-1.
293  These values are later converted in 0-255 like values through a simple converter.
294  The PDF spec says that the Decode is a "linear mapping of sample
295  values into the range appropriate for the functions output values." If we were
296  to use the Range as the Decode for these Lab color spaces we would be producing
297  values outside the 0 to 1 range. So this loop is really designed to catch Lab-like
298  Ranges and turn them back into the correct thing for us. See Watson bug#1285479
299  for an example of a file that we would draw incorrectly without this code.
300 
301  But note that we can only do this for Functions that are being used to convert
302  image-like data. When these same Functions are used to convert real_coord data
303  we need to return the 0-100, -128-127 like values. We do this by storing both a
304  "normalized" array of decode values and an original array. The "normalized" values
305  are used for converting image-like values. See Watson bug#1383197 for an example of
306  a file that uses Lab colors with real_coord conversions.
307  */
308  real_coord* normDecodePtr = m_normalizedDecode;
309  std::copy( this->m_range, this->m_range + dimRangeTimes2, normDecodePtr );
310  for ( i = 0; i < this->m_dimRange; i++ )
311  {
312  if ( normDecodePtr[i*2] < 0 || normDecodePtr[i*2] > 1 || normDecodePtr[i*2+1] < 0 || normDecodePtr[i*2+1] > 1 ) {
313  normDecodePtr[i*2] = 0;
314  normDecodePtr[i*2+1] = 1;
315  }
316  }
317 
318  size_t dimDomainTimes2 = t4::multiply<size_t>( this->m_dimDomain, 2 );
319  m_encode = t4::make_transient_array<real_coord>( this->m_heap, dimDomainTimes2 );
320  real_coord* encodePtr = m_encode;
321  sizePtr = m_size;
322 
323  obj_type encodeObj = funcDict.Get( "Encode" );
324  if ( encodeObj.Type() == store::kObjectTypeNull )
325  {
326  for ( i = 0; i < this->m_dimDomain; i++ )
327  {
328  encodePtr[i*2] = 0;
329  encodePtr[i*2+1] = real_coord( sizePtr[i] - 1 );
330  }
331  }
332  else
333  {
334  array_type encode = encodeObj.ArrayValue();
335  for ( i = 0; i < this->m_dimDomain*2; i++ )
336  encodePtr[i] = encode.Get(i).RealValue();
337  }
338  size_t totValues = this->m_dimRange;
339  for ( i = 0; i < this->m_dimDomain; i++ )
340  totValues = t4::multiply<size_t>( totValues, sizePtr[i] );
341 
342  m_values = t4::make_transient_array<real_coord>( this->m_heap, totValues );
343  real_coord* valuesPtr = m_values;
344  const unsigned char* usamples = reinterpret_cast<const unsigned char*>( samples );
345  switch ( bitsPerSample )
346  {
347  case 1:
348  for ( i = 0; i < totValues; i++ )
349  {
350  unsigned char t = usamples[i/8];
351  valuesPtr[i] = real_services::RealFrom1BitSample(ArgType<real_coord>(), t,i);
352  }
353  break;
354 
355  case 2:
356  for ( i = 0; i < totValues; i++ )
357  {
358  unsigned char t = usamples[i/4];
359  valuesPtr[i] = real_services::RealFrom2BitSample(ArgType<real_coord>(), t,i);
360  }
361  break;
362 
363  case 4:
364  for ( i = 0; i < totValues; i++ )
365  {
366  unsigned char t = usamples[i/2];
367  valuesPtr[i] = real_services::RealFrom4BitSample(ArgType<real_coord>(), t,i);
368  }
369  break;
370 
371  case 8:
372  for ( i = 0; i < totValues; i++ )
373  valuesPtr[i] = real_services::RealFrom8BitSample( ArgType<real_coord>(), usamples[i] );
374  break;
375 
376  case 12:
377  for ( i = 0; i < totValues; i++ )
378  {
379  size_t indx = i*3/2;
380  valuesPtr[i] = real_services::RealFrom12BitSample( ArgType<real_coord>(), &usamples[indx], i );
381  }
382  break;
383 
384  case 16:
385  for ( i = 0; i < totValues; i++ )
386  valuesPtr[i] = real_services::RealFrom16BitSample( ArgType<real_coord>(), &usamples[i*2] );
387  break;
388 
389  case 24:
390  for ( i = 0; i < totValues; i++ )
391  valuesPtr[i] = real_services::RealFrom24BitSample( ArgType<real_coord>(), &usamples[i*3] );
392  break;
393 
394  case 32:
395  for ( i = 0; i < totValues; i++ )
396  valuesPtr[i] = real_services::RealFrom32BitSample( ArgType<real_coord>(), &usamples[i*4] );
397  break;
398 
399  default:
400  throw t4::corrupt_data();
401  }
402 
403  m_inputVecBuf = t4::make_transient_array<real_coord>( this->m_heap, this->m_dimDomain );
404 
405  if(this->m_dimDomain >= 8*sizeof(size_t))
406  throw std::out_of_range( "" );
407 
408  size_t valuesCount = t4::multiply<size_t>( this->m_dimRange, 1 << this->m_dimDomain );
409  m_valuesBuf = t4::make_transient_array<real_coord>( this->m_heap, valuesCount );
410  m_coordsBuf = t4::make_transient_array<int_coord>( this->m_heap, this->m_dimDomain );
411 }

413 template<typename app_traits>
414 typename app_traits::real_geometry_traits::coord_type Type0Function<app_traits>::DoInput( real_coord inp, size_t i )
415 {
416  real_coord* domainPtr = this->m_domain;
417  int_coord* sizePtr = m_size;
418  if ( inp < domainPtr[i*2] )
419  inp = domainPtr[i*2];
420  if ( inp > domainPtr[i*2+1] )
421  inp = domainPtr[i*2+1];
422  if ( domainPtr[i*2+1] != domainPtr[i*2] )
423  {
424  real_coord* encodePtr = m_encode;
425  inp = encodePtr[i*2] + (inp - domainPtr[i*2])*(encodePtr[i*2+1] - encodePtr[i*2])/(domainPtr[i*2+1] - domainPtr[i*2]);
426  }
427  if ( inp < 0 )
428  inp = 0;
429  if ( inp > sizePtr[i] - 1 )
430  inp = real_coord( sizePtr[i] - 1 );
431  return inp;
432 }

455 template<typename app_traits>
456 template<typename signal_traits>
457 void Type0Function<app_traits>::Interpolate( typename signal_traits::signal_type* dst, ptrdiff_t dstSignalStride,
458  const typename signal_traits::signal_type* src, ptrdiff_t srcSignalStride )
459 {
460  real_coord* inputVec = m_inputVecBuf;
461  real_coord* p = m_valuesBuf;
462  int_coord* coordsPtr = m_coordsBuf;
463 
464  size_t i;
465  for ( i = 0; i < this->m_dimDomain; i++ )
466  {
467  inputVec[i] = DoInput( signal_traits::signal_to_real( *src ), i );
468  src = t4::add_bytes( src, srcSignalStride );
469  }
470 
471  for ( i = 0; i < this->m_dimDomain; i++ )
472  coordsPtr[i] = real_services::FloorToInt( ArgType<int_coord>(), inputVec[i] );
473 
474  /*
475  load the pixel values of the 2^m_dimDomain corners that surround the src color.
476  This works by using the values in m_coordsBuff, which are set (more or less)
477  to the integer portion of each value in the inputVec (src color). We then
478  take special action each time the bit we are interested in changes between
479  i and i+1.
480 
481  And note that there is some special handling for the case where coord[i] == size[i] - 1.
482  That is, the color value for the i'th dimension is exactly equal to the largest
483  possible value in that dimension. In that case we still consider the coord[i]
484  to be going between the color value (size[i]-1) and one bigger than that -- which
485  is an invalid value in that dimension. To make that work GetValue pins these
486  cases to size[i]-1, and then looks up that value. This causes us to get two
487  copies of the same color value in the values buffer. The interpolation loop
488  below then interpolates between those two same values and gets the original
489  value.
490  */
491  size_t top = (1<< this->m_dimDomain);
492  for ( i = 0; i < top; i++ ) {
493  GetValue( p );
494  size_t next_i = i + 1;
495  for ( size_t j = 0; j < this->m_dimDomain; j++ ) {
496  size_t i_bit = (i >> j) & 1;
497  size_t next_i_bit = (next_i >> j) & 1;
498  if ( i_bit != next_i_bit ) {
499  if ( next_i_bit == 0 ) {
500  tetraphilia_assert( coordsPtr[j] > 0 );
501  --coordsPtr[j];
502  } else {
503  ++coordsPtr[j];
504  }
505  }
506  }
507  }
508 
509  const real_coord* vp = inputVec;
510  real_coord* p_write = m_valuesBuf;
511  const real_coord* p_read = p_write;
512  real_coord* end_p = &p_write[this->m_dimRange * (static_cast<size_t>(1) << this->m_dimDomain)];
513  do
514  {
515  real_coord mult = real_services::Fraction( *vp++ );
516  while ( p_read != end_p ) {
517  for ( i = 0; i < this->m_dimRange; i++ ) {
518  p_write[0] = p_read[0] + (p_read[this->m_dimRange] - p_read[0]) * mult;
519  ++p_write;
520  ++p_read;
521  }
522  p_read += this->m_dimRange;
523  }
524  end_p = p_write;
525  p_read = p_write = m_valuesBuf;
526  } while ( end_p != &p_read[this->m_dimRange] );
527 
528  real_coord* rangePtr = this->m_range;
529  if ( rangePtr != nullptr ) {
530  real_coord* decodePtr = signal_traits::kUsesFullRangeForZeroToOne ? m_normalizedDecode : m_decode;
531  for ( i = 0; i < this->m_dimRange; i++ ) {
532  p_write[i] = decodePtr[i*2] + p_write[i]*(decodePtr[i*2+1] - decodePtr[i*2]);
533  if ( p_write[i] < rangePtr[i*2] )
534  p_write[i] = rangePtr[i*2];
535  else if ( p_write[i] > rangePtr[i*2+1] )
536  p_write[i] = rangePtr[i*2+1];
537  }
538  }
539 
540  for ( i = 0; i < this->m_dimRange; i++ )
541  {
542  *dst = signal_traits::real_to_signal( p_read[i] );
543  dst = t4::add_bytes( dst, dstSignalStride );
544  }
545 }
