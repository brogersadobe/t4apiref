121  void ConvertPixel( signal_type * dst, ptrdiff_t dstSignalStride,
122  const signal_type * src, ptrdiff_t srcSignalStride )
123  {
124  typedef ColorHelper<signal_traits> color_helper;
125  tetraphilia_assert( src );
126  tetraphilia_assert( dst );
127  size_t index;
128 
129  m_colorCache.NewSearch();
130  if( m_ranges.size() )
131  {
132  for( index = 0; index < m_nSrcSignals; ++index )
133  {
134  m_buf[ index ] = color_helper::PinAndScale(
135  *src,
136  m_ranges[ 2 * index ],
137  m_ranges[ 2 * index + 1 ] );
138  // TODO: should the cache contain the pinned and scaled value?
139  m_colorCache.SetKey( index, *src );
140  src = t4::add_bytes( src, srcSignalStride );
141  }
142  }
143  else
144  {
145  for( index = 0; index < m_nSrcSignals; ++index )
146  {
147  m_buf[ index ] = signal_traits::signal_to_real( color_helper::Pin( *src ) );
148  m_colorCache.SetKey( index, *src );
149  src = t4::add_bytes( src, srcSignalStride );
150  }
151  }
152  for( index = m_nSrcSignals; index < 4; ++index )
153  m_buf[ index ] = 0;
154 
155  if ( m_colorCache.Found() )
156  {
157  for( index = 0; index < m_nDstSignals; ++index )
158  {
159  *dst = m_colorCache.GetResult( index );
160  dst = t4::add_bytes( dst, dstSignalStride );
161  }
162  }
163  else
164  {
165  // if were caching here don't use the convertor's cache
166  // generally this is because were processing an image
167  if (colorcache_type::kColorCacheNever)
168  m_converter->Convert( m_buf.data() );
169  else
170  m_converter->ConvertNoCache(m_buf.data());
171 
172  m_colorCache.NewResult();
173  for( index = 0; index < m_nDstSignals; ++index )
174  {
175  *dst = signal_traits::real_to_signal( m_buf[ index ] );
176  m_colorCache.SetResult( index, *dst );
177  dst = t4::add_bytes( dst, dstSignalStride );
178  }
179  }

337  }

348  srcCS.CacheConverter( dstCS, renderIntent, m_converter );
349  if (updater)
350  updater->set_update_required();
351  }
352  }

479  }
