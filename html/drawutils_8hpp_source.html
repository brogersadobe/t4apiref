296 public:
297  template<typename iterator_type>
298  ConstantOpaqueBackdrop( const color_space_ptr_type& cs,
299  iterator_type it, iterator_type end )
300  : parent_type( cs ),
301  m_store( *t4::get_thread_heap() )
302  {
303  size_t numSignals = cs->GetNumSignals();
304  GraphicLayoutDescriptor desc = MakeGraphicLayoutDescriptor( numSignals,
305  GraphicLayoutDescriptor::kShapeIgnored,
306  GraphicLayoutDescriptor::kOpacityImplicitlyOne,
307  GraphicLayoutDescriptor::kColorPerChannel,
308  true, true );
309  m_store.Init( InfiniteRect<int_rect_type>(), desc, false, *t4::get_thread_heap() );
310  auto const& color = *m_store.GetWritableBuffers().m_color;
311  auto const& layout = *color.m_layout;
312  signal_type* pixel = color.GetPixels();
313  size_t index = 0;
314 
315  while ( it != end )
316  {
317  if ( index == numSignals )
318  throw t4::corrupt_data();
319 
320  signal_type val = signal_traits::real_to_signal( *it );
321  signal_type* signal = t4::add_bytes( pixel, layout.m_firstSignalOffset + static_cast<size_t>( layout.m_signalStride ) * index );
322  *signal = val;
323 
324  ++index;
325  ++it;
326  }
327  }
328 
329 private:
330  raster_painter_ptr_type GetRemovalBackdropImpl( const constraints_type& constraints,
331  const color_space_ptr_type& dstCs, bool /* childIsIsolated */ ) override
332  {
333  raster_painter_ptr_type result = t4::make_brief<FastPixelBufferRasterPainter<signal_traits>>( constraints, m_store.GetBuffers() );
334  return ConvertRasterPainter( std::move( result ), constraints, this->GetColorSpace(), dstCs );
