32 /*
33  Wrapper object for T4 globals
34 
35  Global variables can make leak tracking difficult. One useful way to detect
36  leaks is to verify at shutdown that there exists no outstanding allocated memory,
37  but this is made more difficult if global objects are permitted to allocate memory
38  (as is common with constructs like caches).
39 
40  The wrap_global template encapsulates a global entity of the specified type and
41  a functor object that is used to construct that entity. The initializations don't
42  actually occur until construct_wrapped_globals() is called.
43 
44  At shutdown, destroy_wrapped_globals() should be called, and it is from within this
45  call that all wrapped globals' destructors are called, in the reverse order that
46  they were constructed.
47 
48  When defining a global variable that holds on to t4-allocated resources, we recommend
49  you use the T4_DEFINE_GLOBAL or T4_DEFINE_STATIC macros.
50 */
51 template<typename T>
52 class wrap_global;
53 
54 void construct_wrapped_globals( void );
55 void destroy_wrapped_globals( void );

91  void construct( void )
92  {
93  // If this fires then your program is not calling construct_wrapped_globals() at startup.
94  // You must do this exactly once, from somewhere threadsafe, presumably main(). Do not
95  // forget also to call destroy_wrapped_globals() at termination.
96  assert( s_in_construct_wrapped_globals );
97  construct_impl();
98  }
99 
100  void destroy( void )
101  {
102  // We don't lock here because this can only be called from destroy_wrapped_globals, which takes the lock
103  s_dtor_chain = m_next_dtor;
104  m_next_dtor = nullptr;
105  destroy_impl();
106  }
107 
108  virtual void construct_impl( void ) = 0;
109  virtual void destroy_impl( void ) = 0;
110 
111  friend void construct_wrapped_globals( void );
112  friend void destroy_wrapped_globals( void );
113 };

115 template<typename T>
116 class wrap_global : private wrap_global_base
117 {
118  boost::optional<T> m_wrapped_global;
119  boost::optional<std::function<void(boost::optional<T>&)>> m_initializer;
120 
121 public:
122  wrap_global( void )
123  {
124  using namespace boost;
125  m_initializer = []( optional<T>& val ) { val.emplace(); };
126  link();
127  }
128 
129  template<typename F>
130  wrap_global( F initializer )
131  : m_initializer( std::move( initializer ) )
132  {
133  link();
134  }
135 
136  ~wrap_global( void ) override
137  {
138  /*
139  If a wrapped global's dtor is called, we are presumably in some OS-provided cleanup
140  code which is tasked with invoking all the destructors of all globals in the module
141  being unloaded/terminated. Just in case destroy_wrapped_globals() hasn't already been
142  called, call it here. We'll wind up calling it once for every instance of wrap_global,
143  but after the first call all calls will do nothing.
144  */
145  if ( !wrap_global_base::s_in_destroy_wrapped_globals )
146  destroy_wrapped_globals();
147  }
148 
149  T& operator*( void ) { return *get(); }
150  T* operator->( void ) { return get(); }
151  T* get( void )
152  {
153  if( !m_wrapped_global )
154  construct();
155  return m_wrapped_global.get_ptr();
156  }
157 
158 private:
159  void construct_impl( void ) override
160  {
161  if ( !m_wrapped_global )
162  {
163  assert( m_initializer );
164  (*m_initializer)( m_wrapped_global );
165  link_dtor();
166  }
167  }
168 
169  void destroy_impl( void ) override
170  {
171  m_wrapped_global.reset();
172  }
173 };

177 /*
178  Define a global variable that holds onto t4 resources (and as a result should be constructed/destroyed
179  after/before the memory context, and whatever other t4 globals it requires).
180 
181  Usage:
182 
183  T4_DEFINE_GLOBAL( shared_ptr<font_type>, g_last_used_font, () );
184 
185  The third parameter is a parameter list that is passed to optional::emplace to construct
186  the global inside an optional. Use ref/cref on parameters if necessary. The parameter list is
187  NOT evaluated at global startup. It is evaluated when the global is constructed during
188  construct_wrapped_globals.
189 */
190 #define T4_DEFINE_GLOBAL( type, name, ctor_params ) \
191  ::t4::wrap_global<type> name( []( ::boost::optional<type>& val ){ val.emplace ctor_params; } )
192

195 */
196 #define T4_DECLARE_GLOBAL( type, name ) extern ::t4::wrap_global<type> name
