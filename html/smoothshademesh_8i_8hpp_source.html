438 template <typename app_traits>
439 void GouraudEdgeCrossingInfo<app_traits>::Reset(const GouraudVertexWithZ<app_traits>* pTop, const GouraudVertexWithZ<app_traits>* pBot, int_coord_type y, size_t numSignals, bool zIsInteresting)
440 {
441  tetraphilia_assert(m_isInvalid);
442  m_isInvalid = false;
443 
444  m_p0.Set(*pTop, 0);
445  m_p1.Set(*pBot, 1);
446  m_y = y;
447 
448  real_coord_type deltay = m_p1.m_y - m_p0.m_y;
449  tetraphilia_assert(deltay >= 0);
450 
451  // make sure we have enough total delta y more the math to work out OK. The claim here is that
452  // as long as we reseve the beginning and ending scanlines untouched, it doesn't matter much whether the
453  // edge is truly flat, or just pretty darn flat.
454  real_coord_type mindeltay = real_coord_type(1.0/8);
455  if (deltay < mindeltay)
456  {
457  deltay = mindeltay;
458  // skootch the points around to match.
459  if (real_services::Fraction(m_p0.m_y) + mindeltay < 1)
460  {
461  // both in same scanline
462  m_p1.m_y = m_p0.m_y + mindeltay;
463  }
464  else
465  {
466  // maybe same, maybe different scanlines, but near an edge
467  if (real_services::Floor(m_p0.m_y) == real_services::Floor(m_p1.m_y))
468  {
469  m_p1.m_y = real_services::Ceil(m_p0.m_y);
470  m_p0.m_y = m_p1.m_y - mindeltay;
471  }
472  else
473  {
474  m_p1.m_y = m_p0.m_y + mindeltay;
475  }
476  }
477  }
478 
479  m_pa.SetToInterpolated(m_p0, m_p1, (y - m_p0.m_y) / deltay, numSignals, zIsInteresting);
480  m_pb.SetToInterpolated(m_p0, m_p1, (y+1 - m_p0.m_y) / deltay, numSignals, zIsInteresting);
481  m_pTop = &m_pa;
482  m_pBottom = &m_pb;
483  if (m_pTop->m_x <= m_pBottom->m_x)
484  {
485  m_pLeft = m_pTop;
486  m_pRight = m_pBottom;
487  }
488  else
489  {
490  m_pLeft = m_pBottom;
491  m_pRight = m_pTop;
492  }
493 }

495 template <typename app_traits>
496 void GouraudEdgeCrossingInfo<app_traits>::Advance(size_t numSignals, bool zIsInteresting)
497 {
498  tetraphilia_assert(!m_isInvalid);
499 
500  m_y++;
501  boost::swap(m_pTop, m_pBottom);
502  boost::swap(m_pLeft, m_pRight);
503 
504  m_pBottom->SetToInterpolated(m_p0, m_p1, (m_y+1 - m_p0.m_y) / (m_p1.m_y - m_p0.m_y), numSignals, zIsInteresting);

812  {

822  signal_type* shapeScanlinePos = ShapeBaseAddr();
823  for (int_coord_type y = ymin; y < ymax; y++)
824  {
825  signal_type* shadeChannelPixelPos = shadeChannelScanlinePos;
826  signal_type* shapePixelPos = shapeScanlinePos;

838  {
839  *shapeSignalPos = signal_traits::one();

842  shadeChannelPixelPos = t4::add_bytes(shadeChannelPixelPos, ShadeChannelPixelStride());
843  shapePixelPos = t4::add_bytes(shapePixelPos, ShapePixelStride());
844  }
845  this->m_samplerContext.Yield(100 * ( xmax - xmin ) );
846  shadeChannelScanlinePos = t4::add_bytes(shadeChannelScanlinePos, ShadeChannelScanLineStride());
847  shapeScanlinePos = t4::add_bytes(shapeScanlinePos, ShapeScanLineStride());
848  }

1141 template<typename _signal_traits>
1142 void GouraudPolygonSampler<_signal_traits>::DrawInterpolatedRun( const GouraudEdgeCrossingInfo<app_traits>& startInfo, const GouraudEdgeCrossingInfo<app_traits>& endInfo, size_t numSignals )
1143 {
1144  tetraphilia_assert(startInfo.GetY() == endInfo.GetY());
1145 
1146  int_coord_type y = startInfo.GetY();
1147  const GouraudEdgePoint<app_traits>& startPoint = startInfo.GetLeftmostPoint();
1148  const GouraudEdgePoint<app_traits>& endPoint = endInfo.GetRightmostPoint();
1149 
1150  int_coord_type startx = real_services::FloorToInt(ArgType<int_coord_type>(), startPoint.m_x);
1151  int_coord_type endx = real_services::CeilToInt(ArgType<int_coord_type>(), endPoint.m_x);
1152  // there is a rare condition which needs a special case, that of two vertical lines on a pixel boundary. In this case, both
1153  // the floor and ceil yield the same number, so we need to be careful. I arbitrarily set the pixel to the right of the boundary.
1154  int_coord_type endxinc = endx > startx ? endx-1 : startx;
1155 
1156  tetraphilia_assert( startx <= endxinc);
1157  if (startx > get_xmax(this->m_samplerContext.m_constraints.m_interestBounds) || endxinc < get_xmin(this->m_samplerContext.m_constraints.m_interestBounds))
1158  return; // nothing to do!
1159 
1160  // If Z buffering is enabled, clip xmin/xmaxinc/y to the zbuffer's bounding rectangle
1161  auto const& bounds = ZBufferingIsEnabled() ? m_ZBufBounds : this->m_samplerContext.m_constraints.m_interestBounds;
1162  auto xmin = t4::max<int_coord_type>(get_xmin(bounds), startx);
1163  auto xmaxinc = t4::min<int_coord_type>(t4::subtract<int_coord_type>(get_xmax(bounds), 1), endxinc);
1164 
1165  if ( ZBufferingIsEnabled() )
1166  y = t4::pin<int_coord_type>(get_ymin(m_ZBufBounds), y, get_ymax(m_ZBufBounds));
1167 
1168  if (xmin <= xmaxinc)
1169  {
1170  int_coord_type channelLen = endxinc - startx;
1171  z_signal_type curZ =0;
1172  z_signal_type deltaZ =0;
1173 
1174  tetraphilia_assert(channelLen >= 0);
1175  // calc cur ShadeChannel and deltas at xmin (may be bigger than startx)
1176  size_t s;
1177  switch (channelLen)
1178  {
1179  case 0:
1180  // both ends in same pixel, use avg.
1181  for (s = 0; s < numSignals; s++)
1182  {
1183  m_drawInterpRunCurSigBuf[s] = real_services::Average(startPoint.m_signal[s], endPoint.m_signal[s]);
1184  m_drawInterpRunDeltaSigBuf[s] = 0;
1185  if (ZBufferingIsEnabled())
1186  {
1187  curZ = MakeZSignal(real_services::Average(startPoint.m_z, endPoint.m_z));
1188  deltaZ = 0;
1189  }
1190  }
1191  break;
1192  case 1:
1193  // no intermediate pixels, no div necessary
1194  for (s = 0; s < numSignals; s++)
1195  {
1196  m_drawInterpRunCurSigBuf[s] = (xmin == startx) ? startPoint.m_signal[s] : endPoint.m_signal[s];
1197  m_drawInterpRunDeltaSigBuf[s] = endPoint.m_signal[s]-startPoint.m_signal[s]; // delta should be used at most once.
1198  }
1199  if (ZBufferingIsEnabled())
1200  {
1201  curZ = MakeZSignal((xmin == startx) ? startPoint.m_z : endPoint.m_z);
1202  deltaZ = MakeZDelta(MakeZSignal(startPoint.m_z), MakeZSignal(endPoint.m_z), 1); // delta should be used at most once.
1203  }
1204  break;
1205  default:
1206  // need to work for a living
1207  if (xmin == startx)
1208  {
1209  // common enough to merit faster code
1210  for (s = 0; s < numSignals; s++)
1211  m_drawInterpRunCurSigBuf[s] = startPoint.m_signal[s];
1212  if (ZBufferingIsEnabled())
1213  curZ = MakeZSignal(startPoint.m_z);
1214  }
1215  else
1216  {
1217  // need to offset ShadeChannel from startPoint.m_signal
1218  real_coord_type frac = real_coord_type(xmin - startx) / channelLen;
1219  for (s = 0; s < numSignals; s++)
1220  m_drawInterpRunCurSigBuf[s] = startPoint.m_signal[s] + (endPoint.m_signal[s] - startPoint.m_signal[s]) * frac;
1221  if (ZBufferingIsEnabled())
1222  curZ = MakeZSignal(startPoint.m_z + (endPoint.m_z -startPoint.m_z) * frac);
1223  }
1224 
1225  for (s = 0; s < numSignals; s++)
1226  m_drawInterpRunDeltaSigBuf[s] = (endPoint.m_signal[s]-startPoint.m_signal[s]) / channelLen;
1227  if (ZBufferingIsEnabled())
1228  deltaZ = MakeZDelta(MakeZSignal(startPoint.m_z), MakeZSignal(endPoint.m_z), static_cast<size_t>(channelLen));
1229  break;
1230  }
1231 
1232  // calc channel, z, and signal positions at beginning of shade
1233  signal_type* shadeChannelSignalPos = ShadeChannelFirstSignalPtr(xmin, y);
1234  signal_type* shapeSignalPos = ShapeFirstSignalPtr(xmin, y);
1235  z_signal_type* zSignalPos = ZFirstSignalPtr(xmin, y);
1236 
1237 #ifdef TETRAPHILIA_MESHDEBUG
1238  if (doSkipRightOrBottomMostPixel && xmaxinc > xmin)
1239  --xmaxinc;
1240 #endif
1241 
1242  for (int_coord_type x = xmin; x <= xmaxinc; x++)
1243  {
1244  bool skipSetPixel = false;
1245 #ifdef TETRAPHILIA_MESHDEBUG
1246  if (doMeshOutlineOnly && x != xmin && x != xmaxinc)
1247  skipSetPixel = true;
1248 #endif
1249 
1250  // check z order first if needed
1251  if (zSignalPos)
1252  {
1253  if (*zSignalPos > curZ
1254 #ifdef TETRAPHILIA_MESHDEBUG
1255  && !disableZBuffer
1256 #endif
1257  )
1258  skipSetPixel = true;
1259  else
1260  {
1261  *zSignalPos = curZ;
1262  }
1263  ++zSignalPos;
1264  curZ = z_signal_type(curZ + deltaZ);
1265  }
1266 
1267  if (!skipSetPixel)
1268  {
1269  // fill in the current channel
1270  this->m_channelFunc.GetPixel(shadeChannelSignalPos, ShadeChannelSignalStride(), ranges::make_span(m_drawInterpRunCurSigBuf.data(), t4::convert<ptrdiff_t>(m_drawInterpRunCurSigBuf.size())));
1271 
1272  // fill in the shape
1273  signal_type* tmpSignalPos = shapeSignalPos;
1274  for (s = 0; s < ShapeNumStoredSignals(); s++)
1275  {
1276  *tmpSignalPos = signal_traits::one();
1277  tmpSignalPos = t4::add_bytes(tmpSignalPos, ShapeSignalStride());
1278  }
1279  }
1280 
1281  // advance color
1282  for (s = 0; s < numSignals; s++)
1283  m_drawInterpRunCurSigBuf[s] += m_drawInterpRunDeltaSigBuf[s];
1284 
1285  // advance positions
1286  shadeChannelSignalPos = t4::add_bytes(shadeChannelSignalPos, ShadeChannelPixelStride());
1287  shapeSignalPos = t4::add_bytes(shapeSignalPos, ShapePixelStride());
1288  }
1289  this->m_samplerContext.Yield( 200 * ( xmaxinc-xmin ) );
1290  }
