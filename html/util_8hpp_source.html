106 /*
107  Compare two dereferenceable types, deeply. Returns true if both are null (i.e., convert-to-bool returns
108  false for both), or both are non-null and cmp() returns true when passed references to the result of
109  operator* on both operands. Otherwise returns false.
110 
111  deep_compare will short-circuit and return true if both pointers are pointing to the same object.
112 
113  Useful when comparing (deeply) shared_ptrs, unique_ptrs, optionals, etc.
114 */
115 template<typename Ptr, typename deep_comparator = std::equal_to<decltype(std::declval<Ptr>().operator*())>>
116 bool deep_compare( Ptr const& lhs, Ptr const& rhs, deep_comparator cmp = deep_comparator() )
117 {
118  if ( !lhs )
119  return !rhs;
120  else if ( !rhs )
121  return false;
122  else if ( boost::get_pointer( lhs ) == boost::get_pointer( rhs ) )
123  return true;
124  else
125  return cmp( *lhs, *rhs );
126 }

151 class ptr_holder

160  {}
161 
162  ptr_holder( ptr_holder&& rhs )
163  : m_p( nullptr )
164  {
165  boost::swap( m_p, rhs.m_p );
166  }
167 
168  ~ptr_holder( void )

173

176  boost::swap( m_p, rhs.m_p );
177  return *this;
178  }
179

181  {
182  return m_p != nullptr;
183  }
184

189  }

200  }
