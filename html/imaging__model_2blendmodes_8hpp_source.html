63 /*
64  Helper template for providing a subtractive version of complex blend modes.
65  Given a blend mode suboperation:
66 
67  blend_subexpr( qb, Cb, qji, Cji ) = qb * qji * B( Cb / qb, Cji / qji )
68 
69  this template produces a blend mode operation that produces the subtractive
70  version of that result (with, most likely, a suboptimal implementation).
71  This result is given by the following expression:
72 
73  over( qb, Cji, qb * qji - blend_subexpr( qb, qb - Cb, qji, qji - Cji ) )
74 
75  which is the correct result for a subtractive color space.
76 */
77 template<typename _signal_traits, typename blend_mode_suboperation>
78 struct SubtractiveBlendModeOperationHelper
79 {
80  typedef _signal_traits signal_traits;
81  typedef typename signal_traits::signal_type signal_type;
82 
83  SubtractiveBlendModeOperationHelper( blend_mode_suboperation subop )
84  : m_subop( subop )
85  {}
86 
87  signal_type operator()( signal_type qb, signal_type cb,
88  signal_type qj, signal_type cj )
89  {
90  signal_type subexpr = m_subop( qb, signal_traits::subtract( qb, cb ), qj, signal_traits::subtract( qj, cj ) );
91  tetraphilia_assert( subexpr <= qj && subexpr <= qb );
92  return signal_traits::over(
93  qb,
94  cj,
95  signal_traits::subtract(
96  signal_traits::multiply(
97  qb,
98  qj ),
99  subexpr ) );
100  }
101 
102  blend_mode_suboperation m_subop;
103 };

105 /*
106  Helper template for providing a additive version of complex blend modes.
107  Given a blend mode suboperation:
108 
109  blend_subexpr( qb, Cb, qji, Cji ) = qb * qji * B( Cb / qb, Cji / qji )
110 
111  this template produces a blend mode operation that produces the subtractive
112  version of that result (with, most likely, a suboptimal implementation).
113  This result is given by the following expression:
114 
115  over( qb, Cji, blend_subexpr( qb, Cb, qji, Cji ) )
116 
117  which is the correct result for an additive color space.
118 */
119 template<typename _signal_traits, typename blend_mode_suboperation>
120 struct AdditiveBlendModeOperationHelper
121 {
122  typedef _signal_traits signal_traits;
123  typedef typename signal_traits::signal_type signal_type;
124 
125  AdditiveBlendModeOperationHelper( blend_mode_suboperation subop )
126  : m_subop( subop )
127  {}
128 
129  signal_type operator()( signal_type qb, signal_type cb,
130  signal_type qji, signal_type cji )
131  {
132  signal_type subexpr = m_subop( qb, cb, qji, cji );
133  tetraphilia_assert( subexpr <= qji && subexpr <= qb );
134  return signal_traits::over(
135  qb,
136  cji,
137  subexpr );
138  }
139 
140  blend_mode_suboperation m_subop;

143 template<typename signal_traits>
144 struct MultiplyBlendMode_additive
145 {
146  typedef typename signal_traits::signal_type signal_type;
147 
148 /*
149  Cr = lerp( qb, Cji, qji * B( Cb / qb, Cji / qji ) )
150  = lerp( qb, Cji, qji * ( Cb / qb ) * ( Cji / qji ) ) )
151  = lerp( qb, Cji, ( Cb / qb ) * Cji )
152  = over( qb, Cji, Cb * Cji )
153  = Cji * over( qb, 1, Cb )
154  = Cji * ( 1 - ( qb - Cb ) )
155 */
156  signal_type operator()( signal_type qb, signal_type cb,
157  signal_type /*qj*/, signal_type cj )
158  {
159  return signal_traits::multiply( cj, signal_traits::complement( signal_traits::subtract( qb, cb ) ) );
160  }
161 };

163 template<typename signal_traits>
164 struct MultiplyBlendMode_subtractive
165 {
166  typedef typename signal_traits::signal_type signal_type;
167 
168 /*
169  Cr = lerp( qb, Cji, qji * B( Cb / qb, Cji / qji ) )
170  = lerp( qb, Cji, qji * ( ( Cb / qb ) U ( Cji / qji ) ) ) )
171  = lerp( qb, Cji, qji * ( ( Cb / qb ) + ( Cji / qji ) - ( Cb / qb ) * ( Cji / qji ) ) )
172  = over( qb, Cji, qji * ( Cb + ( qb * Cji ) / qji - ( Cb * Cji ) / qji ) )
173  = over( qb, Cji, qji * Cb + qb * Cji - Cb * Cji )
174  = ( 1 - qb ) * Cji + qji * Cb + qb * Cji - Cb * Cji
175  = Cji - qb * Cji + qji * Cb + qb * Cji - Cb * Cji
176  = Cji + qji * Cb - Cb * Cji
177  = ( 1 - Cb ) * Cji + Cb * qji
178  = lerp( Cb, Cji, qji )
179 */
180  signal_type operator()( signal_type /*qb*/, signal_type cb, signal_type qj, signal_type cj )
181  {
182  return signal_traits::lerp( cb, cj, qj );
183  }

205 template<typename signal_traits>
206 struct OverlaySubexpression
207 {
208  typedef typename signal_traits::signal_type signal_type;
209 
210  /*
211  We only compute the blend mode subexpression:
212 
213  Cr = qb * qji * B( Cb / qb, Cji / qji )
214 
215  And use the AdditiveBlendModeOperationHelper and SubtractiveBlendModeOperationHelper templates.
216 
217  Overlay (in additive space)
218  Cr = qji * qb * B( Cb / qb, Cji / qji ) )
219 
220  if ( Cb / qb ) <= 0.5: (i.e., 2 * Cb < qb )
221  Cr = qb * qji * ( Cji / qji ) * 2 * ( Cb / qb ) )
222  = Cji * 2 * Cb
223  else
224  Cr = qb * qji * ( Cji / qji ) U ( 2 * Cb / qb - 1 )
225  = qb * qji * ( ( Cji / qji ) + ( 2 * Cb / qb - 1 ) - ( Cji / qji ) * ( 2 * Cb / qb - 1 ) )
226  = qb * Cji + qji * ( 2 * Cb - qb ) - Cji * ( 2 * Cb - qb )
227  = qb * qji + - qb * qji + qb * Cji + qji * ( 2 * Cb - qb ) - Cji * ( 2 * Cb - qb )
228  = qb * qji - ( qb - ( 2 * Cb - qb ) ) * ( qji - Cji )
229  = inverted_premultiply( qb, 2 * Cb - qb, qji, Cji )
230  */
231  signal_type operator()( signal_type qb, signal_type cb,
232  signal_type qji, signal_type cji )
233  {
234  typedef typename signal_traits::intermediate_type intermediate_type;
235 
236  intermediate_type cbtimes2 = 2 * cb;
237  if ( cbtimes2 <= qb )
238  return signal_traits::multiply( static_cast<signal_type>( cbtimes2 ), cji );
239  else
240  return signal_traits::inverted_premultiply( qb, static_cast<signal_type>( cbtimes2 - qb ), qji, cji );
241  }
242 };

349 template<typename signal_traits>
350 struct ColorDodgeBlendMode_additive
351 {
352  typedef typename signal_traits::signal_type signal_type;
353 
354  /*
355  Cr = lerp( qb, Cji, qji * B( Cb / qb, Cji / qji ) )
356 
357  if ( Cji != qji )
358  Cr = lerp( qb, Cji, qji * min( 1, ( Cb / qb ) / ( 1 - ( Cji / qji ) ) ) )
359  = over( qb, Cji, qji * min( qb, Cb / ( 1 - ( Cji / qji ) ) ) )
360  = over( qb, Cji, qji * min( qb, qji * Cb / ( qji - Cji ) ) )
361  else
362  Cr = lerp( qb, Cji, qji )
363  */
364  signal_type operator()( signal_type qb, signal_type cb,
365  signal_type qji, signal_type cji )
366  {
367  if ( cji == qji )
368  return signal_traits::lerp( qb, cji, qji );
369  else
370  {
371  signal_type qjicb = signal_traits::multiply( qji, cb );
372  signal_type qjiminuscji = signal_traits::subtract( qji, cji );
373 
374  signal_type pinnedValue;
375  if ( qjicb < qjiminuscji )
376  pinnedValue = t4::min<signal_type>( qb, signal_traits::divide( qjicb, qjiminuscji ) );
377  else
378  pinnedValue = qb;
379 
380  return signal_traits::over(
381  qb,
382  cji,
383  signal_traits::multiply( qji, pinnedValue ) );
384  }
385  }
386 };

388 template<typename signal_traits>
389 struct ColorBurnBlendMode_additive
390 {
391  typedef typename signal_traits::signal_type signal_type;
392 
393  /*
394  Cr = lerp( qb, Cji, qji * B( Cb / qb, Cji / qji ) )
395 
396  if ( Cji == 0 )
397  Cr = 0
398  else
399  Cr = lerp( qb, Cji, qji * ( 1 - min( 1, ( 1 - Cb / qb ) / ( Cji / qji ) ) )
400  = ( 1 - qb ) * Cji + qb * qji * ( 1 - min( 1, ( 1 - Cb / qb ) / ( Cji / qji ) ) )
401  = ( 1 - qb ) * Cji + qji * ( qb - min( qb, ( qb - Cb ) / ( Cji / qji ) ) )
402  = ( 1 - qb ) * Cji + qji * ( qb - min( qb, qji * ( qb - Cb ) / Cji ) )
403  */
404  signal_type operator()( signal_type qb, signal_type cb,
405  signal_type qji, signal_type cji )
406  {
407  if ( cji == signal_traits::zero() )
408  return signal_traits::multiply( signal_traits::complement( qb ), cji );
409 
410  signal_type tmp = signal_traits::multiply( qji, signal_traits::subtract( qb, cb ) );
411  if ( tmp > cji )
412  tmp = qb;
413  else
414  tmp = t4::min<signal_type>( qb, signal_traits::divide( tmp, cji ) );
415  return signal_traits::over( qb, cji, signal_traits::multiply( qji,
416  signal_traits::subtract( qb, tmp ) ) );
417  }

438 template<typename signal_traits>
439 struct SoftLightSubexpression
440 {
441  typedef typename signal_traits::signal_type signal_type;
442 
443  /*
444  We only compute the blend mode subexpression:
445 
446  Cr = qb * qji * B( Cb / qb, Cji / qji )
447 
448  And use the AdditiveBlendModeOperationHelper and SubtractiveBlendModeOperationHelper templates.
449  */
450  signal_type operator()( signal_type qb, signal_type cb,
451  signal_type qji, signal_type cji )
452  {
453  typedef typename signal_traits::intermediate_type intermediate_type;
454 
455  intermediate_type cjitimes2 = 2 * cji;
456  if ( cjitimes2 <= qji )
457  {
458  /*
459  Cr = qb * qji * B( Cb / qb, Cji / qji )
460  = qb * qji * ( Cb / qb - ( 1 - 2 * Cji / qji ) * ( Cb / qb ) * ( 1 - Cb / qb ) )
461  = qji * ( Cb - ( 1 - 2 * Cji / qji ) * Cb * ( 1 - Cb / qb ) )
462  = qji * Cb - ( qji - 2 * Cji ) * Cb * ( 1 - Cb / qb )
463  = Cb * ( qji - ( qji - 2 * Cji ) * ( 1 - Cb / qb ) )
464  */
465  signal_type qjiminus2timescji = signal_traits::subtract( qji, static_cast<signal_type>( cjitimes2 ) );
466  signal_type oneminuscboverqb = signal_traits::complement( signal_traits::divide( cb, qb ) );
467  return signal_traits::multiply(
468  cb,
469  signal_traits::subtract( qji,
470  signal_traits::multiply( qjiminus2timescji, oneminuscboverqb ) ) );
471  }
472  else
473  {
474  signal_type cbdivqb = signal_traits::divide( cb, qb );
475  intermediate_type cbtimes4 = 4 * cb;
476  signal_type Dsubexpr;
477 
478  if ( cbtimes4 <= qb )
479  {
480  /*
481  Cr = qb * ( qji * ( Cb / qb + ( 2 * Cji / qji - 1 ) * ( ( ( 16 * Cb / qb - 12 ) * Cb / qb + 4 ) * Cb / qb - Cb / qb ) ) ) )
482  = qb * ( qji * Cb / qb + ( 2 * Cji - qji ) * ( ( ( 16 * Cb / qb - 12 ) * Cb / qb + 4 ) * Cb / qb - Cb / qb ) ) )
483  = qji * Cb + ( 2 * Cji - qji ) * ( ( ( 16 * Cb / qb - 12 ) * Cb / qb + 4 ) * Cb - Cb )
484  = qji * Cb + ( 2 * Cji - qji ) * Cb * ( ( ( 16 * Cb / qb - 12 ) * Cb / qb + 4 ) - 1 )
485  = qji * Cb + ( 2 * Cji - qji ) * Cb * ( ( ( 16 * Cb / qb - 12 ) * Cb / qb + 4 ) - 1 )
486  = qji * Cb + ( 2 * Cji - qji ) * Cb * ( 4 * ( ( 4 * Cb / qb - 3 ) * Cb / qb + 1 ) - 1 )
487  */
488  Dsubexpr = static_cast<signal_type>( 2 * cbdivqb ); // Dsubexpr = 2 * Cb / qb
489  Dsubexpr = signal_traits::multiply( Dsubexpr, Dsubexpr ); // Dsubexpr = 4 * ( Cb / qb ) * ( Cb / qb )
490  Dsubexpr = static_cast<signal_type>( static_cast<intermediate_type>( Dsubexpr ) - 3 * cbdivqb + signal_traits::one() ); // Dsubexpr = ( 4 * Cb / qb - 3 ) * Cb / qb + 1
491  Dsubexpr = signal_traits::multiply( static_cast<signal_type>( cbtimes4 ), Dsubexpr ); // Dsubexpr = 4 * Cb * ( ( 4 * Cb / qb - 3 ) * Cb / qb + 1 )
492  // = Cb * ( ( 16 * Cb / qb - 12 ) * Cb / qb + 4 )
493  // = qb * D(x)
494  }
495  else
496  {
497  typedef typename signal_traits::app_traits::real_geometry_traits::coord_type real_type;
498 
499  real_type realVal = signal_traits::signal_to_real( cbdivqb );
500  realVal = real_services::Sqrt( realVal );
501  Dsubexpr = signal_traits::real_to_signal( realVal ); // Dsubexpr = sqrt( Cb / qb )
502  Dsubexpr = signal_traits::multiply( qb, Dsubexpr ); // Dsubexpr = qb * D(x)
503  }
504 
505  /*
506  Cr = qb * qji * ( Cb / qb + ( 2 * Cji / qji - 1 ) * ( D( Cb / qb ) - ( Cb / qb ) ) )
507  = qb * qji * ( Cb / qb + ( 2 * Cji / qji - 1 ) * ( Dsubexpr / qb - Cb / qb ) )
508  = qji * ( Cb + ( 2 * Cji / qji - 1 ) * ( Dsubexpr - Cb ) )
509  = qji * Cb + ( 2 * Cji - qji ) * ( Dsubexpr - Cb )
510  */
511  Dsubexpr = signal_traits::subtract( Dsubexpr, cb ); // Dsubexpr = qb * D(x) - Cb
512 
513  signal_type cjitimes2minusqji = static_cast<signal_type>( cjitimes2 - qji );
514 
515  signal_type crmax = signal_traits::multiply( qji, qb );
516  signal_type qjitimescb = signal_traits::multiply( qji, cb );
517 
518  Dsubexpr = signal_traits::multiply( cjitimes2minusqji, Dsubexpr ); // Dsubexpr = ( 2 * Cji - qji ) * ( qb * D( x) - Cb )
519 
520  // The add can overflow in rare cases. We know the subtraction in the test won't overflow
521  // because qb >= cb.
522  if ( Dsubexpr > signal_traits::subtract( crmax, qjitimescb ) )
523  return crmax;
524  else
525  return signal_traits::add( qjitimescb, Dsubexpr );
526  }
527  }
528 };

545 template<typename signal_traits>
546 struct DifferenceSubexpression
547 {
548  typedef typename signal_traits::signal_type signal_type;
549 
550  /*
551  We only compute the blend mode subexpression:
552 
553  Cr = qb * qji * B( Cb / qb, Cji / qji )
554 
555  And use the AdditiveBlendModeOperationHelper and SubtractiveBlendModeOperationHelper templates.
556 
557  Cr = qb * qji * | cb / qb - cji /qji | )
558  = | qji * cb - qb * cji |
559  */
560  signal_type operator()( signal_type qb, signal_type cb,

664 template<typename real_type>
665 RGBColorVal<real_type> ClipColor( const RGBColorVal<real_type>& color )
666 {
667  using namespace t4::literals;
668 
669  RGBColorVal<real_type> result = color;
670  real_type l = Lum( result );
671  real_type n = t4::min<real_type>( result.r, result.g, result.b );
672  real_type x = t4::max<real_type>( result.r, result.g, result.b );
673 
674  if ( n < 0 )
675  {
676  result.r = l + ( ( ( result.r - l ) * l ) / ( l - n ) );
677  result.g = l + ( ( ( result.g - l ) * l ) / ( l - n ) );
678  result.b = l + ( ( ( result.b - l ) * l ) / ( l - n ) );
679  }
680 
681  if ( x > 1 )
682  {
683  result.r = l + ( ( ( result.r - l ) * ( 1 - l ) ) / ( x - l ) );
684  result.g = l + ( ( ( result.g - l ) * ( 1 - l ) ) / ( x - l ) );
685  result.b = l + ( ( ( result.b - l ) * ( 1 - l ) ) / ( x - l ) );
686  }
687 
688  result.r = t4::pin<real_type>( 0_r, result.r, 1_r );
689  result.g = t4::pin<real_type>( 0_r, result.g, 1_r );
690  result.b = t4::pin<real_type>( 0_r, result.b, 1_r );
691 
692  return result;
693 }

750 {

752  real_type* cmin = &result.r;
753  real_type* cmid = &result.g;
754  real_type* cmax = &result.b;
755 
756  // Sort of a hard-coded 3-element bubble sort
757  if ( *cmax < *cmid )
758  boost::swap( cmax, cmid );
759  if ( *cmid < *cmin )
760  {
761  boost::swap( cmid, cmin );
762  if ( *cmax < *cmid )
763  boost::swap( cmax, cmid );
764  }
765 
766  tetraphilia_assert( *cmax >= *cmid && *cmid >= *cmin );
767 
768  if ( *cmax > *cmin )
769  {
770  *cmid = ( ( ( *cmid - *cmin ) * s ) / ( *cmax - *cmin ) );
771  *cmax = s;
772  }
773  else
774  *cmid = *cmax = 0;
775  *cmin = 0;
776 
777  return result;
