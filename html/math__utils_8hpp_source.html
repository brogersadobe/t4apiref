68 template <typename matrix_type>
69 void MatrixUtils<matrix_type>::CalcRadiiOfMappedUnitCircle(const matrix_type& m, real_type* pmajor, real_type* pminor)
70 {
71  // The general solution for the Radii is
72  // _____________________________________________________________________________
73  // / (a^2+b^2) + (c^2+d^2) +/- sqrt( 4*(ac+bd)^2 + ((a^2+b^2)-(c^2+d^2))^2 )
74  // \ / ----------------------------------------------------------------------------
75  // \/ 2
76  //
77  // NOTE: b and c can really be swapped (as long as you do it EVERYWHERE!)
78  // since the transposed matrix generates the same ellipse (though possibly rotated differently)
79  //
80  // Of course, the typical cases when b=c=0 or the orthogonal case when ac+bd=0 are best
81  // handled separately since they don't require nearly as much work to compute.
82 
83  if (get_b(m) == 0 && get_c(m) == 0)
84  {
85  // fast common scale case, may be swapped
86  *pmajor = AbsVal(get_a(m));
87  *pminor = AbsVal(get_d(m));
88  }
89  else
90  {
91  // make the scale near 1 so computations don't blow up...
92  matrix_type scaledm = m;
93  real_type scale;
94  bool wasscaled = ScaleMatrixToAvoid4thPowerOverflow(&scaledm, &scale);
95 
96  // this is a guess at the radii, correct for an orthogonal matrix. We
97  // will fall back to this in the rare case that the compuations below
98  // degenerate.
99  real_type sum_ab_squared = get_a(scaledm)*get_a(scaledm) + get_b(scaledm)*get_b(scaledm);
100  real_type sum_cd_squared = get_c(scaledm)*get_c(scaledm) + get_d(scaledm)*get_d(scaledm);
101 
102  // See if the matrix is orthogonal by seeing if a*c+b*d == 0.
103  real_type ac = get_a(scaledm) * get_c(scaledm);
104  real_type bd = get_b(scaledm) * get_d(scaledm);
105  real_type row_dot = ac + bd;
106 
107  if (AbsVal(row_dot) < real_services::Epsilon(ac)) // want equal to zero, but life isn't always what you want...
108  {
109  // orthogonal matrix
110  // NOTE: this is strictly an optimization since the code below
111  // in the 'else' block should give the same result (up to floating point error)
112  *pmajor = real_services::Sqrt(sum_ab_squared);//real_services::VectorLength(get_a(scaledm), get_b(scaledm));
113  *pminor = real_services::Sqrt(sum_cd_squared);//real_services::VectorLength(get_c(scaledm), get_d(scaledm));
114  }
115  else
116  {
117  // matrix has shear, math is nastier
118  real_type square_sum = sum_ab_squared + sum_cd_squared;
119  real_type square_diff = sum_ab_squared - sum_cd_squared;
120  real_type square_diff_squared = square_diff*square_diff;
121  real_type row_dot_squared = row_dot*row_dot;
122 
123  // OK, we can do the computation. Go ahead and commit it.
124  real_type sqrt_mess = real_services::Sqrt(4 * row_dot_squared + square_diff_squared);
125  *pmajor = real_services::Sqrt( DivideBy2(square_sum + sqrt_mess) );
126 
127  // note that roundoff error can produce a sqrt_mess that is slightly larger than
128  // square_sum, tempting us to call sqrt with a negative value. We never want to do
129  // that
130  *pminor = square_sum > sqrt_mess ? real_services::Sqrt( DivideBy2(square_sum - sqrt_mess) ) : 0;
131  }
132 
133  // rescale the radii if we scaled the matrix above.
134  if (wasscaled)
135  {
136  *pmajor *= scale;
137  *pminor *= scale;
138  }
139  }
140 
141  if (*pminor > *pmajor)
142  boost::swap(*pminor, *pmajor);
143 }
