30 /*
31  Given a model of "data_block_stream device," create a data_block_stream. "device" must be
32  MoveConstructible, and it must support the following calls:
33 
34  data_block get_next_block( Device& d );
35 
36  get_next_block must return the next block in the data_block_stream's sequence. If the
37  data_block_stream is exhausted a zero-length block should be returned.
38 
39  void rewind( Device&& d );
40 
41  If the device is rewindable, reset it to its original state; otherwise throw an exception.
42 
43  bool at_eof( Device&& d );
44 
45  Return true if the device is at the end of its stream. If true, subsequent calls to get_next_block()
46  should return zero-length blocks.
47 
48  Implementers of Devices may presume they will be called in a single-threaded fashion.
49 */
50 template<typename Device>
51 data_block_stream make_data_block_stream( Device&& device );
52 
53 /*
54  Class representing a sequence of data_blocks.
55 
56  Clients call get_next_block on a data_block_stream to fetch a sequence of data_blocks. This
57  model is used for decompression of data in T4.
58 
59  data_block_stream is stateful and single-threaded. As such it is also movable but not copyable,
60  and its methods are all non-const.
61 */
62 class data_block_stream
63 {
64  t4::unique_ptr<dbs_detail::data_block_stream_impl_base> m_impl;
65 
66 public:
67  data_block_stream( void ) = default;
68  data_block_stream( data_block_stream&& rhs ) = default;
69  data_block_stream& operator=( data_block_stream&& rhs ) = default;
70 
71  explicit operator bool( void ) const { return bool( m_impl ); }
72 
73  /*
74  Return the next data_block in the stream. A zero-length block is returned iff
75  the stream is at EOF.
76  */
77  data_block get_next_block( void ) { return m_impl->get_next_block_impl(); }
78 
79  /*
80  Rewind the data_block_stream. An exception is thrown if the underlying device
81  does not support this operation.
82  */
83  void rewind( void ) { return m_impl->rewind_impl(); }
84 
85  /*
86  Returns true if we are at the end of the stream, and subsequent calls to get_next_block
87  will return zero-length blocks.
88  */
89  bool at_eof( void ) { return m_impl->at_eof_impl(); }
90 
91 private:
92  friend std::ostream& operator<<( std::ostream& stm, data_block_stream&& ths );
93 
94  data_block_stream( t4::unique_ptr<dbs_detail::data_block_stream_impl_base> impl ) : m_impl( std::move( impl ) ) {}
95 
96  template<typename Device>
97  friend data_block_stream make_data_block_stream( Device&& device );
98 };
