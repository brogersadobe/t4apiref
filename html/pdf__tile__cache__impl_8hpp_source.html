60 {

62 
63  try
64  {
65  auto pg = get_cached_page_geometry( *m_pdf_tile_cache.get_document(), m_paint_job_params.m_page );
66 // std::cout << "painting page " << m_paint_job_params.m_page << " subgrid " << subgrid_to_paint << std::endl;
67  int_rect paint_bounds = m_tile_cache->tile_grid_to_bounds( subgrid_to_paint, m_paint_job_params.m_draw_params.m_scale );
68 
69  im::GraphicLayoutDescriptor desc( 0, 3, 4, true, im::GraphicLayoutDescriptor::kChunky, false, false,
70  im::GraphicLayoutDescriptor::kShapeIgnored, im::GraphicLayoutDescriptor::kOpacityReplicated, im::GraphicLayoutDescriptor::kColorPerChannel );
71  byte_graphic_store gstore( paint_bounds, desc );
72 
73  auto doc = m_pdf_tile_cache.get_document();
74  auto page_to_device_xform = make<affine_transform>( get_x( m_paint_job_params.m_draw_params.m_scale ),
75  0,
76  0,
77  -get_y( m_paint_job_params.m_draw_params.m_scale ),
78  -get_xmin( pg.m_cropped_media_box ) * get_x( m_paint_job_params.m_draw_params.m_scale ),
79  get_ymax( pg.m_cropped_media_box ) * get_y( m_paint_job_params.m_draw_params.m_scale ) );
80 
81  // @todo gdow: get the real screen resolution somehow
82  static const real screen_scale = real( 4.444 );
83  auto base_page_xform = convert<affine_transform>( invert( pg.m_rotation ) ) * make_scale<affine_transform>( screen_scale, screen_scale );

191 template<typename PDFTCClient>
192 void painter<PDFTCClient>::flush_raster_painter_into_grid( t4::brief_ptr<im::RasterPainter<byte_signal_traits>> source,
193  progress p,
194  const constraints& constraints )
195 {
196  using signal_traits = byte_signal_traits;
197  using signal = signal_traits::signal_type;
198  using segment = im::const_GraphicYWalker<signal_traits>;
199  using machine = im::RasterMachine<signal_traits>;
200  using encoder = im::Encoder<signal_traits>;
201 
202  establish_thread_heap eth;
203  auto& heap = *get_thread_heap();
204 
205  transient_vector<encoder*> encode_procs( heap );
206  encode_procs.reserve( source->GetNumSegments() );
207  std::transform( SegmentsBegin( get_ptr( source ) ), SegmentsEnd( get_ptr( source ) ), begin( encode_procs ), im::MakeEncoder<signal_traits, surface_traits_argb, im::EncoderCopy>() );
208 
209  int_rect update_rect = empty<int_rect>();
210 
211  machine m;
212  SetMachine( &m, source );
213 
214  auto tiling = m_tile_grid.get_tiling();
215 
216  SetY( &m, get_ymin( constraints.m_interestBounds ) );
217  int_rect subgrid = tiling.device_rect_to_subgrid( constraints.m_interestBounds );
218 
219  transient_vector<tile_buffer_walker> surf_walkers( heap );
220  surf_walkers.reserve( static_cast<size_t>( get_width( subgrid ) ) );
221  for ( int i = 0; i < get_width( subgrid ); ++i )
222  surf_walkers.push_back( {} );
223 
224  int_point grid_loc;
225  int_rect tile_device_bounds;
226  set_y( grid_loc, get_ymin( subgrid ) );
227  set_ymin( tile_device_bounds, tiling.get_row_start( get_y( grid_loc ) ) );
228  for ( ; get_y( grid_loc ) < get_ymax( subgrid ); ++get_y( grid_loc ) )
229  {
230  set_ymax( tile_device_bounds, tiling.get_row_start( get_y( grid_loc ) + 1 ) );
231  get_x( grid_loc ) = get_xmin( subgrid );
232  set_xmin( tile_device_bounds, tiling.get_column_start( get_x( grid_loc ) ) );
233  int_coord y = std::max( get_ymin( constraints.m_interestBounds ), get_ymin( tile_device_bounds ) );
234  int_coord y_row_stop = std::min( get_ymax( constraints.m_interestBounds ), get_ymax( tile_device_bounds ) );
235 
236  auto surf_it = begin( surf_walkers );
237  for ( ; get_x( grid_loc ) < get_xmax( subgrid ); ++get_x( grid_loc ), ++surf_it )
238  {
239  set_xmax( tile_device_bounds, tiling.get_column_start( get_x( grid_loc ) + 1 ) );
240  auto const& tile = m_tile_grid.get_tile( grid_loc );
241 
242  int_rect overlap = intersection( tile_device_bounds, constraints.m_interestBounds );
243  auto p2 = tile->get_progress();
244  if ( is_empty( overlap ) )
245  {
246  if ( p2 >= p )
247  *surf_it = { get_xmin( overlap ), get_xmax( overlap ) };
248  else
249  tile->set_progress( p );
250  }
251  else
252  {
253  update_rect = union_( overlap, tile_device_bounds );
254  *surf_it = { overlap };
255  }
256 
257  set_xmin( tile_device_bounds, get_xmax( tile_device_bounds ) );
258  }
259 
260  t3::CostBasedConditionalYielder yielder;
261 
262  for ( ; y < y_row_stop; ++y )
263  {
264  surf_it = begin( surf_walkers );
265  auto x = std::max( get_xmin( constraints.m_interestBounds ), surf_it->get_xstart() );
266  for ( get_x( grid_loc ) = get_xmin( subgrid ); get_x( grid_loc ) < get_xmax( subgrid ); ++get_x( grid_loc ), ++surf_it )
267  {
268  auto x_col_stop = std::min( get_xmax( constraints.m_interestBounds ), surf_it->get_xstop() );
269  if ( surf_it->get_pixels() )
270  {
271  while ( x < x_col_stop )
272  {
273  auto xstop = SetX( get_ptr( source ), x, x_col_stop );
274  tetraphilia_assert( xstop <= x_col_stop );
275  auto seg_id = source->GetCurrentSegmentID();
276  tetraphilia_assert( xstop > x );
277  const segment* seg = CurrentSegment( get_ptr( source ) );
278  const signal* color = seg->m_color->m_scanLine;
279  color = add_bytes( color, seg->m_color->m_pixelBuffer.m_layout->m_pixelStride * static_cast<size_t>( x - get_xmin( *seg->m_color->m_pixelBuffer.m_bounds ) ) );
280  encode_procs[seg_id]->Encode( x, xstop, y, color, surf_it->get_xstart(), surf_it->get_pixels() );
281  x = xstop;
282  }
283 
284  yielder.Yield( 2000 );
285  tetraphilia_assert( x == x_col_stop );
286  surf_it->IncrementY();
287  }
288  else
289  x = x_col_stop;
290  }
291 
292  IncrementY( &m );
293  }
294 
295  surf_it = begin( surf_walkers );
296  get_x( grid_loc ) = get_xmin( subgrid );
297  set_xmin( tile_device_bounds, tiling.get_column_start( get_x( grid_loc ) ) );
298  for ( ; get_x( grid_loc ) < get_xmax( subgrid ); ++get_x( grid_loc ), ++surf_it )
299  {
300  set_xmax( tile_device_bounds, tiling.get_column_start( get_x( grid_loc ) + 1 ) );
301  auto const& tile = m_tile_grid.get_tile( grid_loc );
302 
303  int_rect overlap = intersection( tile_device_bounds, constraints.m_interestBounds );
304  if ( surf_it->get_pixels() )
305  {
306  #ifdef SHOW_TILE_BOUNDARIES
307  add_red_tile_boundary( get_size( tile_device_bounds ), surf_it->get_pixels_base(), surf_it->get_scan_line_stride() );
308  #endif
309  tile->set_bitmap( p, m_pdf_tile_cache.create_bitmap( get_size( overlap ),
310  surf_it->get_pixels_base(), surf_it->get_scan_line_stride() ) );
311  }
312  set_xmin( tile_device_bounds, get_xmax( tile_device_bounds ) );
313  }
314 
315  set_ymin( tile_device_bounds, get_ymax( tile_device_bounds ) );
316  }
317 
318  m_tile_grid.update_progress( p );
319  m_tile_cache->broadcast_tile_update();
320 }

324 {

474 template<typename PDFTCClient>
475 void pdf_tile_cache<PDFTCClient>::work( void )
476 {
477  thread* th = thread_from_id( this_thread::get_id() );
478 
479  do
480  {
481  try
482  {
483  fpu_control fc;
484  auto pr = wait_next_request();
485  if ( pr && pr->claim( th ) )
486  {
487  pdf_detail::painter<PDFTCClient> p( *this, pr, pr->get_paint_job_params() );
488  auto tg = p();
489  pr->complete( th, tg );
490  }
491  else
492  {
493  // wait_next_request() blocks until data shows up or the tile cache is destroyed.
494  assert( m_terminating );
495  return;
496  }
497  }
498  catch( ... )
499  {}
500  }
501  while( !m_terminating );
502 }
