52 template<typename Range>
53 CharacterCollection GetCharacterCollection(const Range& registry, const Range& ordering )
54 {
55  if ( text_detail::eq( registry, "Adobe" ) )
56  {
57  if( text_detail::eq( ordering, "Japan1" ) )
58  return kCCJapanese;
59  else if( text_detail::eq( ordering, "CNS1" ) )
60  return kCCTraditionalChinese;
61  else if( text_detail::eq( ordering, "GB1" ) )
62  return kCCSimplifiedChinese;
63  else if( text_detail::eq( ordering, "Korea1" ) )
64  return kCCKorean;
65  }
66  return kCCUnknown;
67 }
