30 template<typename Result, typename T1, typename T2, typename = void>
31 struct average_impl_helper
32 {
33  /*
34  This is the fully-general version of average that works for all combinations of integer types. It
35  promotes all operands to integral types of the same size as the largest of the inputs, while
36  preserving signedness.
37 
38  After the same-size version of average_impl::operator() returns, the result is converted to the actual
39  requested return type. That conversion may also potentially fail.
40  */
41  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
42  {
43  constexpr auto promotion_size = std::max( { sizeof(T1), sizeof(T2), sizeof(Result) } );
44  auto r = average<boost::optional<extension_type<promotion_size, Result>>>( extend<promotion_size>( x ), extend<promotion_size>( y ) );
45  if ( r )
46  return convert<boost::optional<Result>>( *r );
47  return {};
48  }
49 };

51 /*
52  Specialization of average where all operands are the same size.
53 
54  This is the real meat of the checked average implementation. All calls go through
55  the uniformly-sized implementation.
56 */
57 template<typename Result, typename T1, typename T2>
58 struct average_impl_helper<Result, T1, T2, std::enable_if_t<sizeof(T1) == sizeof(T2) && sizeof(T2) == sizeof(Result)>>
59 {
60  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
61  {
62  static_assert( static_cast<unsigned int>( -1 ) == (~static_cast<unsigned int>( 1 ) + 1u), "This implementation presumes two's complement arithmetic" );
63 
64  using up_t = typename sm_detail::make_unsigned<Result>::type;
65 
66  /*
67  Perform the computation using all unsigned types so as not to generate UB.
68  */
69  auto ux = static_cast<up_t>( x );
70  auto uy = static_cast<up_t>( y );
71  auto ur = static_cast<up_t>( static_cast<up_t>( ux & uy ) + static_cast<up_t>( static_cast<up_t>( x >> 1 ) ^ static_cast<up_t>( y >> 1 ) ) );
72 
73  /*
74  We can partition the representation space up into three segments: negative (from -2^(n-1) to -1),
75  low (from 0 to 2^(n-1)-1), and high (from 2^(n-1) to (2^n)-1. We know which partition a value falls
76  into by looking at the high bit and the signed property of the input types:
77 
78  negative = hi_bit(val) && signed<val>
79  lo = !hi_bit(val)
80  high = hi_bit(x) && unsigned<x>
81 
82  We also know where the average should fall based on the partition of the inputs:
83 
84  x y avg
85 
86  1 neg neg neg
87  2 neg lo neg or lo
88  3 neg hi lo
89  4 lo neg neg or lo
90  5 lo lo lo
91  6 lo hi lo or hi
92  7 hi neg lo
93  8 hi lo lo or hi
94  9 hi hi hi
95 
96  In the cases where correct answers can fall inside one of two partitions (2, 4, 6, 8), we
97  can identify which partition the result fell into simply by looking at the high bit of the
98  result.
99 
100  Note that signed types can only represent values in the neg or lo partition, and unsigned types
101  can only represent values in the lo or hi partition. So the above table determines validity thus:
102 
103  x y signed result unsigned result
104 
105  1 neg neg valid invalid
106  2 neg lo valid !high_bit_set(ur)
107  3 neg hi valid valid
108  4 lo neg valid !high_bit_set(ur)
109  5 lo lo valid valid
110  6 lo hi !high_bit_set(ur) valid
111  7 hi neg valid valid
112  8 hi lo !high_bit_set(ur) valid
113  9 hi hi invalid valid
114 
115  The logic below captures this.
116  */
117  bool valid = true;
118  if ( is_signed<Result>::value )
119  {
120  if ( is_huge( x ) && is_huge( y ) )
121  valid = false;
122  else if ( ( is_huge( x ) && is_positive_low( y ) ) || ( is_positive_low( x ) && is_huge( y ) ) )
123  valid = !high_bit_set( ur );
124  }
125  else
126  {
127  if ( is_negative( x ) && is_negative( y ) )
128  valid = false;
129  else if ( ( is_negative( x ) && is_positive_low( y ) ) || ( is_positive_low( x ) && is_negative( y ) ) )
130  valid = !high_bit_set( ur );
131  }
132 
133  if ( valid )
134  return static_cast<Result>( ur );
135  return {};
136  }
137 };
