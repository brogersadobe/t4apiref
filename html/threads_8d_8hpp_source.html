49 // countdown from this value to below zero. When below zero thread should attempt to yield
50 enum { kInitialCostCounterValue = 1000000 };

52 struct TLSVars
53 {
54  // This object stores the current (cooperative, ContextLock-holding) thread's cost counter.
55  // When it drops below zero it should yield control to another thread.
56  //
57  // While logically thread-specific, g_costCounter has the interesting property that its value
58  // isn't interesting across yields--in other words, it can be shared across cooperative threads.
59  // It is reset to kInitialCostCounterValue whenever a context switch occurs. So there's no need
60  // to use thread_specific_ptr here, or even an atomic.
61  static int g_costCounter;
62 
63  static t4::thread_specific_ptr<ContextLock> tl_currentLock;
64 };

90 class ContextLock
91 {
92  struct lock_with_hooks
93  {
94  t4::cooperative_lock m_lock = { thread_detail::g_t3_cooperative_context };
95  t4::cooperative_lock::yield_hooks m_hooks = m_lock.add_yield_hooks(
96  []() { return pdf::transactions::TransactionManager::GetCurrentTransactionNumber(); },
97  []( size_t prev ) { pdf::transactions::TransactionManager::SetCurrentTransactionNumber( prev ); } );
98  };
99 
100  boost::optional<lock_with_hooks> m_lock_with_hooks;
101  t4::fpu_control m_fpu;
102  t4::establish_thread_heap m_eth;
103 #ifndef T4_SINGLE_THREADED
104  boost::optional<t4::this_thread::disable_interruption> m_disable_interruption;
105 #endif // T4_SINGLE_THREADED
106 public:
107  ContextLock( void )
108  {
109 #ifndef T4_SINGLE_THREADED
110  t4::check_stack_depth();
111  ContextLock* lock = thread_detail::TLSVars::tl_currentLock.get();
112  if ( !lock )
113  {
114  m_lock_with_hooks.emplace( );
115  m_disable_interruption.emplace();
116  thread_detail::TLSVars::tl_currentLock.reset( this );
117  }
118 #endif // T4_SINGLE_THREADED
119  }
120 
121  ~ContextLock( void )
122  {
123 #ifndef T4_SINGLE_THREADED
124  if ( m_lock_with_hooks )
125  thread_detail::TLSVars::tl_currentLock.reset( nullptr );
126 #endif // T4_SINGLE_THREADED
127  }
128 
129  t4::cooperative_lock& GetCooperativeLock( void )
130  {
131  tetraphilia_assert( m_lock_with_hooks );
132  return m_lock_with_hooks->m_lock;
133  }
134 
135  void Yield( void )
136  {
137 #ifndef T4_SINGLE_THREADED
138  t4::this_thread::restore_interruption ri( *m_disable_interruption );
139  m_lock_with_hooks->m_lock.yield();
140 #endif // T4_SINGLE_THREADED
141  }

160 /*
161  Special flavor of ContextLock which temporarily "transfers" an existing ContextLock's
162  properties to another thread. This is intended to support algorithms which are executing
163  on one thread, but briefly need to execute some operation on another thread while holding
164  onto the context lock. The thread's transient_heap is also transferred over (without
165  snapshotting).
166 
167  Currently this is only used by bridge_coroutine_to_jni, so that code executing on a jni-safe
168  thread still has access to all the facilities associated with the calling thread's
169  context_lock.
170 */
171 class ChildThreadContextLock
172 {
173  t4::fpu_control m_fpu;
174  t4::this_thread::disable_interruption m_disable_interruption;
175 
176 public:
177  ChildThreadContextLock( ContextLock& parent_lock, t4::transient_heap* heap )
178  {
179  thread_detail::TLSVars::tl_currentLock.reset( &parent_lock );
180  if ( heap )
181  t4::set_thread_heap( heap );
182  }
183 };
