337 template<typename signal_traits, typename blend_mode>
338 t4::brief_ptr<RasterPainter<signal_traits>> MakeBlendedResult( std::false_type /* nonseparable */,
339  blend_mode blendMode,
340  const Constraints& constraints,
341  const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& dstCS,
342  t4::brief_ptr<const GraphicYWalker<signal_traits>> dstSeg,
343  t4::brief_ptr<RasterPainter<signal_traits>> backdrop,
344  t4::brief_ptr<RasterPainter<signal_traits>> source )
345 {
346  if ( dstCS->IsSubtractive() )
347  {
348  if ( dstCS->GetNumSignals() < 4 )
349  throw t4::corrupt_data();
350 
351  return MakeOperationRasterPainter( constraints, MakeNonSeparableCMYKBlendOperation<signal_traits>( blendMode ), std::move( dstSeg ), std::move( backdrop ), std::move( source ) );
352  }
353  else if ( dstCS->GetNumSignals() == 1 )
354  {
355  // device gray
356  return MakeOperationRasterPainter( constraints, MakeNonSeparableGrayBlendOperation<signal_traits>( blendMode ), std::move( dstSeg ), std::move( backdrop ), std::move( source ) );
357  }
358  else
359  {
360  if ( dstCS->GetNumSignals() < 3 )
361  throw t4::corrupt_data();
362 
363  return MakeOperationRasterPainter( constraints, MakeNonSeparableRGBBlendOperation<signal_traits>( blendMode ), std::move( dstSeg ), std::move( backdrop ), std::move( source ) );
364  }
365 }
