67 template<typename BezierPath>
68 std::enable_if_t<is_bezier_path<BezierPath>::value,
69 boost::optional<rectangle_type_t<BezierPath>>> get_rectangular_path( BezierPath&& path )
70 {
71  /*
72  This algorithm chops the bezier path up into three subsequences:
73 
74  1) All moveto/closepaths at the front of the path
75  2) A contiguous sequence of 4 linetos, or 3 linetos + a closepath
76  3) All movetos/closepaths in the remainder of the path
77 
78  If the sequence of path points can be divided up into three subsequences matching the above. We use
79  the iterator-based std::find to identify these three subsequences, then we use range-v3 to analyze the
80  individual subranges and attempt to build an array of four corner points whose geometry we then
81  test for rectangularity.
82  */
83 
84  using point = point_type_t<BezierPath>;
85 
86  // Returns true if a bezier_path_point is a lineto or a curveto
87  auto is_segment = []( auto const& p ){ return p.type() == path_op::lineto || p.type() == path_op::curveto; };
88 
89  // As we scan the path we will write our "corner" candidates here.
90  using mutable_point = std::remove_const_t<point>;
91  std::array<mutable_point, 4> corners;
92 
93  // First subsequence
94  auto ci = begin( corners );
95 
96  auto vw = ranges::views::common( ranges::views::all( std::forward<BezierPath>( path ) ) );
97  auto start = vw.begin();
98  auto stop = vw.end();
99 
100  auto it = std::find_if( start, stop, [&]( auto const& seg )
101  {
102  auto moveto = [&]( point const& p )
103  {
104  corners[0] = p;
105  if ( ci == begin( corners ) )
106  ++ci;
107  return false;
108  };
109 
110  auto closepath = [&]() { return false; };
111 
112  auto others = []( auto const&... pts ) { return true; };
113 
114  return seg.visit( moveto, others, others, closepath );
115  } );
116 
117  if ( ci == begin( corners ) )
118  return {};
119 
120  // Second subsequence
121  bool success = true;
122  it = std::find_if( it, stop, [&]( auto const& seg )
123  {
124  auto lineto = [&]( point const& pt )
125  {
126  if ( ci == end( corners ) )
127  {
128  success = ( pt == corners[0] );
129  return true;
130  }
131 
132  *ci++ = pt;
133  return false;
134  };
135 
136  auto others = [&]( auto const&... )
137  {
138  success = false;
139  return true;
140  };
141 
142  auto closepath = [&]()
143  {
144  success = ci == end( corners );
145  return true;
146  };
147 
148  return seg.visit( others, lineto, others, closepath );
149  } );
150 
151  // If we didn't find four more linetos (or 3 + closepath), fail
152  if ( ci != end( corners ) || !success )
153  return {};
154 
155  // If we stopped on a closepath or a lineto, advance 1
156  if ( it != stop )
157  ++it;
158 
159  // Third subsequence: make sure none of the remaining points are segments
160  it = std::find_if( it, stop, is_segment );
161  if ( it != stop )
162  return {};
163 
164  // We may have a rectangle!
165  // Check for each point only changing in X or Y but not both
166  // this also catches X like patterns
167 
168  auto is_same_x_or_y = [&]( size_t l, size_t r ){ return get_x( corners[l] ) == get_x( corners[r] ) || get_y( corners[l] ) == get_y( corners[r] ); };
169  if (is_same_x_or_y(0, 1))
170  if (is_same_x_or_y(1, 2))
171  if (is_same_x_or_y(2, 3))
172  if (is_same_x_or_y(3, 0))
173  // we now have either a rectangle or an "L" check for points 2 and 4 being different and pts 1 and 3
174  if (!is_same_x_or_y(1, 3))
175  if (!is_same_x_or_y(0, 2))
176  {
177  // compute path bounds
178  using rectangle = typename rectangle_type<std::remove_const_t<point>>::type;
179  return make<rectangle>( std::min( get_x( corners[0] ), get_x( corners[2] ) ),
180  std::min( get_y( corners[0] ), get_y( corners[2] ) ),
181  std::max( get_x( corners[0] ), get_x( corners[2] ) ),
182  std::max( get_y( corners[0] ), get_y( corners[2] ) ) );
183  }
184 
185  return {};
186 }

249 template<typename BezierPath, typename F>
250 std::enable_if_t<is_bezier_path<BezierPath>::value,
251 bool> clean( BezierPath&& path, F&& f )
252 {
253  using point = point_type_t<BezierPath>;
254 
255  struct visitor
256  {
257  F&& m_f;
258  boost::optional<basic_moveto<point>> m_deferred_moveto;
259  bool m_active_path = false;
260 
261  visitor( F&& f ) : m_f( std::forward<F>( f ) ) {}
262 
263  bool operator()( basic_moveto<point> const& m )
264  {
265  m_deferred_moveto = m;
266  m_active_path = true;
267  return true;
268  }
269 
270  bool operator()( basic_lineto<point> const& l )
271  {
272  if ( !m_active_path )
273  return false;
274 
275  if ( m_deferred_moveto )
276  {
277  std::forward<F>( m_f )( *m_deferred_moveto );
278  m_deferred_moveto.reset();
279  }
280 
281  std::forward<F>( m_f )( l );
282  return true;
283  }
284 
285  bool operator()( basic_curveto<point> const& c )
286  {
287  if ( !m_active_path )
288  return false;
289 
290  if ( m_deferred_moveto )
291  {
292  std::forward<F>( m_f )( *m_deferred_moveto );
293  m_deferred_moveto.reset();
294  }
295 
296  std::forward<F>( m_f )( c );
297  return true;
298  }
299 
300  bool operator()( basic_closepath const& h )
301  {
302  if ( m_active_path && !m_deferred_moveto )
303  std::forward<F>( m_f )( h );
304  m_active_path = false;
305  return true;
306  }
307  };
308 
309  visitor v( std::forward<F>( f ) );
310  for ( auto it = path.begin(); it != path.end(); ++it )
311  {
312  if ( !(*it).visit( v ) )
313  return false;
314  }
315 
316  return true;
317 }
