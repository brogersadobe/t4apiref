89 /*
90  Convert a value to a specified type. If the correct result can't be expressed
91  in the specified type, call the provided fail functor (defaults to throw corrupt_data().
92 */
93 template<typename To, typename From>
94 constexpr std::enable_if_t<!is_instantiation_of<To, boost::optional>::value,
95 To> convert( From&& x )
96 {
97  auto result = convert<boost::optional<To>>( std::forward<From>( x ) );
98  if ( !result )
99  convert_detail::signal_conversion_failure<To, From>();
