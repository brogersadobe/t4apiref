82 template<typename Event, typename... Events>
83 constexpr size_t index_of_best_match( void )
84 {
85  using namespace boost::mp11;
86  using boost::mp11::_1;
87 
88  static_assert( event_detail::exactly_one_match_exists<Event, Events...>, "" );
89  if ( mp_contains<mp_list<Events...>, Event>() )
90  return mp_find<mp_list<Events...>, Event>();
91 
92 
93  constexpr auto result = mp_find_if<mp_list<Events...>, mp_bind<is_convertible_t, Event, _1>::template fn>();
94  static_assert( result != mp_size<mp_list<Events...>>(), "no match found in Event list" );
95  return result;
96 }

98 /*
99  Because events use overloading on parameter type for dispatch, you can't use C++'s built-in
100  vtables to implement this sort of runtime polymorphism. The ctor takes an Impl (as a hana
101  basic_type) and produces a table of functions that essentially "expand" to this:
102 
103  void (*)( event_stream_impl_base<Events...> const*, first_event_type );
104  void (*)( event_stream_impl_base<Events...> const*, second_event_type );
105 
106  etc., etc.
107 
108  In their implementation, the entries in this vtable cast their argument to an event_stream_impl<Impl, Events...>,
109  extract the contained Impl object, and dispatch the event to that object.
110 */
111 template<typename... Events>
112 struct event_vtable
113 {
114  /*
115  This produces a hana::tuple of function pointers, each with a signature differing in type
116  in the second parameter (the event type).
117  */
118  boost::hana::tuple<void (*)( event_stream_impl_base<Events...> const& impl, Events const& )...> m_vtable;
119 
120  template<typename Impl>
121  event_vtable( boost::hana::basic_type<Impl> )
122  : m_vtable( boost::hana::make_tuple( &dispatch_event<Impl, Events, Events...>... ) )
123  {}
124 
125  template<typename Event>
126  auto get_notify_proc( void ) const
127  {
128  return m_vtable[ std::integral_constant<size_t, index_of_best_match<Event, Events...>()>() ];
129  }
130 };
