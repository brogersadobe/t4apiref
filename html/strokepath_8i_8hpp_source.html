78 template <typename stroker_traits>
79 template<typename Rng>
80 std::pair<StrokePathResult, t4::transient_bezier_path> Stroker<stroker_traits>::GenerateStroke( Rng&& bez_segments )
81 {
82  auto& heap = *t4::get_thread_heap();
83  t4::transient_bezier_path result(heap);
84 
85  DelayedPath<stroker_traits> delayed_result( result, m_params.GetScaleFactor() );
86  if (m_params.GetActuallyDoThinStroke() && m_params.GetDashArrayLen() == 0)
87  return std::make_pair( kThinStrokeOriginalPath, result ); // do a quick thin stroke
88 
89  contour_type* pcontour = t4::make_transient<contour_type>( heap, m_params);
90  contour_stroker_base_type* pstroker;
91  if (m_params.GetActuallyDoThinStroke())
92  pstroker = t4::make_transient<contour_stroker_thin_type>( heap );
93  else
94  pstroker = t4::make_transient<contour_stroker_thick_type>( heap, m_params );
95 
96  if (m_params.GetDashArrayLen() > 0 && m_params.GetDashArray() != nullptr)
97  pstroker = t4::make_transient<contour_stroker_dashed_type>( heap, m_params, pstroker );
98 
99  // clear the existing contour
100  pcontour->ClearSegs();
101 
102  bool hadAMoveTo = false;
103  point_type openPt;
104  point_type curPt;
105 
106  auto vw = ranges::views::all( std::forward<Rng>( bez_segments ) );
107  for ( auto seg = vw.begin(); seg != vw.end(); ++seg )
108  {
109  auto moveto = [&]( t4::real_point const& p0 )
110  {
111  if (pcontour->GetNumSegs() > 0)
112  {
113  pstroker->GenerateStroke(*pcontour, delayed_result);
114  pcontour->ClearSegs();
115  }
116  hadAMoveTo = true;
117  curPt = convertPoint(p0);
118  openPt = curPt;
119  return true;
120  };
121 
122  auto lineto = [&]( t4::real_point const& p0 )
123  {
124  if ( !hadAMoveTo )
125  return false;
126 
127  point_type p = convertPoint(p0);
128  pcontour->AppendUserLineSegment(curPt, p);
129  curPt = p;
130  return true;
131  };
132 
133  auto curveto = [&]( t4::real_point const& pp1, t4::real_point const& pp2, t4::real_point const& pp3 )
134  {
135  if ( !hadAMoveTo )
136  return false;
137 
138  auto p1 = convertPoint(pp1);
139  auto p2 = convertPoint(pp2);
140  auto p3 = convertPoint(pp3);
141 
142  if (m_params.GetUserSpaceFlatness() != 0)
143  pcontour->AppendUserFlattenedBezierSegment(curPt, p1, p2, p3);
144  else
145  pcontour->AppendUserBezierSegment(curPt, p1, p2, p3);
146  curPt = p3;
147 
148  return true;
149  };
150 
151  auto closepath = [&]( void )
152  {
153  // if not yet started, the PS behavior is to just ignore the closepath. Bizarre.
154  if (hadAMoveTo)
155  {
156  pcontour->Close(curPt, openPt);
157  pstroker->GenerateStroke(*pcontour, delayed_result);
158  pcontour->ClearSegs();
159  curPt = openPt;
160  }
161 
162  return true;
163  };
164 
165  if ( !(*seg).visit( moveto, lineto, curveto, closepath ) )
166  {
167  // Watson bug#1297644
168  // permissive builds allow invalid paths to be constructed. If we detect one here we act as if the path were empty
169  #ifdef TETRAPHILIA_PDF_PERMISSIVE
170  break;
171  #else
172  throw t4::corrupt_data();
173  #endif
174  }
175  }
176 
177  pstroker->GenerateStroke(*pcontour, delayed_result);
178  delayed_result.Flush();
179 
180  if (m_params.GetActuallyDoThinStroke())
181  return std::make_pair( kThinStrokeResultPath, std::move( result ) );
182  else
183  return std::make_pair( kFillResultPath, std::move( result ) );
184 }
