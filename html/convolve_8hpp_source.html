149 inline void FCD_Clamp(int32x4_t vM128[], unsigned char* output)
150 {
151  _mm_unpack_epi32( vM128[0], vM128[2], vM128[0], vM128[2] );
152  int32x4_t acSumM128 = _mm_add_epi32( vM128[0], vM128[2] );
153  _mm_unpack_epi32( vM128[1], vM128[3], vM128[1], vM128[3] );

156  acSumM128 = _mm_add_epi32( acSumM128, bdSumM128 );
157  acSumM128 = _mm_add_epi32( acSumM128, fcd_half_M128 );
158  acSumM128 = _mm_srai_epi32(acSumM128, 22);
159  acSumM128 = _mm_max_epi32(acSumM128, kZeroS );

161  output[0] = static_cast<unsigned char>(_mm_extract_epi32(acSumM128, 0));
162  output[1] = static_cast<unsigned char>(_mm_extract_epi32(acSumM128, 1));
163  output[2] = static_cast<unsigned char>(_mm_extract_epi32(acSumM128, 2));

390  {

405  tetraphilia_assert( -1<=temp[j] && temp[j]<=1 );

407  s += rowStride;
408  }
409 
410  // perform one col of convolution
