174  /*
175  Returns the size in pixels of a tile at the specified scale. This is real-valued because
176  we need tile edges at different scales to be coincident. The actual tiling will snap edges to
177  the nearest integral coordinate, so even unclipped tile sizes within a single tier may vary in
178  dimension by one pixel.
179  */
180  static real_point compute_scaled_tile_size( int_coord minimum_tile_size, real_point const& scale, int tier )
181  {
182  auto uts = minimum_tile_size / static_cast<real_coord>( std::pow( 2, tier ) );
183  return make<real_point>( get_x( scale ) * uts, get_y( scale ) * uts );
184  }
185 
186  static real_point compute_scaled_tile_size( int_coord minimum_tile_size, real_point const& scale )
187  {
188  int tier = scale_to_tier( scale );
189  auto uts = minimum_tile_size / static_cast<real_coord>( std::pow( 2, tier ) );
190  return make<real_point>( get_x( scale ) * uts, get_y( scale ) * uts );
191  }
