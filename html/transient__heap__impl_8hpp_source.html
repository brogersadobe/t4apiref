33 /*
34  An instance of "canary" monitors unused chunks of a transient_block and makes sure that the
35  bytes in those ranges are not changed, which would indicate a buffer overrun. On creation
36  a canary initializes its byte range to transient_heap_guardrail_signature (0x55). On destruction
37  it first verifies that those bytes are unchanged, and then writes deallocated_memory_signature
38  (0xaa) into them.
39 */
40 class transient_heap::canary
41 {
42  void* m_mem_start = nullptr;
43  void* m_mem_stop = nullptr;
44 
45 public:
46  canary( void ) = default;
47 
48  canary( void* mem_start, void* mem_stop )
49  : m_mem_start( mem_start ),
50  m_mem_stop( mem_stop )
51  {
52  // On construction, fill the range with the guardrail signature
53  memory_detail::set_mem_range( mem_start, mem_stop, memory_detail::transient_heap_guardrail_signature );
54  }
55 
56  // Move ctor.
57  canary( canary&& rhs )
58  {
59  boost::swap( m_mem_start, rhs.m_mem_start );
60  boost::swap( m_mem_stop, rhs.m_mem_stop );
61  }
62 
63  ~canary( void )
64  {
65  if ( m_mem_start )
66  {
67  // Verify that the range still contains the guardrail signature
68  check();
69 
70  // Now fill it with the deallocated memory signature
71  memory_detail::set_mem_range( m_mem_start, m_mem_stop, memory_detail::deallocated_memory_signature );
72  }
73  }
74 
75  // Move assign
76  canary& operator=( canary&& rhs )
77  {
78  boost::swap( m_mem_start, rhs.m_mem_start );
79  boost::swap( m_mem_stop, rhs.m_mem_stop );
80  return *this;
81  }
82 
83  // Assert that the memory range still contains the guardrail signature
84  void check( void ) const
85  {
86  // If this fires then we're looking at a range that most likely was deallocated already.
87  T4_MEMORY_ASSERT( !memory_detail::check_mem_range( m_mem_start, m_mem_stop, memory_detail::deallocated_memory_signature ) );
88 
89  // If this fires then there is a memory smasher somewhere.
90  T4_MEMORY_ASSERT( memory_detail::check_mem_range( m_mem_start, m_mem_stop, memory_detail::transient_heap_guardrail_signature ) );
91  }
92 };

160 /*
161  transient_ptr_holder is used by the various flavors of make_transient to "reset" a
162  transient_heap if an exception is thrown during the object's construction. "Resetting"
163  really just amounts to rewinding the pointer to the top of free block memory to its
164  location before the call.
165 
166  The conditions under which "releasing" is a correct thing to do are a little subtle:
167 
168  1) If the heap is being shared across multiple threads, then we can't release. Due
169  to other thread activity we can't reliably reset without locking.
170 
171  2) If, during the construction of the object, other allocations are made from the same
172  heap, we can't reset because there's no way of knowing for sure that those follow-on
173  allocations will need to persist after this allocation. Most of the time they won't,
174  but it's not something we can count on. This is the reason for the update() method--
175  we record the state of the heap after space for the object has been allocated but
176  before the object is constructed; if it is unchanged when we hit the dtor during
177  unwinding, then we know we can safely reset the heap.
178 */
179 #if !T4_HARDENED_TRANSIENT_HEAP
180 class transient_ptr_holder
181 {
182  typedef transient_heap::block_iterator block_iterator;
183 
184  transient_heap* m_heap;
185  block_iterator m_old_current_block;
186  size_t m_old_current_block_top;
187  block_iterator m_new_current_block;
188  size_t m_new_current_block_top;
189 
190 #if T4_TRANSIENT_HEAP_BUFFER_GUARDS
191  transient_heap::canary_iterator m_old_canary_end;
192 #endif
193 
194 public:
195  transient_ptr_holder( transient_heap& heap )
196  : m_heap( heap.m_share_count ? nullptr : &heap )
197  {
198  if ( m_heap )
199  {
200  m_old_current_block = heap.m_current_block;
201  m_old_current_block_top = heap.m_current_block_top;
202  #if T4_TRANSIENT_HEAP_BUFFER_GUARDS
203  m_old_canary_end = heap.m_canaries.end();
204  #endif
205  }
206  }
207 
208  ~transient_ptr_holder( void )
209  {
210  if ( m_heap && m_heap->m_current_block == m_new_current_block && m_heap->m_current_block_top == m_new_current_block_top )
211  {
212  m_heap->m_current_block = m_old_current_block;
213  m_heap->m_current_block_top = m_old_current_block_top;
214  #if T4_TRANSIENT_HEAP_BUFFER_GUARDS
215  m_heap->m_canaries.erase( m_old_canary_end, m_heap->m_canaries.end() );
216  #endif
217  }
218  }
219 
220  void update( void )
221  {
222  if ( m_heap )
223  {
224  m_new_current_block = m_heap->m_current_block;
225  m_new_current_block_top = m_heap->m_current_block_top;
226  }
227  }
228 
229  void commit( void )
230  {
231  m_heap = nullptr;
232  }
233 };
234 #endif /* T4_HARDENED_TRANSIENT_HEAP */

238 /*
239  Implementation of various flavors of make_transient. The amount of work necessary depends
240  heavily on whether or not the type being constructed has a trivial dtor or not.
241 */
242 template<typename T, typename... TArgs>
243 std::enable_if_t<!std::is_trivially_destructible<T>::value,
244 T*> make_transient( transient_heap& heap, TArgs&&... args )
245 {
246  using namespace t4::literals;
247 
248 #if !T4_HARDENED_TRANSIENT_HEAP
249  using transient_heap_detail::dtor_record;
250  using transient_heap_detail::transient_ptr_holder;
251 
252  static_assert( std::is_trivially_destructible<dtor_record>::value, "" );
253 
254  transient_ptr_holder ph( heap );
255 
256  dtor_record* drec = static_cast<dtor_record*>( heap.malloc( sizeof( dtor_record ), alignof( dtor_record ) ) );
257  void* p = heap.malloc( sizeof( T ), alignof( T ) );
258  ph.update();
259 
260  new (p) T( std::forward<TArgs>( args )... );
261  heap.commit_dtor( drec, static_cast<T*>( p ), 1 );
262  ph.commit();
263  return static_cast<T*>( p );
264 #else
265  using namespace t4::literals;
266  auto p = boost::allocate_unique<T>( t4::allocator<T>(), std::forward<TArgs>( args )... );
267  heap.commit_destruction( std::addressof( *p ), 1_z );
268  return std::addressof( *p.release() );
269 #endif /* T4_HARDENED_TRANSIENT_HEAP */
270 }
