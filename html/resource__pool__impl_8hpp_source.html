47 template<typename Condition, typename Granularity>
48 bool resource_cache_manager::purge_until( Condition&& condition, Granularity&& granularity )
49 {
50  if ( condition() )
51  return true;
52 
53  return boost::with_lock_guard( m_purge_mutex, [&]
54  {
55  cache_time_point now = current_cache_time();
56  auto ose = on_scope_exit( [now]()
57  {
58  cache_detail::report_purge( current_cache_time() - now );
59  } );
60 
61  size_t now_tick = cache_detail::current_cache_tick();
62 
63  size_t purge_count = 0;
64  bool succeeded = false;
65  while( true )
66  {
67  /*
68  Update every entry's weight (by calling compute_size on all resource entries). The
69  lighter an object is, the more likely it is to be purged.
70  */
71  deletion_queue dq;
72  size_t this_purge_count = 0;
73 
74  boost::with_lock_guard( m_map_mutex, [&]()
75  {
76  m_entry_vector.resize( 0 );
77  for ( auto it = m_entries.begin(); it != m_entries.end(); ++it )
78  {
79  if ( it->cache_entry_is_unused() )
80  {
81  it->update_weight( now_tick );
82  m_entry_vector.push_back( it );
83  }
84  }
85 
86  std::sort( m_entry_vector.begin(), m_entry_vector.end(), []( entry_set::iterator const& lhs, entry_set::iterator const& rhs )
87  {
88  return lhs->m_weight < rhs->m_weight;
89  } );
90 
91  size_t trash_goal = granularity();
92  size_t trash_size = 0;
93 
94  for ( auto it = begin( m_entry_vector ); it != end( m_entry_vector ) && trash_size < trash_goal; ++it )
95  {
96  if ( ( *it )->attempt_eject( dq ) )
97  {
98  trash_size += ( *it )->get_size();
99  ++this_purge_count;
100  }
101  }
102  } );
103 
104  m_entry_vector.clear();
105  purge_count += this_purge_count;
106 
107  if ( condition() )
108  {
109  succeeded = true;
110  break;
111  }
112 
113  if ( this_purge_count == 0 )
114  break;
115  }
116 
117  cache_detail::report_element_purge( purge_count );
118  if ( !succeeded )
119  cache_detail::report_failed_purge( purge_count == 0 );
120 
121  m_amount_in_use_after_last_purge = amount_in_use();
122  return succeeded;
123  } );
124 }
