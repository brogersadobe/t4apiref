385 template<typename app_traits>
386 void ObjectGraph<app_traits>::processObject(const obj_type& object, stack_type& stack)
387 {
388  t4::check_stack_depth();
389 
390  //Processing Object
391  ObjectType type = object.Type();
392  if (type == kObjectTypeDictionary)
393  {
394  const dict_type& dict = object.DictionaryValue_NoTypeCheck();
395  typename dict_type::const_iterator_noresolve_type iter = dict.begin_noresolve();
396  while (iter != dict.end_noresolve())
397  {
398  const obj_type& subObj = iter->GetValue();
399  processObject(subObj, stack);
400  ++iter;
401  }
402  }
403  else if (type == kObjectTypeArray)
404  {
405  const array_type& array = object.ArrayValue_NoTypeCheck();
406  typename array_type::const_iterator_noresolve_type iter = array.begin_noresolve();
407  while (iter != array.end_noresolve())
408  {
409  const obj_type& subObj = *iter;
410  processObject(subObj, stack);
411  ++iter;
412  }
413  }
414  else if (type == kObjectTypeReference)
415  {
416  const ref_type& refObj = object.ReferenceValue_NoTypeCheck();
417 
418  // If the reference points to an object which doesn't exist, bail out without doing anything. Example - when the reference's
419  // generation number is not equal to the existing valid object's generation number
420  const obj_type& referredObj = m_store->ResolveReference(refObj);
421  if (referredObj.Type() == kObjectTypeNull)
422  return;
423 
424  //Make sure that a node gets created for the object referred by this reference object
425  node_type& refObjNode = m_object_map[refObj];
426 
427  optional_ref_type idoRefOpt = object.GetContainerIDOReference();
428  tetraphilia_assert(idoRefOpt);
429  if (idoRefOpt)
430  {
431  ref_type& idoRef = *idoRefOpt;
432  node_type& node = m_object_map[idoRef]; //This will create the node if it doesn't exist in the map.
433  node.addOutgoingEdge(refObj);
434 
435  bool strongRef = true;
436  if (referredObj.Type() == kObjectTypeDictionary)
437  {
438  const dict_type& dict = referredObj.DictionaryValue_NoTypeCheck();
439  optional_name_type optName = dict.GetName("Type");
440  if (optName)
441  {
442  size_t len = 0;
443  const char* name = optName->get(len);
444  if ((len == 4) && (strncmp(name, "Page", 4) == 0))
445  {
446  strongRef = false;
447  const obj_type& referringObj = m_store->ResolveReference(idoRef);
448  if (referringObj.Type() == kObjectTypeDictionary)
449  {
450  const dict_type& dict2 = referringObj.DictionaryValue_NoTypeCheck();
451  optional_name_type optTypeName = dict2.GetName("Type");
452  if (optTypeName)
453  {
454  len = 0;
455  name = optTypeName->get(len);
456  if ((len == 5) && (strncmp(name, "Pages", 5)) == 0)
457  {
458  strongRef = true;
459  }
460  }
461  }
462  }
463  }
464  }
465 // TODO_READER_MOBILE<vmarda>: Theoretically, it's possible that the array containing references to Page objects is an Indirect Object. Though we haven't yet found any such PDF. Consider adding that logic too just to be safe.
466 
467  if (strongRef)
468  refObjNode.addIncomingEdgeStrong(idoRef);
469  else
470  refObjNode.addIncomingEdgeWeak(idoRef);
471 
472  if (!refObjNode.processed()) //This is just a filter, the check which guarantees single processing of all nodes is in 'populate' function of ObjectGraph.
473  stack.push(refObj);
474  }
475  else
476  {
477  tetraphilia_assert(false);
478  throw t4::corrupt_data();
479  }
480  }
481 }
