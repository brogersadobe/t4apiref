186 template<typename SignalTraits, typename Allocator>
187 void PixelStore<SignalTraits, Allocator>::Init( const t4::int_rect& bounds,
188  const PixelBufferLayoutDescriptor& descriptor,
189  bool zeroInit,
190  Allocator allocator )
191 {
192  m_bounds = bounds;
193  t4::int_coord tileWidth = descriptor.m_tileX ? 1 : get_width( bounds );
194  t4::int_coord tileHeight = descriptor.m_tileY ? 1 : get_height( bounds );
195 
196  if ( tileWidth < 0 || tileHeight < 0 )
197  throw std::invalid_argument( "" );
198 
199  size_t numStoredSignals = descriptor.m_replicateSignals ? 1 : descriptor.m_pixelSize;
200  size_t sz = t4::multiply<size_t>( sizeof( signal_type ), static_cast<size_t>( tileWidth ) );
201  if ( !descriptor.m_planar )
202  sz = t4::multiply<size_t>( sz, numStoredSignals );
203  size_t scanLineSize = RoundUp( sz );
204  size_t allocSize = t4::multiply<size_t>( scanLineSize, static_cast<size_t>( tileHeight ) );
205  m_layout.m_scanLineStride = descriptor.m_tileY ? 0 : scanLineSize;
206  m_layout.m_padSignalsAreUnused = true;
207 
208  if ( descriptor.m_planar )
209  {
210  m_layout.m_signalStride = descriptor.m_replicateSignals ? 0 : static_cast<ptrdiff_t>( allocSize );
211  allocSize = t4::multiply<size_t>( allocSize, numStoredSignals );
212  }
213  else
214  m_layout.m_signalStride = descriptor.m_replicateSignals ? 0 : sizeof( signal_type );
215 
216  if ( descriptor.m_littleEndian )
217  {
218  m_layout.m_firstSignalOffset = ( numStoredSignals - 1 - descriptor.m_padSignals ) * static_cast<size_t>( m_layout.m_signalStride );
219  m_layout.m_signalStride = -m_layout.m_signalStride;
220  }
221  else
222  m_layout.m_firstSignalOffset = descriptor.m_padSignals;
223 
224  m_layout.m_pixelStride = descriptor.m_tileX ? 0 : ( sizeof( signal_type ) * ( descriptor.m_planar ? 1 : numStoredSignals ) );
225  m_layout.m_numSignals = descriptor.m_numSignals;
226  m_storage.SetSize( allocSize, std::move( allocator ) );
227 
228  if ( zeroInit )
229  memset( m_storage.GetBuffer(), 0, allocSize );
230 }
