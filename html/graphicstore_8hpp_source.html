134  const GraphicLayoutDescriptor& descriptor,
135  bool zeroInit,
136  Allocator allocator )
137 {

139

151  break;
152  }
153 
154  if ( zeroInit )
155  m_pixelStorage.ZeroInit();

158 template<typename SignalTraits, typename Allocator>
159 void GraphicStore<SignalTraits, Allocator>::SetUpChunkyBuffers( const GraphicLayoutDescriptor& descriptor, Allocator allocator )
160 {
161  tetraphilia_assert( descriptor.m_interleaving == GraphicLayoutDescriptor::kChunky );
162  tetraphilia_assert( descriptor.m_pixelSize >= ( descriptor.m_padComponents
163  + GraphicLayoutDescriptor::StorageSize( descriptor.m_shapeRep, descriptor.m_numComponents )
164  + GraphicLayoutDescriptor::StorageSize( descriptor.m_opacityRep, descriptor.m_numComponents )
165  + GraphicLayoutDescriptor::StorageSize( descriptor.m_colorRep, descriptor.m_numComponents ) ) * sizeof( signal_type ) );
166 
167  ptrdiff_t signalStride = static_cast<ptrdiff_t>( sizeof( signal_type ) ) * ( descriptor.m_littleEndian ? -1 : 1 );
168  size_t firstSignalOffset = ( descriptor.m_littleEndian ? descriptor.m_pixelSize - ( descriptor.m_padComponents + 1 ) : descriptor.m_padComponents ) * sizeof( signal_type );
169  size_t pixelStride = descriptor.m_tileH ? static_cast<size_t>(0) : descriptor.m_pixelSize;
170  size_t scanLineSize = RoundUp( t4::multiply<size_t>( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ), descriptor.m_pixelSize * sizeof( signal_type ) ) );
171  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
172 
173  m_pixelStorage.SetSize( t4::multiply<size_t>( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ), scanLineSize ), std::move( allocator ) );
174 
175  int numChannelsInUse = 3;
176 
177  switch( descriptor.m_shapeRep )
178  {
179  case GraphicLayoutDescriptor::kShapeIgnored:
180  m_buffers.m_shape = nullptr;
181  m_writableBuffers.m_shape = nullptr;
182  --numChannelsInUse;
183  break;
184  case GraphicLayoutDescriptor::kShapeImplicitlyOne:
185  m_buffers.m_shape = &m_onesBuff;
186  m_writableBuffers.m_shape = nullptr;
187  --numChannelsInUse;
188  break;
189  case GraphicLayoutDescriptor::kShapeReplicated:
190  m_layouts.m_shape = PixelBufferLayout( PixelBufferLayout::kReplicatedSignal, firstSignalOffset, 0, pixelStride, scanLineStride, false );
191  m_bufferStorage.m_shape = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_shape );
192  m_writableBuffers.m_shape = &m_bufferStorage.m_shape;
193  m_buffers.m_shape = m_writableBuffers.m_shape;
194  firstSignalOffset += static_cast<size_t>( signalStride );
195  break;
196  default:
197  tetraphilia_assert( descriptor.m_shapeRep == GraphicLayoutDescriptor::kShapePerChannel );
198  m_layouts.m_shape = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, false );
199  m_bufferStorage.m_shape = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_shape );
200  m_writableBuffers.m_shape = &m_bufferStorage.m_shape;
201  m_buffers.m_shape = m_writableBuffers.m_shape;
202  firstSignalOffset += descriptor.m_numComponents * static_cast<size_t>( signalStride );
203  break;
204  }
205 
206  switch( descriptor.m_opacityRep )
207  {
208  case GraphicLayoutDescriptor::kOpacityIgnored:
209  m_buffers.m_opacity = nullptr;
210  m_writableBuffers.m_opacity = nullptr;
211  --numChannelsInUse;
212  break;
213  case GraphicLayoutDescriptor::kOpacityImplicitlyOne:
214  m_buffers.m_opacity = &m_onesBuff;
215  m_writableBuffers.m_opacity = nullptr;
216  --numChannelsInUse;
217  break;
218  case GraphicLayoutDescriptor::kOpacitySameAsShape:
219  m_buffers.m_opacity = m_buffers.m_shape;
220  m_writableBuffers.m_opacity = m_writableBuffers.m_shape;
221  --numChannelsInUse;
222  break;
223  case GraphicLayoutDescriptor::kOpacityReplicated:
224  m_layouts.m_opacity = PixelBufferLayout( PixelBufferLayout::kReplicatedSignal, firstSignalOffset, 0, pixelStride, scanLineStride, false );
225  m_bufferStorage.m_opacity = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_opacity );
226  m_writableBuffers.m_opacity = &m_bufferStorage.m_opacity;
227  m_buffers.m_opacity = m_writableBuffers.m_opacity;
228  firstSignalOffset += static_cast<size_t>( signalStride );
229  break;
230  default:
231  tetraphilia_assert( descriptor.m_opacityRep == GraphicLayoutDescriptor::kOpacityPerChannel );
232  m_layouts.m_opacity = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, false );
233  m_bufferStorage.m_opacity = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_opacity );
234  m_writableBuffers.m_opacity = &m_bufferStorage.m_opacity;
235  m_buffers.m_opacity = m_writableBuffers.m_opacity;
236  firstSignalOffset += descriptor.m_numComponents * static_cast<size_t>( signalStride );
237  break;
238  }
239 
240  switch( descriptor.m_colorRep )
241  {
242  case GraphicLayoutDescriptor::kColorIgnored:
243  m_buffers.m_color = nullptr;
244  m_writableBuffers.m_color = nullptr;
245  --numChannelsInUse;
246  break;
247  default:
248  tetraphilia_assert( descriptor.m_colorRep == GraphicLayoutDescriptor::kColorPerChannel );
249  m_layouts.m_color = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, false );
250  m_bufferStorage.m_color = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_color );
251  m_writableBuffers.m_color = &m_bufferStorage.m_color;
252  m_buffers.m_color = m_writableBuffers.m_color;
253  break;
254  }
255 
256  if ( numChannelsInUse == 1 )
257  {
258  m_layouts.m_shape.m_padSignalsAreUnused = true;
259  m_layouts.m_opacity.m_padSignalsAreUnused = true;
260  m_layouts.m_color.m_padSignalsAreUnused = true;
261  }

264 template<typename SignalTraits, typename Allocator>
265 void GraphicStore<SignalTraits, Allocator>::SetUpChannelPlanes( const GraphicLayoutDescriptor& descriptor, Allocator allocator )
266 {
267  tetraphilia_assert( descriptor.m_interleaving == GraphicLayoutDescriptor::kChannelPlanes );
268 
269  size_t totalSize = 0;
270  size_t opacityStart;
271  size_t colorStart;
272 
273  switch( descriptor.m_shapeRep )
274  {
275  case GraphicLayoutDescriptor::kShapeIgnored:
276  case GraphicLayoutDescriptor::kShapeImplicitlyOne:
277  break;
278  case GraphicLayoutDescriptor::kShapeReplicated:
279  {
280  size_t pixelStride = descriptor.m_tileH ? 0 : sizeof( signal_type );
281  size_t scanLineSize = RoundUp( t4::multiply<size_t>( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ), sizeof( signal_type ) ) );
282  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
283  m_layouts.m_shape = PixelBufferLayout( PixelBufferLayout::kReplicatedSignal, 0, 0, pixelStride, scanLineStride, true );
284  totalSize += ( ( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ) ) * scanLineSize );
285  break;
286  }
287  default:
288  {
289  tetraphilia_assert( descriptor.m_shapeRep == GraphicLayoutDescriptor::kShapePerChannel );
290  size_t pixelStride = descriptor.m_tileH ? 0 : descriptor.m_pixelSize;
291  size_t scanLineSize = RoundUp( t4::multiply<size_t>( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ), descriptor.m_pixelSize ) );
292  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
293  size_t firstSignalOffset;
294  ptrdiff_t signalStride;
295 
296  if ( descriptor.m_littleEndian )
297  {
298  firstSignalOffset = descriptor.m_pixelSize - ( descriptor.m_padComponents + 1 ) * sizeof( signal_type );
299  signalStride = -static_cast<ptrdiff_t>( sizeof( signal_type ) );
300  }
301  else
302  {
303  firstSignalOffset = descriptor.m_padComponents * sizeof( signal_type );
304  signalStride = sizeof( signal_type );
305  }
306 
307  m_layouts.m_shape = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, true );
308  totalSize += ( ( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ) ) * scanLineSize );
309  break;
310  }
311  }
312 
313  opacityStart = totalSize;
314  switch( descriptor.m_opacityRep )
315  {
316  case GraphicLayoutDescriptor::kOpacityIgnored:
317  case GraphicLayoutDescriptor::kOpacityImplicitlyOne:
318  case GraphicLayoutDescriptor::kOpacitySameAsShape:
319  break;
320  case GraphicLayoutDescriptor::kOpacityReplicated:
321  {
322  size_t pixelStride = descriptor.m_tileH ? 0 : sizeof( signal_type );
323  size_t scanLineSize = RoundUp( t4::multiply<size_t>( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ), sizeof( signal_type ) ) );
324  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
325  m_layouts.m_opacity = PixelBufferLayout( PixelBufferLayout::kReplicatedSignal, 0, 0, pixelStride, scanLineStride, true );
326  totalSize += ( ( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ) ) * scanLineSize );
327  break;
328  }
329  default:
330  {
331  tetraphilia_assert( descriptor.m_opacityRep == GraphicLayoutDescriptor::kOpacityPerChannel );
332  size_t pixelStride = descriptor.m_tileH ? 0 : descriptor.m_pixelSize;
333  size_t scanLineSize = RoundUp( t4::multiply<size_t>( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ), descriptor.m_pixelSize ) );
334  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
335  size_t firstSignalOffset;
336  ptrdiff_t signalStride;
337 
338  if ( descriptor.m_littleEndian )
339  {
340  firstSignalOffset = descriptor.m_pixelSize - ( descriptor.m_padComponents + 1 ) * sizeof( signal_type );
341  signalStride = -static_cast<ptrdiff_t>( sizeof( signal_type ) );
342  }
343  else
344  {
345  firstSignalOffset = descriptor.m_padComponents * sizeof( signal_type );
346  signalStride = sizeof( signal_type );
347  }
348 
349  m_layouts.m_opacity = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, true );
350  totalSize += ( ( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ) ) * scanLineSize );
351  break;
352  }
353  }
354 
355  colorStart = totalSize;
356  if ( descriptor.m_colorRep == GraphicLayoutDescriptor::kColorPerChannel )
357  {
358  size_t pixelSize = descriptor.m_pixelSize * sizeof( signal_type );
359  size_t pixelStride = descriptor.m_tileH ? 0 : pixelSize;
360  size_t scanLineSize = RoundUp( t4::multiply<size_t>( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ), pixelSize ) );
361  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
362  size_t firstSignalOffset;
363  ptrdiff_t signalStride;
364 
365  if ( descriptor.m_littleEndian )
366  {
367  firstSignalOffset = descriptor.m_pixelSize - ( descriptor.m_padComponents + 1 ) * sizeof( signal_type );
368  signalStride = -static_cast<ptrdiff_t>( sizeof( signal_type ) );
369  }
370  else
371  {
372  firstSignalOffset = descriptor.m_padComponents * sizeof( signal_type );
373  signalStride = sizeof( signal_type );
374  }
375  m_layouts.m_color = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, true );
376  totalSize += ( ( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ) ) * scanLineSize );
377  }
378 
379  m_pixelStorage.SetSize( totalSize, std::move( allocator ) );
380 
381  switch( descriptor.m_shapeRep )
382  {
383  case GraphicLayoutDescriptor::kShapeIgnored:
384  m_buffers.m_shape = nullptr;
385  m_writableBuffers.m_shape = nullptr;
386  break;
387  case GraphicLayoutDescriptor::kShapeImplicitlyOne:
388  m_buffers.m_shape = &m_onesBuff;
389  m_writableBuffers.m_shape = nullptr;
390  break;
391  case GraphicLayoutDescriptor::kShapeReplicated:
392  m_bufferStorage.m_shape = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_shape );
393  m_writableBuffers.m_shape = &m_bufferStorage.m_shape;
394  m_buffers.m_shape = m_writableBuffers.m_shape;
395  break;
396  default:
397  tetraphilia_assert( descriptor.m_shapeRep == GraphicLayoutDescriptor::kShapePerChannel );
398  m_bufferStorage.m_shape = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_shape );
399  m_writableBuffers.m_shape = &m_bufferStorage.m_shape;
400  m_buffers.m_shape = m_writableBuffers.m_shape;
401  break;
402  }
403 
404  switch( descriptor.m_opacityRep )
405  {
406  case GraphicLayoutDescriptor::kOpacityIgnored:
407  m_buffers.m_opacity = nullptr;
408  m_writableBuffers.m_opacity = nullptr;
409  break;
410  case GraphicLayoutDescriptor::kOpacityImplicitlyOne:
411  m_buffers.m_opacity = &m_onesBuff;
412  m_writableBuffers.m_opacity = nullptr;
413  break;
414  case GraphicLayoutDescriptor::kOpacitySameAsShape:
415  m_buffers.m_opacity = m_buffers.m_shape;
416  m_writableBuffers.m_opacity = m_writableBuffers.m_shape;
417  break;
418  case GraphicLayoutDescriptor::kOpacityReplicated:
419  m_bufferStorage.m_opacity = pixel_buffer( t4::add_bytes( m_pixelStorage.GetBuffer(), opacityStart ), &m_bounds, &m_layouts.m_opacity );
420  m_writableBuffers.m_opacity = &m_bufferStorage.m_opacity;
421  m_buffers.m_opacity = m_writableBuffers.m_opacity;
422  break;
423  default:
424  tetraphilia_assert( descriptor.m_opacityRep == GraphicLayoutDescriptor::kOpacityPerChannel );
425  m_bufferStorage.m_opacity = pixel_buffer( t4::add_bytes( m_pixelStorage.GetBuffer(), opacityStart ), &m_bounds, &m_layouts.m_opacity );
426  m_writableBuffers.m_opacity = &m_bufferStorage.m_opacity;
427  m_buffers.m_opacity = m_writableBuffers.m_opacity;
428  break;
429  }
430 
431  switch( descriptor.m_colorRep )
432  {
433  case GraphicLayoutDescriptor::kColorIgnored:
434  m_buffers.m_color = nullptr;
435  m_writableBuffers.m_color = nullptr;
436  break;
437  default:
438  tetraphilia_assert( descriptor.m_colorRep == GraphicLayoutDescriptor::kColorPerChannel );
439  m_bufferStorage.m_color = pixel_buffer( t4::add_bytes( m_pixelStorage.GetBuffer(), colorStart ), &m_bounds, &m_layouts.m_color );
440  m_writableBuffers.m_color = &m_bufferStorage.m_color;
441  m_buffers.m_color = m_writableBuffers.m_color;
442  break;
443  }
444 }

446 template<typename SignalTraits, typename Allocator>
447 void GraphicStore<SignalTraits, Allocator>::SetUpSignalPlanes( const GraphicLayoutDescriptor& descriptor, Allocator allocator )
448 {
449  using namespace t4::literals;
450 
451  tetraphilia_assert( descriptor.m_interleaving == GraphicLayoutDescriptor::kSignalPlanes );
452  tetraphilia_assert( descriptor.m_pixelSize >= sizeof( signal_type ) );
453 
454  size_t pixelSize = sizeof( signal_type );
455  size_t pixelStride = descriptor.m_tileH ? 0_z : descriptor.m_pixelSize;
456  size_t scanLineSize = RoundUp( t4::multiply<size_t>( ( descriptor.m_tileH ? 1u : static_cast<size_t>( get_width( m_bounds ) ) ), pixelSize ) );
457  size_t scanLineStride = descriptor.m_tileV ? 0 : scanLineSize;
458  size_t planeSize = t4::multiply<size_t>( descriptor.m_tileV ? 1u : static_cast<size_t>( get_height( m_bounds ) ), scanLineSize );
459  size_t numPlanes = descriptor.ShapeStorageSize() + descriptor.OpacityStorageSize() + descriptor.ColorStorageSize();
460  ptrdiff_t signalStride = t4::multiply<ptrdiff_t>( planeSize, descriptor.m_littleEndian ? -1 : 1 );
461  size_t firstSignalOffset = t4::multiply<size_t>( ( descriptor.m_littleEndian ? descriptor.m_pixelSize - ( descriptor.m_padComponents + 1 ) : descriptor.m_padComponents ), planeSize );
462  m_pixelStorage.SetSize( t4::multiply<size_t>( numPlanes, planeSize ), std::move( allocator ) );
463 
464  int numChannelsInUse = 3;
465 
466  switch( descriptor.m_shapeRep )
467  {
468  case GraphicLayoutDescriptor::kShapeIgnored:
469  m_buffers.m_shape = nullptr;
470  m_writableBuffers.m_shape = nullptr;
471  --numChannelsInUse;
472  break;
473  case GraphicLayoutDescriptor::kShapeImplicitlyOne:
474  m_buffers.m_shape = &m_onesBuff;
475  m_writableBuffers.m_shape = nullptr;
476  --numChannelsInUse;
477  break;
478  case GraphicLayoutDescriptor::kShapeReplicated:
479  m_layouts.m_shape = PixelBufferLayout( PixelBufferLayout::kReplicatedSignal, firstSignalOffset, 0, pixelStride, scanLineStride, true );
480  m_bufferStorage.m_shape = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_shape );
481  m_writableBuffers.m_shape = &m_bufferStorage.m_shape;
482  m_buffers.m_shape = m_writableBuffers.m_shape;
483  firstSignalOffset += static_cast<size_t>( signalStride );
484  break;
485  default:
486  tetraphilia_assert( descriptor.m_shapeRep == GraphicLayoutDescriptor::kShapePerChannel );
487  m_layouts.m_shape = PixelBufferLayout( descriptor.m_numComponents, firstSignalOffset, signalStride, pixelStride, scanLineStride, false );
488  m_bufferStorage.m_shape = pixel_buffer( m_pixelStorage.GetBuffer(), &m_bounds, &m_layouts.m_shape );
489  m_writableBuffers.m_shape = &m_bufferStorage.m_shape;
490  m_buffers.m_shape = m_writableBuffers.m_shape;
491  firstSignalOffset += descriptor.m_numComponents * static_cast<size_t>( signalStride );
492  break;
493  }
494 
495  switch( descriptor.m_opacityRep )
496  {
497  case GraphicLayoutDescriptor::kOpacityIgnored:
498  m_buffers.m_opacity = nullptr;
499  m_writableBuffers.m_opacity = nullptr;
500  --numChannelsInUse;
