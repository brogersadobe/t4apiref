37 // int divide normally truncates result toward zero. template plus power of two specializations to floor the result
38 template<int constVal>
39 class FloorDivideByConst
40 {
41 public:
42  template <typename int_type>
43  static inline int_type DoDivide(int_type n)
44  {
45  return n >= 0 ? n/constVal : (n+1)/constVal-1;
46  }
47 };

380 template<typename sampler_def, int kNumScanlines, typename edgeadder_type,typename tolerance_traits>
381 template<typename Rng>
382 void SuperSamplingBezierRasterPainter<sampler_def,kNumScanlines,edgeadder_type,tolerance_traits>::InitializePoints(
383  const typename edgeadder_type::edge_adder_params& adderParams,
384  Rng&& bez_segments)
385 {
386  static_assert(kZero == 0 && kOne == 1 && kPixels == 2, "");
387 
388  m_segments[0] = segment_type( m_zeroWalker, {}, {} );
389  m_segments[1] = segment_type( m_oneWalker, {}, {} );
390  m_segments[2] = segment_type( m_pixelWalker, {}, {} );
391 
392  bool boundsIsEmpty = RectIsEmpty(m_constraints.m_interestBounds) || bez_segments.begin() == bez_segments.end();
393  if (!boundsIsEmpty)
394  {
395  edgeadder_type adder(adderParams, m_sampler, kNumScanlines, m_constraints);
396 
397  // adds edges and computes userspace path bounds
398  // only one call to AddEdges permitted
399  // TODO: support curves w/o flattening
400  auto vw = ranges::views::common( bez_segments );
401  auto curPoint = vw.begin();
402  adder.AddEdges(curPoint,vw.end());
403 
404  boundsIsEmpty = adder.UpdateConstraints(m_constraints,m_samplerBounds);
405  }
406 
407  if (boundsIsEmpty) // this path will not be drawn
408  {
409  MakeEmptyRect(m_constraints.m_interestBounds);
410  return;
411  }
412 
413  // vector only - this must be an absolute worst case computation
414  size_t initialTransitionVectorLength = sampler_def::kTransitionListInitialMaxSize;
415 
416  // create transition data
417  {
418  for (size_t idx = 0; idx < kNumScanlines; idx++)
419  m_transData.push_back( t4::make_brief<sampler_transition_data>( initialTransitionVectorLength ) );
420  }
421 
422  m_sampler.FinishInitialize();
423 }

441 template<typename sampler_def, int kNumScanlines, typename edgeadder_type,typename tolerance_traits>
442 typename SuperSamplingBezierRasterPainter<sampler_def,kNumScanlines,edgeadder_type,tolerance_traits>::int_type
443 SuperSamplingBezierRasterPainter<sampler_def,kNumScanlines,edgeadder_type,tolerance_traits>::SetXImpl( int_type xstart, int_type xstop )
444 {
445 
446  m_xstart = xstart;
447  if (xstart < m_xstop)
448  return t4::min<int_type>( xstop, m_xstop );
449 
450  tetraphilia_assert(m_insideSamplersNonzeroYBounds);
451 
452  if (xstart < get_xmin(m_constraints.m_interestBounds))
453  {
454  this->SetCurrentSegmentID( kZero );
455  m_xstop = get_xmin(m_constraints.m_interestBounds);
456  return t4::min<int_type>( xstop, m_xstop );
457  }
458 
459  if (xstart >= get_xmax(m_constraints.m_interestBounds))
460  {
461  this->SetCurrentSegmentID( kZero );
462  m_xstop = limits<int_type>::max();
463  return t4::min<int_type>( xstop, m_xstop );
464  }
465 
466  if (!m_hasValidTransData)
467  BuildTransitionData();
468 
469  if (!m_transNextIsReady || (xstart+1) * kNumScanlines >= m_transAAXStop)
470  BuildNextTransData(xstart);
471 
472  if (m_transAACount == 0)
473  {
474  this->SetCurrentSegmentID( kZero );
475  // floor division
476  m_xstop = FloorDivideByConst<kNumScanlines>::DoDivide(m_transAAXStop);
477  m_transNextIsReady = false;
478  return t4::min<int_type>( xstop, m_xstop );
479  }
480  else if (m_transAACount == kNumScanlines*kNumScanlines)
481  {
482  this->SetCurrentSegmentID( kOne );
483  // floor division
484  m_xstop = FloorDivideByConst<kNumScanlines>::DoDivide(m_transAAXStop);
485  m_transNextIsReady = false;
486  return t4::min<int_type>( xstop, m_xstop );
487  }
488  else
489  {
490  // AA segment
491  this->SetCurrentSegmentID( kPixels );
492 
493  m_xstop = m_xstart;
494  unsigned char *pixels = m_pixelStore.GetPixelBuffer().GetPixels() + ( m_xstop - get_xmin(m_pixelStore.GetBounds()) );
495  // unsigned char *ppp = m_pixelWalker.GetWritablePixels();
496  // tetraphilia_assert(pixels == ppp);
497 #ifdef BEZSAMPLER_DEBUG
498  unsigned char *begpixels = pixels;
499 #endif
500  while ( m_xstop < xstop )
501  {
502  // optimize for powers of two
503  if (kNumScanlines == 2)
504  *pixels++ = static_cast<unsigned char>(m_transAACount*64);
505  else if (kNumScanlines == 4)
506  *pixels++ = static_cast<unsigned char>(m_transAACount*16);
507  else if (kNumScanlines == 8)
508  *pixels++ = static_cast<unsigned char>(m_transAACount*4);
509  else if (kNumScanlines == 16)
510  *pixels++ = static_cast<unsigned char>(m_transAACount);
511  else if (kNumScanlines == 32)
512  *pixels++ = static_cast<unsigned char>(m_transAACount/4);
513  else if (kNumScanlines == 64)
514  *pixels++ = static_cast<unsigned char>(m_transAACount/16);
515  else if (kNumScanlines == 128)
516  *pixels++ = static_cast<unsigned char>(m_transAACount/64);
517  else
518  {
519  // not a power of two
520  unsigned int rslt = 255 * m_transAACount / (kNumScanlines*kNumScanlines);
521  *pixels++ = static_cast<unsigned char>( rslt );
522  }
523  ++m_xstop;
524  if (m_xstop == get_xmax(m_constraints.m_interestBounds))
525  {
526  m_transNextIsReady = false;
527  break;
528  }
529  BuildNextTransData(m_xstop);
530  if (m_transAACount == 0 || m_transAACount == kNumScanlines*kNumScanlines)
531  break;
532  }
533 
534 #ifdef BEZSAMPLER_DEBUG
535  if (detail::debugging)
536  {
537  int_type begxstart = m_xstart;
538  cache_logger(_T("SEG y %d type %d xon %d xoff %d"),m_y,this->GetCurrentSegmentID(),begxstart,m_xstop);
539  while (begxstart++ < m_xstop)
540  cache_logger(TEXT(" %x"),*begpixels++);
541  cache_logger(TEXT("\n"));
542  }
543 #endif
544 
545  return t4::min<int_type>( xstop, m_xstop );
546  }
547 }

549 template<typename sampler_def, int kNumScanlines, typename edgeadder_type,typename tolerance_traits>
550 void SuperSamplingBezierRasterPainter<sampler_def,kNumScanlines,edgeadder_type,tolerance_traits>::BuildNextTransData(int_type xstart)
551 {
552  const int_type aaxstart = kNumScanlines * xstart + m_sampler.getDeviceMatrixXOffset(); // beginning of this pixel
553  const int_type aaxnext = aaxstart + kNumScanlines; // beginning of the next pixel
554 
555  // calculate these in temporaries
556  int_type transAAXStop = t4::add<int_type>(t4::multiply<int_type>(kNumScanlines, get_xmax(m_constraints.m_interestBounds)), m_sampler.getDeviceMatrixXOffset());
557  int_type transAACount = 0;
558 
559  for (size_t idx = 0; idx < kNumScanlines; idx++)
560  {
561  sampler_transition_data_ptr pData = m_transData[idx];
562  if (pData->m_empty)
563  continue;
564 
565  // advance to the beginning of the pixel
566  while (aaxstart >= *pData->m_nextTrans)
567  {
568  // toggle the segment type
569  pData->m_currSeg ^= kOne;
570  ++pData->m_nextTrans;
571  }
572 
573  // count how many subpixels this pixel and advance to the end
574  int_type aapixelpartstart = aaxstart;
575  int_type nextTrans = *pData->m_nextTrans;
576  for (;;)
577  {
578  if (pData->m_currSeg)
579  {
580  tetraphilia_assert(pData->m_currSeg == kOne);
581  int_type aa = t4::min<int_type>(nextTrans,aaxnext)-aapixelpartstart;
582  tetraphilia_assert(aa >= 0 && aa <= kNumScanlines);
583  transAACount += aa;
584  }
585  // all done if next transition is in another pixel
586  if (nextTrans >= aaxnext)
587  break;
588 
589  // this part of the pixel begins here
590  aapixelpartstart = *pData->m_nextTrans;
591 
592  // advance the segment
593  pData->m_currSeg ^= kOne;
594  ++pData->m_nextTrans;
595  nextTrans = *pData->m_nextTrans;
596  }
597  if (nextTrans < transAAXStop)
598  transAAXStop = nextTrans;
599  }
600
