216 template<typename tree_traits>
217 class tree_pos
218 {
219 private:
220 
221  typedef std::pair<t4::pdf::array,t4::optional<size_t>> node_data;
222 
223  // Holds positional state for Kids arrays as the tree is traversed. Each item is a pdf array and an index into that
224  // array. During traversal as an intermediate node is encountered it is pushed onto the stack, so higher-indexed
225  // items are deeper in the tree. For flat name trees this will always be empty.
226  t4::vector<node_data> m_kids;
227 
228  // Holds positional state for the current Names/Nums array as the tree is traversed. Has a value if and only if
229  // the traversal has entered the Names/Nums array. The size_t index value always is that of a key and so will always
230  // be even. (This implies that code doing the actual Names/Nums array iteration actually does so in jumps of 2.)
231  t4::optional<node_data> m_names;
232 
233 // State management APIs (invoked by the traversal code).
234 public:
235 
236  void push_kids( t4::pdf::array& kids )
237  {
238  m_kids.push_back( node_data( { kids, t4::none } ) );
239  }
240 
241  void pop_kids()
242  {
243  m_kids.pop_back();
244  }
245 
246  void set_kids_index( size_t array_index )
247  {
248  node_data& node_data = m_kids.back();
249  node_data.second.emplace(array_index);
250  }
251 
252  void set_names(const t4::pdf::array& names)
253  {
254  m_names = node_data( { names, t4::none } );
255  }
256 
257  void clear_names()
258  {
259  m_names = boost::none;
260  }
261 
262  void set_names_index( size_t array_index )
263  {
264  m_names->second.emplace(array_index);
265  }
266 
267 
268 // State access APIs (invoked by clients of the traversal code).
269 public:
270 
271  t4::pdf::array get_names_array() const
272  {
273  if (!m_names)
274  throw std::invalid_argument("");
275  return m_names->first;
276  }
277 
278  size_t get_names_index() const
279  {
280  if (!m_names || !m_names->second)
281  throw std::invalid_argument("");
282  return *m_names->second;
283  }
284 
285  typedef typename tree_traits::tree_key_type tree_key_type;
286 
291  const tree_key_type get_key() const
292  {
293  size_t array_index = get_names_index();
294  return tree_traits::to_key( m_names->first[array_index] );
295  }
296 
302  const t4::pdf::object get_value( t4::pdf::autoresolve ar ) const
303  {
304  size_t array_index = get_names_index();
305  if ( ar )
306  return m_names->first.Get(array_index+1);
307  else
308  return m_names->first.Get_NoResolve(array_index+1);
309  }
310 };

320 template<typename tree_traits, typename kids_func, typename names_func>
321 class tree_node_walker
322 {
323 public:
324 
325  typedef tree_pos<tree_traits> tree_pos;
326 
334  static void for_each(const t4::pdf::dictionary& tree, kids_func kids_proc, names_func names_proc)
335  {
336  tree_pos pos;
337  for_each(tree, kids_proc, names_proc, pos);
338  }
339 
340 private:
341 
342  static void for_each(const t4::pdf::dictionary& node, kids_func kids_proc, names_func names_proc, tree_pos& pos)
343  {
344  t4::check_stack_depth();
345 
346  boost::optional<t4::pdf::array> kids = node.GetArray( "Kids" );
347  if ( kids )
348  {
349  if ( node.Get( tree_traits::dict_key ).Type() != store::ObjectType::kObjectTypeNull )
350  throw t4::corrupt_data();
351  pos.push_kids( *kids );
352  size_t length = kids->Length();
353  for ( size_t array_index = 0; array_index < length; ++array_index )
354  {
355  t4::pdf::dictionary kid = kids->Get(array_index).DictionaryValue();
356  if ( kids_proc( kid ) )
357  {
358  pos.set_kids_index( array_index );
359  for_each(kid, kids_proc, names_proc, pos);
360  }
361  }
362  pos.pop_kids();
363  }
364  else
365  {
366  t4::pdf::array names = node.GetRequiredArray( tree_traits::dict_key );
367  if (names.Length() % 2 != 0)
368  throw t4::corrupt_data();
369 
370  names_proc( pos, names );
371  }
372  }
373 };

392 template<typename search_traits>
393 std::pair<size_t,bool> TreeFind(const t4::pdf::array& names, const typename search_traits::tree_traits::param_key_type& key)
394 {
395  typedef typename search_traits::tree_traits tree_traits;
396 
397  size_t start = 0;
398  size_t end = names.Length() / 2;
399 
400  while (start < end)
401  {
402  size_t n = (start + end) / 2;
403 
404  int cmp = search_traits::key_compare( tree_traits::to_key(names[2*n]), key );
405  if (cmp < 0)
406  {
407  start = n+1;
408  }
409  else if (cmp > 0)
410  {
411  end = n;
412  }
413  else
414  {
415  return std::pair<size_t,bool>( { 2 * n, true} );
416  }
417  }
418 
419  return std::pair<size_t,bool>( { 2 * start, false} );
420 };

437 template<typename search_traits>
438 inline boost::optional<tree_pos<typename search_traits::tree_traits>> TreeFindFast(
439  const t4::pdf::dictionary& tree,
440  const typename search_traits::tree_traits::param_key_type& key )
441 {
442  typedef typename search_traits::tree_traits tree_traits; // abstracts out name tree vs. number tree
443  typedef tree_pos<tree_traits> tree_pos;
444 
445  class ex{};
446 
447  boost::optional<tree_pos> result = boost::none;
448 
449  // Only visit intermediate nodes if the key is in the range defined by "Limits."
450  auto kids_proc = [&key]( const t4::pdf::dictionary& intermediate_node )
451  {
452  t4::pdf::array limits = intermediate_node.GetRequiredArray( "Limits" );
453  return ( search_traits::is_in_range( limits, key ) );
454  };
455 
456  auto names_proc = [&key,&result]( tree_pos& pos, const t4::pdf::array& names )
457  {
458  auto find_result = TreeFind<search_traits>( names, key );
459  if (find_result.second)
460  {
461  pos.set_names( names );
462  pos.set_names_index( find_result.first );
463  result = pos;
464  }
465  throw ex();
466  };
467 
468  typedef tree_node_walker<tree_traits, decltype(kids_proc), decltype(names_proc)> tree_node_walker;
469  try
470  {
471  tree_node_walker::for_each(tree, kids_proc, names_proc);
472  }
473  catch (const ex&){}
474 
475  return result;
476 }

493 template<typename search_traits>
494 inline boost::optional<std::pair<tree_pos<typename search_traits::tree_traits>,size_t>> TreeFindSlow(
495  const t4::pdf::dictionary& tree,
496  const typename search_traits::tree_traits::param_key_type& key )
497 {
498  typedef typename search_traits::tree_traits tree_traits; // abstracts out name tree vs. number tree
499  typedef tree_pos<tree_traits> tree_pos;
500  typedef boost::optional<std::pair<tree_pos,size_t>> result_type;
501 
502  result_type result = boost::none;
503  size_t item_index = 0;
504  bool excluded = false;
505  class ex{};
506 
507  auto kids_proc = [&key,&excluded]( const t4::pdf::dictionary& intermediate_node )
508  {
509  t4::pdf::array limits = intermediate_node.GetRequiredArray( "Limits" );
510  excluded = !( search_traits::is_in_range( limits, key ) );
511  return true;
512  };
513 
514  auto names_proc = [&]( tree_pos& pos, t4::pdf::array& names )
515  {
516  size_t length = names.Length();
517 
518  if (excluded)
519  {
520  // According to Limits, the key is not in this names/nums array. Increment the counter, but otherwise skip.
521  item_index += (length / 2);
522  }
523  else
524  {
525  // According to Limits, if the key exists it is in this names/nums array. Find it, if there, and return.
526  auto find_result = TreeFind<search_traits>(names, key);
527  if (find_result.second)
528  {
529  size_t array_index = find_result.first;
530  item_index += (array_index / 2);
531  pos.set_names( names );
532  pos.set_names_index( array_index );
533  result.emplace( pos, item_index );
534  }
535  throw ex();
536  }
537  };
538 
539  typedef tree_node_walker<tree_traits, decltype(kids_proc), decltype(names_proc)> tree_node_walker;
540  try
541  {
542  tree_node_walker::for_each(tree, kids_proc, names_proc);
543  }
544  catch(const ex&){}
545 
546  return result;
547 }

622 template <>
623 inline boost::optional<tree_pos<typename CaseInsensitiveNameTreeLookup::tree_traits>> TreeFindByKey<CaseInsensitiveNameTreeLookup>(
624  const t4::pdf::dictionary& tree,
625  const typename CaseInsensitiveNameTreeLookup::tree_traits::param_key_type& key )
626 {
627  // Searching for an exact match is fast. And most callers will probably in fact be passing an exact match for the
628  // key. So try that first.
629  auto result1 = TreeFindFast<StandardNameTreeLookup>( tree, key );
630  if (result1)
631  return result1;
632 
633  // CaseInsensitiveNameTreeLookup converts the keys it finds in the tree, but assumes that the caller has converted
634  // the key being searched for. So do that here.
635  t4::string key2_str( key.first, key.second );
636  detail::to_lower( key2_str );
637  const char* keyStart = key2_str.c_str();
638  const char* keyEnd = keyStart + key2_str.length();
639  std::pair<const char*, const char*> key2( keyStart, keyEnd );
640 
641  // Perform the case-insensitive search.
642  // This is a brute-force search that steps through each item in the tree. This could be optimized somewhat.
643  // For example the code could do a "Limits"-based search to find the first and last nodes that might contain the key,
644  // and then perform the actual enumeration within these bounds.
645  auto result2 = TreeFindSlow<CaseInsensitiveNameTreeLookup>( tree, key2 );
646  if (result2)
647  return result2->first;
648 
649  return boost::none;
650 }

843 template <typename search_traits, typename obj_type>
844 inline obj_type TreeAddValue(
845  const t4::pdf::dictionary& tree,
846  const typename search_traits::tree_traits::param_key_type& key,
847  std::function<obj_type(t4::pdf::array&,size_t)> create_proc)
848 {
849  typedef typename search_traits::tree_traits tree_traits; // abstracts out name tree vs. number tree
850  typedef tree_pos<tree_traits> tree_pos;
851 
852  class ex{};
853  boost::optional<obj_type> result = boost::none;
854 
855  // Callback proc tells tree_node_walker to only visit intermediate nodes if the key is in the range defined by
856  // "Limits."
857  auto kids_proc = [&key]( const t4::pdf::dictionary& intermediate_node )
858  {
859  // SDK_TODO<crosby> 6/30/16: Untested; only invoked when adding to hierarchical name tree. Will be addressed
860  // with Jira task PDFNM-102 "Implement NameTree, Part 3: modification for hierarchical NameTree."
861  t4::pdf::array limits = intermediate_node.GetRequiredArray( "Limits" );
862  return ( search_traits::is_in_range( limits, key ) );
863  };
864 
865  // Callback proc invoked by tree_node_walker for the names/nums array where 'key' is supposed to go (according to
866  // 'kids_proc').
867  auto names_proc = [&]( const tree_pos&, t4::pdf::array& names )
868  {
869  auto find_result = TreeFind<search_traits>( names, key );
870 
871  size_t array_index = find_result.first;
872  if (find_result.second)
873  {
874  // Key already exists. Delete the value, but reuse the key.
875  names.DeleteObjectAtIndex(++array_index);
876  }
877  else
878  {
879  // Key doesn't exist. Add it.
880  tree_traits::insert_key(names, array_index++, key);
881  }
882  result = create_proc(names, array_index);
883  throw ex();
884  };
885 
886  typedef tree_node_walker<tree_traits, decltype(kids_proc), decltype(names_proc)> tree_node_walker;
887  try
888  {
889  tree_node_walker::for_each(tree, kids_proc, names_proc);
890  }
891  catch(const ex&){}
892 
893  if (!result)
894  {
895  // This can only happen in a hierarchical name tree where the key falls outside the range of every Limits
896  // array. Currently this case is not handled. Best solution is to modify 'kids_proc' so that it checks for this
897  // condition and expands the Limits for an appropriate node. Then the new value would be inserted as normal.
898  // TODO_SDK<crosby> 6/27/16: Handle this case. This will be addressed with Jira task PDFNM-102 "Implement
899  // NameTree, Part 3: modification for hierarchical NameTree."
900  throw t4::corrupt_data();
901  }
902 
903  return *result;
904 }
