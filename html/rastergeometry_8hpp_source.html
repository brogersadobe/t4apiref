53 /*
54  Helper functions for container objects that need hash/comparison functions
55  for point/rectangle/matrix types
56 
57  Rather than implement operator< and hash_compute for Point, Rectangle and
58  Matrix, we provide free functions that work on any model of point, rectangle
59  and matrix. We do this because we can't presume these functions exist for
60  all models of these concepts, but we don't want to introduce name collisions
61  when they do.
62 
63  To create functor objects, we recommend using t4::static_mem_ptr. E.g.:
64 
65  std::set<MyRectangle, t4::static_mem_ptr<less_than_rectangle<MyRectangle>>> MySet;
66  std::unordered_set<MyRectangle, t4::static_mem_ptr<hash_rectangle<MyRectangle>>> MyUnorderedSet;
67 
68  Define sets of MyRectangles with inlinable comparison and hash operators.
69 */

706 // See DoNonAxisAlignedRectsIntersect for details:
707 template<typename rect_type, typename matrix_type>
708 bool DoNonAxisAlignedRectsIntersectInternal(
709  const rect_type & rect1,
710  const rect_type & rect2,
711  const matrix_type & transformRect2SpaceToRect1Space,
712  bool recurse )
713 {
714  typedef typename RectangleTypeHelper<rect_type>::coord_type coord_type;
715  enum
716  {
717  QuadrantTop = 1,
718  QuadrantRight = 2,
719  QuadrantBottom = 4,
720  QuadrantLeft = 8
721  };
722 
723  coord_type xRect1[ 4 ]; // Point coordinates,
724  coord_type yRect1[ 4 ]; // in clockwise order.
725  xRect1[ 0 ] = xRect1[ 1 ] = get_xmin( rect1 );
726  xRect1[ 2 ] = xRect1[ 3 ] = get_xmax( rect1 );
727  yRect1[ 0 ] = yRect1[ 3 ] = get_ymin( rect1 );
728  yRect1[ 1 ] = yRect1[ 2 ] = get_ymax( rect1 );
729  coord_type xRect2[ 4 ]; // Point coordinates,
730  coord_type yRect2[ 4 ]; // in clockwise order.
731  xRect2[ 0 ] = xRect2[ 1 ] = get_xmin( rect2 );
732  xRect2[ 2 ] = xRect2[ 3 ] = get_xmax( rect2 );
733  yRect2[ 0 ] = yRect2[ 3 ] = get_ymin( rect2 );
734  yRect2[ 1 ] = yRect2[ 2 ] = get_ymax( rect2 );
735  int quadrants[ 4 ] = { 0, 0, 0, 0 };
736 
737  // Transform the non-aligned rect points into the local coordinate system
738  // for the aligned rect:
739  for( unsigned int index = 0; index < 4; ++index )
740  {
741  MatrixTransform( transformRect2SpaceToRect1Space, xRect2[ index ], yRect2[ index ] );
742  // Quick check to see if rect1 contains a vertex of rect2:
743  if( RectContainsPoint( rect1, xRect2[ index ], yRect2[ index ] ) )
744  return true;
745  if( xRect2[ index ] > xRect1[ 2 ] )
746  quadrants[ index ] |= QuadrantRight;
747  else if( xRect2[ index ] < xRect1[ 0 ] )
748  quadrants[ index ] |= QuadrantLeft;
749  if( yRect2[ index ] > yRect1[ 1 ] )
750  quadrants[ index ] |= QuadrantTop;
751  else if( yRect2[ index ] < yRect1[ 0 ] )
752  quadrants[ index ] |= QuadrantBottom;
753  }
754 
755  // Quick check to see if rect2 lies entirely outside of rect1. Note that
756  // this will not catch all cases (for example when rect2 lies entirely
757  // outside of rect1 but extends both over and to the right of rect1):
758  if( quadrants[ 0 ] & quadrants[ 1 ] & quadrants[ 2 ] & quadrants[ 3 ] )
759  return false;
760 
761  // Check for line intersections:
762  coord_type xIntersect, yIntersect;
763  for( unsigned int rect2Index1 = 0; rect2Index1 < 4; ++rect2Index1 )
764  {
765  unsigned int rect2Index2 = rect2Index1 == 0 ? 3 : rect2Index1 - 1;
766  for( unsigned int rect1Index1 = 0; rect1Index1 < 4; ++rect1Index1 )
767  {
768  unsigned int rect1Index2 = rect1Index1 == 0 ? 3 : rect1Index1 - 1;
769  if( real_services::ComputeLineSegmentIntersection(
770  &xIntersect, &yIntersect,
771  xRect2[ rect2Index1 ], yRect2[ rect2Index1 ],
772  xRect2[ rect2Index2 ], yRect2[ rect2Index2 ],
773  xRect1[ rect1Index1 ], yRect1[ rect1Index1 ],
774  xRect1[ rect1Index2 ], yRect1[ rect1Index2 ] ) )
775  return true;
776  }
777  }
778 
779  // If we haven't done it yet, do the same checks after swapping the rects
780  // (rect2 might fully enclose rect1):
781  if ( !recurse )
782  return false;
783  matrix_type tmpMat;
784  if ( !MatrixInvert( tmpMat, transformRect2SpaceToRect1Space ) )
785  return false;
786  return DoNonAxisAlignedRectsIntersectInternal( rect2, rect1, tmpMat, false/*recurse*/ );
787 }
