68 template <typename app_traits>
69 void StructureDebugFile<app_traits>::DrawStringInfo(const string_summary_type& ss)
70 {
71  drr_type drr = ss.GetDRR();
72  // The DRR can be invalid for multi-show strings or invisible text:
73  if( ss.GetNumVisibleGlyphs() == 0 || drr.IsInvalid() )
74  return;
75 
76  const matrix_type& cm = ss.m_info.m_charMat;
77  real_coord_type cmdet = get_a(cm)*get_d(cm) - get_b(cm)*get_c(cm);
78  matrix_type invcma;
79  imaging_model::MakeMatrix(invcma, get_d(cm)/cmdet, -get_b(cm)/cmdet, -get_c(cm)/cmdet, get_a(cm)/cmdet, 0, 0);
80 
81  GSave();
82  Concat(ss.m_info.m_userSpaceToDeviceSpaceMat);
83  SetColor("DocText", real_coord_type( 0.5 ) );
84  size_t nadv = strlen(ss.m_dbgStrASCII);
85  real_coord_type* xadv = new real_coord_type[nadv];
86  real_coord_type* yadv = new real_coord_type[nadv];
87  for (size_t i = 0; i < nadv-1; i++)
88  {
89  point_type start = ss.m_info.GetUserSpaceBaselinePos( ss.m_contentPoint.m_stringNum, i, 0, 1 );
90  point_type end = ss.m_info.GetUserSpaceBaselinePos( ss.m_contentPoint.m_stringNum, i, 1, 1 );
91  xadv[i] = get_x( end ) - get_x( start );
92  yadv[i] = get_y( end ) - get_y( start );
93  imaging_model::MatrixTransform(invcma, xadv[i], yadv[i]); // take the delta back to "glyph space"
94  }
95  xadv[nadv-1] = yadv[nadv-1] = 0; // not used
96 
97  XYShow(get_x(ss.GetStartPoint()), get_y(ss.GetStartPoint()), xadv, yadv, nadv, ss.m_info.m_charMat, ss.m_info.m_isVertical, ss.m_dbgStrASCII);
98  delete[] xadv;
99  delete[] yadv;
100 
101  real_coord_type x0 = get_x(ss.GetStartPoint());
102  real_coord_type y0 = get_y(ss.GetStartPoint());
103  real_coord_type x1 = get_x(ss.GetEndPoint());
104  real_coord_type y1 = get_y(ss.GetEndPoint());
105 
106  // now draw the full baseline and covered region
107  SetColor("CurShow", real_coord_type( 0.5 ) );
108  SetPathForDRR(drr, x0, y0);
109  SetLineWidth(real_coord_type(0.1));
110  Stroke();
111 
112  SetColor("TextLocs", real_coord_type( 0.5 ) );
113  char buf[16];
114  sprintf(buf, "%zu-%zu:%s%zu", ss.m_contentPoint.m_dlEntryNum, ss.m_contentPoint.m_stringNum, ss.m_info.m_isVertical ? "v" : "", ss.m_info.GetNumGlyphsInSubstring(ss.m_contentPoint.m_stringNum));
115  Show(x0, y0, m_strBaseCharMat, false, buf);
116 
117  MoveTo(x0, y0);
118  LineTo(x1, y1);
119  for (size_t i = 0; i < ss.m_info.GetNumGlyphsInSubstring(ss.m_contentPoint.m_stringNum); i++)
120  {
121  x0 = get_x(ss.m_info.GetUserSpaceBaselinePos(ss.m_contentPoint.m_stringNum, i, 0, 1));
122  y0 = get_y(ss.m_info.GetUserSpaceBaselinePos(ss.m_contentPoint.m_stringNum, i, 0, 1));
123  // tick mark per glyph
124  MoveTo(x0 + drr.GetCWDirX(), y0 + drr.GetCWDirY());
125  LineTo(x0, y0);
126  }
127  SetLineWidth(0);
128  Stroke();
129  GRestore();
130 }

145 template <typename app_traits>
146 void StructureDebugFile<app_traits>::SetPathForDRR(const drr_type& drr)
147 {
148  tetraphilia_assert(!drr.IsInvalid());
149 
150  real_coord_type ox = drr.GetDirX() * drr.GetMinInDir() + drr.GetCWDirX() * drr.GetMinInCWDir();
151  real_coord_type oy = drr.GetDirY() * drr.GetMinInDir() + drr.GetCWDirY() * drr.GetMinInCWDir();
152  real_coord_type adx = drr.GetDirX() * drr.Length();
153  real_coord_type ady = drr.GetDirY() * drr.Length();
154  real_coord_type bdx = drr.GetCWDirX() * drr.Width();
155  real_coord_type bdy = drr.GetCWDirY() * drr.Width();
156 
157  MoveTo(ox, oy);
158  if (drr.SkepticalOfDirection())
159  {
160  CurveTo(ox + bdx/6 + bdy/6, oy + bdy/6 - bdx/6, ox + bdx/3 + bdy/6, oy + bdy/3 - bdx/6, ox + bdx/2 + bdy/6 , oy + bdy/2 - bdx/6);
161  CurveTo(ox + 2*bdx/3 + bdy/6, oy + 2*bdy/3 - bdx/6, ox + 5*bdx/6 + bdy/6, oy + 5*bdy/6 - bdx/6, ox + bdx , oy + bdy);
162  }
163  else
164  {
165  LineTo(ox + bdx/2 + bdy/6 , oy + bdy/2 - bdx/6);
166  LineTo(ox + bdx, oy + bdy);
167  }
168  LineTo(ox + adx + bdx, oy + ady + bdy);
169  LineTo(ox + adx, oy + ady);
170  ClosePath();
171 }
