33 // This computes the avg position, assuming the "num" objects are equally sized, spaced from
34 // the first one's left side at "from" to the last one's right side at "to". It actually
35 // averages the left sides, easily changed for centers.
36 template<typename real_type>
37 real_type GetAvgAsIfSpreadEvenly( size_t num, const real_type& from, const real_type& to )
38 {
39  return from + ( static_cast<real_type>( static_cast<std::uint32_t>( num-1 ) ) / static_cast<std::uint32_t>(2*num) ) * ( to-from );
40 }

42 // This blends two averages,given the number of samples of each
43 template<typename real_type>
44 real_type CombineAverages( size_t num1, const real_type& avg1, size_t num2, const real_type& avg2 )
45 {
46  return avg1 + ( static_cast<real_type>(static_cast<std::uint32_t>( num2 ) ) / static_cast<std::uint32_t>(num1+num2) ) * ( avg2-avg1 );

153  real_type GetStdDev( void ) const;
154 
155  // this tries to encapsulate the "unconfident of stats on small samples but quite
156  // confident as sample size increases". It computes the standard deviation, but makes sure it
157  // is at least as big as a rapidly decreasing (with count) number.
158  // The important thing here is that with only one sample, the std deviation is 0, so
159  // any two different number will be outside of the error bars. We need a bit of slop until
160  // the stats build confidence.
161  real_type GetErrorBars( const real_type& numStdDevs, const real_type& fractionIfStatsUnreliable ) const
162  {
163  real_type minbars = GetAvg() * fractionIfStatsUnreliable / static_cast<std::uint32_t>(m_count*m_count);
164  if( minbars < 0 )
165  minbars = -minbars;
166  real_type stddevbars = numStdDevs * GetStdDev();
167  return minbars < stddevbars ? stddevbars : minbars;
168  }
169 
170  bool IsWithinErrorBars( const real_type& n, const real_type& numStdDevs, const real_type& fractionIfStatsUnreliable ) const
171  {
172  if( m_count == 0 )
173  return false;
174  real_type avg = GetAvg();
175  if( n == avg )
176  return true;
177  if( n > avg )
178  return n < avg + GetErrorBars( numStdDevs, fractionIfStatsUnreliable );
179  else
180  return n > avg - GetErrorBars( numStdDevs, fractionIfStatsUnreliable );
181  }
182 
183  bool IsWithinErrorBars( const SequenceStats& rhs, const real_type& numStdDevs, const real_type& fractionIfStatsUnreliable ) const
184  {
185  if( m_count == 0 )
186  return false;
187  real_type myavg = GetAvg();
188  real_type myerror = GetErrorBars( numStdDevs, fractionIfStatsUnreliable );
189  real_type rhsavg = rhs.GetAvg();
190  real_type rhserror = rhs.GetErrorBars( numStdDevs, fractionIfStatsUnreliable );
191  if( myavg == rhsavg )
192  return true;
193  if( myavg < rhsavg )
194  return( myavg + myerror >= rhsavg - rhserror );
195  else
196  return( rhsavg + rhserror >= myavg - myerror );
197  }
198 
199  bool IsAllOneValue( void ) const
200  {
201  return m_max == m_min;
202  }

247 template<typename real_type>
248 void SequenceStats<real_type>::Add( const SequenceStats& rhs )
249 {
250  if( rhs.m_count == 0 )
251  return;
252 
253  if( m_count == 0 )
254  {
255  *this = rhs;
256  return;
257  }
258 
259  if( rhs.m_min < m_min )
260  m_min = rhs.m_min;
261  if( rhs.m_max > m_max )
262  m_max = rhs.m_max;
263 
264  // this "d" is a useful construct for making updating the avg and sum of deltas squareds
265  // the math is nontrivial, but doable, lots and lots of cancellation.
266  // Basically:
267  // avg += rhs.avg/(sum of counts)
268  // sumdeltasquared += rhs.sumdeltasquared + (product of counts)*d*d/(sum of counts);
269  real_type d = ( rhs.m_avg - m_avg );
270  real_type k = ( static_cast<real_type>(static_cast<std::uint32_t>( m_count ) ) / static_cast<std::uint32_t>( m_count + rhs.m_count ) )
271  * static_cast<std::uint32_t>( rhs.m_count ); // we are being careful here to avoid overflow
272 
273  // need to update avg here since the old avg is needed for d.
274  m_avg = CombineAverages(m_count, m_avg, rhs.m_count, rhs.m_avg);
275 
276  // this is the NOT dangerous case
277  m_sumDeltasSquared += rhs.m_sumDeltasSquared + k*d*d;
278 
279  m_count += rhs.m_count;
280 
281  #ifdef SANITY_CHECK_SEQUENCE_STAT_COMPUTATIONS
282  if( rhs.m_minFloat < m_minFloat )
283  m_minFloat = rhs.m_minFloat;
284  if( rhs.m_maxFloat > m_maxFloat )
285  m_maxFloat = rhs.m_maxFloat;
286  m_sumFloat += rhs.m_sumFloat;
287  m_sumSquaresFloat += rhs.m_sumSquaresFloat;
288  #endif
289 }

291 template<>
292 inline void SequenceStats<Fixed16_16>::Add( const SequenceStats& rhs )
293 {
294  // In order to avoid overflows later on, when these counts are
295  // used in formulae that require they be converted to Fixed16_16,
296  // we impose this restriction upfront. Our statistical objects
297  // will accummulate information only while the number of samples
298  // doesn't exceed 32767. For reflow, this is equivalent to saying
299  // that at most 32767 gyphs on a page will contribute their
300  // values and almost all of the existing files will have much
301  // fewer glyphs per page. Also, note that, in general, the
302  // more samples you have, the less significant their individual
303  // contributions are for measures like an average, so we're not
304  // losing much.
305  if( rhs.m_count == 0 || ( m_count + rhs.m_count ) > 32767 )
306  return;
307 
308  if( m_count == 0 )
309  {
310  m_count = rhs.m_count;
311  m_min = rhs.m_min;
312  m_max = rhs.m_max;
313  m_avg = rhs.m_avg;
314  m_sumDeltasSquared = rhs.m_sumDeltasSquared;
315  m_sumDeltasSquaredShift = rhs.m_sumDeltasSquaredShift;
316 #ifdef SANITY_CHECK_SEQUENCE_STAT_COMPUTATIONS
317  m_minFloat = rhs.m_minFloat;
318  m_maxFloat = rhs.m_maxFloat;
319  m_sumFloat = rhs.m_sumFloat;
320  m_sumSquaresFloat = rhs.m_sumSquaresFloat;
321 #endif
322  return;
323  }
324 
325  if( rhs.m_min < m_min )
326  m_min = rhs.m_min;
327  if( rhs.m_max > m_max )
328  m_max = rhs.m_max;
329 
330  // this "d" is a useful construct for making updating the avg and sum of deltas squareds
331  // the math is nontrivial, but doable, lots and lots of cancellation.
332  // Basically:
333  // avg += rhs.avg/(sum of counts)
334  // sumdeltasquared += rhs.sumdeltasquared + (product of counts)*d*d/(sum of counts);
335  Fixed16_16 d = ( rhs.m_avg - m_avg );
336  Fixed16_16 k = ( static_cast<Fixed16_16>(static_cast<std::uint32_t>(m_count)) / static_cast<std::uint32_t>( m_count + rhs.m_count ) )
337  * static_cast<std::uint32_t>( rhs.m_count ); // we are being careful here to avoid overflow
338 
339  // need to update avg here since the old avg is needed for d.
340  m_avg = CombineAverages(m_count, m_avg, rhs.m_count, rhs.m_avg);
341 
342  // we have to be careful and scale
343  // scale to same size, store this scale as our new scale
344  Fixed16_16 rhsNormalizedSumDeltasSquared = rhs.m_sumDeltasSquared;
345  if( rhs.m_sumDeltasSquaredShift != m_sumDeltasSquaredShift )
346  {
347  if( rhs.m_sumDeltasSquaredShift > m_sumDeltasSquaredShift )
348  {
349  m_sumDeltasSquared.fVal >>= ( rhs.m_sumDeltasSquaredShift - m_sumDeltasSquaredShift );
350  m_sumDeltasSquaredShift = rhs.m_sumDeltasSquaredShift;
351  }
352  else
353  {
354  rhsNormalizedSumDeltasSquared.fVal >>= ( m_sumDeltasSquaredShift - rhs.m_sumDeltasSquaredShift );
355  }
356  }
357 
358  // now we need to play with d*d for updating the sumDeltasSquared
359  // we can take the abs(d) with impunity, and need to scale along with
360  // sumDeltasSquared
361  d = AbsVal(d);
362  d.fVal >>= ( m_sumDeltasSquaredShift / 2 );
363  while ( real_services::SquareWillBeDangerouslyLarge( d ) )
364  {
365  d.fVal >>= 1;
366  rhsNormalizedSumDeltasSquared.fVal >>= 2;
367  m_sumDeltasSquared.fVal >>= 2;
368  m_sumDeltasSquaredShift += 2;
369  }
370 
371  Fixed16_16 dsquared = d*d;
372  // what we really need is k*d*d, so we have to be careful here too.
373  while ( k > 1 && dsquared > limits<Fixed16_16>::max() / k )
374  {
375  //d >>= 1; not used again
376  dsquared.fVal >>= 2;
377  rhsNormalizedSumDeltasSquared.fVal >>= 2;
378  m_sumDeltasSquared.fVal >>= 2;
379  m_sumDeltasSquaredShift += 2;
380  }
381 
382  Fixed16_16 kdsquared = k*dsquared;
383  // now we need to make sure we can add these guys without overflowing
384 
385  size_t addOverflow1 = false;
386  size_t addOverflow2 = false;
387  Fixed16_16 partialSum1 = real_services::Add( rhsNormalizedSumDeltasSquared, kdsquared, &addOverflow1 );
388  Fixed16_16 partialSum2 = real_services::Add( partialSum1, m_sumDeltasSquared, &addOverflow2 );
389 
390  while( addOverflow1 || addOverflow2 )
391  {
392  //d >>= 1; not used again
393  // dsquared >>= 2; not used again
394  kdsquared.fVal >>= 2;
395  rhsNormalizedSumDeltasSquared.fVal >>= 2;
396  m_sumDeltasSquared.fVal >>= 2;
397  m_sumDeltasSquaredShift += 2;
398 
399  addOverflow1 = addOverflow2 = false;
400  partialSum1 = real_services::Add( rhsNormalizedSumDeltasSquared, kdsquared, &addOverflow1 );
401  partialSum2 = real_services::Add( partialSum1, m_sumDeltasSquared, &addOverflow2 );
402  }
403 
404  m_sumDeltasSquared = partialSum2;
405 
406  m_count += rhs.m_count;
407 
408  #ifdef SANITY_CHECK_SEQUENCE_STAT_COMPUTATIONS
409  if( rhs.m_minFloat < m_minFloat )
410  m_minFloat = rhs.m_minFloat;
411  if( rhs.m_maxFloat > m_maxFloat )
412  m_maxFloat = rhs.m_maxFloat;
413  m_sumFloat += rhs.m_sumFloat;
414  m_sumSquaresFloat += rhs.m_sumSquaresFloat;
415  #endif
416 }

431 template<>
432 inline Fixed16_16 SequenceStats<Fixed16_16>::GetStdDev() const
433 {
434  tetraphilia_assert( m_count > 0 );
435  if( m_min == m_max )
436  return 0;
437 
438  tetraphilia_assert( m_count > 1 );
439  Fixed16_16 res = real_services::Sqrt( m_sumDeltasSquared / static_cast<std::uint32_t>( m_count-1 ) );
440 
441  res.fVal <<= ( m_sumDeltasSquaredShift / 2 );
442 
443  #ifdef SANITY_CHECK_SEQUENCE_STAT_COMPUTATIONS
444  float varFloat = ( m_sumSquaresFloat - m_sumFloat * m_sumFloat / (std::uint32_t)m_count );
445  if( varFloat < 0 )
446  varFloat = 0;
447  float resFloat = (float)sqrt( varFloat / (std::uint32_t)(m_count-1) );
448  Fixed16_16 adjEpsilon = real_services::Epsilon( res ).fVal;
449  adjEpsilon.fVal <<= ( m_sumDeltasSquaredShift / 2 );
450  real_type realTypeResFloat( resFloat );
451  assertClose( res, realTypeResFloat, adjEpsilon );
452  #endif
453 
454  return res;
