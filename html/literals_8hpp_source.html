74  User-defined literal for size_t literals.
75 
76  Example: 64_z == size_t( 64 )
77 */
78 constexpr size_t operator"" _z( unsigned long long n )
79 {
80  #pragma clang diagnostic push
81  #pragma clang diagnostic ignored "-Wtautological-type-limit-compare"
82  if ( n > std::numeric_limits<size_t>::max() )
83  throw std::logic_error( T4_EXCEPTION_STR( "Invalid size_t literal" ) );
84  #pragma clang diagnostic pop
85  return static_cast<size_t>( n );
86 }
