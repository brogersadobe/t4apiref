130  void VerifyAddressRangePixelCount( const signal_type* start, int_type numPix ) const
131  {
132  #ifndef NDEBUG
133  const signal_type* end = t4::add_bytes( start, ( numPix - 1 ) * m_layout->m_pixelStride +
134  (NumStoredSignals( *m_layout ) - 1) * m_layout->m_signalStride + sizeof(signal_type)); // byte after last signal in last pixel
135  end = t4::max<const signal_type*>( end, t4::add_bytes( start, numPix * m_layout->m_pixelStride ) ); // byte after last pixel in non-planar case (pixels could have padding at end)
136  VerifyAddressRange( start, end );
137  #endif
138  }

140  void VerifyAddressRange( const signal_type* start_, const signal_type* end_ ) const
141  {
142  #ifndef NDEBUG
143  const unsigned char* start = static_cast<const unsigned char*>( static_cast<const void*>( start_ ) );
144  const unsigned char* end = static_cast<const unsigned char*>( static_cast<const void*>( end_ ) );
145 
146  if ( t4::is_infinite( *m_bounds ) )
147  {
148  tetraphilia_assert( m_layout->m_scanLineStride == 0
149  && m_layout->m_pixelStride == 0 );
150  const unsigned char* buffStart = static_cast<const unsigned char*>( static_cast<const void*>( this->GetPixels() ) );
151  const unsigned char* buffEnd = buffStart; // start of last scan line
152  if ( m_layout->m_pixelStride > 0 )
153  buffEnd += m_layout->m_pixelStride;
154  else
155  buffEnd += NumStoredSignals( *m_layout );
156  }
157  else if ( static_cast<ptrdiff_t>(m_layout->m_pixelStride) >= m_layout->m_signalStride )
158  {
159  // Chunky
160  const unsigned char* buffStart = static_cast<const unsigned char*>( static_cast<const void*>( this->GetPixels() ) );
161  const unsigned char* buffEnd = buffStart + m_layout->m_scanLineStride * static_cast<size_t>( get_height( *m_bounds ) - 1 ); // start of last scan line
162  buffEnd += m_layout->m_pixelStride * static_cast<size_t>( get_width( *m_bounds ) - 1 ); // start of last pixel in last scan line
163 
164  if ( m_layout->m_pixelStride > 0 )
165  buffEnd += m_layout->m_pixelStride; // start of first pixel "just past the end"
166  else
167  buffEnd += sizeof( signal_type ) * NumStoredSignals( *m_layout ); // start of first signal "just past the end"
168 
169  tetraphilia_assert( start <= end
170  && buffStart <= start && start < buffEnd
171  && buffStart <= end && end <= buffEnd );
172  }
173  else
174  {
175  // Planar
176  const unsigned char* buffStart = static_cast<const unsigned char*>( static_cast<const void*>( this->GetPixels() ) );
177  const unsigned char* buffEnd = buffStart + m_layout->m_scanLineStride * static_cast<size_t>( get_height( *m_bounds ) - 1 ); // start of last scan line
178  buffEnd += m_layout->m_signalStride * static_cast<ptrdiff_t>( NumStoredSignals( *m_layout ) - 1 ); // start of last plane
179  buffEnd += m_layout->m_pixelStride * static_cast<size_t>( get_width( *m_bounds ) - 1 ); // start of last pixel in last plane
180 
181  if ( m_layout->m_pixelStride > 0 )
182  buffEnd += m_layout->m_pixelStride; // start of first pixel "just past the end"
183  else
184  buffEnd += sizeof( signal_type ); // start of first signal "just past the end"
185 
186  tetraphilia_assert( start <= end
187  && buffStart <= start && start < buffEnd
188  && buffStart <= end && end <= buffEnd );
189  }
190  #endif
191  }
