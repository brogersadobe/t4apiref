38 /*
39  Raw entry in an optional_content_state instance. Identifies an OCG by indirect reference and specifies whether
40  the group is "on" (visible) or not. Early entries take priority over later ones.
41 */
42 struct oc_item

44  boost::optional<reference> m_ref;
45  bool m_is_on;
46 };

48 /*
49  Class whose instances describe the visibility state of some or all of the OCGs in a document. This is literally
50  just a wrapper around a vector of oc_items.
51 */
