368 template<typename app_traits>
369 char Type1<app_traits>::GetC()
370 {
371  if ( m_numBufferedChars > 0 ) {
372  --m_numBufferedChars;
373  return m_bufferedChars[m_numBufferedChars];
374  }
375  char c;
376  if ( m_decrypting ) {
377  if ( m_binaryCrypt )
378  c = BinaryDecrypt( m_bds->PeekAssert() );
379  else {
380  while ( !AsciiDecrypt( &c, m_bds->PeekAssert() ) )
381  m_bds->Advance();
382  }
383  } else
384  c = m_bds->PeekAssert();
385  m_bds->Advance();
386  return c;
387 }

389 template<typename app_traits>
390 typename app_traits::real_geometry_traits::coord_type Type1<app_traits>::ScanReal( const char*& p )
391 {
392  std::int32_t mantissa = 0;
393  std::int32_t fraction = 0;
394  std::int32_t exponent = 0;
395 
396  std::int32_t* curInt = &mantissa;
397  size_t fractionDigits = 0;
398 
399  bool mantissaNeg = false;
400  bool exponentNeg = false;
401  bool* curNeg = &mantissaNeg;
402  bool first = true;
403 
404  while ( true )
405  {
406  char c = *p++;
407  if ( !c ) {
408  --p;
409  break;
410  }
411  if ( first && (c == '-') ) {
412  *curNeg = true;
413  first = false;
414  }
415  else if ( first && (c == '+') )
416  {
417  first = false;
418  }
419  else if ( curInt == &mantissa && c == '.' ) {
420  curInt = &fraction;
421  }
422  else if ( c >= '0' && c <= '9' ) {
423  if ( curInt != &fraction || fractionDigits < real_services::DivisionByPowersOfTen<std::int32_t, real_type>::kMaxDecimalDigits ) {
424  *curInt = *curInt * 10 + (c - '0');
425  if ( curInt == &fraction )
426  ++fractionDigits;
427  first = false;
428  }
429  }
430  else if ( c == 'E' || c == 'e' )
431  {
432  curInt = &exponent;
433  curNeg = &exponentNeg;
434  first = true; // A sign is acceptable again
435  }
436  // We continue scanning for a valid number start unless we're parsing the exponent
437  else if ( !first || curInt == &exponent )
438  break;
439  }
440  real_type ret = real_type(mantissa);
441  if ( fractionDigits > 0 )
442  ret += real_services::DivisionByPowersOfTen<std::int32_t, real_type>::Compute( fraction, fractionDigits );
443 
444  if ( exponent != 0 )
445  {
446  if ( exponentNeg )
447  exponent = -exponent;
448  ret *= real_services::Pow( real_type{10}, static_cast<real_type>(exponent) );
449  }
450 
451  return mantissaNeg ? -ret : ret;
452 }

454 template<typename app_traits>
455 std::int32_t Type1<app_traits>::ScanInt( const char*& p )
456 {
457  std::int32_t accum = 0;
458  bool neg = false;
459  bool first = true;
460  while ( true )
461  {
462  char c = *p++;
463  if ( c == 0 ) {
464  --p;
465  break;
466  }
467  if ( first && (c == '-') ) {
468  neg = true;
469  first = false;
470  }
471  else if ( first && (c == '+') )
472  {
473  first = false;
474  }
475  else if ( c >= '0' && c <= '9' ) {
476  accum = accum * 10 + (c - '0');
477  first = false;
478  }
479  else if ( !first )
480  break;
481  }
482  return neg ? -accum : accum;
483 }

519 {

523  else if ( *p == '{' )
524  term = '}';
525  else

540 template<typename app_traits>
541 template<typename termination>
542 void Type1<app_traits>::CopyUntil( termination& term )
543 {
544  size_t index = 0;
545  if ( !AtEOF() ) {
546  char c = Peek();
547  while ( true ) {
548  bool terminates = term.Terminates( c );
549  if ( terminates && !termination::includeTermChar )
550  break;
551  if ( index + 1 >= m_tmpLength ) {
552  size_t newLength = t4::multiply<size_t>(m_tmpLength, 2 );
553  auto newBuf = t4::make_transient_array<char>( get_TransientHeap( this->GetContext() ), newLength );
554  memcpy( newBuf, m_tmpBuffer, index );
555  m_token.value = m_tmpBuffer = newBuf;
556  m_tmpLength = newLength;
557  }
558  m_tmpBuffer[index] = c;
559  ++index;
560  Advance();
561  if ( terminates )
562  break;
563  else {
564  if ( AtEOF() )
565  break;
566  c = Peek();
567  }
568  }
569  }
570  m_tmpBuffer[index] = 0;
571 }

648 {

805 template<typename app_traits>
806 const Type1Token& Type1<app_traits>::GetNextToken()
807 {
808  /* Skip comments and whitespace */
809  while ( true ) {
810  Skip( pdf::store::Parser::kWhiteSpace );
811  if ( AtEOF() ) {
812  m_token.type = Type1Token::pstTokenCnt;
813  return m_token;
814  }
815  if ( Peek() == '%' )
816  SkipUntil( pdf::store::Parser::kLineEnd );
817  else
818  break;
819  }
820 
821  switch( Peek() ) {
822 
823  case '/': {
824  Advance();
825  TokenEndTerminator<app_traits> term;
826  CopyUntil( term );
827  m_token.type = Type1Token::pstLiteral;
828  } break;
829 
830  case '{': {
831  BalancedCharTerminator<app_traits> term( '{', '}' );
832  CopyUntil( term );
833  m_token.type = Type1Token::pstProcedure;
834  } break;
835 
836  case '(': {
837  BalancedCharTerminator<app_traits> term( '(', ')' );
838  CopyUntil( term );
839  m_token.type = Type1Token::pstString;
840  } break;
841 
842  case '[': {
843  BalancedCharTerminator<app_traits> term( '[', ']' );
844  CopyUntil( term );
845  m_token.type = Type1Token::pstArray;
846  } break;
847 
848  case '<': {
849  BalancedCharTerminator<app_traits> term( '<', '>' );
850  CopyUntil( term );
851  switch ( m_token.value[1] ) {
852  case '<':
853  m_token.type = Type1Token::pstDictionary;
854  break;
855  case '~':
856  m_token.type = Type1Token::pstASCII85;
857  break;
858  default:
859  m_token.type = Type1Token::pstHexString;
860  break;
861 
862  }
863  } break;
864 
865  case '+': case '-': case '.': {
866  char c = GetC();
867  char nc = Peek();
868  bool isNumber = nc >= '0' && nc <= '9';
869  UngetC( c );
870  if ( isNumber )
871  HandleNumber();
872  else
873  HandleOperator();
874  } break;
875 
876  case '0': case '1': case '2': case '3': case '4':
877  case '5': case '6': case '7': case '8': case '9':
878  HandleNumber();
879  break;
880 
881  default:
882  HandleOperator();
883  break;
884  }

1249 void Type1<app_traits>::DoCharStrings( void )
1250 {
1251  // find the "begin" operator
1252  while ( true ) {
1253  const Type1Token& tok = GetNextToken();
1254  if ( tok.type == Type1Token::pstOperator && strcmp( tok.value, "begin" ) == 0 )
1255  break;
1256  if ( tok.type == Type1Token::pstTokenCnt )
1257  throw t4::corrupt_data();
1258  }
1259 
1260  while ( true ) {
1261 
1262  const Type1Token& tok = GetNextToken();
1263  if ( tok.type == Type1Token::pstOperator && strcmp( tok.value, "end" ) == 0 )
1264  break;
1265  else if ( tok.type != Type1Token::pstLiteral )
1266  throw t4::corrupt_data();
1267 
1268  // copy the glyph name
1269  size_t len = strlen( tok.value ) + 1;
1270  char* v = t4::make_transient_array<char>( m_heap, len );
1271  memcpy( v, tok.value, len );
1272 
1273  char* cstr;
1274  size_t length;
1275  ReadCString( cstr, length, GetNextToken() );
1276 
1277  m_cstrs.push_back( CStringRec( v, cstr, length ) );
1278 
1279  /* PDF_SLANG
1280 
1281  Partially validate the charstring -- check the last byte
1282  This code was probably added to the original t1read logic because the original
1283  DoType1InterpretCharString does not have a mechanism to tell when it has read
1284  past the end of the supplied charstring data. So this sort of verification
1285  will catch some invalid charstrings and declare the font to be invalid instead
1286  of letting DoType1InterpretCharString have unpredictable behavior. But there
1287  are easy ways to spoof this check if you know what you are doing. The
1288  tetraphilia version of the charstring interpreter doesn't really need this
1289  check since it does make sure it never reads past the end of the charstring.
1290  But if an error is detected during charstring interpretation the behavior is
1291  to throw an exception, which stops the page from rendering. This code from
1292  t1read executes during font load, and if the error is detected at that point
1293  the result is to use a faux font instead of using this font. Watson bug#1590159
1294  contains a font that triggers this behavior. Without this check we throw
1295  an exception and stop rendering.
1296  */
1297  typedef typename app_traits::pdf_traits pdf_traits;
1298  if ( pdf_traits::permissive ) {
1299 
1300  /*
1301  Occasionally we see charstrings (see bug DCMCD-799) which terminate one character
1302  too early. For example, the .notdef entry in the aforementioned bugfile concludes
1303  with these characters:
1304 
1305  cmdENDCHAR 0xbf
1306 
1307  So this loop allow us to test two possibilities: the end sequence appears where it
1308  should, or one byte earlier in the sequence.
1309  */
1310  bool okay = false;
1311  for ( size_t offset = 0; !okay && offset < 2; ++offset )
1312  {
1313  switch ((length > offset)? cstr[length - offset - 1]: 0) {
