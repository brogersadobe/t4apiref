101 template<typename sampler_def, typename edgeadder_type, typename tolerance_traits>
102  template<typename path_point_iterator_type>
103 ThinStrokeBezierRasterPainter<sampler_def,edgeadder_type,tolerance_traits>::ThinStrokeBezierRasterPainter( const constraints_type& constraints,
104  const path_point_iterator_type& start,
105  const path_point_iterator_type& end,
106  const matrix_type& mat, bool /*useEvenOddFill*/ )
107  : m_zeroWalker( IdentityPixelBuffers<signal_traits>::ZerosBuffer() ),
108  m_oneWalker( IdentityPixelBuffers<signal_traits>::OnesBuffer() ),
109  m_zeroSegment( &m_zeroWalker, 0, 0 ),
110  m_oneSegment( &m_oneWalker, 0, 0 ),
111  m_constraints( constraints ),
112  m_hasValidTransData(false),
113  m_transData(*t4::get_thread_heap()),
114  m_sampler(false)
115 {
116 
117  // thin stroker must use an overscanner
118  tetraphilia_assert(ThinStrokeBezierSampler<sampler_def>::kSamplerIsOverscan);
119 #ifndef NDEBUG
120  typedef typename sampler_def::sampler_type sampler_type;
121  typedef typename sampler_def::filtering_sampler_type filtering_sampler_type;
122 #endif
123  tetraphilia_assert(sampler_type::kSamplerIsOverscan);
124  tetraphilia_assert(filtering_sampler_type::kSamplerIsOverscan);
125 
126  tetraphilia_assert(kZero == 0 && kOne == 1 && kPixels == 2);
127 
128  m_segments[0] = &m_zeroSegment;
129  m_segments[1] = &m_oneSegment;
130 
131  path_point_iterator_type curPoint = start;
132 
133  // Watson bug#1297644
134  // permissive builds allow invalid paths to be constructed. If we detect one here we act as if the path were empty
135  if ( sampler_def::permissive ) {
136  if ( !(curPoint == end || is_moveto( get_pathOp(*curPoint) ) || is_close( get_pathOp(*curPoint) )) )
137  curPoint = end;
138  } else {
139  tetraphilia_assert( curPoint == end || is_moveto( get_pathOp(*curPoint) ) || is_close( get_pathOp(*curPoint) ) );
140  }
141 
142  bool boundsIsEmpty = RectIsEmpty(m_constraints.m_interestBounds);
143  if (!boundsIsEmpty)
144  {
145  real_type tolerance = tolerance_traits::GetFlattenerTolerance();
146  typename edgeadder_type::edge_adder_params params(mat, tolerance, true);
147  edgeadder_type adder(params, m_sampler, 1, m_constraints);
148 
149  // adds edges and computes userspace path bounds
150  // only one call to AddEdges permitted
151  // TODO: support curves w/o flattening
152  adder.template AddEdges<path_point_iterator_type>(curPoint,end);
153 
154  boundsIsEmpty = adder.UpdateConstraints(m_constraints,m_samplerBounds);
155  }
156 
157  if (boundsIsEmpty) // this path will not be drawn
158  {
159  MakeEmptyRect(m_constraints.m_interestBounds);
160  return;
161  }
162 
163  this->Initialize( 2, m_segments, m_constraints.m_interestBounds );
164 
165  m_sampler.FinishInitialize();
166 
167  #ifdef _DEBUG
168  this->SetName( "ThinStroke" );
169  #endif
170 }

173 void ThinStrokeBezierRasterPainter<sampler_def,edgeadder_type,tolerance_traits>::SetYImpl( int_type y )
174 {
175 
176  m_y = y;
177  m_hasValidTransData = false;
178 
179  // check y bounds
180  m_insideSamplersNonzeroYBounds = ( y >= get_ymin(m_constraints.m_interestBounds) && y < get_ymax(m_constraints.m_interestBounds) );
181 
182  // set current segment
183  this->SetCurrentSegID( kZero );
184  m_xstop = m_insideSamplersNonzeroYBounds ? limits<int_type>::min() : limits<int_type>::max();

187 template<typename sampler_def,typename edgeadder_type,typename tolerance_traits>
188 typename ThinStrokeBezierRasterPainter<sampler_def,edgeadder_type,tolerance_traits>::int_type
189 ThinStrokeBezierRasterPainter<sampler_def,edgeadder_type,tolerance_traits>::SetXImpl( int_type xstart, int_type xstop )
190 {
191 
192  m_xstart = xstart;
193  if (xstart < m_xstop)
194  return;
195 
196  tetraphilia_assert(m_insideSamplersNonzeroYBounds);
197 
198  if (xstart < get_xmin(m_constraints.m_interestBounds))
199  {
200  m_xstop = get_xmin(m_constraints.m_interestBounds);
201  this->SetCurrentSegmentID( kZero );
202  return t4::min<int_type>( m_xstop, xstop );
203  }
204 
205  if (xstart >= get_xmax(m_constraints.m_interestBounds))
206  {
207  m_xstop = limits<int_type>::max();
208  this->SetCurrentSegmentID( kZero );
209  return t4::min<int_type>( m_xstop, xstop );
210  }
211 
212  if (!m_hasValidTransData)
213  {
214  // empty the array
215  m_transData.m_transitions.Clear();
216 
217  m_sampler.SetY(m_y);
218 
219  // calculate transition table
220  m_sampler.CalcTransitionTable(m_transData.m_transitions, m_samplerBounds);
221 
222  // initialize for SetXImpl calls
223  m_transData.m_nextTrans = m_transData.m_transitions.begin();
224  m_transData.m_currSeg = kZero;
225 
226  m_hasValidTransData = true;
227  }
228 
229  while (xstart >= *m_transData.m_nextTrans)
230  {
231  m_transData.m_currSeg ^= kOne;
232  m_transData.m_nextTrans++;
233  }
234 
235  m_xstop = *m_transData.m_nextTrans;
236  this->SetCurrentSegmentID( m_transData.m_currSeg );
237  return t4::min<int_type>( m_xstop, xstop );
238 }
