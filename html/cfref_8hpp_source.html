91 /*
92  RAII wrapper class for Apple types that need to be retained and released.
93 
94  This template should be instantiated on the fundamental reference type. For
95  example:
96 
97  cfref<CGContextRef> cgc = UIGraphicsGetCurrentContext();
98 
99  To take ownership of a retained object (e.g., the result of a create call)
100  pass no_retain to the ctor:
101 
102  cfref<CGContextRef> cgc{ CGBitmapContextCreate( ...params... ), no_retain };
103 */

234  /*
235  Convert a cfref to some other type.
236 
237  If U is an integer or floating point type:
238 
239  First we verify that the contained CF object has CFNumber's CFTypeID. If that fails bad_cast
240  is thrown.
241 
242  If the contained type is a CFNumberRef, then we map the requested type to a CFNumberType and
243  call CFNumberGetValue requesting the specified type. If that fails we throw range_error
244  (unfortunately CF doesn't given us any information about what really went wrong, but it's
245  most likely that the contained value can't be converted to the requested one, so range_error
246  is a good answer).
247 
248  If U is bool:
249 
250  If the contained CF object is a CFBooleanRef, we return its contained value. Otherwise
251  bad_cast is thrown.
252 
253  If U is a CF type:
254 
255  T must be CFTypeRef, or a mutable -> const conversion (e.g., CFMutableStringRef to CFStringRef).
256  The contained object's CFTypeID is compared to the requested type's CFTypeID; if they
257  do not match, bad_cast is thrown. Otherwise the value is casted and returned.
258  */
259  template<typename U>
260  U as( void ) const
261  {
262  return cf_convert<U>( m_val );
263  }
264 
265  /*
266  Returns true if a cfref can be converted to the requested type. This simply returns false
267  if calling as<U>() would throw a bad_cast or range_error exception; otherwise it returns true.
268  */
269  template<typename U>
270  bool is( void ) const
271  {
272  return cf_can_convert<U>( m_val );
273  }

309 template<typename F>
310 void for_each_applier( void const* value, void* context )
311 {
312  static_assert( std::is_reference<F>::value, "" );
313  auto f = static_cast<std::remove_reference_t<F>*>( context );
314  std::forward<std::remove_reference_t<F>>(*f)( static_cast<CFTypeRef>( value ) );
315 }
