437 edc_decrypter::edc_decrypter( const ref_type& ref, C_EDCHelper pEDCHelperStruct, int encryptionAlg)
438  : m_encryptionAlg( encryptionAlg )
439 
440 {
441  // Munge the decryption key according to the Acrobat encryption scheme
442  // Do it here in the constructor, so that we don't have to remember
443  // the ref and the C_EDCDecryptor
444  auto& heap = *t4::get_thread_heap();
445  size_t keyLen = C_EDCHelper_GetKeyLength(pEDCHelperStruct);
446  size_t buffSize = keyLen + 5;
447 
448  unsigned char* buff = t4::make_transient_array<unsigned char>( heap, buffSize );
449  memcpy( buff, C_EDCHelper_GetKeyBufferPointer(pEDCHelperStruct), keyLen );
450 
451  unsigned char* ptr = buff + keyLen;
452 
453  *ptr++ = (unsigned char)(ref.objNum & 0xFF);
454  *ptr++ = (unsigned char)((ref.objNum >> 8) & 0xFF);
455  *ptr++ = (unsigned char)((ref.objNum >> 16) & 0xFF);
456  *ptr++ = (unsigned char)(ref.generation & 0xFF);
457  *ptr++ = (unsigned char)((ref.generation >> 8) & 0xFF);
458 
459  CMD5 md5;
460  md5.Update( (const char*)buff, buffSize );
461 
462  // Oh, my. The AESV2 algorithm for some reason uses the old "sAlT" string
463  // that was in Standard encryption V3.
464  if ( encryptionAlg == kAES )
465  md5.Update( "sAlT", 4);
466 
467  size_t keySize = buffSize < 16 ? buffSize : 16;
468  m_decrypt_key.SetNumElements( keySize );
469  md5.Final();
470  md5.GetDigest( (char*)m_decrypt_key.GetBuffer(), keySize );
471 }
