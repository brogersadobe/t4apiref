274  template<typename dl_type>
275  void EnumerateDisplayListWithClipImpl( dl_type* dl,
276  const store_dict_type* groupDict,
277  const real_rect_type& bBox)
278  {
279  auto& heap = *t4::get_thread_heap();
280  {
281  TransientSnapShot<app_traits> ss( heap );
282  raster_painter_ptr_type softClip;
283  hardclip_ptr_type hardClip = ComputeGroupClips( bBox, softClip );
284 
285  group_type* group = MakeChildGroup( m_rc.m_group->GetConstraints(), groupDict, m_rc.m_group, softClip, *hardClip );
286 
287  if ( group == m_rc.m_group )
288  {
289  tetraphilia_assert( !softClip );
290  SimpleValuePusher<app_traits, hardclip_type*> pushClip( m_rc.m_hardClip, hardClip );
291  dl->EnumerateContent( *this );
292  }
293  else
294  {
295  SimpleValuePusher<app_traits, group_type*> pushSamplingGroup( m_rc.m_samplingGroup, ( m_rc.m_samplingGroup == m_rc.m_group ? group : m_rc.m_samplingGroup ) );
296  SimpleValuePusher<app_traits, group_type*> pushGroup( m_rc.m_group, group );
297  SimpleValuePusher<app_traits, hardclip_type*> pushClip( m_rc.m_hardClip, hardClip );
298 
299  {
300  derived_gstate_ptr_type clonedGState;
301  if ( pushGroup.GetOldValue() != group )
302  {
303  clonedGState = m_rc.m_gstateConsumerContext.CloneGState( this->m_pGState );
304  clonedGState->SetContentRecord(&dl->GetContentRecord());
305  clonedGState->ResetTransparencyGState();
306  }
307  else
308  clonedGState = static_cast<derived_gstate_type*>(this->m_pGState );
309 
310  SimpleValuePusher<app_traits, gstate_ptr_type> pushGState( this->m_pGState, clonedGState );
311 
312  dl->EnumerateContent( *this );
313  }
314 
315  EndTransparencyGroup( group->GetConstraints(), softClip, pushGroup.GetOldValue(), group, false );
316  }
317  }

388  template<typename show_data_type>
389  void HandleTextShow( show_data_type const& showData,
390  pdf::content::TextBBoxInfo & bBoxInfo )
391  {
392 
393  // TODO: we need to do something about enabling skipping of Type3
394  // glyphs. see the comment in SkipTextRender Bug 1434787 - when using
395  // the DL bounds the SkipTextRender text wasn't using the adjusted
396  // glyph positions created in AdjustForReflow. For reflow mode never
397  // use (or update) the bounds in the DL. Instead always execute
398  // AdjustForReflow and compute the bounds:
399  bool boundsIsInfiniteRect = false;
400  if( !this->InReflowMode() )
401  {
402  TextSkipReason skipReason = SkipTextRender( boundsIsInfiniteRect, bBoxInfo );
403  if ( skipReason == kUnusableFont || ( !AccumulatingBBox() && skipReason == kOutsideBBox ))
404  {
405  PerformSkipText(bBoxInfo);
406  return;
407  }
408 
409  ShowInfo showInfo( GetGState(), this->GetUserToDeviceTransform(), this->GetTextMatrix() );
410  this->FillShowInfo( showData, showInfo, bBoxInfo );
411 
412  if( boundsIsInfiniteRect || AccumulatingBBox() )
413  {
414  if ( boundsIsInfiniteRect )
415  this->ComputeTextBounds( showInfo, bBoxInfo.m_bounds );
416  if ( AccumulatingBBox() )
417  AccumulateBBox( showInfo.GetDefaultUserSpaceBoundingBox() );
418  if( (SkipTextRender( boundsIsInfiniteRect, bBoxInfo ) != kDontSkip ) )
419  {
420  PerformSkipText(bBoxInfo);
421  return;
422  }
423  }
424 
425  ShowText( showInfo );
426  }
427  else
428  {
429  if( this->m_reflowLayoutIter == this->m_reflowLayoutEnd )
430  return;
431 
432  text::PDFFont::smart_ptr_type pFont = this->m_pGState->GetTextFont();
433  if( !pFont->IsUsable() )
434  {
435  // reflow mode skip if font not usable b4 calling FillShowInfoFrom
436  m_eventStream << unusable_font_event();
437  PerformSkipText( bBoxInfo );
438  return;
439  }
440 
441  ShowInfo showInfo( GetGState(), this->GetUserToDeviceTransform(), this->GetTextMatrix() );
442  this->FillShowInfo( showData, showInfo, bBoxInfo );

704 template <typename signal_traits>
705 t4::brief_ptr<imaging_model::RasterPainter<signal_traits>> RenderConsumer<signal_traits>::GetColorRasterPainter( bool isStroke,
706  bool isText,
707  const imaging_model::Constraints& constraints,
708  raster_painter_ptr_type const& softClip /* = nullptr */)
709 {
710  using namespace imaging_model;
711 
712  if ( m_rc.m_group->GetLayout().m_colorRep == GraphicLayoutDescriptor::kColorIgnored )
713  return MakeIdentityMasker<signal_traits>( constraints );
714 
715  if ( m_colorServerOverride )
716  return m_colorServerOverride->GetColor( constraints, m_rc.m_group->GetColorSpace() );
717 
718  pdfcolor::PDFColorSpace srcCS = this->m_pGState->GetColorSpace( isStroke );
719 
720  // TODO: It would be good if we could prevent any non-marking cases from even calling this method
721  // so that we could prevent even trying to paint when no marking is desired. However for now we will
722  // return a Zero raster painter in such cases, and unfortunately we currently have a few of them sprinkled around.
723 
724  // IsNonMarking does not directly apply to pattern color spaces, but rather to their underlying color space
725  if ( srcCS.IsNonMarking() && srcCS.GetType() != pdfcolor::kPDFCSTPattern )
726  {
727  // return raster painter with zero shape
728  return MakeZeroRasterPainter<signal_traits>( constraints );
729  }
730 
731  pdf::render::GStateExtractedColor color;
732 
733  if( srcCS.HasColorSpace() || srcCS.GetType() == pdfcolor::kPDFCSTPattern )
734  this->m_pGState->GetColor( isStroke, color );
735  else
736  {
737  srcCS = pdfcolor::PDFColorSpace::CreateUnaliasedDeviceGraySpace( pdfcolor::GetColorSpaceCache(m_rc.m_documentCaches) );
738  color.SetBlack();
739  }
740 
741  if ( srcCS.GetType() == pdfcolor::kPDFCSTPattern )
742  {
743  if ( color.GetColorName().empty() )
744  {
745  // return raster painter with zero shape (we do this to cover the initial/default pattern color which paints nothing)
746  return MakeZeroRasterPainter<signal_traits>( constraints );
747  }
748 
749  optional_store_dict_type patternDict = content::GetResourceDictionary( this->GetContentRecord()->GetResources().get_ptr(), "Pattern", color.GetColorName().c_str() );
750  if ( patternDict )
751  {
752  typename app_traits::pdf_traits::int_type patternType = patternDict->GetRequiredInteger( "PatternType" );
753  if ( patternType == 1 )
754  {
755  // TILING PATTERN
756 
757  // The "initial" matrix of the current consumer's m_gsContext defines the space in
758  // which the pattern is to be drawn, modulo any matrix specified on the pattern itself
759  std::int32_t paintType = patternDict->GetRequiredInteger( "PaintType" );
760  if ( paintType == 2 ) {
761  if ( !this->m_pGState->HasColorValues( isStroke ) )
762  throw t4::corrupt_data();
763 
764  if ( srcCS.IsNonMarking() )
765  {
766  // uncolored pattern with non-marking underlying color space does not mark so return zero raster painter
767  return MakeZeroRasterPainter<signal_traits>( constraints );
768  }
769  }
770 
771  // NOTE: Pattern could be colored or uncolored at this point.
772  return GetTilingPatternRasterPainter( isStroke, *patternDict, constraints, *this->m_pDpsToDeviceTransform, softClip );
773  }
774  else if ( patternType == 2 )
775  {
776  ::tetraphilia::color::RenderIntent renderIntent = this->m_pGState->GetRenderingIntent();
777  // patterns do not honor the gstate matrix, rather the initial matrix plus any imbedded in their
778  // dictionaries.
779  matrix_type transform = *this->m_pDpsToDeviceTransform;
780  if (patternDict->Get("Matrix").Type() != store::kObjectTypeNull)
781  {
782  store_array_type marray = patternDict->GetRequiredArray("Matrix");
783  matrix_type m;
784  MakeMatrix(m, marray.GetRequiredReal(0), marray.GetRequiredReal(1), marray.GetRequiredReal(2), marray.GetRequiredReal(3), marray.GetRequiredReal(4), marray.GetRequiredReal(5));
785  transform = m * transform;
786  }
787 
788  if (patternDict->Get("ExtGState").Type() != store::kObjectTypeNull)
789  {
790  //tetraphilia_assert_always(); // TODO CHIP
791  }
792  auto pShadingRP = CreateSmoothShadeRasterPainter(m_rc, constraints, transform, renderIntent, false, this->GetContentRecord(), patternDict->GetRequiredDictionary("Shading"));
793  if (pShadingRP)
794  return pShadingRP;
795 
796  // fall through to black
797  srcCS = pdfcolor::PDFColorSpace::CreateUnaliasedDeviceGraySpace( pdfcolor::GetColorSpaceCache(m_rc.m_documentCaches) );
798  color.ClearColorNameIfSetBlackAsserts();
799  color.SetBlack();
800  }
801  else
802  throw t4::corrupt_data();
803  } else if ( !srcCS.HasColorSpace() ) {
804  // This is a pattern with no dict
805  throw t4::corrupt_data();
806  }
807  }
808 
809  // The solid color case
810  // convert colors to destination color space
811  typedef typename RealSignalTraitsHelper<app_traits,real_type>::signal_traits real_signal_traits;
812  auto converter = srcCS.GetColorSpace()->template CreateConverter<real_signal_traits>(
813  *m_rc.m_group->GetColorSpace(),
814  this->m_pGState->GetRenderingIntent(), color::kConverterCachingPersistent );
815  size_t numSignals = m_rc.m_group->GetColorSpace()->GetNumSignals();
816  t4::small_vector<real_type, 4> curColor( numSignals );
817  converter->Convert( curColor.data(), sizeof( real_type ), color.GetColor().data(), sizeof( real_type ) );
818 
819  // This is a very simple scheme to move all color values toward black in a reflow view. The basic purpose
820  // of this is to make sure that things like white text (on a black backround) don't display white in
821  // the reflowed view. The black background would be missing, and so the text wouldn't show up.
822  // It isn't at all clear that this is the best way to map the colors, but it is a very simple, and
823  // frequently effective way of doing it.
824  if( isText && this->InReflowMode() )
825  {
826  size_t i;
827  if( m_rc.m_group->GetColorSpace()->IsSubtractive() )
828  for ( i = 0; i < numSignals; i++ )
829  {
830  curColor[i] += real_type( 0.2 );
831  if( curColor[i] > 1 )
832  curColor[i] = 1;
833  }
834  else
835  for ( i = 0; i < numSignals; i++ )
836  curColor[i] /= 2;
837  }
838 
839  return imaging_model::MakeSolidColorPainter<signal_traits>( constraints, curColor.data(), curColor.data() + numSignals );
840 }

1588 {
1589  /*

1591 
1592  If a path is both stroked and filled, we might have to create a knockout group, because
1593  the stroke and the fill are not supposed to interact. The blend mode, stroke and fill
1594  constant alphas, and the knockout attribute of the current group affect this decision.
1595  There are a few interesting cases here.
1596 
1597  There are some cases where we could optimize for constant alpha = 0, but in practice
1598  testing for this case is not worth the bother--constant alpha is rarely 0.
1599 
1600  If the current group is already knockout, no need to create a new group.
1601 
1602  Otherwise:
1603 
1604  If the blend mode is normal:
1605 
1606  If the stroke alpha is 1, no need to create a new group.
1607  If the constant stroke alpha and constant fill alpha are identical, a new
1608  group is needed, but it does not need to be knockout. The path will
1609  be rendered completely opaquely into this group, and the constant alpha
1610  will be applied at ETG time.
1611 
1612  Otherwise a new, isolated knockout group must be created for the fill and stroke.
1613  */

1615  bool newGroupIsKO = true;
1616 
1617  if ( fillRule == pdf::content::kNoFill || !stroke || m_rc.m_group->IsKnockout() )
1618  needsNewGroup = false;

1620  {
1621  if ( GetGState()->GetConstantAlpha( true ) == TETRAPHILIA_STATIC_TYPENAME_REAL( real_type, 1.0 ) )
1622  needsNewGroup = false;
1623  else
1624  newGroupIsKO = GetGState()->GetConstantAlpha( true ) != GetGState()->GetConstantAlpha( false );
1625  }
1626 
1627  if ( needsNewGroup )
1628  {
1629  if ( newGroupIsKO )
1630  {
1631  group_type* newGroup = MakeChildGroup( m_rc.m_group->GetConstraints(), m_rc.m_group, false, true, content::kNormal, m_rc.m_group->GetColorSpace() );
1632  SimpleValuePusher<app_traits, group_type*> pushGroup( m_rc.m_group, newGroup );
1633  functor( stroke, fillRule );
1634  EndTransparencyGroup( newGroup->GetConstraints(), {}, pushGroup.GetOldValue(), m_rc.m_group, true );
1635  }
1636  else
1637  {
1638  // Shortcut for when the alpha values are equal. The group is _isolated_ and non-knockout. Blend mode is applied
1639  // at ETG time for an equivalent result without having to track shape.
1640  group_type* newGroup = MakeChildGroup( m_rc.m_group->GetConstraints(), m_rc.m_group, true, false, GetGState()->GetBlendMode(), m_rc.m_group->GetColorSpace() );
1641  SimpleValuePusher<app_traits, group_type*> pushGroup( m_rc.m_group, newGroup );

1645  clonedGState1->ResetTransparencyGState();

1647  functor( stroke, fillRule );
1648  }
1649

1652  }
1653  else
1654  functor( stroke, fillRule );
1655 }
