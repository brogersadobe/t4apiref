33 /*
34  Special base class for never/maybe_null specializations on raw pointer types,
35  since you can't inherit from a fundamental type.
36 */
37 template<typename T, typename = void>
38 class raw_ptr_base : boost::totally_ordered<raw_ptr_base<T*>, T*>
39 {
40  T* m_p;
41 
42 protected:
43  raw_ptr_base( void ) : m_p( nullptr ) {}
44  raw_ptr_base( T* p ) : m_p ( p ) {}
45 
46  raw_ptr_base& operator=( T* rhs ) { m_p = rhs; return *this; }
47 
48 public:
49  operator T*( void ) const { return m_p; }
50  bool operator==( T* rhs ) const { return m_p == rhs; }
51  bool operator<( T* rhs ) const { return m_p < rhs; }
52 
53  explicit operator bool( void ) const { return m_p; }
54  T* operator->( void ) const { return m_p; }
55  T& operator*( void ) const { return *m_p; }
56 };
