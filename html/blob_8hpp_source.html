70 /*
71  A size-aware moveable (but not copyable) block of memory. More efficient
72  than vector<uint8_t>, as it does not bother default-initializing allocated
73  memory, and keeps its size in the allocated block so it is no larger than
74  a unique_ptr (which is usually just the size of a pointer).
75 
76  Usage:
77 
78  blob make_hello( void )
79  {
80  blob x( strlen( "Hello" ) );
81  memcpy( x.data(), "Hello", x.size() );
82  return x;
83  }
84 
85  Notes:
86 
87  There is no special optimization for a zero-length blob, which is distinct
88  from an "invalid" blob. Calling size() or data() on an invalid blob will assert.
89 */
90 class blob : boost::noncopyable
91 {
92  t4::unique_ptr<blob_detail::blob_impl> m_impl;
93 
94 public:
95  // Construct an invalid blob
96  blob( void ) {}
97 
98  // Construct a blob with the specified number of bytes. The memory is
99  // not initialized in any way.
100  blob( size_t sz )
101  : m_impl( blob_detail::blob_impl::make( sz ) )
102  {}
103 
104  // Construct a blob of the specified size and copy the provided data into it.
105  blob( void const* src, size_t sz )
106  : m_impl( blob_detail::blob_impl::make( sz ) )
107  {
108  memcpy( data(), src, sz );
109  }
110 
111  // Move ctor
112  blob( blob&& rhs )
113  : m_impl( std::move( rhs.m_impl ) )
114  {}
115 
116  // Returns true if the blob is in a valid state, and pointing to real data
117  explicit operator bool( void ) const { return static_cast<bool>( m_impl ); }
118 
119  // Assignment
120  blob& operator=( blob rhs )
121  {
122  boost::swap( *this, rhs );
123  return *this;
124  }
125 
126  // Swap

128  {

131 
132  // Fetch the size of the blob. Illegal to call on an invalid blob.
133  size_t size( void ) const { assert( m_impl ); return m_impl->size(); }
134 
135  // Fetch a pointer to the blob's bytes. Illegal to call on an invalid blob.
136  std::uint8_t* data( void ) const { assert( m_impl ); return m_impl->data(); }
137 };
