63 template<typename _signal_traits>
64 void TransparencyGroup<_signal_traits>::CheckOptimalLayout( void )
65 {
66  typedef IdealWalkerClusterTypes<signal_traits> ideal_walker_cluster_types;
67  typedef typename ideal_walker_cluster_types::dest_buffer_color dest_buffer_color;
68 
69  t4::transient_snapshot ss( *t4::get_thread_heap() );
70 
71  RasterMachine<signal_traits> machine;
72  GraphicYWalker<signal_traits> walker( MakeWalker( m_destination.m_shape ),
73  MakeWalker( m_destination.m_opacity ),
74  MakeWalker( m_destination.m_color ) );
75 
76  if ( !dest_buffer_color::CanSpecialize( &walker ) )
77  {
78  // gdow 10/18/2010
79  //
80  // I added this pseudo-assertion in debug builds to help engineers porting T3 to new platforms verify that
81  // their preferred pixel layouts are configured optimally (or at least as optimally as possible, given the
82  // constraints of T3's implementation). It is _not_ an error to hit this line of code, it's simply a signal
83  // that a TransparencyGroup with a suboptimal layout has been created.
84  //
85  // Sometimes this is unavoidable--color-managed files with CMYK color spaces, knockout groups, etc. But you
86  // should _not_ hit this line for really basic files--for example, pure text in device RGB. If you do then you
87  // need to revisit your values of app_traits::imaging_model_traits::optimize_for_transparency and/or
88  // app_traits::imaging_model::preferred_pixel_layouts.
89  //
90  // In the long term we really should refactor how this pixel-layout stuff is done; right now T3 can only be
91  // configured for one optimal layout and there's no good reason for this. At TransparencyGroup creation time
92  // we should simply check if the layout is one of several common cases, and instantiate a blitter factory for
93  // that particular layout. This might have consequences for app size though.
94 
95  T3_LOG_FLOW("CheckOptimalLayout - TransparencyGroup with a suboptimal pixel layout detected\n" );
96  DumpPB( "shape", m_destination.m_shape );
97  DumpPB( "opacity", m_destination.m_opacity );
98  DumpPB( "color", m_destination.m_color );
99  }
100 }

147 template<typename _signal_traits>
148 void TransparencyGroup<_signal_traits>::EndTransparencyGroup( const constraints_type& constraints,
149  HardClip<signal_traits>* hardClip,
150  raster_painter_ptr_type clip,
151  raster_painter_ptr_type mask,
152  group_type* child,
153  boost::optional<const blender_type&> blender )
154 {
155  raster_painter_ptr_type obj = child->GetResult( constraints, this->GetColorSpace() );
156  auto backdrop = t4::make_brief<TransparencyBackdropRasterPainter<signal_traits>>( this, constraints );
157 
158  if ( m_layout.m_colorRep != GraphicLayoutDescriptor::kColorIgnored && ( blender || this->GetColorSpace() != child->GetColorSpace() ) )
159  {
160  // Even if the target group doesn't have opacity, we might need it for intermediate
161  // computations. Create all temporary segments with opacity.
162  GraphicLayoutDescriptor desc = MakeGroupGraphicLayoutDescriptor( this->GetColorSpace()->GetNumSignals(), m_layout, m_knockout );
163 
164  // Passing true instead of childIsIsolated here. This is correct! This is the "nonisolated backdrop
165  // removal" step that occurs at the end of a group composition to compute the group's "true" color.
166  //
167  // agn * ( correct group result color ) = Cnbar - ( 1 - agn U ag0 ) * C0bar
168  raster_painter_ptr_type childRemovalBackdrop = this->GetRemovalBackdrop( constraints, this->GetColorSpace(), true );
169  raster_painter_ptr_type correctedSource = MakeOperationRasterPainter(
170  constraints, MakeSeparableOperation( RemoveBackdropOperation<signal_traits>() ),
171  MakeTempBuffer<signal_traits>( desc, constraints ),
172  childRemovalBackdrop, obj );
173 
174  if ( blender )
175  {
176  raster_painter_ptr_type correctedBackdrop = GetCorrectedBackdrop( constraints );
177  correctedSource = blender->Blend( constraints, this->GetColorSpace(),
178  MakeTempBuffer<signal_traits>( desc, constraints ),
179  correctedBackdrop, correctedSource );
180  }
181 
182  obj = MakeOperationRasterPainter( constraints,
183  MakeSeparableOperation( RestoreBackdropOperation<signal_traits>() ),
184  MakeTempBuffer<signal_traits>( desc, constraints ),
185  childRemovalBackdrop, correctedSource );
186  }
187 
188  ETGHelper( constraints, hardClip, std::move( clip ), std::move( mask ), std::move( backdrop ), std::move( obj ) );
189 }

221 template<typename _signal_traits>
222 t4::brief_ptr<RasterPainter<_signal_traits>> TransparencyGroup<_signal_traits>::GetRemovalBackdropImpl( const constraints_type& constraints,
223  const color_space_ptr_type& dstCs,
224  bool childIsIsolated )
225 {
226  if ( childIsIsolated )
227  {
228  if ( m_knockout )
229  return m_backdrop->GetRemovalBackdrop( constraints, dstCs, childIsIsolated );
230  else
231  {
232  const_pixel_buffer_type zeros = IdentityPixelBuffers<signal_traits>::ZerosBuffer();
233  raster_painter_ptr_type result = MakeFastPixelBufferRasterPainter( constraints,
234  &zeros, &zeros, m_source.m_color );
235  return ConvertRasterPainter( result, constraints, this->GetColorSpace(), dstCs );
236  }
237  }
238  else
239  {
240  raster_painter_ptr_type myBackdrop = m_backdrop->GetRemovalBackdrop( constraints, dstCs, m_isolated );
241  raster_painter_ptr_type cbbar;
242 
243  if ( m_knockout )
244  cbbar = myBackdrop;
245  else
246  {
247  cbbar = MakeFastPixelBufferRasterPainter( constraints, m_source.m_shape, m_source.m_opacity, m_source.m_color );
248  cbbar = ConvertRasterPainter( cbbar, constraints, this->GetColorSpace(), dstCs );
249  }
250 
251  GraphicLayoutDescriptor layout = MakeGraphicLayoutDescriptor( dstCs->GetNumSignals(), GraphicLayoutDescriptor::kShapeIgnored, m_layout.m_opacityRep, m_layout.m_colorRep, false, true );
252  return MakeOperationRasterPainter( constraints, MakeSeparableOperation( NonisolatedBackdropOperation<signal_traits>() ),
253  MakeTempBuffer<signal_traits>( layout, constraints ),
254  myBackdrop, cbbar );
255  }

259 t4::brief_ptr<RasterPainter<_signal_traits>> TransparencyGroup<_signal_traits>::GetResultImpl( const constraints_type& constraints, const color_space_ptr_type& dstCs )
260 {
261  constraints_type myConstraints( constraints );
262  myConstraints.ConstrainFurther( m_constraints.m_interestBounds );
263  raster_painter_ptr_type result = MakeFastPixelBufferRasterPainter( myConstraints, m_source.m_shape, m_source.m_opacity, m_source.m_color );
264  result = ConvertRasterPainter( result, myConstraints, this->GetColorSpace(), dstCs );
265 
266  if ( !RectContainsRect( m_constraints.m_interestBounds, constraints.m_interestBounds ) )
267  {
268  typedef TransparencyResultRasterPainter<signal_traits> result_type;
269  typedef TransparencyResultRasterPainter<signal_traits> result_type;
270  raster_painter_ptr_type backdrop = m_backdrop->GetResult( constraints, dstCs );
271  result = t4::make_brief<result_type>( constraints, backdrop, std::move( result ), m_constraints.m_interestBounds );
272  }
273 
274  return result;
