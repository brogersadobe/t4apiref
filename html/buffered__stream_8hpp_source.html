307 // Overload of BytesToInteger for single-byte integral types. Some compilers
308 // complain about the "ret <<= 8" expression--which is UB for single-byte types--
309 // in the generic BytesToInteger implementation, below.
310 template<typename Char>
311 std::enable_if_t<std::is_integral<Char>::value && sizeof(Char) == 1,
312 Char> BytesToInteger( data_io::BufferedStream& bds, size_t bytesPerInteger )
313 {
314  if ( bytesPerInteger == 0 )
315  return 0;
316 
317  // Make sure the requested number of bytes isn't larger than the requested data type
318  tetraphilia_assert( bytesPerInteger == 1 );
319 
320  const char* bytes;
321  bds.PeekBytesAssert( bytes, bytesPerInteger );
322  const unsigned char* ub = reinterpret_cast<const unsigned char*>( bytes );
323 
324  using UChar = std::make_unsigned_t<Char>;
325  auto ret = static_cast<UChar>( static_cast<Char>( ub[0] ) );
326  bds.Advance( bytesPerInteger );
327  return static_cast<Char>( ret );
328 }

330 template<typename Integer>
331 std::enable_if_t<std::is_integral<Integer>::value && (sizeof(Integer) > 1),
332 Integer> BytesToInteger( data_io::BufferedStream& bds, size_t bytesPerInteger )
333 {
334  // Make sure the requested number of bytes isn't larger than the requested data type
335  tetraphilia_assert( sizeof( Integer ) >= bytesPerInteger );
336 
337  if ( bytesPerInteger == 0 )
338  return 0;
339 
340  const char* bytes;
341  bds.PeekBytesAssert( bytes, bytesPerInteger );
342  const unsigned char* ub = reinterpret_cast<const unsigned char*>( bytes );
343 
344  using uint_type = std::make_unsigned_t<Integer>;
345  typedef boost::mp11::mp_if<std::is_unsigned<Integer>, unsigned char, signed char> char_type;
346  uint_type ret = static_cast<uint_type>( static_cast<char_type>( ub[0] ) );
347 
348  for ( size_t i = 1; i < bytesPerInteger; i++ )
349  {
350  ret <<= 8;
351  ret += ub[i];
352  }
353 
354  bds.Advance( bytesPerInteger );
355  return static_cast<Integer>( ret );
356 }

358 template<typename Integer>
359 std::enable_if_t<std::is_integral<Integer>::value,
360 void> BytesToIntegers( data_io::BufferedStream& bds, size_t bytesPerInteger, size_t n, Integer* ints )
361 {
362  tetraphilia_assert( sizeof(Integer) >= bytesPerInteger );
363 
364  typedef boost::mp11::mp_if<std::is_unsigned<Integer>, unsigned char, signed char> char_type;
365  using uint_type = std::make_unsigned_t<Integer>;
366 
367  const char* bytes;
368  bds.PeekBytesAssert( bytes, n * bytesPerInteger );
369  auto ub = reinterpret_cast<const unsigned char*>( bytes );
370  uint_type* uints = reinterpret_cast<uint_type*>( ints );
371  for ( size_t i = 0; i < n; i++ )
372  {
373  uints[i] = static_cast<uint_type>( static_cast<char_type>( ub[0] ) );
374  // Should be pure compile-time check: to make compiler happy and stop compaining about shift's value == type's size
375  if ( sizeof(Integer) > 1 )
376  {
377  for ( size_t j = 1; j < bytesPerInteger; ++j )
378  {
379  uints[i] <<= 8;
380  uints[i] += ub[j];
381  }
382  }
383  ub += bytesPerInteger;
384  }
385 
386  bds.Advance( n * bytesPerInteger );
387 }
