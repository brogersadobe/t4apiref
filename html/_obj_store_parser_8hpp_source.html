166 template<typename object_traits>
167 class DictParser : public ObjectStoreParser<object_traits>
168 {
169  typedef typename object_traits::app_traits app_traits;
170 
171 public:
172  DictParser( IndirectObject<app_traits>* pIDO, t4::pdf::decryption_context dc, const t4::pdf::decrypter* decrypter )
173  : ObjectStoreParser<object_traits>( pIDO, false/*stopAfterFirstObject*/, true/*eofIsError*/, dc, decrypter )
174  {}
175 
176  bool EndDict() override
177  {
178  this->m_parserStream->Advance();
179  if ( this->m_parserStream->PeekAssert() != '>' )
180  throw t4::corrupt_data();
181  this->m_parserStream->Advance();
182  return false;
183  }
184 
185  bool DoName_Dispatch() override
186  {
187  using t4::pdf::decryption_context;
188  auto result = ObjectStoreParser<object_traits>::DoName_Dispatch();
189  if ( this->m_decryption_context == decryption_context::trailer_top || this->m_decryption_context == decryption_context::id_entry_container )
190  {
191  if ( this->m_stack->size() % 2 == 1 ) // We just pushed a key
192  {
193  if ( strcmp( this->m_stack->Top().NameValue(), "ID" ) == 0 )
194  this->m_decryption_context = decryption_context::id_entry_container;
195  else
196  this->m_decryption_context = decryption_context::trailer_top;
197  }
198  }
199 
200  return result;
201  }
202 
203  bool DoInvalidTokenOrUnexpectedEOF( bool isUnexpectedEOF ) override
204  {
205  if (isUnexpectedEOF)
206  {
207  throw t4::corrupt_data();
208  }
209  else
210  {
211  /*
212  A dictionary stream in PDF can be specified in one of two formats :
213 
214  1. pure PDF (<< & >> brackets)
215 
216  /CIDSystemInfo <<
217  /Registry (ABCDEE+Myriad_Web_Pro+0)
218  /Ordering (T1UV)
219  /Supplement 0
220  >> def
221 
222  2. pure Postscript (begin & end keywords)
223 
224  /CIDSystemInfo 3 dict dup begin
225  /Registry (ABCDEE+Myriad_Web_Pro+0) def
226  /Ordering (T1UV) def
227  /Supplement 0 def
228  end def
229 
230  There is an old bug in MakeAccessible which generates the dictionary
231  stream in hybrid format i.e. "def" keyword inside a stream following
232  pure PDF syntax. We have chosen to ignore this token while parsing
233  dictionary streams in pure PDF syntax.
234 
235  Reference : https://jira.corp.adobe.com/browse/PDFNXTPC-595
236  */
237 
238  // Since an invalid token has already been detected, we wish to check
239  // if it was due to "def" token before aborting. If we find it, we wish
240  // to return "true" and continue parsing, else we throw an exception
241  // and abort parsing.
242  //
243  // We achieve this by calling SkipExpectedToken() below which throws an
244  // exception internally if it doesn't find the "def" keyword, else we
245  // continue parsing by returning true.
246  Parser::SkipExpectedToken(*(this->m_parserStream), "def");
247  return true;
248  }
249  }
