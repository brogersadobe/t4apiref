31 struct right_shift_impl
32 {
33  template<typename T1, typename T2>
34  using promotion_type = decltype(std::declval<T1>() << std::declval<T2>());
35 
36  template<typename Result, typename T1, typename T2>
37  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
38  {
39  // negative right shift is undefined (it is not just a left shift)
40  if ( y < 0 )
41  return {};
42 
43  // It is UB to shift by more than the size of the left-hand type, but in T4 we relax that restriction:
44  // any right shift greater than that amount simply produces 0 (or -1 for negative values).
45  if ( !lt( y, 8 * sizeof( T1 ) ) )
46  return convert_optional<Result>( is_negative( x ) ? -1 : 0 );
47  return convert_optional<Result>( static_cast<T1>( x >> y ) );
48  }
49 };
50 
51 } // end namespace sm_detail
