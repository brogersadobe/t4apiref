118 template<typename signal_traits>
119 imaging_model::TransparencyGroup<signal_traits>* MakeSoftMaskLuminosityTransparencyGroup(
120  const store::Dictionary<store::StoreObjTraits<typename signal_traits::app_traits> >& softMaskDict,
121  const imaging_model::Constraints& constraints,
122  pdfcolor::ColorSpaceCache& csCache,
123  const store::Dictionary<store::StoreObjTraits<typename signal_traits::app_traits> >* resources,
124  const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& groupCS )
125 {
126  typedef typename signal_traits::app_traits app_traits;
127  typedef TransientAllocator<app_traits> allocator_type;
128  typedef imaging_model::TransientGraphicStore<signal_traits> graphic_store_type;
129  typedef imaging_model::TransparencyGroup<signal_traits> group_type;
130  typedef store::StoreObjTraits<app_traits> obj_traits;
131  typedef store::Dictionary<obj_traits> dict_type;
132  typedef store::Array<obj_traits> array_type;
133  typedef typename app_traits::real_geometry_traits::coord_type real_type;
134  typedef imaging_model::GraphicLayoutDescriptor descriptor_type;
135  typedef typename color::ColorSpace<app_traits>::smart_ptr_type color_space_ptr_type;
136  typedef typename imaging_model::RealSignalTraitsHelper<app_traits, real_type>::signal_traits real_signal_traits;
137 
138  auto& heap = *t4::get_thread_heap();
139  allocator_type allocator( heap );
140 
141  /*
142  The "real" CS is the color space specified in the PDF file. Because this is a luminosity color space,
143  it is required to exist. The "group" CS will not necessarily match the real CS if
144  display_traits::kRespectTransparencyGroupColorSpaces is false.
145  */
146  dict_type formDict = softMaskDict.GetRequiredDictionary( "G" );
147  pdfcolor::PDFColorSpace pdfCS = GetPageOrFormPDFColorSpace( csCache, formDict, resources );
148  color_space_ptr_type realCs = pdfCS.GetColorSpace();
149 
150  // Luminosity-based soft mask dictionaries are required to have /Group dictionaries with /CS entries
151  if ( !realCs )
152  throw t4::unimplemented();
153  size_t realNumSignals = realCs->GetNumSignals();
154 
155  dict_type groupDict = formDict.GetRequiredDictionary( "Group" );
156  bool knockout = groupDict.GetBoolean( "K", false );
157  bool isolated = groupDict.GetBoolean( "I", false );
158 
159  size_t groupNumSignals = groupCS->GetNumSignals();
160 
161  // Attempt to build a "preferred" color layout, if at all possible
162  descriptor_type parentDesc = MakeGraphicLayoutDescriptor( groupNumSignals,
163  imaging_model::GraphicLayoutDescriptor::kShapeIgnored,
164  imaging_model::GraphicLayoutDescriptor::kOpacityIgnored,
165  imaging_model::GraphicLayoutDescriptor::kColorPerChannel );
166  descriptor_type layout = imaging_model::MakeGroupGraphicLayoutDescriptor( groupNumSignals, parentDesc, knockout );
167 
168  typedef imaging_model::ConstantOpaqueBackdrop<signal_traits> backdrop_type;
169  boost::optional<array_type> bc = softMaskDict.GetArray( "BC" );
170  t4::transient_vector<real_type> vals( realNumSignals, heap );
171  backdrop_type* backdrop;
172 
173  auto vIt = vals.begin();
174  if ( bc )
175  {
176  typename array_type::const_iterator_type it = bc->begin();
177  typename array_type::const_iterator_type end = bc->end();
178  if ( bc->Length() < realNumSignals )
179  throw t4::corrupt_data();
180  while ( it != end )
181  {
182  *vIt++ = it->RealValue();
183  ++it;
184  }
185  }
186  else
187  {
188  if ( pdfCS.GetNumSignals() != realCs->GetNumSignals() )
189  throw t4::corrupt_data();
190  const real_type* it = pdfCS.GetInitialColor();
191  const real_type* end = it + realNumSignals;
192  while ( it != end )
193  *vIt++ = *it++;
194  }
195 
196  /*
197  If the real and group CS's are different, then we need to convert the backdrop color. If
198  they match, this code won't do anything, but it's a nice cheap validation of our identity color
199  conversion facilities.
200  */
201  tetraphilia_assert( groupCS );
202  t4::transient_vector<real_type> overrideVals( groupNumSignals, heap );
203  auto conv = realCs->template CreateConverter<real_signal_traits>( *groupCS, color::kRelColorimetric, color::kConverterCachingNone );
204  conv->Convert( overrideVals.data(), sizeof( real_type ), vals.data(), sizeof( real_type ) );
205  backdrop = t4::make_transient<backdrop_type>( heap, groupCS, overrideVals.begin(), overrideVals.end() );
206 
207  graphic_store_type* gStore = t4::make_transient<graphic_store_type>( heap,
208  constraints.GetAlignedInterestBounds(), layout, false, heap );
209 
210  return t4::make_transient<group_type>( heap, constraints, gStore->GetWritableBuffers(), groupCS, layout, backdrop, isolated, knockout );
211 }

213 template<typename signal_traits>
214 imaging_model::TransparencyGroup<signal_traits>* MakePageTransparencyGroup(
215  const imaging_model::Constraints& constraints,
216  store::Store<typename signal_traits::app_traits>& store_,
217  const t4::pdf::dictionary& pageDict,
218  const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& cs,
219  const typename signal_traits::signal_type* clipMap = nullptr )
220 
221 {
222  typedef typename signal_traits::app_traits app_traits;
223  typedef imaging_model::TransientGraphicStore<signal_traits> graphic_store_type;
224  typedef imaging_model::TransparencyGroup<signal_traits> group_type;
225  typedef store::StoreObjTraits<app_traits> obj_traits;
226  typedef store::Dictionary<obj_traits> dict_type;
227  typedef imaging_model::GraphicLayoutDescriptor descriptor_type;
228 
229  auto& heap = *t4::get_thread_heap();
230 
231  size_t numSignals = cs->GetNumSignals();
232 
233  bool knockout = false;
234  boost::optional<dict_type> groupDict = pageDict.GetDictionary( "Group" );
235  if ( groupDict && groupDict->GetRequiredName( "S" ) == "Transparency" )
236  knockout = groupDict->GetBoolean( "K", false );
237 
238  // Attempt to build a "preferred" color layout, if at all possible
239  descriptor_type parentDesc = MakeGraphicLayoutDescriptor( numSignals,
240  imaging_model::GraphicLayoutDescriptor::kShapeIgnored, imaging_model::GraphicLayoutDescriptor::kOpacityIgnored,
241  imaging_model::GraphicLayoutDescriptor::kColorPerChannel );
242  descriptor_type layout = imaging_model::MakeGroupGraphicLayoutDescriptor( numSignals, parentDesc, knockout );
243 
244  // #pragma message( "Do not check this in" )
245  // // This is a good way to test interleaved channels and automatic opacity channel creation
246  // layout.m_opacityRep = imaging_model::GraphicLayoutDescriptor::kOpacityIgnored;
247 
248  typedef imaging_model::PageBackdrop<signal_traits> backdrop_type;
249  backdrop_type* backdrop = t4::make_transient<backdrop_type>( heap, cs );
250  group_type* result;
251 
252  typedef tetraphilia::imaging_model::TransparencyGroup<signal_traits> group_type;
253  graphic_store_type* gStore = t4::make_transient<graphic_store_type>( heap, constraints.GetAlignedInterestBounds(), layout, false, heap );
254  result = t4::make_transient<group_type>( heap, constraints, gStore->GetWritableBuffers(), cs, layout, backdrop, true, knockout );
255 
256  result->SetClipMap( clipMap );
257  return result;
258 }
