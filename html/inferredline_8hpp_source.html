90 void dbgGetAsASCII(
91  char* buf,
92  size_t bufSize,
93  const char* str,
94  const char* end,
95  ptr_to_UCM_type pUCM )
96 {
97  if (!pUCM->IsValid())
98  strncpy(buf, "!! No ToUnicodeMap !!", bufSize-1);
99  else
100  {
101  size_t remaining = bufSize;
102  const char* iter = str;
103  while( iter != end )
104  {
105  bool isCharCode32;
106  text::CharCodeLookup lookup = pUCM->Lookup( iter, end, &isCharCode32 );
107

109  buf = dbgGetAsASCII(buf, &remaining, lookup.start, lookup.end);
110  else
111  {
112  std::uint32_t* start = &lookup.singleValue;
113  buf = dbgGetAsASCII(buf, &remaining, start, start+1);
114  }

266 StringSummary::StringSummary(
267  const char * str,
268  const char * end,
269  const t4::pdf::content_point& contentPoint,
270  render::ShowInfo & info,
271  const t4::sptr<text::CharCodeMap> & umap,
272  bool forceInvalidDRR )
273  : m_info( info )
274 #ifdef TETRAPHILIA_ENABLE_STRUCTURE_DEBUG_FILE
275  , m_contentPoint( contentPoint )
276 #endif
277  , m_leadingInvisibleCount( 0 )
278  , m_betweenVisibleCount( 0 )
279  , m_trailingInvisibleCount( 0 )
280  , m_numVisibleGlyphs( 0 )
281  , m_numPeriods( 0 )
282  , m_firstVisibleChar( 0 )
283  , m_secondToLastVisibleChar( 0 )
284  , m_lastVisibleChar( 0 )
285  , m_firstVisibleCharIsBullet( false )
286  , m_lastVisibleCharIsBullet( false )
287  , m_hasValidUCM( umap->IsValid() )
288  , m_flags( 0 )
289  , m_bBox( info.GetDeviceSpaceBoundingBox() )
290 {
291  using namespace imaging_model;
292 
293 #ifdef TETRAPHILIA_ENABLE_STRUCTURE_DEBUG_FILE
294  dbgGetAsASCII( m_dbgStrASCII, sizeof( m_dbgStrASCII ), str, end, umap );
295 #endif
296 
297  const size_t numGlyphs = info.GetNumGlyphsInSubstring( contentPoint.m_stringNum );
298  if( m_hasValidUCM )
299  {
300  const char * iter = str;
301  while( iter != end )
302  {
303  bool isCharCode32;
304  text::CharCodeLookup lookup = umap->Lookup( iter, end, &isCharCode32 );
305  if( lookup.start )
306  {
307  AccumulateGlyph( lookup.start, lookup.end );
308 
309  for(const std::uint32_t* ptr = lookup.start; ptr != lookup.end; ++ptr)
310  m_str.append(1, static_cast<char>(*ptr));
311  }
312  else
313  {
314  std::uint32_t * start = &lookup.singleValue;
315  AccumulateGlyph( start, start + 1 );
316 
317  m_str.append(1, static_cast<char>(*start));
318  }
319  }
320 
321  ResolveInconsistenciesWithStringGeometryInfo( numGlyphs );
322  UnicodeCategorizer categ;
323  m_lastVisibleCharIsBullet = categ.IsBulletLike( m_lastVisibleChar );
324  m_firstVisibleCharIsBullet = categ.IsBulletLike( m_firstVisibleChar );
325  }
326  if( m_leadingInvisibleCount == 0 && m_betweenVisibleCount == 0 && m_trailingInvisibleCount == 0 )
327  m_betweenVisibleCount = numGlyphs;
328 
329  if( forceInvalidDRR )
330  {
331  m_startPt = m_endPt = t4::make<t4::real_point>( 0, 0 );
332  return; // DRR remains invalid
333  }
334 
335  if( m_betweenVisibleCount > 0 )
336  {
337  m_startPt = m_info.GetDeviceSpaceBaselinePos(
338  contentPoint.m_stringNum, m_leadingInvisibleCount, 0, 1 );
339  m_endPt = m_info.GetDeviceSpaceBaselinePos(
340  contentPoint.m_stringNum, m_leadingInvisibleCount + m_betweenVisibleCount - 1, 1, 1 );
341  }
342  else if( m_leadingInvisibleCount > 0 )
343  {
344  m_startPt = m_info.GetDeviceSpaceBaselinePos(
345  contentPoint.m_stringNum, 0, 0, 1 );
346  m_endPt = m_info.GetDeviceSpaceBaselinePos(
347  contentPoint.m_stringNum, m_leadingInvisibleCount - 1, 1, 1 );
348  }
349  else if( m_trailingInvisibleCount > 0 )
350  {
351  m_startPt = m_info.GetDeviceSpaceBaselinePos(
352  contentPoint.m_stringNum, 0, 0, 1 );
353  m_endPt = m_info.GetDeviceSpaceBaselinePos(
354  contentPoint.m_stringNum, m_trailingInvisibleCount - 1, 1, 1 );
355  }
356  else
357  {
358  m_startPt = m_endPt = t4::make<t4::real_point>( 0, 0 );
359  return; // DRR remains invalid
360  }
361 
362  auto len = real_services::VectorLength( get_x( m_endPt ) - get_x( m_startPt ),
363  get_y( m_endPt ) - get_y( m_startPt ) );
364  if( len < real_services::Epsilon( t4::real_coord( 1.0 ) ) )
365  return; // DRR remains invalid
366 
367  if( !m_hasValidUCM )
368  {
369  m_DRR = DRR(
370  1 /*dirX*/, 0 /*dirY*/,
371  get_xmin( m_bBox ), get_xmax( m_bBox ),
372  get_ymin( m_bBox ), get_ymax( m_bBox ), true );
373  m_trimmedPunctDRR = m_DRR;
374  return;
375  }
376 
377  // Calculate the DRR for this StringSummary:
378  t4::real_point ascentVec, descentVec;
379  m_info.GetDeviceSpaceAscenderDescenderVectors( ascentVec, descentVec );
380  auto ascent =
381  real_services::VectorLength( get_x( ascentVec ), get_y( ascentVec ) );
382  auto descent =
383  -real_services::VectorLength( get_x( descentVec ), get_y( descentVec ) );
384  auto dx = ( get_x( m_endPt ) - get_x( m_startPt ) ) / len;
385  auto dy = ( get_y( m_endPt ) - get_y( m_startPt ) ) / len;
386  const bool skeptical = ( m_betweenVisibleCount < 2 );
387  auto minInDir = dx * get_x( m_startPt ) + dy * get_y( m_startPt );
388  auto maxInDir = dx * get_x( m_endPt ) + dy * get_y( m_endPt );
389  auto valInCWDir = -dy * get_x( m_startPt ) + dx * get_y( m_startPt );
390  size_t overflow = false;
391  auto minInCWDir = real_services::Add( valInCWDir, descent, &overflow );
392  auto maxInCWDir = real_services::Add( valInCWDir, ascent, &overflow );
393  if( overflow )
394  return; // DRR remains invalid
395  m_DRR = DRR( dx, dy, minInDir, maxInDir, minInCWDir, maxInCWDir, skeptical );
396 
397  // Calculate the DRR when excluding leading and trailing punctuation (used when
398  // checking for hanging punctuation):
399  m_trimmedPunctDRR = m_DRR;
400  if( m_flags & ( kBeginsWithPunctuation | kEndsWithPunctuation ) )
401  {
402  if( m_betweenVisibleCount < 2 )
403  m_trimmedPunctDRR =
404  DRR( dx, dy, minInDir, minInDir,
405  valInCWDir + descent, valInCWDir + ascent, skeptical );
406  else
407  {
408  auto startPt = m_startPt;
409  if( m_flags & kBeginsWithPunctuation )
410  startPt = m_info.GetDeviceSpaceBaselinePos(
411  contentPoint.m_stringNum, m_leadingInvisibleCount + 1, 0, 1 );
412  auto endPt = m_endPt;
413  if( m_flags & kEndsWithPunctuation )
414  endPt = m_info.GetDeviceSpaceBaselinePos(
415  contentPoint.m_stringNum, m_leadingInvisibleCount + m_betweenVisibleCount - 2, 1, 1 );
416  minInDir = dx * get_x( startPt ) + dy * get_y( startPt );
417  maxInDir = dx * get_x( endPt ) + dy * get_y( endPt );
418  if( maxInDir < minInDir )
419  maxInDir = minInDir;
420  valInCWDir = -dy * get_x( startPt ) + dx * get_y( startPt );
421  m_trimmedPunctDRR =
422  DRR( dx, dy, minInDir, maxInDir,
423  valInCWDir + descent, valInCWDir + ascent, skeptical );
424  }
425  }
426 } // StringSummary ctor

430 // Process a single glyph, composed of one or more unicode characters, and
431 // update the StringSummary to include information about that glyph:
432 void StringSummary::AccumulateGlyph(
433  const std::uint32_t * start,
434  const std::uint32_t * end )
435 {
436  UnicodeCategorizer categ;
437 
438  bool allInvisible = true;
439  for( const std::uint32_t * pos = start; pos < end; ++pos )
440  {
441  if( categ.IsVisible( *pos ) )
442  {
443  allInvisible = false;
444  ++m_numVisibleGlyphs;
445  }
446 
447  if( !( m_flags & kHasHiragana ) && categ.IsHiragana( *pos ) )
448  m_flags |= kHasHiragana | kHasCJK;
449 
450  if( !( m_flags & kHasKatakana ) && categ.IsKatakana( *pos ) )
451  m_flags |= kHasKatakana | kHasCJK;
452 
453  if( !( m_flags & kHasCJK ) && categ.IsCJK( *pos ) )
454  m_flags |= kHasCJK;
455 
456  if( *pos == '.' )
457  ++m_numPeriods;
458  }
459 
460  if( allInvisible )
461  m_trailingInvisibleCount++;
462  else
463  {
464  if( m_betweenVisibleCount == 0 )
465  m_leadingInvisibleCount = m_trailingInvisibleCount;
466  else
467  m_betweenVisibleCount += m_trailingInvisibleCount;
468  m_trailingInvisibleCount = 0;
469  m_betweenVisibleCount++;
470  if( !m_firstVisibleChar )
471  {
472  m_firstVisibleChar = *start;
473  if( categ.IsPunctuation( m_firstVisibleChar ) )
474  m_flags |= kBeginsWithPunctuation;
475  }
476  m_secondToLastVisibleChar = m_lastVisibleChar;
477  m_lastVisibleChar = *( end - 1 );
478  if( categ.IsPunctuation( m_lastVisibleChar ) )
479  m_flags |= kEndsWithPunctuation;
480  else
481  m_flags &= static_cast<unsigned int>( ~kEndsWithPunctuation );
482  }
483 }

487 void StringSummary::ResolveInconsistenciesWithStringGeometryInfo(
488  const size_t totalGlyphs )
489 {
490  if( m_leadingInvisibleCount + m_betweenVisibleCount + m_trailingInvisibleCount <=
491  totalGlyphs )
492  return;
493 
494  size_t needToShortenBy =
495  m_leadingInvisibleCount + m_betweenVisibleCount + m_trailingInvisibleCount -
496  totalGlyphs;
497 
498  // This means that some of the glyphs in the string are "missing" in the
499  // font, so while we have N characters in the ShowString, we actually have
500  // fewer glyphs. We need to adjust for that:
501  if( m_trailingInvisibleCount >= needToShortenBy )
502  m_trailingInvisibleCount -= needToShortenBy;
503  else
504  {
505  needToShortenBy -= m_trailingInvisibleCount;
506  m_trailingInvisibleCount = 0;
507  if( m_betweenVisibleCount >= needToShortenBy )
508  m_betweenVisibleCount -= needToShortenBy;
509  else
510  {
511  needToShortenBy -= m_betweenVisibleCount;
512  m_betweenVisibleCount = 0;
513  tetraphilia_assert( m_leadingInvisibleCount >= needToShortenBy );
514  m_leadingInvisibleCount -= needToShortenBy;
515  }
516  }
517 }

931  t4::real_point ComputeStartReference( void ) const
932  {
933  using namespace imaging_model;
934 
935  auto refPt = m_lineRefStartPt;
936  auto endPt = m_lineRefEndPt;
937  auto rotateMat = t4::identity<t4::affine_transform>();
938  // Make a local copy of the matrix so that we don't have to worry about
939  // SafeL2Norm side-effects on parameters:
940  auto lineRefMat = m_lineRefMat;
941 
942  // If the angle between the horizontal text's baseline and the horizontal
943  // direction, with a cosine of a / sqrt( a*a + b*b ), or the one between
944  // the vertical text's baseline and the vertical direction, with a cosine
945  // of d / sqrt( c*c + d*d ), is neither 0, nor 90 degrees, align the text's
946  // baseline with the horizontal or vertical directions, respectively.
947  t4::real_coord denom;
948  if( m_lineRefIsVertical && ( AbsVal( get_d( lineRefMat ) ) != 1 ||
949  get_c( lineRefMat ) != 0 ) &&
950  get_d( lineRefMat ) != 0 )
951  {
952  // Note that it is safe for SafeL2Norm to modify mat.m_c and
953  // mat.m_d directly as the matrix mat is not going to be used in
954  // its entirety beyond this point.
955  denom = real_services::SafeL2Norm( get_c( lineRefMat ), get_d( lineRefMat ) );
956  if( AbsVal( get_d( lineRefMat ) ) / denom != 1 )
957  {
958  MakeMatrix( rotateMat, get_d( lineRefMat ) / denom,
959  get_c( lineRefMat ) / denom,
960  -get_c( lineRefMat ) / denom,
961  get_d( lineRefMat ) / denom, 0, 0 );
962  }
963  }
964  else if( !m_lineRefIsVertical && ( AbsVal( get_a( lineRefMat ) ) != 1 ||
965  get_b( lineRefMat ) != 0 ) &&
966  get_a( lineRefMat ) != 0 )
967  {
968  // Note that it is safe for SafeL2Norm to modify mat.m_a and
969  // mat.m_b directly as the matrix mat is not going to be used in
970  // its entirety beyond this point.
971  denom = real_services::SafeL2Norm( get_a( lineRefMat ), get_b( lineRefMat ) );
972  if( AbsVal( get_a( lineRefMat ) ) / denom != 1 )
973  {
974  MakeMatrix( rotateMat, get_a( lineRefMat ) / denom,
975  -get_b( lineRefMat ) / denom,
976  get_b( lineRefMat ) / denom,
977  get_a( lineRefMat ) / denom, 0, 0 );
978  }
979  }
980 
981  // Now rotate the line's direction by this new rotation matrix.
982  auto lineDir = t4::make<t4::real_point>( m_lineWithRubiDRR.GetDirX(), m_lineWithRubiDRR.GetDirY() );
983  lineDir = MatrixTransform( rotateMat, lineDir );
984 
985  // Align the new line's direction with the horizontal (1,0).
986  rotateMat = rotateMat *
987  t4::make<t4::affine_transform>( get_x( lineDir ), -get_y( lineDir ),
988  get_y( lineDir ), get_x( lineDir ), 0, 0 );
989 
990  if( !IsIdentity(rotateMat) )
991  {
992  refPt = MatrixTransform( rotateMat, refPt );
993  endPt = MatrixTransform( rotateMat, endPt );
994  }
995 
996  if( get_x( refPt ) == get_x( endPt ) )
997  {
998  // Somehow, we ended up with a vertical glyph baseline. Set
999  // penStartPt and penEndPt to the lower left corner and the
1000  // lower right corner of the glyph bounds, respectively. To
1001  // do so, we adjust their x coordinates by the ascent and
1002  // descent vectors, while setting their y coordinates to the
1003  // lower of the two y coordinates. We need to be a little
1004  // careful with which of the ascent and descent vectors we add
1005  // to whose x coordinate, as these vectors are not of equal
1006  // magnitude in the case of horizontal text, so they can't
1007  // be used interchangeably
1008 
1009  t4::real_coord ascent, descent;
1010  if( AbsVal( get_x( m_lineRefAscent ) ) < AbsVal( get_y( m_lineRefAscent ) ) )
1011  {
1012  ascent = AbsVal( get_y( m_lineRefAscent ) );
1013  descent = AbsVal( get_y( m_lineRefDescent ) );
1014  }
1015  else
1016  {
1017  ascent = AbsVal( get_x( m_lineRefAscent ) );
1018  descent = AbsVal( get_x( m_lineRefDescent ) );
1019  }
1020 
1021  if( get_y( refPt ) < get_y( endPt ) )
1022  {
1023  // this is the lower left corner
1024  get_x( refPt ) -= ascent;
1025  // this is the lower right corner
1026  get_x( endPt ) += descent;
1027  get_y( endPt ) = get_y( refPt );
1028  }
1029  else
1030  {
1031  // this is the lower left corner
1032  get_x( refPt ) -= descent;
1033  get_y( refPt ) = get_y( endPt );
1034  // this is the lower right corner
1035  get_x( endPt ) += ascent;
1036  }
1037  }
1038 
1039  // Now rotate refPt back.
1040  auto invRotateMat = rotateMat;
1041  // Since this is a rotation its inverse can be easily
1042  // computed without calling the expensive MatrixInvert.
1043  get_b( invRotateMat ) = -get_b( invRotateMat );
1044  get_c( invRotateMat ) = -get_c( invRotateMat );
1045 
1046  if( !IsIdentity(invRotateMat) )
1047  refPt = MatrixTransform( invRotateMat, refPt );
1048 
1049  return refPt;
1050  }

1052  void ComputeMaxAboveAndBelowBaseline(
1053  t4::real_coord& maxAboveBaseline, t4::real_coord& maxBelowBaseline ) const
1054  {
1055  // We define "above" as that portion of the line that advances
1056  // in the direction of the CCW vector (determined by rotating
1057  // the line direction by 90 degrees counterclockwise). In some
1058  // cases, that direction is *opposite* the direction of the
1059  // ascent vector (for example when the text is upside-down).
1060  // In such cases, GetMaxInCWDir() will actually describe the
1061  // portion of the line that's below, and not above the baseline.
1062 
1063  auto refPt = ComputeStartReference();
1064  auto baseline =
1065  -m_lineWithRubiDRR.GetDirY() * get_x( refPt )
1066  + m_lineWithRubiDRR.GetDirX() * get_y( refPt );
1067  auto dirX = m_lineWithRubiDRR.GetDirX();
1068  auto dirY = m_lineWithRubiDRR.GetDirY();
1069 
1070  // We compute the cosine of the angle between the CCW
1071  // direction and the ascent vector, and we detect the
1072  // case when it's -1 (i.e. the angle is 180 degrees)
1073  auto lineRefAscent = m_lineRefAscent;
1074  auto cosAngleBetweenAscentAndCCWDir =
1075  ( -dirY * get_x( lineRefAscent ) + dirX * get_y( lineRefAscent ) ) /
1076  ( real_services::SafeL2Norm( dirX, dirY ) *
1077  real_services::SafeL2Norm( get_x( lineRefAscent ), get_y( lineRefAscent ) ) );
1078  auto delta = AbsVal( cosAngleBetweenAscentAndCCWDir + 1 );
1079  if( delta < 16 * real_services::Epsilon(cosAngleBetweenAscentAndCCWDir) )
1080  {
1081  maxBelowBaseline = AbsVal( m_lineWithRubiDRR.GetMaxInCWDir() - baseline );
1082  maxAboveBaseline = AbsVal( baseline - m_lineWithRubiDRR.GetMinInCWDir() );
1083  }
1084  else
1085  {
1086  maxAboveBaseline = AbsVal( m_lineWithRubiDRR.GetMaxInCWDir() - baseline );
1087  maxBelowBaseline = AbsVal( baseline - m_lineWithRubiDRR.GetMinInCWDir() );
1088  }
1089  }

1317 ReadOrderType InferredLine::DoesStripFitInLine( bool checkForRubi,
1318  size_t numGlyphsInStrip )
1319 {
1320  using namespace t4::literals;
1321 
1322  if( m_unionedLineDRR.IsInvalid() )
1323  return kStripIncompatible;
1324 
1325  auto avgTextAdvance = AbsVal( GetAvgTextAdvance() );
1326 
1327  // We need to get compatible rotations first just in case the union
1328  // changed directions (we may be skeptical of direction):
1329  DRR lineDRRCompat = m_lineDRR.CalcCompatibleRotation( m_unionedLineDRR );
1330  m_stripDRRCompat = m_stripDRR.CalcCompatibleRotation( m_unionedLineDRR );
1331  const auto lineWidth = lineDRRCompat.Width();
1332  const auto stripWidth = m_stripDRRCompat.Width();
1333 
1334  // Do a check for Rubi, which is smaller explanatory text over the main text
1335  if( checkForRubi )
1336  {
1337  // We would like to also claim Rubi should not stick out on either side,
1338  // of the existing text, but sadly that rule is violated by ID PDF gen,
1339  // it puts the Rubi in first. Really all we can require is that the
1340  // text be "small and adjacent in non-baseline direction" (above or below)
1341  if( lineWidth > 8 * stripWidth / 5 ) // smaller
1342  if( m_unionedLineDRR.Width() < lineWidth + 2 * stripWidth )
1343  // above or below, with at most a "rubi sized" gap
1344  return kStripExtendsAsRubi;
1345  }
1346 
1347  // Make sure the two heights are compatible, in other words that one is not
1348  // significantly shorter than the other. Note that, in general, we can't test
1349  // to see if the strip is significantly shorter than the existing line because
1350  // drop caps are supported and they wouldn't pass the test. However, we will
1351  // only support the case in which drop caps are actual text, as opposed to
1352  // images. Otherwise, we could end up accepting text on the same line with
1353  // standalone images in a situation that did *not* involve drop caps (for an
1354  // example file, see Watson #1736533.) Should we want to support drop caps
1355  // that are images, we would need to further enhance this logic.
1356  if( 2 * lineWidth < stripWidth ||
1357  ( m_beginsWithFigure && 2 * stripWidth < lineWidth ) )
1358  return kStripIncompatible; // One of the strips was less than half as tall as the other
1359 
1360  // Make sure the two rectangles at least overlapped somewhat in the minor
1361  // direction:
1362  const auto widthSum = lineWidth + stripWidth;
1363  const auto unionWidth = m_unionedLineDRR.Width();
1364  if( widthSum + real_services::Epsilon( unionWidth ) < unionWidth )
1365  return kStripIncompatible; // They did not overlap at all in the minor direction
1366 
1367  // Make sure we're not encountering a linefeed:
1368  if( unionWidth / widthSum > t4::real_coord(0.9f) )
1369  return kStripIncompatible; // They were on separate lines
1370 
1371  if( avgTextAdvance == 0 )
1372  {
1373  // If we got here, it must mean that we have nothing but figure(s) on
1374  // this line. Unfortunately the code below relies on avgTextAdvance
1375  // not being zero.
1376 
1377  // Attempt to estimate avgTextAdvance from the new strip:
1378  if( numGlyphsInStrip > 0 )
1379  avgTextAdvance = m_stripDRR.Length() / static_cast<std::uint32_t>(numGlyphsInStrip);
1380 
1381  if( avgTextAdvance == 0 )
1382  {
1383  if( m_stripDRRCompat.GetMinInDir() < lineDRRCompat.GetMaxInDir() )
1384  return kStripIncompatible;
1385  else
1386  return kStripSimplyExtends;
1387  }
1388  }
1389 
1390  // Compute how close they were to abutting in the major direction.
1391  // the result below will be positive if there is a gap between the
1392  // right edge of a and the left edge of b, or negative if they
1393  // overlap.
1394  auto overlapOrGap = m_stripDRRCompat.GetMinInDir() - lineDRRCompat.GetMaxInDir();
1395 
1396  // and they can skip ahead a few glyphs but not too many, except after a bullet
1397  auto maxGap = ( m_lastVisibleCharIsBullet ) ?
1398  avgTextAdvance * 10 : avgTextAdvance * 3;
1399  auto inferredSpaceGap = ( 4 * avgTextAdvance ) / 10; // Estimated skinny space width
1400 
1401  // The strings overlap if overlapOrGap is negative:
1402  if( overlapOrGap < 0 )
1403  {
1404  // Let characters kern by slightly more than a whole glyph (rotated
1405  // vertical is problematic here):
1406  const auto maxOverlap = avgTextAdvance * 1.375_r;
1407  if( -overlapOrGap > maxOverlap )
1408  return kStripIncompatible;
1409  }
1410  else if( overlapOrGap > maxGap ) // We have a gap
1411  return kStripExtendsWithLargeGap;
1412 
1413  return ( overlapOrGap >= inferredSpaceGap ) ?
1414  kStripExtendsWithInferredSpace : kStripSimplyExtends;
1415 } // InferredLine::DoesStripFitInLine

1675 ReadOrderType InferredLine::TryAddString(
1676  const StringSummary& strSummary,
1677  const t4::pdf::content_point& curContentPt,
1678  const t4::pdf::content_point& prevContentPt,
1679  bool allowRubi )

1682 
1683  if( m_isEmpty )
1684  return kStripBegins;
1685 
1686  // Both the line and the prospective string must have valid Unicode maps,
1687  // or neither must have valid Unicode maps:
1688  if( strSummary.HasValidUnicodeMap() == m_beginsWithInconvertibleString )
1689  return kStripIncompatible;
1690 
1691  // Don't add visible glyphs to an invisible line (gets rid of leading
1692  // whitespace):
1693  if( strSummary.GetNumVisibleGlyphs() != 0 && m_beginsWithInvisibleString )
1694  return kStripIncompatible;
1695

1697  || ( !strSummary.HasValidUnicodeMap()
1698  && m_contentRanges.back().m_cr.m_to.m_dlEntryNum == curContentPt.m_dlEntryNum ) )
1699  {
1700  // Watson 2329507: don't add disjoint invisible strings to figures:
1701  if( m_isFigureLike )
1702  return kStripIncompatible;
1703  else
1704  return kStripExtendsAsInvisibleOrInvalid;
1705  }
1706 
1707  if( !m_blockContext.IsCompatible( strSummary.m_info.m_userSpaceToDeviceSpaceMat ) )
1708  return kStripIncompatible;
1709 
1710  // Only permit adding the string as TCY text if the text so far includes
1711  // CJK characters:
1712  const bool allowTCY = m_flags & kHasCJK;
1713  SetStrip( strSummary.GetDRR(), strSummary.GetTrimmedPunctDRR(), allowTCY );
1714 
1715  // This should only happen when there are no real glyphs and that case
1716  // should've been caught above:
1717  tetraphilia_assert( !m_stripDRR.IsInvalid() );
1718  tetraphilia_assert( get_x( strSummary.GetStartPoint() ) != 0 || get_y( strSummary.GetStartPoint() ) != 0 ||
1719  get_x( strSummary.GetEndPoint() ) != 0 || get_y( strSummary.GetEndPoint() ) != 0 );
1720 
1721  ReadOrderType result;
1722  if( m_unionedTextDRR.IsInvalid()
1723  || m_unionedLineDRR.IsInvalid()
1724  || m_unionedLineTrimmedPunctDRR.IsInvalid()
1725  || m_unionedLineWithRubiDRR.IsInvalid() )
1726  result = kStripIncompatible;
1727  else
1728  {
1729  const bool checkForRubi = allowRubi
1730  && strSummary.HasValidUnicodeMap()
1731  && ( ( strSummary.GetFlags() & ( kHasHiragana | kHasKatakana ) )
1732  || ( strSummary.GetFirstVisibleChar() ==
1733  0xd7 /* multiplication symbol */ ) ); // Watson 1889737
1734  result = DoesStripFitInLine( checkForRubi, strSummary.GetNumVisibleGlyphs() );
1735

1737  {
1738  // We treat inferred spaces as special ligatures ( [<prevChar>|<inferredSpace>] ) where for the inferred space the unicodeNum is
1739  // equal to the totalNumUnicode.
1740  auto inferredSpaceContentPt = prevContentPt;
1741  inferredSpaceContentPt.AdvanceUnicodeNum();

1743  }
1744  }

1746  {
1747  if( result == kStripIncompatible )
1748  return kStripDisjointAndInvisible;
1749  else
1750  return kStripExtendsAsInvisibleOrInvalid;
1751  }
1752  else
