33 /********************************************************************************************
34 
35  Number parsing and pushing
36 
37  This code deals with several independent problems. First, the number type, as expressed
38  by the ascii we are parsing may indicate:
39 
40  1) an integer that can be represented in 32 bits
41  2) an integer that can be represented in 64 bits, but not 32
42  3) an integer whose representation requires more than 64 bits
43  4) a real
44 
45  Second, the compile environment may have:
46 
47  1) pdf_traits::int_type and pdf_traits::large_int_type set to 32 bits
48  2) pdf_traits::int_type set to 32 bits and pdf_traits::large_int_traits set to 64 bits
49  3) pdf_traits::int_type and pdf_traits::large_int_type set to 64 bits
50 
51  Third, the compile environment may have:
52 
53  1) real_type set to an IEEE float or double (overflow not possible)
54  2) real_type set to Fixed16_16 (overflow is possible)
55 
56  Some assumptions:
57 
58  a) It is always preferable to parse an integer as an int_type over a large_int_type
59  b) If int_type and large_int_type are the same thing we don't want have two duplicates
60  of the same code -- one trying to parse as an int_type, and another trying to parse
61  as a large_int_type
62 
63  The code below uses template specialization to try to produce optimal code in
64  all of the above possibilities while correctly handling all of the issues. It uses
65  template specialization strategies to get the code it wants.
66 
67 ********************************************************************************************/

137 /*
138  The next 4 templates are used in a List, where the one chosen from the
139  list is the one that meets several criteria. They each have a method
140  named Push that takes a stack (the thing we are going to push onto) and
141  an integer-like value. The implementation of each one calls either
142  PushInt or PushLargeInt depending on which one is the correct thing to
143  call.
144 */
145 
146 /*
147  This is the one used when the value is known to be too large to be represented
148  in either the int_type or large_int_type. So we just push the min/max large_int_type.
149  The OverflowIntPusher is chosen when the sizeof(large_int_type) == 4 and the
150  number is larger than the largest 32 bit integer.
151 */
152 struct OverflowIntPusher
153 {
154  typedef std::int32_t type;
155  enum { valid = 2 };
156 
157  template<typename stack_type, typename T>
158  static void Push( stack_type& stack, T val )
159  {
160  if ( val < 0 )
161  PushLargeInt( stack, limits<std::int64_t>::min() );
162  else
163  PushLargeInt( stack, limits<std::int64_t>::max() );
164  }
165 };

173  typedef std::int32_t type;

231 template<int intSize, bool doIt>
232 struct ParseAndPushNumber
233 {
234  template<typename stack_type>
235  static bool Do( const char* p, size_t len, size_t tlen, bool negate, data_io::BufferedStream& stream, stack_type& stack )
236  {
237  using namespace boost::mp11;
238  typedef typename NumberParserHelper<intSize>::int_type int_type;
239  typedef t4::real real_type;
240 
241  typedef IntPusherList int_pusher_list_type;
242  typedef mp_at<int_pusher_list_type, mp_find_if_q<int_pusher_list_type, CanPush_f<intSize>>> int_pusher_type;
243 
244  if ( NumberParserHelper<intSize>::LessThanMax( p, tlen ) ) {
245  int_type result = 0;
246  for ( size_t i = 0; i < tlen; i++ )
247  result = result * 10 + (*p++ - '0');
248  stream.Advance( len );
249  if ( !stream.AtEOF() && (stream.Peek() == '.') ) {
250 
251  stream.Advance();
252  len = stream.PeekBytes( p, 0, Parser::m_ByteTypes, Parser::kNotDigit );
253  real_type real_res;
254 
255  tetraphilia_assert( result >= 0 );
256  if ( IntegerToRealConversion<int_type, real_type>::ConversionOfNonNegativeIntToRealWillOverflow( result ) )
257  {
258  // too large to fit in real_type, use MAX or MIN real_type value
259  real_res = limits<real_type>::max();
260 
261  } else {
262 
263  tlen = len;
264  if ( tlen > real_services::DivisionByPowersOfTen<std::int32_t, real_type>::kMaxDecimalDigits )
265  tlen = real_services::DivisionByPowersOfTen<std::int32_t, real_type>::kMaxDecimalDigits;
266 
267  real_res = ParseIntToReal<int_type, real_type>::Do(result);
268  if( tlen != 0 ) {
269  std::int32_t frac_res = 0;
270  for ( size_t i = 0; i < tlen; i++ )
271  frac_res = frac_res * 10 + *p++ - '0';
272 
273  real_res += real_services::DivisionByPowersOfTen<std::int32_t, real_type>::Compute( frac_res, tlen );
274  }
275 
276  }
277 
278  if ( negate )
279  real_res = -real_res;
280  stream.Advance( len );
281 
282  PushReal( stack, real_res );
283 
284  } else {
285  if ( negate )
286  result = -result;
287 
288  int_pusher_type::Push( stack, result );
289  }

292  PDF_SLANG
293 
294  Watson bug#1303216 contains a pdf file created by the program FineReader, which, according
295  to the Abbyy website is an "award winning" program for creating pdf files. It contains
296  a bunch of numbers which look something like:
297 
298  /F8 8.-480 Tf
299 
300  (note the '-' after the '8.') It isn't clear what kind of bug generated this sort of output.
301  But the desktop Acrobat number parser essentially skips to the end of the token once it finds
302  something that doesn't make sense in its number parsing. In permissive builds we do the same.
303  */
304  #ifdef TETRAPHILIA_PDF_PERMISSIVE
305  Parser::SkipToTokenEnd( stream );
306  #else
307  if ( !stream.AtEOF() && !Parser::IsDelimiterOrWhiteSpace(stream.Peek()) )
308  throw t4::corrupt_data();
309  #endif

312  }
313  return false;
314  }

335 template<typename stack_type>
336 void Parser::ParseNumber( data_io::BufferedStream& stream,
337  stack_type& stack )
338 {
339  char c = stream.Peek();
340  bool negate;
341 
342  if ( c == '-' || c == '+' )
343  {
344  negate = ( c == '-' );
345  stream.Advance();
346  }
347  else
348  negate = false;
349 
350  const char* p;
351  size_t len = stream.PeekBytes( p, 0, m_ByteTypes, kNotDigit );
352  size_t tlen = len;
353  while ( (tlen > 1) && (*p == '0') ) {
354  ++p;
355  --tlen;
356  }
357 
358  static constexpr auto kWorthTrying = sizeof(std::int64_t) > sizeof(std::int32_t);
359  if ( ParseAndPushNumber<sizeof(std::int32_t), true>::Do( p, len, tlen, negate, stream, stack )
360  || ParseAndPushNumber<sizeof(std::int64_t), kWorthTrying>::Do( p, len, tlen, negate, stream, stack ) )
361  return;
362 
363  // Integer portion of the number is too big to fit into whatever the largest integer size we are using
364  // just push the biggest/smallest number and skip to the end of the token. This doesn't quite handle
365  // certain invalid files correctly (that is, it won't throw in all the "correct" cases). But it is simple
366  // to implement and probably matches desktop Acrobat
367  if ( negate )
368  PushLargeInt( stack, limits<std::int64_t>::min() );
369  else
370  PushLargeInt( stack, limits<std::int64_t>::max() );
371  SkipToTokenEnd( stream );
372 }

374 template<typename stack_type>
375 bool Parser::DoStringImpl( stack_type& stack )
376 {
377  int openParenCount = 1;
378 
379  const char* ptr;
380  size_t numPeeked = 1;
381 
382  while ( true )
383  {
384  numPeeked = m_parserStream->PeekBytes( ptr, numPeeked, m_ByteTypes, kStringSpecialChar );
385  size_t avail = m_parserStream->PeekBytes( ptr, numPeeked + 1 );
386  if ( avail <= numPeeked ) // EOF
387  throw t4::corrupt_data();
388 
389  if ( ptr[numPeeked] == ')' ) {
390  if ( --openParenCount == 0 )
391  break;
392  } else if ( ptr[numPeeked] == '\\' )
393  // whether the next character is a paren or not, it cannot
394  // possibly terminate the string, so we skip it.
395  numPeeked++;
396  else {
397  tetraphilia_assert( ptr[numPeeked] == '(' );
398  openParenCount++;
399  }
400 
401  numPeeked++;
402  }
403 
404  // skip the open paren
405  size_t rawLen = numPeeked - 1;
406  t4::small_vector<char, 64> strBuff( rawLen );
407  size_t strLen = 0;
408  const char* start = ptr + 1;
409  const char* end = start + rawLen;
410  char* buf = strBuff.data();
411  int result;
412  t4::pdf::string_decrypter* strDecrypt = CreateStringDecrypter();
413  while( ( result = ScanLiteralChar( start, end ) ) != -1 )
414  {
415  *buf++ = static_cast<char>(result);
416  ++strLen;
417  }
418  buf = strBuff.data();
419  if( strDecrypt )
420  strDecrypt->decrypt( reinterpret_cast<unsigned char *>(buf), strLen, reinterpret_cast<unsigned char *>(buf), strLen );
421 
422  auto strObj = MakeStringObj( GetHeap( stack ), { strBuff.data(), strLen }, hex_format::no );
423  PushString( stack, strObj );
424  m_parserStream->Advance( numPeeked + 1 );
425  return m_continueParsing;
426 }

428 template<typename stack_type>
429 bool Parser::DoNameImpl( stack_type& stack )
430 {
431  const char* pstart;
432  size_t numPeeked = m_parserStream->PeekBytes( pstart, 1/* skip the leading "/" */, m_ByteTypes, kWhiteSpace | kDelimiter );
433  if (numPeeked == 0) // EOF
434  throw t4::corrupt_data();
435  numPeeked -= 1;
436  t4::small_vector<char, 64> nameBuff( numPeeked );
437 
438  //TODO: use PeekBytes to do the hex processing since there we already need to traverse the name character by character
439  //This would require adding another mask bit for kHexDigit and passing the mask to LocateChar.
440  const char* p = pstart+1;
441  const char* pend = p + numPeeked;
442  char* b = nameBuff.data();
443  size_t len = numPeeked;
444  while( p < pend )
445  {
446  if( *p == '#' )
447  {
448  ++p;
449  int result = 0;
450  if( p < pend && IsHexDigit( *p ) )
451  {
452  result = GetHexValue( *p++ );
453  --len;
454  if( p < pend && IsHexDigit( *p ) )
455  {
456  result = 16 * result + GetHexValue( *p++ );
457  --len;
458  }
459  }
460  tetraphilia_assert( 0 <= result && result <= 255 );
461  *b++ = static_cast<char>(result);
462  }
463  else
464  *b++ = *p++;
465  }
466 
467  auto nameObj = MakeStringObj( GetHeap( stack ), { nameBuff.data(), len }, hex_format::no );
468  PushName( stack, nameObj );
469  m_parserStream->Advance( numPeeked+1 );
470  return m_continueParsing;
471 }

514 {

516  size_t len = m_parserStream->PeekBytes( pstart, 0, m_ByteTypes, kNotHexDigit );

524  bool even = true;
525  size_t strLen = 0;

527  {
528  if( even )
529  {

531  if( IsHexDigit( *p ) )

538  }
