40 /*
41  Count the number of "on" bits in a byte (this is only used at compile time to build the s_bit_counts table,
42  so it doesn't need to be efficient).
43 */
44 constexpr size_t do_bit_count( std::uint8_t b )
45 {
46  // Kernighan method
47  size_t result = 0;
48  for ( ; b != 0; ++result )
49  b &= b - 1;
50  return result;
51 }

53 /*
54  Shift the given integral type d bits to the left and return the result as a Word (the vanilla operator<<
55  does not necessarily return a Word, it treats this like any other binary operation and does integral promotion).
56 */
57 template<typename Word>
58 constexpr Word shift_left( Word w, size_t d )
59 {
60  return static_cast<Word>( w << d );
61 }

63 /*
64  Shift the given integral type d bits to the right and return the result as a Word (the vanilla operator>>
65  does not necessarily return a Word, it treats this like any other binary operation and does integral promotion).
66 */
67 template<typename Word>
68 constexpr Word shift_right( Word w, size_t d )
69 {
70  return static_cast<Word>( w >> d );

73 /*
74  Return the complement of w as a Word (operator~ will promote to an int, which is not always what we want).
75 */
76 template<typename Word>
77 constexpr Word complement( Word w )

79  return static_cast<Word>( ~w );

102 /*
103  Get the value of the bit at the specified index in the provided word.
104 */
105 template<typename Word>
106 constexpr bool bit_value( Word word, size_t bit_index_into_word )
107 {
108  constexpr Word one = 1;
109  return bool( word & shift_left( one, bit_index_into_word ) );
110 }

112 /*
113  class that behaves like a "reference" to a single bit in a bitset.
114 */
115 template<size_t N, typename Word>
116 class reference
117 {
118  constexpr static Word one = 1;
119  Word* m_word;
120  size_t m_bit_index_into_word;
121 
122 public:
123  constexpr reference( void ) noexcept
124  : m_word( nullptr ),
125  m_bit_index_into_word( 0 )
126  {}
127 
128  constexpr reference( Word* word, size_t bit_index_into_word ) noexcept
129  : m_word( word ),
130  m_bit_index_into_word( bit_index_into_word )
131  {}
132 
133  constexpr operator bool( void ) const noexcept
134  {
135  return bit_value( *m_word, m_bit_index_into_word );
136  }
137 
138  constexpr reference operator=( bool value ) noexcept
139  {
140  if ( value )
141  *m_word |= shift_left( one, m_bit_index_into_word );
142  else
143  *m_word &= complement( shift_left( one, m_bit_index_into_word ) );
144  return *this;
145  }
146 };

148 /*
149  Private class that provides all the algorithmic work for bitset.
150 */
151 template<size_t N, typename Word>
152 class bitset_impl
153 {
154  static_assert( std::is_integral<Word>::value && std::is_unsigned<Word>::value, "" );
155 
156  constexpr static size_t bits_per_word = 8 * sizeof( Word );
157  constexpr static size_t num_words = ( N + bits_per_word - 1 ) / bits_per_word;
158  constexpr static Word word_none = 0;
159  constexpr static Word word_all = complement( word_none );
160  constexpr static Word one = 1;
161  constexpr static Word zero = 0;
162  constexpr static Word last_word_mask = shift_right( word_all, ( bits_per_word - ( N % bits_per_word ) ) % bits_per_word );
163 
164  // As a matter of policy, we keep all the unused bits set to 0 to keep
165  // comparison fast.
166  std::array<Word, num_words> m_words;
167 
168 public:
169  using reference = reference<N, Word>;
170 
171  constexpr bitset_impl( void ) noexcept
172  : m_words{}
173  {}
174 
175  constexpr bitset_impl( std::initializer_list<bool> const& list ) noexcept
176  : m_words{}
177  {
178  assert(( list.size() == N ));
179  size_t i = N;
180  for ( auto b : list )
181  (*this)[--i] = b;
182  }

184  template<typename UInteger, std::enable_if_t<std::is_unsigned<UInteger>::value, int> = 0>
185  constexpr explicit bitset_impl( UInteger bits )
186  : m_words{}
187  {
188  using namespace t4::literals;
189  if ( sizeof( Word ) < sizeof( UInteger ) )
190  {
191  constexpr auto stop_index = std::min( sizeof( UInteger ) / sizeof( Word ), num_words );
192  for ( auto i = 0_z; i < stop_index; ++i )
193  {
194  auto word_mask = ( i == stop_index - 1 ) ? last_word_mask : word_all;
195  get_word_at_index( i ) = static_cast<Word>( bits & word_mask );
196  bits >>= bits_per_word;
197  }
198  }
199  else
200  get_word_at_index( 0 ) = static_cast<Word>( bits & last_word_mask );
201  }

267  constexpr size_t count( void ) const noexcept
268  {
269  size_t result = 0;
270  for ( size_t i = 0; i < num_words; ++i )
271  {
272  auto w = m_words[i];
273  for ( size_t j = 0; j < sizeof( Word ); ++j )
274  {
275  result += bit_count_of_byte( static_cast<std::uint8_t>( w & Word( 0xFF ) ) );
276  w = shift_right( w, 8 );

278  }

322  constexpr bitset_impl operator~( void ) const noexcept
323  {
324  size_t bits_remaining = N;
325  bitset_impl result;
326 
327  size_t i = 0;
328 
329  while ( bits_remaining )
330  {

332  {

334  bits_remaining -= bits_per_word;
335  }
336  else

347  constexpr bitset_impl& operator>>=( size_t pos ) noexcept
348  {
349  size_t word_shift = pos / bits_per_word;
350  size_t bit_shift = pos % bits_per_word;
351 
352  for ( size_t i = 0; i < num_words; ++i )
353  {
354  if ( ( i + word_shift ) < num_words )
355  {
356  get_word_at_index( i ) = shift_right( get_word_at_index( i + word_shift ), bit_shift );
357  if ( ( i + word_shift + 1 ) < num_words && bit_shift > 0 )
358  get_word_at_index( i ) |= shift_left( get_word_at_index( i + word_shift + 1 ), bits_per_word - bit_shift );
359  }
360  else
361  get_word_at_index( i ) = 0;
362  }
363 
364  return *this;
365  }

367  constexpr bitset_impl& operator<<=( size_t pos ) noexcept
368  {
369  size_t word_shift = pos / bits_per_word;
370  size_t bit_shift = pos % bits_per_word;
371 
372  for ( size_t i = num_words; i-- > 0; )
373  {
374  if ( i >= word_shift )
375  {
376  get_word_at_index( i ) = ( shift_left( get_word_at_index( i - word_shift ), bit_shift ) );
377  if ( i > word_shift && bit_shift > 0 )
378  get_word_at_index( i ) |= shift_right( get_word_at_index( i - word_shift - 1 ), bits_per_word - bit_shift );
379  }
380  else
381  get_word_at_index( i ) = 0;
382  }
383 
384  // clear any trailing bits
385  if ( num_words > 0 )
386  get_word_at_index( num_words - 1 ) &= last_word_mask;
387 
388  return *this;
389  }
