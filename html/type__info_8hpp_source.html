294 // List of supported Java integer types
295 typedef mp11::mp_list<jbyte, jchar, jshort, jint, jlong, jsize> jint_list;
296 
297 // ordering function for the sorted list
298 template<typename T1, typename T2>
299 struct sz_lt : public std::integral_constant<bool, ( sizeof( T1 ) < sizeof( T2 ) )> {};
300 
301 // Sort the list by size
302 typedef mp11::mp_sort<jint_list, sz_lt> sorted_int_list;
303 
304 // Helper function for is_superset
305 template<typename To, typename From>
306 using is_superset_base = std::integral_constant<bool,
307  std::is_signed<From>::value
308  ? std::is_signed<To>::value && sizeof( To ) >= sizeof( From )
309  : std::is_signed<To>::value
310  ? ( sizeof( To ) > sizeof( From ) )
311  : ( sizeof( To ) >= sizeof( From ) )>;
312 
313 // Template metafunction that returns true if all values of type From are
314 // representable by type To
315 template<typename To, typename From>
316 struct is_superset : public is_superset_base<std::decay_t<To>, std::decay_t<From>>
317 {};

319 // Find the first (smallest) Java integral type that is a superset of the specified type
320 template<typename Integer>
321 struct best_java_integer_type
322 {
323  typedef mp11::mp_find_if_q<sorted_int_list, mp11::mp_bind<is_superset, mp11::_1, Integer>> match_index;
324 
325  // If we couldn't find a compatible type, use jlong
326  typedef mp11::mp_eval_if_c<match_index::value == mp11::mp_size<sorted_int_list>::value,
327  jlong,
328  mp11::mp_at, sorted_int_list, match_index> type;
329 };

461 /*
462  ntype_info for std::array. Converts to and from Java arrays of the java type corresponding
463  to T.
464 */
465 template<typename T, size_t N>
466 struct ntype_info<std::array<T, N>>
467  : public ntype_array_info<T>
468 {
469  typedef ntype_info<T> element_ntype_info;
470  typedef typename element_ntype_info::jtype element_jtype;
471  typedef typename jni_detail::jarray_type<element_jtype>::type jtype;
472 
473  static std::array<T, N> to_native( JNIEnv* env, jtype jarr )
474  {
475  auto sz = env->GetArrayLength( jarr );
476  if ( sz != N )
477  throw std::invalid_argument( "Mismatched array lengths" );
478  java_array_elements<element_jtype> elems( env, jarr );
479  std::array<T, N> result;
480  size_t i = 0;
481  elems.template for_each_element<T>( [&]( T e ){ result[ i++ ] = std::move( e ); } );
482  return result;
483  }
484 
485  static jtype from_native( JNIEnv* env, std::array<T, N> const& arr )
486  {
487  auto result = create_java_array<T>( env, N );
488  java_array_elements<element_jtype> elems( env, result );
489  elems.write_elements( arr.begin(), arr.end() );
490  return result;
491  }
492 };

495  ntype_info for std::vector. Converts to and from Java arrays of the java type corresponding

505  static std::vector<T, Allocator> to_native( JNIEnv* env, jtype jarr )

508  java_array_elements<element_jtype> elems( env, jarr );
509  std::vector<T, Allocator> result;
