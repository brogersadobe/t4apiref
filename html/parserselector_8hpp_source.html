49  // Create the font parser
50  static void CreateFontParser(font_ptr& rslt, typename fonts::Font<app_traits>::FileFormat format, typename app_traits::context_type& context,
51  t4::data_store store, bool verifyDataCanFitInMemory)
52  {
53  typedef typename app_traits::pdf_traits pdf_traits;
54 
55  try {
56 
57  if ( verifyDataCanFitInMemory ) {
58  // if we are asked to verify that the data fits into memory then we loop through
59  // the data, storing pointers to the data blocks in a stack, until we get to the
60  // end. If this runs out of memory it will throw an exception which we catch
61  // below.
62  typedef typename app_traits::file_system_traits::file_size_type file_size_type;
63  Stack<TransientAllocator<app_traits>, t4::data_block> stack{ TransientAllocator<app_traits>(*t4::get_thread_heap()) };
64  file_size_type pos = 0;
65  while ( true ) {
66  t4::data_block block = store.get_block( pos );
67  if ( block.get_size() == 0 )
68  break;
69  stack.Push( block );
70  pos += block.get_size();
71  }
72  }
73 
74  if( format == fonts::Font<app_traits>::kCFF ) {
75 #ifdef TETRAPHILIA_USE_FREETYPE
76  rslt = t4::make_sptr<fonts::parsers::FreeType<app_traits>>(format, context, store, fonts::parsers::GetFreeTypeLib());
77 #else
78  rslt = t4::make_sptr<fonts::parsers::CFF<app_traits>>(context, store);
79 #endif
80  }
81  else if( format == fonts::Font<app_traits>::kTrueType)
82  {
83 #ifdef TETRAPHILIA_USE_FREETYPE
84  rslt = t4::make_sptr<fonts::parsers::FreeType<app_traits>>(format, context, store, fonts::parsers::GetFreeTypeLib());
85 #else
86  rslt = t4::make_sptr<fonts::parsers::TrueType>(store);
87 #endif
88  }
89  else if ( format == fonts::Font<app_traits>::kType1 )
90 #ifdef TETRAPHILIA_USE_FREETYPE
91  rslt = t4::make_sptr<fonts::parsers::FreeType<app_traits>>(format, context, store, fonts::parsers::GetFreeTypeLib());
92 #else
93  rslt = t4::make_sptr<fonts::parsers::Type1<app_traits>>(context, store);
94 #endif
95  else
96  {
97  if ( !pdf_traits::permissive )
98  throw t4::corrupt_data();
99  }
100  }
101  /*
102  In permissive builds we swallow all corrupt data errors that occur during font parsing. This makes it
103  more like desktop Acrobat. Watson bug#1280599 contains an example of a pdf file with a damaged
104  TrueType font embedded in it. In this case the font is damanged to the point where we can't use it at all.
105  */
106  catch( const t4::corrupt_data& )
107  {
108  if ( !pdf_traits::permissive )
109  throw;
110  // else swallow the error
111  }
112  catch( std::bad_alloc )
113  {
114  if ( !verifyDataCanFitInMemory )
115  throw;
116  // else swallow the error
117  }
118  }
119 
120 };
