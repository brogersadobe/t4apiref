152 class write_stream_impl : public write_stream_impl_base
153 {
154  Device m_device;
155  bool m_closed = false;
156  bool m_error = false;
157  size_t m_bytes_written = 0;
158 
159 public:

161  : m_device(std::move(device))
162  {
163  }
164 
165  ~write_stream_impl() override
166  {
167  if (!m_closed)
168  close_impl();
169  }
170 
171 protected:
172  void write_impl(const void* buffer, size_t length) override
173  {
174  if (m_closed || m_error)
175  throw std::runtime_error(T4_EXCEPTION_STR("writing to a closed stream"));
176 
177  try
178  {
179  write(m_device, buffer, length);
180  m_bytes_written += length;
181  }
182  catch (...)
183  {
184  t4::exception_ptr ex = t4::current_exception();
185  try
186  {
187  set_error(m_device, ex);
188  }
189  catch (...)
190  {
191  }
192  t4::rethrow_exception(ex);
193  }
194  }
195 
196  void flush_impl() override
197  {
198  if (m_closed || m_error)
199  throw std::runtime_error(T4_EXCEPTION_STR("flushing a closed stream"));
200 
201  try
202  {
203  flush(m_device);
204  }
205  catch (...)
206  {
207  t4::exception_ptr ex = t4::current_exception();
208  try
209  {
210  set_error(m_device, ex);
211  }
212  catch (...)
213  {
214  }
215  t4::rethrow_exception(ex);
216  }
217  }
218 
219  size_t bytes_written_impl() const override
220  {
221  return m_bytes_written;
222  }
223 
224  void set_error_impl(t4::exception_ptr const& ex) override
225  {
226  if (!ex)
227  return;
228 
229  if (!m_closed && !m_error)
230  {
231  m_error = true;
232  try
233  {
234  // Disable interruptions. This way we can eat all exceptions thrown and not worry about missing an
235  // interruption.
236  boost::this_thread::disable_interruption di;
237  set_error(m_device, ex);
238  }
239  catch (...)
240  {
241  }
242  }
243  }
244 
245  void close_impl() override
246  {
247  if (!m_closed)
248  {
249  // Disable interruptions. This way we can eat all exceptions thrown and not worry about missing an
250  // interruption.
251  boost::this_thread::disable_interruption di;
252 
253  m_closed = true;
254  if (!m_error)
255  {
256  try
257  {
258  flush(m_device);
259  }
260  catch (...)
261  {
262  }
263  }
