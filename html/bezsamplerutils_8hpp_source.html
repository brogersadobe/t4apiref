409 template<typename sampler_def, typename edgeadder_type>
410 class OverscanSamplerEdgeAdderHelper
411 {
412  typedef typename sampler_def::sampler_edge_type Edge;
413 public:
414  static void AddNewSamplerEdge(edgeadder_type *ea, Edge *newEdge)
415  {
416  if (!ea->GetFirstEdge())
417  {
418  ea->SetFirstEdge(newEdge);
419  ea->SetPrevEdge(newEdge);
420  }
421  else
422  {
423  Edge *prevEdge = ea->GetPrevEdge();
424  if (prevEdge->getDirection() == 0 || prevEdge->getDirection() == newEdge->getDirection())
425  {
426  prevEdge->updateDirection(newEdge->getDirection());
427  if (prevEdge->getDirection() == 1)
428  newEdge->setSuccessor(prevEdge);
429  else
430  prevEdge->setSuccessor(newEdge);
431  }
432  else if (newEdge->getDirection() == 0)
433  {
434  newEdge->updateDirection(prevEdge->getDirection());
435  if (prevEdge->getDirection() == 1)
436  newEdge->setSuccessor(prevEdge);
437  else
438  prevEdge->setSuccessor(newEdge);
439  }
440  else
441  {
442  if (prevEdge->getDirection() == 1) // going up
443  {
444 #ifdef BEZSAMPLER_DEBUG
445  if (detail::debugging)
446  cache_logger(TEXT("Add local minima %x %x\n"),prevEdge,newEdge);
447 #endif
448  ea->AddEdgeToEdgeVector(prevEdge);
449  ea->AddEdgeToEdgeVector(newEdge); // local minima
450  }
451  }
452  tetraphilia_assert(ea->GetPrevEdge() == prevEdge); // not changed since it was fetched
453  ea->SetPrevEdge(newEdge);
454  }
455  }
456 
457  static void ClosePath(edgeadder_type *ea)
458  {
459  // firstEdge/preveEdge NULL if all horizontal segments (can that happen?)
460  Edge *firstEdge = ea->GetFirstEdge();
461  if (firstEdge)
462  {
463  Edge *prevEdge = ea->GetPrevEdge();
464  if (prevEdge->getDirection() == 1 && firstEdge->getDirection() != 1)
465  {
466  ea->AddEdgeToEdgeVector(firstEdge); // local minima
467  ea->AddEdgeToEdgeVector(prevEdge); // local minima
468  }
469  else if (firstEdge == prevEdge)
470  {
471  // one flat edge in a single scanline
472  ea->AddEdgeToEdgeVector(firstEdge);
473  }
474  else if (prevEdge->getDirection() == firstEdge->getDirection())
475  {
476  if (prevEdge->getDirection() == 1)
477  firstEdge->setSuccessor(prevEdge);
478  else if (firstEdge->getSuccessor() == prevEdge)
479  {
480  // don't make a successor loop instead this is a two edge path this can be created in
481  // many different ways. generally these edges come from flattening curves in device space
482  // and not clipping them to the box. make the edges into an up/down path
483  // See bug 1430629 for one of the cases generating this condition
484  if (prevEdge->isHorizontalEdge())
485  {
486  prevEdge->setDirection(1);
487  // both edges can be horizontal
488  firstEdge->updateDirection(-1);
489  }
490  else
491  {
492  firstEdge->setDirection(1);
493  tetraphilia_assert(prevEdge->getDirection() == -1);
494  }
495  firstEdge->setSuccessor(nullptr);
496  ea->AddEdgeToEdgeVector(firstEdge); // local minima
497  ea->AddEdgeToEdgeVector(prevEdge); // local minima
498  }
499  else
500  prevEdge->setSuccessor(firstEdge);
501  }
502  // else its a local maxima and no work needs doing
503  tetraphilia_assert(firstEdge == ea->GetFirstEdge());
504  tetraphilia_assert(prevEdge == ea->GetPrevEdge());
505  }
506  }
507 };
