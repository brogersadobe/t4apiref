128 template<typename text_click_client_type>
129 void TextClickDLConsumer<text_click_client_type>::HandleUnicodeChar(
130  std::uint32_t /*c*/, bool /*isLastInShow*/ )
131 {
132  const content_point_type& contentPoint = this->m_curContentPoint;
133  point_type start = this->m_pShowInfo->GetDeviceSpaceBaselinePos( contentPoint.m_stringNum, contentPoint.m_glyphNum, contentPoint.m_unicodeNum, contentPoint.m_totalNumUnicode);
134  point_type end = this->m_pShowInfo->GetDeviceSpaceBaselinePos( contentPoint.m_stringNum,contentPoint.m_glyphNum, 1, 1);
135  if( ClickIsInside( this->m_pShowInfo->GetDeviceSpaceRoughBoundingBox( start, end ) ) )
136  {
137  real_type dStart = real_services::VectorLength(get_x(m_clickPoint) - get_x(start), get_y(m_clickPoint) - get_y(start) );
138  real_type dEnd = real_services::VectorLength(get_x(m_clickPoint) - get_x(end), get_y(m_clickPoint) - get_y(end) );
139  // I beieve all text points coming into here are set to before the glyph, so I don't work too hard here...
140  tetraphilia_assert(contentPoint.m_isAfter == false);
141  point_type ascentVec, descentVec;
142  this->m_pShowInfo->GetDeviceSpaceAscenderDescenderVectors( ascentVec, descentVec );
143  if (dStart < dEnd)
144  {
145  m_client.HandleTextClickFound(contentPoint, start, ascentVec, descentVec);
146  }
147  else
148  {
149  content_point_type after = contentPoint;
150  after.m_isAfter = true;
151  m_client.HandleTextClickFound(after, end, ascentVec, descentVec);
152  }
153  // do NOT stop consumption, there may be more than one glyph under the cursor.
154  }
155 }

161 {

248 struct TextSelectionRectangle
249 {
250  typedef document::PageID page_id_type;
251  typedef typename app_traits::real_geometry_traits::coord_type real_type;
252  typedef typename app_traits::real_geometry_traits::rect_type real_rect_type;
253  typedef typename app_traits::real_geometry_traits::point_type real_point_type;
254  typedef typename app_traits::pdf_traits::display_traits display_traits;
255  typedef reflow::ReflowLayout reflow_layout_type;

257  bool InsideRect( const real_rect_type& glyphRect, const real_rect_type& pageRect, TextSelectionInclusionPolicy policy ) const
258  {
259  switch( policy )
260  {
261  case kIncludeSmart:
262  case kIncludeIntersectingGlyphs:
263  case kIncludeIntersectingGlyphsAndImages:
264  // We can't use the standard RectIntersect here, because we allow pageRect to be empty,
265  // and treat it like a point.
266  return get_xmin( glyphRect ) <= get_xmax( pageRect )
267  && get_ymin( glyphRect ) <= get_ymax( pageRect )
268  && get_xmax( glyphRect ) >= get_xmin( pageRect )
269  && get_ymax( glyphRect ) >= get_ymin( pageRect );
270  default:
271  tetraphilia_assert( policy == kIncludeContainedGlyphs );
272  return imaging_model::RectContainsRect( pageRect, glyphRect );
273  }
274  }
275 
276  bool Inside( const page_id_type& pageID, const real_rect_type& glyphRect, TextSelectionInclusionPolicy policy ) const
277  {
278  if ( pageID == m_firstPage )
279  return InsideRect( glyphRect, m_firstPageBBox, policy );
280  else if ( pageID == m_lastPage && m_lastPage != m_firstPage )
281  return InsideRect( glyphRect, m_lastPageBBox, policy );
282  else
283  return ( m_firstPage < pageID && pageID < m_lastPage );
284  }
285 
286  #ifdef _MSC_VER
287  // For some reason VC.NET has a bug where it refuses to do Koenig lookup for real_rect_type and real_point_type
288  // in this one particular file.
289  template<typename rect_type>
290  typename imaging_model::RectangleTypeHelper<rect_type>::coord_type get_xmin( const rect_type& r ) const { return imaging_model::get_xmin( r ); }
291  template<typename rect_type>
292  typename imaging_model::RectangleTypeHelper<rect_type>::coord_type get_ymin( const rect_type& r ) const { return imaging_model::get_ymin( r ); }
293  template<typename rect_type>
294  typename imaging_model::RectangleTypeHelper<rect_type>::coord_type get_xmax( const rect_type& r ) const { return imaging_model::get_xmax( r ); }
295  template<typename rect_type>
296  typename imaging_model::RectangleTypeHelper<rect_type>::coord_type get_ymax( const rect_type& r ) const { return imaging_model::get_ymax( r ); }
297 
298  template<typename point_type>
299  typename imaging_model::PointTypeHelper<point_type>::coord_type get_x( const point_type& p ) const { return imaging_model::get_x( p ); }
300  template<typename point_type>
301  typename imaging_model::PointTypeHelper<point_type>::coord_type get_y( const point_type& p ) const { return imaging_model::get_y( p ); }
302  #endif
303 
304  real_type DistanceFromMouseDown( const page_id_type& pageID, const real_rect_type& glyphRect ) const
305  {
306  if ( pageID == m_mouseDownPage )
307  {
308  // We compute taxicab distance here, to avoid overflow
309  return AbsVal( ( get_xmin( glyphRect ) + get_xmax( glyphRect ) ) / 2 - get_x( m_mouseDownPoint ) )
310  + AbsVal( ( get_ymin( glyphRect ) + get_ymax( glyphRect ) ) / 2 - get_y( m_mouseDownPoint ) );
311  }
312  else
313  return limits<real_type>::max();
314  }
315 
316  real_type DistanceFromMouseUp( const page_id_type& pageID, const real_rect_type& glyphRect ) const
317  {
318  if ( pageID == m_mouseUpPage )
319  {
320  // We compute taxicab distance here, to avoid overflow
321  return AbsVal( ( get_xmin( glyphRect ) + get_xmax( glyphRect ) ) / 2 - get_x( m_mouseUpPoint ) )
322  + AbsVal( ( get_ymin( glyphRect ) + get_ymax( glyphRect ) ) / 2 - get_y( m_mouseUpPoint ) );
323  }
324  else
325  return limits<real_type>::max();
326  }
327 
328  bool IsAPoint( void ) const
329  {
330  return m_firstPage == m_lastPage && imaging_model::RectIsEmpty( m_firstPageBBox );
331  }
332 
333  page_id_type m_firstPage;
334  real_rect_type m_firstPageBBox;
335  reflow_layout_type* m_firstPageReflowLayout;
336  page_id_type m_lastPage; // INCLUSIVE range.
337  real_rect_type m_lastPageBBox;
338  reflow_layout_type* m_lastPageReflowLayout;
339 
340  // These identify exactly where the mouse down and mouse up occurred. They are used
341  // to fine-tune the result derived from the rectangles listed above.
342  page_id_type m_mouseDownPage;
343  real_point_type m_mouseDownPoint;
344  page_id_type m_mouseUpPage;
345  real_point_type m_mouseUpPoint;
346 };

494 template<typename _app_traits>
495 void TextSelectionFinder<_app_traits>::HandleUnicodeChar(
496  std::uint32_t /*c*/, bool /*isLastInShow*/ )
497 {
498  const content_point_type& contentPoint = this->m_curContentPoint;
499  point_type start = this->m_pShowInfo->GetDeviceSpaceBaselinePos( contentPoint.m_stringNum, contentPoint.m_glyphNum, contentPoint.m_unicodeNum, contentPoint.m_totalNumUnicode);
500  point_type end = this->m_pShowInfo->GetDeviceSpaceBaselinePos( contentPoint.m_stringNum,contentPoint.m_glyphNum, 1, 1);
501 
502  real_rect_type glyphRect = this->m_pShowInfo->GetDeviceSpaceRoughBoundingBox( start, end );
503  if ( m_searchRect.Inside( m_pageID, glyphRect, m_policy ) )
504  {
505  if ( !m_first )
506  {
507  m_first.emplace( std::pair<content_point_type, real_rect_type>( contentPoint, glyphRect ) );
508  m_first->first.m_isAfter = false;
509  }
510  m_last.emplace( std::pair<content_point_type, real_rect_type>( contentPoint, glyphRect ) );
511  m_last->first.m_isAfter = true;
512 
513  real_type dist = m_searchRect.DistanceFromMouseDown( m_pageID, glyphRect );
514  if ( dist < limits<real_type>::max() && ( !m_mouseDownPoint || m_mouseDownPoint->second > dist ) )
515  m_mouseDownPoint.emplace( std::pair<content_point_type, real_type>( contentPoint, dist ) );
516 
517  dist = m_searchRect.DistanceFromMouseUp( m_pageID, glyphRect );
518  if ( dist < limits<real_type>::max() && ( !m_mouseUpPoint || m_mouseUpPoint->second > dist ) )
519  m_mouseUpPoint.emplace( std::pair<content_point_type, real_type>( contentPoint, dist ) );
520  }

523 template<typename _app_traits>
524 void TextSelectionFinder<_app_traits>::HandleImage( void )
525 {
526  real_rect_type bBox;
527  matrix_type matrix = this->GetUserToDeviceTransform();
528  if( real_services::CanTransformAndBoundRealRect( get_a( matrix ), get_b( matrix ),
529  get_c( matrix ), get_d( matrix ), get_e( matrix ), get_f( matrix ),
530  0, 0, 1, 1 ) )
531  {
532  bBox = imaging_model::TransformAndBoundRealRect( t4::make<real_rect_type>( 0, 0, 1, 1 ), matrix );
533  }
534  else
535  {
536  // If we got here, it means that we encountered an overflow
537  // and we'll construct an empty bBox. We could have pinned the
538  // corner that overflows, but we would have needed to know which
539  // one. Because we only came across one file that triggers this
540  // condition, I'm declaring that identifying the corner is not
541  // worth the effort.
542  bBox = t4::make<real_rect_type>( 0, 0, 0, 0 );
543  }
544 
545  if ( m_searchRect.Inside( m_pageID, bBox, m_policy ) )
546  {
547  if ( !m_first )
548  {
549  m_first.emplace( std::pair<content_point_type, real_rect_type>( this->m_curContentPoint, bBox ) );
550  m_first->first.m_isAfter = false;
551  }
552  m_last.emplace( std::pair<content_point_type, real_rect_type>( this->m_curContentPoint, bBox ) );
553  m_last->first.m_isAfter = true;
554 
555  real_type dist = m_searchRect.DistanceFromMouseDown( m_pageID, bBox );
556  if ( dist < limits<real_type>::max() && ( !m_mouseDownPoint || m_mouseDownPoint->second > dist ) )
557  m_mouseDownPoint.emplace( std::pair<content_point_type, real_type>( this->m_curContentPoint, dist ) );
558 
559  dist = m_searchRect.DistanceFromMouseUp( m_pageID, bBox );
560  if ( dist < limits<real_type>::max() && ( !m_mouseUpPoint || m_mouseUpPoint->second > dist ) )
561  m_mouseUpPoint.emplace( std::pair<content_point_type, real_type>( this->m_curContentPoint, dist ) );
562  }
563 }

567 template<typename app_traits>
568 boost::optional<content::ContentRange> GetContentRange(
569  typename app_traits::context_type& context,
570  document::DocumentCaches<app_traits>& dc,
571  const TextSelectionRectangle<app_traits>& selectionRectangle,
572  TextSelectionInclusionPolicy policy )
573 {
574  typedef typename app_traits::real_geometry_traits::rect_type real_rect_type;
575  typedef reflow::ReflowLayout reflow_layout_type;
576  typedef content::ContentPoint content_point_type;
577 
578  boost::optional<std::pair<content_point_type, real_rect_type> > first;
579  boost::optional<std::pair<content_point_type, real_rect_type> > last;
580  boost::optional<content_point_type> mouseDownContentPoint;
581  boost::optional<content_point_type> mouseUpContentPoint;
582 
583  auto& store = dc.GetStore();
584  auto firstPage = t4::pdf::get_page_num( store.GetDocument(), selectionRectangle.m_firstPage );
585  auto lastPage = t4::pdf::get_page_num( store.GetDocument(), selectionRectangle.m_lastPage );
586 
587  DocumentContentPointLessThan_StructureOrder cmp( store.GetDocument() );
588 
589  for ( auto page = firstPage; page <= lastPage; ++page )
590  {
591  auto pageID = t4::pdf::get_page_id( store.GetDocument(), page );
592  TransientSnapShot<app_traits> ss( get_TransientHeap( context ) );
593  auto& heap = *t4::get_thread_heap();
594  textextract::TextExtractContext* tecontext = t4::make_transient<textextract::TextExtractContext>( heap, dc );
595 
596  store.PrefetchPage( page );
597  auto dlHolder = document::GetPageDisplayList( dc.GetDocument(), t4::pdf::get_page_dictionary( dc.GetDocument(), page ) );
