90 /*
91  Function that manages all the complex interactions of optional and reference-valued parameter types.
92 
93  The value type of a parameter tag may be a value or a reference to a value. The actual type of the
94  variable in the parameters dictionary depends on both the value type _and_ whether or not the corresponding
95  parameter descriptor is optional. This leads to the following matrix for the appropriate return type
96  for get():
97 
98  non-optional optional
99 
100  value type
101 
102  int& int& boost::optional<int&> const&
103  int int const& boost::optional<int> const&
104 
105  extract_from_ref takes the lvalue ref from inside the hana tuple, and produces one of the above types
106  based on the value type and optional-ness of the parameter (as derived from the "actual" type of the
107  variable in the parameters dictionary).
108 */
109 template<typename Actual, typename Value>
110 constexpr decltype(auto) extract_from_ref( Value& val )
111 {
112  return extract_from_ref_impl<Actual, Value>()( val );
113 }

340 /*
341  Given:
342 
343  - required: a tuple of parameter tags
344  - formals: the hana::map inside a formal_parameter_list
345  - actuals: a map built from a list of actual parameters provided to formal_parameter_list::operator().
346 
347  Produce:
348 
349  An actual_parameter_list mapping all parameter takes in required to values derived from formals and
350  actuals.
351 */
352 template<typename Requireds, typename Formals, typename Actuals>
353 auto build_parameters( Requireds const& required, Formals const& formals, Actuals&& actuals )
354 {
355  auto result = hana::fold( required, hana::make_tuple(), [&]( auto&& l, auto const& e )
356  {
357  using actual_value_type = typename std::decay_t<decltype(formals[e])>::actual_value_type;
358  return hana::append( std::move( l ), hana::pair<std::decay_t<decltype(e)>, actual_value_type>( e, extract_value( e, formals, std::move( actuals ) ) ) );
359  } );
360  return make_actual_parameter_list( std::move( result ) );
361 }

375  template<typename Key, typename FormalValueDesc>
376  struct to_actual_parameter_impl<hana::pair<Key, FormalValueDesc>>
377  {
378  using type = hana::pair<Key, typename FormalValueDesc::actual_value_type>;
379  };

382  using to_actual_parameter = typename to_actual_parameter_impl<KeyValue>::type;

385 
386  /*
387  Produce an actual_parameter_list from a formal_parameter_list and a sequence of parameters
388  provided by the caller. Every parameter tag in m_map will be mapped to something, or a compile-time
389  error will ensue.
390  */
391  template<typename... TArgs>
392  auto operator()( TArgs&&... args ) const&
393  {
394  auto requireds = hana::to_set( hana::transform( hana::to_tuple( m_map ), hana::first ) );
395  auto argument_map = hana::to_map( hana::transform( hana::make_tuple( std::forward<TArgs>( args )... ),
396  []( auto&& e )
397  {
398  using key_type = typename std::decay<decltype(e)>::type::key;
399  auto key = hana::type_c<key_type>;
400  using actual_value_type = typename std::decay_t<decltype( m_map[ key ] )>::actual_value_type;
401  using parameter_value_type = typename key_type::value;
402  return hana::pair<decltype(key), actual_value_type>( key, extract_from_ref<actual_value_type, parameter_value_type>( e.m_value ) );
403  } ) );
404  return build_parameters( requireds, m_map, argument_map );
405  }
