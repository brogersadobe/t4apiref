29 /*
30  a single instance of CharCodeLookup represents either a single glyph ID (unicode value, etc...)
31  or a list of such things. The typical case is a single value. In that case both start and end
32  will be zero and the value is stored in "singleValue". If the CharCodeLookup is representing
33  more than one value, then it will represent an array of such values. The array starts at
34  "start" and ends (one before) "end".
35 
36  Instances of CharCodeLookup are returned by CharCodeMap::Lookup.
37 */
38 struct CharCodeLookup
39 {
40  std::uint32_t singleValue;
41  const std::uint32_t* start;

43 };

60 /*
61  Enumerates each Unicode value in a CharCodeLookup.
62  Note that CharCodeLookup holds utf16 values, but stores them as 32-bit. This method resolves utf16 surrogate pairs
63  into their corresponding 32-bit Unicode equivalents.
64 */
65 template<typename F>
66 void for_each_character( const text::CharCodeLookup& lookup, F&& f )
67 {
68  if( !lookup.start )
69  {
70  f( lookup.singleValue );
71  }
72  else
73  {
74  size_t count = t4::convert<size_t>(lookup.end - lookup.start);
75  const uint32_t* utf16 = lookup.start;
76  for (size_t i = 0; i < count; ++i)
77  {
78  uint32_t value = utf16[i];
79  if (0xD800 <= value && value <= 0xDBFF)
80  {
81  ++i;
82  if (i >= count)
83  throw t4::corrupt_data("truncated utf16");
84  std::uint32_t value2 = utf16[i];
85  if (0xDC00 <= value2 && value2 <= 0xDFFF)
86  value = (((value & 0x3FF) << 10) | (value2 & 0x3FF)) + 0x10000;
87  else
88  throw t4::corrupt_data("invalid utf16");
89  }
90  f( value );
91  }
92  }
93 }
