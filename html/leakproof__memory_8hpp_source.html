49 /*
50  Class used to support memory allocation with third-party libraries with unreliable
51  memory resource management.
52 
53  Third-party libraries vary in reliability when it comes to memory allocation. Most
54  libraries are fine, and won't leak as long as they're used correctly, but we use a
55  few that will leak memory under exceptional circumstances. Typically these leaks
56  occur when a corrupt data or failed memory allocation occurs inside code that isn't
57  implemented with RAII techniques.
58 
59  An instance of leakproof_heap provides a simple malloc/free style interface, but
60  it also maintains a linked list of the blocks it has allocated but not freed. At
61  destruction, any blocks still in this list are automatically freed. The client may
62  also elect to call free_leaked_memory at times deemed safe by the client.
63 
64  leakproof_heap is _not_ threadsafe. Please use shared_leakproof_heap if you need
65  multithreaded access to the same leakproof_heap.
66 */
67 class leakproof_heap : boost::noncopyable
68 {
69  /*
70  A leakproof_heap needs to record all allocations, because we can't know in advance which
71  onces might leak. These allocations are stored in a linked list of unique_ptr<block>. A
72  "block" is a heap-allocated object whose operator new() allocates enough space "past the
73  end" to fulfill a client's allocation request. This allows us to do our bookkeeping in
74  the same memory allocation which we return from malloc().
75  */
76  leakproof_detail::block_list_type m_blocks;
77  bool m_failure_latch = false;
78 
79 public:
80  /*
81  Allocate memory from the leakproof_heap.
82 
83  If free() is not called on the returned pointer, it will be automatically freed
84  on the next call to free_leaked_memory or this leakproof_heap's dtor. It is
85  illegal to call free on the pointer after that happens.
86 
87  Returns nullptr if memory is unavailable.
88  */
89  void* malloc(size_t sz) BOOST_NOEXCEPT;
90 
91  /*
92  Same as malloc, but throws std::bad_alloc if no memory is availble.
93  */
94  void* malloc_throw( size_t sz );
95 
96  /*
97  Free memory from the leakproof_heap.
98 
99  This removes the pointer from the leakproof_heap's linked list and frees the
100  associated memory.
101  */
102  void free(void* p) BOOST_NOEXCEPT;
103 
104  /*
105  Free all memory that hasn't been explicitly freed.
106  */
107  void free_leaked_memory(void) BOOST_NOEXCEPT;
108 
109  /*
110  "Unlink" an allocation from the leakproof_heap. This pointer will not be freed when the
111  leakproof_heap is destroyed (or free_leaked_memory is called).
112 
113  After this call the client _must_ call leakproof_heap::free on the pointer, otherwise it will leak.
114  */
115  void unlink(void* p) BOOST_NOEXCEPT;
116 
117  /*
118  Convenience function for making a leakproof_unique_ptr to an instance of a class
119  */
120  template<typename T, typename... TArgs>
121  leakproof_unique_ptr<T, leakproof_heap> make_unique( TArgs&&... args )
122  {
123  return leakproof_unique_ptr<T, leakproof_heap>( new (*this) T( std::forward<TArgs>( args )... ), *this );
124  }
125 
126  template<typename T>
127  void delete_obj(T* p) BOOST_NOEXCEPT
128  {
129  p->~T();
130  free( p );
131  }
132  /*
133  Failure detection: if, during the lifetime of this leakproof_heap, it ever failed to satisfy a memory
134  request since its creation or the last call to test_and_clear_failure, it returns true. Otherwise returns
135  false.
136 
137  We use this to get a first approximation at why JBIG2 failed. When JBIG2 fails to decode an image, it returns
138  -1 and a nullptr for the data. But this doesn't necessarily mean memory wasn't available; for example, the
139  input data might have been malformed. This bit at least allows us to make an educated guess.
140  */
141  bool test_and_clear_failure( void )
142  {
143  auto result = m_failure_latch;
144  m_failure_latch = false;
145  return result;
146  }
147 };

158 /*
159  Wrapper around a leakproof_heap that can be shared by multiple unreliable allocators
160  and/or by multiple threads.
161 
162  An instance of lock_shared_leakproof_heap _must_ surround any call to malloc/free
163  on an instance of shared_leakproof_heap.
164 
165  lock_shared_leakproof_heaps assert a "use count" on their associated shared_leakproof_heap.
166  When this use count drops to 0 all outstanding memory is presume leaked and automatically
167  freed, so be sure to keep an instance of lock_shared_leakproof_heap for the lifetime of
168  any allocation from that heap. This behavior differs dramatically from leakproof_heap.
169 */
170 class shared_leakproof_heap
171 {
172  mutex m_mutex;
173  leakproof_heap m_heap;
174  int m_use_count;
175 
176 public:

180  /*
181  Allocate a block of memory from the shared_leakproof_heap. An instance of lock_shared_leakproof_heap
182  must exist when making this call, and it must exist until the corresponding free call.
183  */
184  void* malloc( size_t sz );
185 
186  /*
187  Same as malloc, but throws std::bad_alloc if no memory is availble.

189  void* malloc_throw( size_t sz );
