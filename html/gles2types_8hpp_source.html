66 template<void genFunc( GLsizei n, GLuint* buff ), void delFunc( GLsizei n, const GLuint* buff )>
67 class GLName : private boost::noncopyable
68 {
69 public:
70  GLName( boost::none_t )
71  : m_name( 0 ),
72  m_owner( false )
73  {}
74 
75  GLName( void )
76  : m_name( 0 ),
77  m_owner( true )
78  {
79  GL_CALL( genFunc( 1, &m_name ) );
80  }
81 
82  GLName( GLuint name )
83  : m_name( name ),
84  m_owner( false )
85  {
86  }
87 
88  GLName( GLName&& rhs )
89  : m_name( 0 ),
90  m_owner( false )
91  {
92  boost::swap( *this, rhs );
93  }
94 
95  ~GLName( void )
96  {
97  if( m_owner )
98  GL_CALL( delFunc( 1, &m_name ) );
99  }
100 
101  friend void swap( GLName& lhs, GLName& rhs )
102  {
103  boost::swap( lhs.m_name, rhs.m_name );
104  boost::swap( lhs.m_owner, rhs.m_owner );
105  }
106 
107  explicit operator bool( void ) const
108  {
109  return m_name != 0;
110  }
111 
112  GLName& operator=( GLName&& rhs )
113  {
114  boost::swap( *this, rhs );
115  rhs = boost::none;
116  return *this;
117  }
118 
119  GLName& operator=( boost::none_t )
120  {
121  GLName tmp( boost::none );
122  boost::swap( *this, tmp );
123  return *this;
124  }
125 
126  operator GLuint( void ) const
127  {
128  return m_name;
129  }
130 
131  bool IsOwner()
132  {
133  return m_owner;
134  }
135 
136 private:
137  GLuint m_name;
138  bool m_owner;

145 template<void genFunc( GLsizei n, GLuint* buff ), void delFunc( GLsizei n, const GLuint* buff ), typename recycleKey, typename recycler>
146 class GLRecycledName
147 {
148 public:

150  : m_name( boost::none )
151  , m_recycleKey( pRecycleKey )
152  , m_recycled( false )
153 
154  {
155  auto name = recycler::getRecycledName(m_recycleKey);
156  if ( name )
157  {
158  m_name = std::move( name );
159  m_recycled = true;

161  else
162  m_name = GLName<genFunc, delFunc>();
163  }
164 
165  GLRecycledName( GLuint name )

168  {
169  }
170

172  {
173  if( m_name.IsOwner() )
174  recycler::recycle(m_recycleKey, std::move(m_name));
175  }
176

178  GLRecycledName& operator=(GLRecycledName&&) = default;
179 
180  operator GLuint( void ) const
181  {
182  return m_name;
183  }
184 
185  bool IsRecycled()
186  {
187  return m_recycled;
188  }
189 
190 private:

236 const size_t MAX_RECYCLED_TEXTURES = 12;
237 const size_t MAX_RECYCLED_TEXTURE_MEMORY = 32 * 1024 * 1024;
238 const size_t MAX_TEXTURE_SIZE_TO_BE_CACHED = 16 * 1024 * 1024;
239 
240 class texture2DRecycler
241 {
242 public:
243  typedef GLName<glGenTextures, glDeleteTextures> glNameType;
244  typedef std::list<std::pair<glNameType, texture2DRecycleKey>> mruListType;
245  typedef std::map<texture2DRecycleKey, std::vector<typename mruListType::iterator>> textureMapType;
246 
247  static glNameType getRecycledName(const texture2DRecycleKey& pRecycleKey)
248  {
249  if(pRecycleKey.m_width <= 0 || pRecycleKey.m_height <= 0)
250  return boost::none;
251 
252  if(s_textureMap.find(pRecycleKey) != s_textureMap.end())
253  return removeFirst(pRecycleKey);
254 
255  return boost::none;
256  }
257 
258  static bool recycle(const texture2DRecycleKey& pRecycleKey, glNameType pName)
259  {
260  if(pRecycleKey.m_width <= 0 || pRecycleKey.m_height <= 0)
261  return false;
262 
263  size_t lSize = getTextureSize(pRecycleKey);
264 
265  if(lSize > MAX_TEXTURE_SIZE_TO_BE_CACHED)
266  return false;
267 
268  while(!s_textureMruList.empty() && ((s_textureMruList.size() >= MAX_RECYCLED_TEXTURES) || ((lSize + s_memoryUsed) > MAX_RECYCLED_TEXTURE_MEMORY)))
269  removeFirst(s_textureMruList.rbegin()->second);
270 
271  s_textureMruList.emplace_front(std::move(pName), pRecycleKey);
272  s_textureMap[pRecycleKey].push_back(s_textureMruList.begin());
273 
274  s_memoryUsed += lSize;
275 
276  return true;
277  }
278 
279  static void purgeAll()
280  {
281  s_textureMap.clear();
282  s_textureMruList.clear();
283  s_memoryUsed = 0;
284  }
285 
286 private:
287  static size_t getTextureSize(const texture2DRecycleKey& pRecycleKey)
288  {
289  return static_cast<size_t>(pRecycleKey.m_width * pRecycleKey.m_height) * getFormatSizePerPixel(pRecycleKey.m_format);
290  }
291 
292  static size_t getFormatSizePerPixel(GLint pFormat)
293  {
294  switch(pFormat)
295  {
296  case GL_ALPHA:
297  return 1;
298  case GL_RGB:
299  return 3;
300  case GL_RGBA:
301  return 4;
302  default:
303  GpuAssert(0);
304  }
305 
306  return 4;
307  }
308 
309  static glNameType removeFirst(texture2DRecycleKey pRecycleKey)
310  {
311  std::vector<typename mruListType::iterator>& lVector = s_textureMap[pRecycleKey];
312  typename mruListType::iterator& lIter = lVector[0];
313 
314  glNameType lName = std::move(lIter->first);
315  size_t lSize = getTextureSize(pRecycleKey);
316 
317  s_textureMruList.erase(lIter);
318 
319  if(lVector.size() == 1)
320  s_textureMap.erase(pRecycleKey);
321  else
322  lVector.erase(lVector.begin());
323 
324  s_memoryUsed -= lSize;
325 
326  return lName;
327  }
328 
329  static size_t s_memoryUsed;
330  static mruListType s_textureMruList;
331  static textureMapType s_textureMap;
332 };
