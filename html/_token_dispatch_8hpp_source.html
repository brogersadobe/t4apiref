64 template<typename app_traits>
65 bool Dispatch( const DispatchStruct<app_traits>* st )
66 {
67  if ( st->stream->AtEOF() ) {
68  if ( st->eofIsError )
69  (st->invalid)( st->inst, true /*isUnexpectedEOF*/ );
70  return false;
71  }
72 
73  unsigned char c = static_cast<unsigned char>( st->stream->Peek() );
74  size_t ci = c;
75  const TokenDispatchTable* table = st->table;
76 
77  while ( true ) {
78  bool invalidToken = table->m_table == nullptr;
79  size_t minChar = 0, maxChar = 0;
80 
81  // Each table entry hosts nullptr or one of the following two valid values.
82  size_t tableIndex = 0;
83  TokenDispatchTable::DispatchProcType dispatchProc = nullptr;
84 
85  if ( !invalidToken ) {
86  // first two elements in the table always describe the valid range
87  // of character values which rest of the table corresponds to.
88  minChar = boost::variant2::get<size_t>( table->m_table[0] );
89  maxChar = boost::variant2::get<size_t>( table->m_table[1] );
90 
91  // check for out of range character
92  if ( ci < minChar || ci > maxChar )
93  invalidToken = true;
94  else
95  {
96  auto& v = table->m_table[ci - minChar + 2];
97 
98  // check for index into another row
99  if ( auto p = boost::variant2::get_if<size_t>( &v ) )
100  tableIndex = *p;
101  // check for dispatch proc for given character
102  else if ( auto p = boost::variant2::get_if<TokenDispatchTable::DispatchProcType>( &v ) )
103  dispatchProc = *p;
104 
105  if ( tableIndex == 0 && dispatchProc == nullptr )
106  invalidToken = true;
107  }
108  }
109 
110  if ( invalidToken )
111  return (st->invalid)( st->inst, false /*isUnexpectedEOF*/ );
112 
113  if ( dispatchProc )
114  return dispatchProc( st->inst );
115 
116  table += tableIndex;
117  st->stream->Advance();
118  if ( table->m_token ) {
119  const char* p = table->m_token;
120  while ( *p ) {
121  if ( *p != st->stream->PeekAssert() )
122  return (st->invalid)( st->inst, false /*isUnexpectedEOF*/ );
123  ++p;
124  st->stream->Advance();
125  }
126  }
127 
128  bool isDelimiter;
129  if ( st->stream->AtEOF() )
130  isDelimiter = true;
131  else {
132  c = static_cast<unsigned char>( st->stream->Peek() );
133  ci = c;
134  isDelimiter = (st->typeArray[ci] & st->mask) != 0;
135  }
136  if ( isDelimiter ) {
137  if (auto p = boost::variant2::get_if<TokenDispatchTable::DispatchProcType>(&( table->m_proc ) ) )
138  return (*p)( st->inst );
139  else
140  return (st->invalid)( st->inst, false /*isUnexpectedEOF*/ );
141  }
142  }
143 }
