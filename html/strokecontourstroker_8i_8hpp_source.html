141 template <typename stroker_traits>
142 void ContourStrokerThick<stroker_traits>::handleNoOffsetContour(const point_type& p, const params_plus_type& params, output_path_type& outPath )
143 {
144  // If we do not have rounds caps, mark nothing.
145  if (!params.IsRoundCap())
146  return;
147 
148  // now build a circle, two half circles which go from 12-6, then 6-12.
149  point_type origin(0, 0);
150  point_type twelveOClock(0, params.GetOffsetWidth());
151  point_type halfcircle[7];
152  getHalfCircle(halfcircle, origin, twelveOClock);
153 
154  outPath.MoveTo(p + halfcircle[0]);
155  outPath.CurveTo(p + halfcircle[1], p + halfcircle[2], p + halfcircle[3]);
156  outPath.CurveTo(p + halfcircle[4], p + halfcircle[5], p + halfcircle[6]);
157  outPath.CurveTo(p - halfcircle[1], p - halfcircle[2], p - halfcircle[3]);
158  outPath.CurveTo(p - halfcircle[4], p - halfcircle[5], p - halfcircle[6]);
159  outPath.ClosePath();
160 }

162 template <typename stroker_traits>
163 void ContourStrokerThick<stroker_traits>::getHalfCircle(point_type arc[7], const point_type& center, const point_type& startOffset)
164 {
165  static const static_coord_type k = TETRAPHILIA_STATIC_TYPENAME_REAL(coord_type, 0.552284749831);
166  static const static_coord_type negk = TETRAPHILIA_STATIC_TYPENAME_REAL(coord_type, -0.552284749831);
167  // TODO: it would be nice to store the points in a static list and use
168  // matrix transform multiple, but this proves tricky if the activities
169  // involve runtime contruction at app start.
170 
171  matrix_type mat;
172  MakeMatrix(mat, startOffset.x, startOffset.y, -startOffset.y, startOffset.x, center.x, center.y);
173  arc[0] = MatrixTransform(mat, point_type(1, 0));
174  arc[1] = MatrixTransform(mat, point_type(1, negk));
175  arc[2] = MatrixTransform(mat, point_type(k, -1));
176  arc[3] = MatrixTransform(mat, point_type(0, -1));
177  arc[4] = MatrixTransform(mat, point_type(negk, -1));
178  arc[5] = MatrixTransform(mat, point_type(-1, negk));
179  arc[6] = MatrixTransform(mat, point_type(-1, 0));
180 }

182 template <typename stroker_traits>
183 void ContourStrokerThick<stroker_traits>::QuickStrokeLine(const contour_seg_type& seg, const params_plus_type& params, output_path_type& outPath )
184 {
185  point_type p0 = seg.GetStartPoint();
186  point_type p1 = seg.GetEndPoint();
187 
188  // get tangent and normalize to length of offset width; make sure not to use p0 and
189  // p1 bebasue this breaks zero length dashes.
190  point_type tangent = seg.GetPointAtParam(1) - seg.GetPointAtParam(0);
191  if (tangent.x == 0)
192  {
193  if (tangent.y == 0)
194  {
195  // no tangent, so we have to resort to the "no offset contour case"
196  handleNoOffsetContour(p0, params, outPath);
197  return;
198  }
199  tangent.y = tangent.y > 0 ? params.GetOffsetWidth() : -params.GetOffsetWidth();
200  }
201  else if (tangent.y == 0)
202  tangent.x = tangent.x > 0 ? params.GetOffsetWidth() : -params.GetOffsetWidth();
203  else
204  tangent *= params.GetOffsetWidth() / tangent.Mag();
205 
206  point_type offset = tangent.Rotate90CCW();
207  if (params.IsButtCap())
208  {
209  outPath.MoveTo(p0+offset);
210  outPath.LineTo(p1+offset);
211  outPath.LineTo(p1-offset);
212  outPath.LineTo(p0-offset);
213  }
214  else if (params.IsSquareCap())
215  {
216  outPath.MoveTo(p0-tangent+offset);
217  outPath.LineTo(p1+tangent+offset);
218  outPath.LineTo(p1+tangent-offset);
219  outPath.LineTo(p0-tangent-offset);
220  }
221  else if (params.IsRoundCap())
222  {
223  point_type cap[7];
224 
225  getHalfCircle(cap, p1, offset);
226  outPath.MoveTo(cap[0]);
227  outPath.CurveTo(cap[1], cap[2], cap[3]);
228  outPath.CurveTo(cap[4], cap[5], cap[6]);
229 
230  getHalfCircle(cap, p0, -offset);
231  outPath.LineTo(cap[0]);
232  outPath.CurveTo(cap[1], cap[2], cap[3]);
233  outPath.CurveTo(cap[4], cap[5], cap[6]);
234  }
235  else
236  {
237  tetraphilia_assert_always();
238  }
239  outPath.ClosePath();
240 }

242 template <typename stroker_traits>
243 void ContourStrokerThick<stroker_traits>::GenerateStroke( const contour_type& contour, output_path_type& outPath )
244 {
245  size_t numsegs = contour.GetNumSegs();
246 
247  if (numsegs == 0)
248  return;
249 
250  if (numsegs == 1 && contour.GetFirst().IsLine())
251  {
252  QuickStrokeLine(contour.GetFirst(), m_params, outPath);
253  return;
254  }
255 
256  // general case
257  bool allok = true;
258  if (contour.IsClosed())
259  {
260  resetOffsetState();
261 
262  // The natural thing to do for a closed path is to have two contours, one for the inside and one
263  // for the outside, which go opoosite directions. This is what was originally here. However, the
264  // GDI printers do not all support this feature, so Chip came up with a hack to make one single path
265  // instead of two. The trick is to make a "cut" in the stroke from the outside to the inside, making
266  // sure that the cut stays within the stroke path. You then carefully move along the cut from the
267  // outside to the inside when you finish the outside curve, and then go backwards along the cut
268  // when you finish the nside curve. This does not change the image, but it does remove the compund
269  // path complexity.
270  // I use the first point in the countour as a point known to be inside of the stroke path as the way
271  // navigate the cut.
272  point_type contourStart = contour.GetFirst().GetStartPoint();
273 
274  // first part of cut.
275  outPath.MoveTo(contourStart);
276 
277  // build the "outer" path, using LineTo to define the cut (not necessarily outer, but doesn't matter)
278  if (allok)
279  allok = addOffsetCurve(contour, outPath, false /*reverse*/, kLineToFirstPoint, false/* stopAfterFirstJoin*/);
280  // close up with a final join on the outer path
281  if (allok)
282  allok = addOffsetCurve(contour, outPath, false /*reverse*/, kIgnoreFirstPoint, true /* stopAfterFirstJoin*/);
283 
284  // LineTo along the cut
285  outPath.LineTo(contourStart);
286 
287  resetOffsetState();
288 
289  // build the "inner" path, using LineTo to define the cut (not necessarily inner, but doesn't matter)
290  if (allok)
291  allok = addOffsetCurve(contour, outPath, true /*reverse*/, kLineToFirstPoint, false /* stopAfterFirstJoin*/);
292 
293  // close up with a final join on the inner path
294  if (allok)
295  allok = addOffsetCurve(contour, outPath, true /*reverse*/, kIgnoreFirstPoint, true /* stopAfterFirstJoin*/);
296 
297  // LineTo along the cut, finishing it where it started above.
298  outPath.LineTo(contourStart);
299  }
300  else
301  {
302  // begin new offset curve
303  resetOffsetState();
304 
305  // Go out along the left
306  if (allok)
307  allok = addOffsetCurve(contour, outPath, false /*reverse*/, kMoveToFirstPoint, false /* stopAfterFirstJoin*/);
308 
309  // come back along the right
310  if (allok)
311  allok = addOffsetCurve(contour, outPath, true /*reverse*/, kIgnoreFirstPoint, false /* stopAfterFirstJoin*/);
312 
313  // close up with a final join/lineto
314  if (allok)
315  allok = addOffsetCurve(contour, outPath, false /*reverse*/, kIgnoreFirstPoint, true /* stopAfterFirstJoin*/);
316  }
317 
318  if (allok)
319  outPath.ClosePath();
320  else // i.e. we had an error....
321  handleNoOffsetContour(contour[0].GetStartPoint(), m_params, outPath);
322 }

324 template <typename stroker_traits>
325 bool ContourStrokerThick<stroker_traits>::addOffsetCurve(const contour_type& contour, output_path_type& outPath, bool reverse, offset_start_type startType, bool stopAfterFirstJoin)
326 {
327  size_t segNum = reverse ? contour.GetNumSegs()-1 : 0;
328  size_t segNumLast = reverse ? 0 : contour.GetNumSegs()-1;
329  ptrdiff_t segNumIncr = reverse ? -1 : 1;
330 
331  bool isFirst = true;
332  bool done = false;
333  coord_type tanTooShortLen = real_services::CalcVectorTooShortToScaleToLengthWithinPrecision(m_params.GetOffsetWidth(), m_params.GetUserSpacePrecision());
334 
335  while (!done)
336  {
337  const contour_seg_type& seg = contour[segNum];
338  if (!seg.IsDegenerate())
339  {
340  // we declare the offset computer on the stack, since we will potentially be
341  // decalring many of these over the course of the contour and want to give the memory
342  // back. If the size of this structure becomes prohibitive, we may nned to be clever.
343  static_assert(sizeof(offset_computer_loop_type) < 64 * sizeof(coord_type), "");
344  offset_computer_loop_type ocloop(m_params, tanTooShortLen, seg, reverse);
345  while (ocloop.GetNextOffset())
346  {
347  if (ocloop.CurOffsetIsValid())
348  {
349  if (isFirst)
350  {
351  switch(startType)
352  {
353  case kMoveToFirstPoint:
354  outPath.MoveTo(ocloop.GetCurOffsetPoint(0));
355  break;
356  case kLineToFirstPoint:
357  outPath.LineTo(ocloop.GetCurOffsetPoint(0));
358  break;
359  case kIgnoreFirstPoint:
360  break;
361  }
362  isFirst = false;
363  }
364 
365  // Now we need to add a join, usually. We will not add a join if there is no
366  // preceding offset curve to join with,
367  // but otherwise we will add either a simple line for "interior joins" or the real
368  // join for "exterior joins". Regardless, all this logic is in AddJoin.
369  // NOTE: addJoin modifies fJoinStatePrevPathPoint, fJoinStatePrevOffsetPoint, and fNextLineJoinType
370  // to prepare for the next join to this offset curve.
371  //
372  // We may want to stop after the first join is emitted, this allows us to use the
373  // same "find a valid offset curve" machinery to handle the final line join on
374  // closed curves, and then stop.
375  if (stopAfterFirstJoin)
376  {
377  addJoin(outPath, ocloop, false);
378  done = true;
379  break;
380  }
381  else
382  addJoin(outPath, ocloop, true);
383 
384  // Now add the actual offset curve
385  if (ocloop.CurOffsetIsBezier())
386  {
387  point_type X;
388  if (ComputeLineSegmentIntersection(&X, ocloop.GetCurOffsetPathStartPoint(), ocloop.GetCurOffsetPoint(0), ocloop.GetCurOffsetPathEndPoint(), ocloop.GetCurOffsetPoint(3)))
389  {
390  // odd bow-tie case where radius of curvature is less than line width
391  outPath.LineTo( X );
392  outPath.LineTo( ocloop.GetCurOffsetPoint(3) );
393  outPath.CurveTo( ocloop.GetCurOffsetPoint(2), ocloop.GetCurOffsetPoint(1), ocloop.GetCurOffsetPoint(0) );
394  outPath.LineTo( X );
395  outPath.LineTo( ocloop.GetCurOffsetPoint(3) );
396  updateOffsetState(ocloop.GetCurOffsetPathEndPoint(), ocloop.GetCurOffsetPoint(3), ocloop.CurOffsetPathEndIsUserEndpoint());
397  }
398  else
399  {
400  outPath.CurveTo(ocloop.GetCurOffsetPoint(1), ocloop.GetCurOffsetPoint(2), ocloop.GetCurOffsetPoint(3));
401  updateOffsetState(ocloop.GetCurOffsetPathEndPoint(), ocloop.GetCurOffsetPoint(3), ocloop.CurOffsetPathEndIsUserEndpoint());
402  }
403  }
404  else
405  {
406  outPath.LineTo(ocloop.GetCurOffsetPoint(1));
407  updateOffsetState(ocloop.GetCurOffsetPathEndPoint(), ocloop.GetCurOffsetPoint(1), ocloop.CurOffsetPathEndIsUserEndpoint());
408  }
409  }
410  else
411  {
412  pathologicalOffsetState();
413  }
414  }
415  }
416 
417  if (segNum == segNumLast && !done)
418  {
419  // We need to put a cap on, but we need to be careful if the whole contour was
420  // degenerate, i.e. no valid offsets.
421  if (m_offsetNumValid == 0)
422  return false; // failure
423 
424  if (!contour.IsClosed())
425  addCap(outPath);
426 
427  done = true;
428  }
429  segNum += static_cast<size_t>( segNumIncr );
430  } // next segment
431  return true; // success!
432 }

460 // Adds caps (as needed) and updates the join state machinery for the next join
461 template <typename stroker_traits>
462 void ContourStrokerThick<stroker_traits>::addCap(output_path_type& outPath)
463 {
464  // need to gather these here because the addJoin modifies them
465  point_type pathPt = m_offsetLastPathPoint;
466  point_type startPt = m_offsetLastPoint;
467  point_type offsetVec = startPt-pathPt;
468  point_type endPt = pathPt - offsetVec;
469 
470  if (m_params.IsButtCap())
471  {
472  outPath.LineTo(endPt);
473  updateOffsetState(pathPt, endPt, false);
474  }
475  else if (m_params.IsSquareCap())
476  {
477  point_type tanVec = offsetVec.Rotate90CW();
478  if (outPath.GetLastPathOpOp() == output_path_point_type::kOpLineTo)
479  {
480  outPath._ChangeLastPathOpEndPoint(startPt + tanVec);
481  outPath.LineTo(endPt + tanVec);
482  }
483  else
484  {
485  outPath.LineTo(startPt + tanVec);
486  outPath.LineTo(endPt + tanVec);
487  outPath.LineTo(endPt);
488  }
489  updateOffsetState(pathPt, endPt, false);
490  }
491  else if (m_params.IsRoundCap())
492  {
493  point_type halfcircle[7];
494  // NEED TO ACCOMODATE EXTRA POINT!
495  getHalfCircle(halfcircle, pathPt, offsetVec);
496  outPath.CurveTo(halfcircle[1], halfcircle[2], halfcircle[3]);
497  outPath.CurveTo(halfcircle[4], halfcircle[5], halfcircle[6]);
498  updateOffsetState(pathPt, endPt, false); // a bit of a lie, but OK....
499  }
500  else
501  {
502  tetraphilia_assert_always();
503  }
504 }

506 // Adds joins (as needed) and updates the join state machinery for the next join
507 template <typename stroker_traits>
508 void ContourStrokerThick<stroker_traits>::addJoin(output_path_type& outPath, const offset_computer_loop_type& nextOffsetData, bool upcomingOffsetWillBeIssued)
509 {
510  if (m_offsetNumValid > 0)
511  {
512  // Thanks to jigglePoints, and the fact that I am
513  // carrying two path points along (we didin't in the past), it can
514  // be true that even WITHOUT intervening pathological curves the
515  // startPath can be different form end path (the same point jiggled
516  // differently by the curves on either side).
517  // Because we know that they SHOULD be the same so long as there is
518  // no intervening pathological, we can snap the new path point to the
519  // old one and make quite a number of nasty edge cases go away
520  point_type nextPathPoint = m_offsetHadPathologicalSinceLast ? point_type(nextOffsetData.GetCurOffsetPathStartPoint()) : point_type(m_offsetLastPathPoint);
521  point_type nextOffsetPoint = nextOffsetData.GetCurOffsetPoint(0);
522  bool startIsUserEndpoint = nextOffsetData.CurOffsetPathStartIsUserEndpoint();
523 
524  // only bother to actually put a join in if the offset points are different;
525  // they are often identical for internal joins
526  if (!point_type::EssentiallyEqual(m_offsetLastPoint, nextOffsetPoint, 4))
527  {
528  // We want to see whether this is an internal or external join,
529  // the best way is cross(oldoffset, newoffset) > 0 but this is slighly error prone in
530  // the case where a line goes out and comes back exactly along itself. There's no harm
531  // done putting a fancy join on an interior join so long as it stays "within the
532  // envelope" of the stroke; this will always be true except for big mitres. Luckily
533  // big mitres get chopped off at exactly the place where this coxdde kicks in, so there is no danger.
534  // So instead of checking whether the cross > 0, I check to see if it is > 0.1 which allows up
535  // to 180.6 degree joins (obviously real ones should be up to 180, but again this is an edge
536  // case
537  //
538  // I no longer "normalize" by dividing by the width since it is possible (through odd means) to
539  // have width of zero make it in here. Therefore I compare the resulting cross to 0.01*width*width
540  // which does not suffer the div by zero problem. This fixes bug #584072
541  if (Cross(m_offsetLastPoint-m_offsetLastPathPoint, nextOffsetPoint-nextPathPoint) >= coord_type(0.01)*m_params.GetOffsetWidthSquared())
542  {
543  addInteriorJoin(outPath, m_offsetLastPathPoint, nextPathPoint, nextOffsetPoint, nextOffsetData, upcomingOffsetWillBeIssued);
544  }
545  else
546  {
547  // An outside join so we have to work.
548 
549  // if we've had problems, or or doing "internal" joins, we want to use
550  // round caps to spackle in the cracks
551  if (!(startIsUserEndpoint || m_offsetLastWasUserEndpoint))
552  {
553  addRoundJoin(outPath, m_offsetLastPathPoint, m_offsetLastPoint, nextPathPoint, nextOffsetPoint);
554  }
555  else if (m_params.IsMiterJoin())
556  {
557  addMiterJoin(outPath, m_offsetLastPathPoint, m_offsetLastPoint, nextPathPoint, nextOffsetPoint, nextOffsetData, upcomingOffsetWillBeIssued);
558  }
559  else if (m_params.IsRoundJoin())
560  {
561  addRoundJoin(outPath, m_offsetLastPathPoint, m_offsetLastPoint, nextPathPoint, nextOffsetPoint);
562  }
563  else
564  {
565  tetraphilia_assert(m_params.IsBevelJoin());
566  addBevelJoin(outPath, nextPathPoint, nextOffsetPoint);
567  }
568  }
569  }
570  }
571 }

573 template <typename stroker_traits>
574 void ContourStrokerThick<stroker_traits>::addBevelJoin(output_path_type& outPath,
575  const point_type& endPath,
576  const point_type& endOffset )
577 {
578  outPath.LineTo( endOffset );

582 template <typename stroker_traits>
583 void ContourStrokerThick<stroker_traits>::addRoundJoin(output_path_type& outPath,
584  const point_type& startPath,
585  const point_type& startOffset,
586  const point_type& endPath,
587  const point_type& endOffset )
588 {
589  // first see if the arc we would generate is so flat as to not deviate from
590  // a straight line by more than device precision; then we can save time by doing so.
591  // The math is taht we can susbtitute a line if:
592  //
593  // delta between line and arc < precision
594  // r*(1-cos(theta/2)) < precision
595  // r*(1-sqrt((1+cos(theta))/2)) < precision
596  // r - precision < r*sqrt((1+cos(theta))/2
597  // (r - precision)^2 < r*r*(1+cos(theta))/2
598  // 2*(r - precision)^2 < r*r + r*r*cos(theta))
599  // 2*(r - precision)^2 - r^2 < dot(v1, v2)
600  // dot(v1,v2) > r^2 - 4*r*precision - 2*precision^2
601  //
602  // So....
603  //
604  // dot(v1, v2) > fArcCanBeApproximatedByLineConstant
605  //
606  // means we can substitute a line (if we haven't been forced
607  // to put full circles there)
608 
609  if (!m_params.GetPutFullCirclesAtRoundJoins() && Dot(startOffset-startPath, endOffset-endPath) > m_arcCanBeApproximatedByLineConstant)
610  {
611  outPath.LineTo(endOffset);
612  updateOffsetState(endPath, endOffset, false);
613  }
614  else
615  {
616  // Put in a real arc.
617  // we declare the offset computer on the stack, since we will potentially be
618  // decalring many of these over the course of the contour and want to give the memory
619  // back. If the size of this structure becomes prohibitive, we may nned to be clever.
620  static_assert(sizeof(offset_computer_loop_type) < 64 * sizeof(coord_type), "");
621  arc_generator_type arc(startPath, startOffset, endPath, endOffset, m_params.GetOffsetWidth());
622  arc.AppendToPath(outPath);
623 
624  if (m_params.GetPutFullCirclesAtRoundJoins())
625  {
626  // UGH. We have been asked to match PS's bugs. Oh well....
627  // This picks up where the previous arc left off, goes all
628  // the way around to whre the previous started, then replays the previous.
629  arc_generator_type otherarc(endPath, endOffset, startPath, startOffset, m_params.GetOffsetWidth());
630  otherarc.AppendToPath(outPath);
631  arc.AppendToPath(outPath);
632  }
633  updateOffsetState(endPath, endOffset, false);
634  }
635 }

637 template <typename stroker_traits>
638 void ContourStrokerThick<stroker_traits>::addMiterJoin(output_path_type& outPath,
639  const point_type& startPath,
640  const point_type& startOffset,
641  const point_type& endPath,
642  const point_type& endOffset,
643  const offset_computer_loop_type& nextOffsetData,
644  bool upcomingOffsetWillBeIssued )
645 {
646  // NOTE: THis algorithm is slighlty different than normal because it can handle cases
647  // where the offsets do not share a common point on the original path; this happens
648  // when offset computations fail due to precision issues. We go ahead an put on a nice miter anyway.
649 
650  // calculate where the offset curve lines would intersect if they were extended (this is just a simple intersection of two lines)
651  point_type startNormal = startOffset - startPath;
652  point_type endNormal = endOffset - endPath;
653  coord_type cross = Cross(startNormal, endNormal);
654  if (AbsVal(cross) < real_services::Epsilon(coord_type(1)))
655  {
656  addBevelJoin( outPath, endPath, endOffset );
657  return;
658  }
659 
660  point_type peak;
661  if (!ComputeLineIntersection(&peak, startOffset, startOffset+startNormal.Rotate90CW(), endOffset, endOffset+endNormal.Rotate90CCW()))
662  {
663  addBevelJoin( outPath, endPath, endOffset );
664  }
665  else
666  {
667  if ((peak - endPath).Mag() > m_params.GetOffsetWidth() * m_params.GetMiterLimit())
668  {
669  addBevelJoin( outPath, endPath, endOffset );
670  }
671  else
672  {
673  /*
674  this is mostly usefull when coord_type is compiled as
675  a fixed point type and there can be significant
676  round-off errors in the computation above.
677  */
678  if ( startPath.x == startOffset.x )
679  peak.y = startOffset.y;
680  else if ( endPath.x == endOffset.x )
681  peak.y = endOffset.y;
682 
683  if ( startPath.y == startOffset.y )
684  peak.x = startOffset.x;
685  else if ( endPath.y == endOffset.y )
686  peak.x = endOffset.x;
687 
688  // this is the code which simplifies strokes a bit; if a miter is a simple extension of
689  // the offset line, then we can discard an interior LineTo (draw a simple picture). Due to
690  // exactly how the curve has progressed so far, we need to make sure that the previous offset segment
691  // was a line, and that the new point is collinear with the previous, then we can replace Two lineto's with
692  // one lineto.
693  bool didOptimize = false;
694  if (outPath.GetLastPathOpOp() == output_path_point_type::kOpLineTo && outPath.GetPenultimatePathOpOp() != output_path_point_type::kOpClosePath)
695  {
696  point_type prevStart = outPath.GetPenultimatePathOpEndPoint();
697  point_type prevEnd = outPath.GetLastPathOpEndPoint();
698 
699  // OK, the already commited segment is a simple line, we can see if the new point
700  // is colinear.
701  point_type perp = (prevEnd-prevStart).Rotate90CCW();
702  coord_type approxLen = perp.TaxicabLength() * coord_type(0.75f);
703  if (approxLen > coord_type(1))
704  {
705  coord_type approxDist = Dot(perp / approxLen, peak-prevEnd);
706  if (approxDist < 0)
707  approxDist = -approxDist;
708  if (approxDist < this->m_params.GetUserSpacePrecision())
709  {
710  outPath._ChangeLastPathOpEndPoint(peak);
711  didOptimize = true;
712  }
713  }
714  }
715 
716  if (!didOptimize)
717  outPath.LineTo(peak);
718 
719  if (nextOffsetData.CurOffsetIsLine() && upcomingOffsetWillBeIssued)
720  ;// do nothing, subsequest lineto will cover this too.
721  else
722  outPath.LineTo(endOffset);
723 
724  updateOffsetState(endPath, endOffset, false);
725  }
726  }
727 }

729 template <typename stroker_traits>
730 void ContourStrokerThick<stroker_traits>::addInteriorJoin(output_path_type& outPath,
731  const point_type& startPath,
732  const point_type& endPath,
733  const point_type& endOffset,
734  const offset_computer_loop_type& nextOffsetData,
735  bool upcomingOffsetWillBeIssued )
736 {
737  if (upcomingOffsetWillBeIssued)
738  {
739  if (outPath.GetLastPathOpOp() == output_path_point_type::kOpLineTo && nextOffsetData.CurOffsetIsLine())
740  {
741  point_type prevStart = outPath.GetPenultimatePathOpEndPoint();
742  point_type prevEnd = outPath.GetLastPathOpEndPoint();
743  point_type cross;
744  if (ComputeLineSegmentIntersection(&cross, prevStart, prevEnd, nextOffsetData.GetCurOffsetPoint(0), nextOffsetData.GetCurOffsetPoint(1)))
745  {
746  outPath._ChangeLastPathOpEndPoint(cross);
747  return;
748  }
749  }
750  }
751 
752  // this is an "inside" join, so we just do a lineto to keep the path whole.
753  // It turns out that this simple "single lineto" is not good enough, because
754  // extremely short segment with thick strokes. Trust me. Rather than concoct
755  // a complex test to see whether we are in this situation, we can remedy
756  // it by always going through the two path points as well, which is what
757  // I do here....
758  outPath.LineTo(startPath);
759  if (startPath != endPath)
760  outPath.LineTo(endPath);
761  outPath.LineTo(endOffset);
762  updateOffsetState(endPath, endOffset, false);
763 }

799 template <typename stroker_traits>
800 ContourStrokerDashed<stroker_traits>::ContourStrokerDashed(const params_plus_type& params, out_stroker_type& outStroker)
801  : m_params(params)
802  , m_outStroker(outStroker)
803  , m_skipDashingCompletely(false)
804  , m_ignoreGapThresholdLine(0)
805  , m_ignoreGapThresholdBezier(0)
806 {
807  tetraphilia_assert(params.GetDashArrayLen() > 0 && params.GetDashArray() != nullptr);
808 
809  if (params.GetActuallyDoThinStroke())
810  {
811  // if we are thin stroking; we can get away with a lot more; no miters or caps,
812  // its just a question of whether the gaps are bigger than the requested precision
813  // we needn't sweat the caps and stroke widths
814  m_ignoreGapThresholdLine = m_ignoreGapThresholdBezier = params.GetApproximatePixelSize();
815  m_skipDashingCompletely = allInternalGapsAreIgnorable(true);
816  return;
817  }
818 
819  // Now we need to validate the dashing a bit and normalize the offset in preparation
820  // to split the contour into subpieces.
821  coord_type dashPatternLength = 0;
822  for (size_t i = 0; i < params.GetDashArrayLen(); i++)
823  {
824  if (params.GetDashArray()[i] < 0)
825  {
826  tetraphilia_assert_always();
827  m_skipDashingCompletely = true;
828  return;
829  }
830  coord_type newDPL = dashPatternLength + params.GetDashArray()[i];
831  if (newDPL < dashPatternLength)
832  {
833  // aprrently had an overflow, the "total length" is not really needed
834  // futher down, just for the initial offset, which is smaller than
835  // the toatal here (since it wrapped), so pinning to MAX_VAL is fine.
836  dashPatternLength = limits<coord_type>::max();
837  break;
838  }
839  else
840  dashPatternLength = newDPL;
841  }
842 
843  // odd dash pattern lengths repeat every two times through
844  if (params.GetDashArrayLen() & 1)
845  {
846  if (dashPatternLength + dashPatternLength < dashPatternLength) // again, overflow check
847  dashPatternLength = limits<coord_type>::max();
848  else
849  dashPatternLength += dashPatternLength;
850  }
851 
852  // make sure the dash pattern is long enough to add up to offset (at least)
853  if (dashPatternLength <= real_services::Epsilon(params.GetDashOffset()))
854  {
855  tetraphilia_assert_always();
856  m_skipDashingCompletely = true;
857  return;
858  }
859 
860  // We have a thick stroked dashed line. We still want to see if we can ignore the dashes.
861  // now we want to see whether the dashing will be "visible". When zoomed way
862  // out (or given silly dashes) the dashes begin to cover up the gaps entirely.
863  // When this is the case, we can ignore gaps which do not span corners. Those
864  // that span corners still need to be sweated about as they result in no join
865  // being put at that corner.
866  // The cases where the gaps are "invisible" are slightly different for the line and
867  // bezier cases, since bezier curvature causes issues. I have taken the most pessimestic
868  // stance on beziers, menaing I will not disregard a gap unless it is impossible for it
869  // to be visible.
870  m_ignoreGapThresholdLine = calcIgnoreGapThresholdLine(params);
871  m_ignoreGapThresholdBezier = calcIgnoreGapThresholdBezier(params);
872 }

874 template <typename stroker_traits>
875 typename ContourStrokerDashed<stroker_traits>::coord_type ContourStrokerDashed<stroker_traits>::calcIgnoreGapThresholdLine(const params_plus_type& params)
876 {
877  // The following code measures how short a gap on a dashed LINE must be to be ignorable.
878  coord_type strokewidth = params.GetActualStrokeWidth();
879  if (params.IsButtCap())
880  return params.GetApproximatePixelSize();
881 
882  if (params.IsSquareCap())
883  return params.GetApproximatePixelSize() + strokewidth;
884 
885  if (params.IsRoundCap())
886  {
887  if (params.GetApproximatePixelSize() < real_services::DivideBy2(strokewidth))
888  // this case is for when the precision is small compared to the "scalloping"
889  // we get as the endcap semi-circles start to move apart. We want
890  // to make sure the depth of these scallops is not visible
891  return 2*real_services::Sqrt(params.GetApproximatePixelSize()*(strokewidth-params.GetApproximatePixelSize()));
892  else
893  // scalloping is invisible, now just worry about the span in between the caps
894  return params.GetApproximatePixelSize() + strokewidth;
895  }
896 
897  tetraphilia_assert_always();
898  return params.GetApproximatePixelSize();
899 }

901 template <typename stroker_traits>
902 typename ContourStrokerDashed<stroker_traits>::coord_type ContourStrokerDashed<stroker_traits>::calcIgnoreGapThresholdBezier(const params_plus_type& params)
903 {
904  // The following code measures how short a gap on a dashed BEZIER must be to be ignorable.
905  coord_type strokewidth = params.GetActualStrokeWidth();
906  if (params.IsButtCap())
907  {
908  // first see whether the "rounded corner" introduced by ignoring the gap is
909  // itself potentially visible. If so, we cannot ignore any gaps
910  if (real_services::DivideBy2(strokewidth) > params.GetApproximatePixelSize())
911  return 0; // can't ignore any but 0
912  else
913  return params.GetApproximatePixelSize() - strokewidth; // the minus is correct; draw a tight 180 with the gap on it...
914  }
915 
916  if (params.IsSquareCap())
917  {
918  // first see whether the "rounded corner" introduced
919  // by ignoring the gap leaves off the sharp corners
920  // of the square caps; this measures how much those
921  // sharp corners poke out (Sqrt(2)-1)*(sw/2) = 0.207*strokewidth
922  if (coord_type(0.207)*strokewidth > params.GetApproximatePixelSize())
923  return 0; // can't ignore any but 0
924  else
925  return params.GetApproximatePixelSize() + strokewidth; // the plus is correct. Since we know that the
926  // square caps cover the rounded corner within precision
927  // we need only worry about the areas between the caps/lines
928  // which may be painted. since the caps and width all are offset
929  // by sw/2, we can safely assume they encroach on the gap by sw.
930  }
931 
932  if (params.IsRoundCap())
933  {
934  // round caps are the same on beziers as lines
935  return calcIgnoreGapThresholdLine(params);
936  }
937 
938  tetraphilia_assert_always();
939  return params.GetApproximatePixelSize();
940 }

953 template <typename stroker_traits>
954 void ContourStrokerDashed<stroker_traits>::GenerateStroke(const contour_type& contour, output_path_type& outPath)
955 {
956  if (contour.GetNumSegs() == 0)
957  return;
958 
959  // see if there is any dashing to be done, otherwise defer to target
960  if (m_skipDashingCompletely)
961  {
962  m_outStroker->GenerateStroke(contour, outPath);
963  return;
964  }
965 
966 
967  if (m_params.GetActuallyDoThinStroke() || contour.GetNumSegs() == 1)
968  {
969  // there are no join issues since we are thin stroking ore there is only
970  // one segment, if all the internal gaps are ignorable we can skip dashing
971  // entirely.for this contour,
972  // NOTE: this check happens in a weaker form above, where it assumes the contours
973  // contain beziers and makes a pessimistic prediction, Here we can be more precise
974  if (allInternalGapsAreIgnorable(contour.HasBeziers()))
975  {
976  m_outStroker->GenerateStroke(contour, outPath);
977  return;
978  }
979  }
980 
981  // OK, we're really doing dashing from here on out.
982  coord_type offsetRemaining = m_params.GetDashOffset();
983  contour_type dashContour(m_params);
984 
985  // find out where to start in the dash array. Start out with the offset, and subtract off
986  // whole dash segs in order until we are inside of a seg, then figure out how much of that
987  // dash seg is left
988  size_t curDashIndex = 0;
989  bool curDashIsGap = false;
990  while(offsetRemaining > 0 && offsetRemaining >= m_params.GetDashArray()[curDashIndex])
991  {
992  offsetRemaining -= m_params.GetDashArray()[curDashIndex];
993  curDashIndex = nextDashIndex(curDashIndex);
994  curDashIsGap = !curDashIsGap;
995  }
996  coord_type curDashLength = m_params.GetDashArray()[curDashIndex] - offsetRemaining;
997 
998  // our current position in the contour
999  contour_pos_type dashSectionStart(0, contour[0].GetStartT());
1000  contour_pos_type dashSectionEnd;
1001 
1002  // if we skip the first dash (closed contour, might need a join) store the
1003  // end point here so we can play it back later.
1004  contour_pos_type repeatFirstSectionStart; // invalid
1005  contour_pos_type repeatFirstSectionEnd; // invalid
1006 
1007  // remember where we were in the contour at the last time we "began" the dash pattern.
1008  // This is so we can dtect later if we have NOT moved after a complete dash pattern, then we
1009  // will never move again (think about it) This is caused by very short dashes on very long curves,
1010  // such that the "delta t" for a dash is too small to even advance the position.
1011  contour_pos_type posAtLastDashPatternStart; // // it is invalid with NULL ctor
1012  coord_type minDeltaTPerDashPattern = real_services::Epsilon(coord_type(1.0));
1013 
1014  // a bit of state
1015  size_t numDashSectionsSoFar = 0;
1016  bool hitEndOfContour = false;
1017  // loop over the contour, advancing dash segs and countour segs as needed until done.
1018  while (1)
1019  {
1020  // find the contour position dashSectionEnd such that the arclength from dashSectionStart to
1021  // dashSectionEnd is up to but no greater than curDashLength.
1022  contour.ComputeEndForArclen(curDashLength, dashSectionStart, &dashSectionEnd, &hitEndOfContour);
1023 
1024  if (hitEndOfContour)
1025  {
1026  // If we need to repeat the first segment, send the current contour and repeated first, then leave
1027  if (contour.PosIsValid(repeatFirstSectionStart))
1028  {
1029  if (!curDashIsGap)
1030  dashContour.AppendContourSubsection(contour, dashSectionStart, dashSectionEnd );
1031  dashContour.AppendContourSubsection(contour, repeatFirstSectionStart, repeatFirstSectionEnd );
1032  m_outStroker->GenerateStroke(dashContour, outPath);
1033  return; // all done
1034  }
1035 
1036  // if not a gap, send down the current seg
1037  if (!curDashIsGap)
1038  {
1039  dashContour.AppendContourSubsection(contour, dashSectionStart, dashSectionEnd );
1040  if (contour.IsClosed() && numDashSectionsSoFar == 0)
1041  // Chip sez: next time we're feling gutsy: dashContour = contour!!!
1042  dashContour.Close(dashContour.GetLast().GetEndPoint(), dashContour.GetFirst().GetStartPoint()); // closed iff only one giant seg, (i.e. none so far)
1043  m_outStroker->GenerateStroke(dashContour, outPath);
1044  }
1045  return;
1046  }
1047 
1048  // OK, didn't hit the end, just a normal contour section
1049  if (!curDashIsGap)
1050  {
1051  if (numDashSectionsSoFar == 0 && contour.IsClosed())
1052  {
1053  // skip the first on segment because we may need a join to it at the end
1054  repeatFirstSectionStart = dashSectionStart;
1055  repeatFirstSectionEnd = dashSectionEnd;
1056  }
1057  else
1058  {
1059  // send the subcontour down to the target.
1060  // we may be zoomed in quite far, but unable to discard input path sections outside the visible
1061  // region because that would screw up the dashing phase. However, we can discard DASHES
1062  // outside of the visible region, since they won't show up. We do that check here instead, at
1063  // least for those dashes which fall entirely within on segment (the check is easier)
1064  bool omitdash = false;
1065  if (dashSectionStart.m_segIndex == dashSectionEnd.m_segIndex)
1066  {
1067  const contour_seg_type& seg = contour[dashSectionStart.m_segIndex];
1068  point_type seg_pt[4];
1069  seg.CopyControlPoints(seg_pt, false);
1070  if (seg.IsBezier())
1071  {
1072  if (m_params.InOneOctantOutsideOfVisible(seg_pt[0], seg_pt[1], seg_pt[2], seg_pt[3]))
1073  {
1074  omitdash = true; // if the whole segment is out, a portion is as well, skip it.
1075  }
1076  else
1077  {
1078  point_type sect_pt[4];
1079  ComputeBezSection(seg_pt[0], seg_pt[1], seg_pt[2], seg_pt[3], dashSectionStart.m_segT, dashSectionEnd.m_segT, sect_pt[0], sect_pt[1], sect_pt[2], sect_pt[3]);
1080  if (m_params.InOneOctantOutsideOfVisible(sect_pt[0], sect_pt[1], sect_pt[2], sect_pt[3]))
1081  omitdash = true; // whole sesction is out, skip it.
1082  }
1083  }
1084  else
1085  {
1086  if (m_params.InOneOctantOutsideOfVisible(seg_pt[0], seg_pt[1]))
1087  {
1088  omitdash = true; // if the whole segment is out, a portion is as well, skip it.
1089  }
1090  else
1091  {
1092  point_type sect_pt[2];
1093  sect_pt[0] = seg.GetPointAtParam(dashSectionStart.m_segT);
1094  sect_pt[1] = seg.GetPointAtParam(dashSectionEnd.m_segT);
1095  if (m_params.InOneOctantOutsideOfVisible(sect_pt[0], sect_pt[1]))
1096  omitdash = true; // whole sesction is out, skip it.
1097  }
1098  }
1099  }
1100 
1101  if (!omitdash)
1102  {
1103  dashContour.AppendContourSubsection(contour, dashSectionStart, dashSectionEnd );
1104  m_outStroker->GenerateStroke(dashContour, outPath);
1105  dashContour.ClearSegs(); // we reuse this contour
1106  }
1107  }
1108  }
1109 
1110  // OK, advance the dash pattern and the current contour position
1111  numDashSectionsSoFar++;
1112  curDashIndex = nextDashIndex(curDashIndex);
1113  curDashLength = m_params.GetDashArray()[curDashIndex];
1114  curDashIsGap = !curDashIsGap;
1115  dashSectionStart = dashSectionEnd;
1116 
1117  // if we are starting the pattern over, we want to make sure we actually
1118  // moved since the last time we started the pattern. Furthermore we need
1119  // to verify that we are advancing by enough for the the "t vals" to eventually
1120  // actually add up to 1.0. (Note that if you add float(1e-14) to itself infinitely
1121  // many times, it converges to a value much less than 1 (10-7???)
1122  if (curDashIndex == 0)
1123  {
1124  // make sure we either changed contour segments or that t moved "significantly"
