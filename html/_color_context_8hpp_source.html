164  {
165  if( family == kDeviceGray )

172  { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ),

174  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ) };
175 
176  return color::ICCColorSpace<app_traits>::CreateGraySpace( m_cache,

179  blackPoint,

182  else if( family == kDeviceRGB )
183  {
184  static const static_color_conv_type mediaWhite[ 3 ] =
185  { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.96420 ),
186  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ),
187  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.82491 ) };

191  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ) };
192  static const static_color_conv_type rColorant[ 3 ] =
193  { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.43607 ),
194  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.22249 ),
195  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.01392 ) };
196  static const static_color_conv_type gColorant[ 3 ] =
197  { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.38515 ),

199  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.09708 ) };
200  static const static_color_conv_type bColorant[ 3 ] =
201  { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.14307 ),

208  return color::ICCColorSpace<app_traits>::CreateRGBSpace( m_cache,
209  family,
