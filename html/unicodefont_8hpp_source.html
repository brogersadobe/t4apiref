85 /*
86  Base class for providing unicode code point-based glyph rendering (and font metrics)
87  from a third party implementation (typically the OS).
88 
89  In practice, a UnicodeFont instance is created from a base font name. Ideally, the returned
90  object will render glyphs using a font with that name where possible, and fallback to fonts
91  with similar properties where not. For example, on iOS we use Core Text services and its
92  built-in font cascading to render most of unicode from a base font name of "Helvetica."
93 */
94 template<typename app_traits>
95 class UnicodeFont
96 {
97  typedef typename app_traits::real_geometry_traits::coord_type real_type;
98  typedef typename app_traits::real_geometry_traits::matrix_type matrix_type;
99  typedef typename app_traits::real_geometry_traits::rect_type real_rect_type;
100 
101 public:
102  virtual ~UnicodeFont( void ) {}
103 
104  /*
105  Get the character info associated with a particular code point. If the
106  implementation can't support this code point, an uninitialized optional
107  should be returned.
108  */
109  boost::optional<CharacterInfo> GetCharacterInfo( std::uint32_t ucp ) const
110  {
111  return GetCharacterInfoImpl( ucp );
112  }
113 
114  /*
115  Get the font metrics for this font.
116  */
117  FontMetrics GetFontMetrics( void ) const
118  {
119  return GetFontMetricsImpl();
120  }
121 
122  /*
123  Create an instance of UnicodeGlyphRenderer for the specified matrix.
124  */
125  t4::unique_ptr<UnicodeGlyphRenderer<app_traits>> GetRenderer( const matrix_type& matrix ) const
126  {
127  return GetRendererImpl( matrix );
128  }
129 
130  /*
131  Returns true if the implementation is capable of retrieving outlines for its glyphs, to
132  support stroked text.
133  */
134  bool SupportsOutlines( void ) const
135  {
136  return SupportsOutlinesImpl();
137  }
138 
139  /*
140  Get a bezier path describing the glyph for the specified code point.
141  */
142  void GetCharacterOutline( t4::transient_bezier_path& path, std::uint32_t ucp ) const
143  {
144  return GetCharacterOutlineImpl( path, ucp );
145  }
146 
147  /*
148  Return true if multithreading is supported.
149  */
150  bool SupportsMT( void ) const
151  {
152  return SupportsMTImpl();
153  }
154 
155  const matrix_type& GetFontMatrix() const
156  {
157  return GetFontMatrixImpl();
158  }
159 
160 private:
161  // Implementation interface
162  virtual boost::optional<CharacterInfo> GetCharacterInfoImpl( std::uint32_t ucp ) const = 0;
163  virtual FontMetrics GetFontMetricsImpl( void ) const = 0;
164  virtual t4::unique_ptr<UnicodeGlyphRenderer<app_traits>> GetRendererImpl( const matrix_type& matrix ) const = 0;
165  virtual bool SupportsOutlinesImpl( void ) const = 0;
166  virtual void GetCharacterOutlineImpl( t4::transient_bezier_path& path, std::uint32_t ucp ) const = 0;
167  virtual bool SupportsMTImpl( void ) const = 0;
168  virtual const matrix_type& GetFontMatrixImpl() const
169  {
170  return imaging_model::IdentityMatrix<matrix_type>();
171  }
172 };
