43 /*
44  This is the maximum size of an image tile, including the fringe. This should be a
45  power of 2.
46  */
47 extern const int32_t kImageAtlasEntryWidth;
48 const int32_t kMaxTileSizeWithFringe = kImageAtlasEntryWidth;
49 
50 /*
51  The fringe size. ImageTiles are padded with replicated pixels (on exterior edges) or
52  neighboring tile pixels (on interior edges) in regions this wide/tall.
53  */
54 static const int32_t kTileFringeSize = 2;
55 
56 /*
57  The size of the actual pixel data represented by a single tile. This is just the
58  full ImageTile minus the fringe.
59  */
60 static const int32_t kMaxTileSize = ( kMaxTileSizeWithFringe - 2 * kTileFringeSize );

62 inline gpuRectInt32 PixelToGridRect( const gpuRectInt32& rectInPixels )
63 {
64  return t4::make<gpuRectInt32>(
65  get_xmin( rectInPixels ) / kMaxTileSize,
66  get_ymin( rectInPixels ) / kMaxTileSize,
67  ( get_xmax( rectInPixels ) + kMaxTileSize - 1 ) / kMaxTileSize,
68  ( get_ymax( rectInPixels ) + kMaxTileSize - 1 ) / kMaxTileSize );
69 }

71 struct LoadedTile
72 {
73  /*
74  m_tile must be loaded. The tile will be larger than m_tileDeviceRect for two reasons: the LoadedTile
75  may only be representing a subset of the tile (for example in obj/mask tilings with different resolutions,
76  hence different tilings), and the tile contains the image fringe.
77  */
78  ImageTile* m_tile;
79  gpuRectFloat m_textureRect; // subrectangle in normalized texture space identifying source pixels
80  gpuRectFloat m_tileDeviceRect; // subrectangle in image device space represented by this tile
81 };

83 inline void AdjustTextureRect( gpu::LoadedTile& loadedTile, const gpu::gpuRectFloat& deviceSubRect )
84 {
85  GpuAssert( tetraphilia::imaging_model::RectContainsRect( loadedTile.m_tileDeviceRect, deviceSubRect ) );
86  float wr = get_width( loadedTile.m_textureRect ) / get_width( loadedTile.m_tileDeviceRect );
87  float hr = get_height( loadedTile.m_textureRect ) / get_height( loadedTile.m_tileDeviceRect );
88 
89  // The four coords are set in this unusual order because we use the min values in the computation
90  set_xmax( loadedTile.m_textureRect,
91  get_xmin( loadedTile.m_textureRect ) + wr * ( get_xmax( deviceSubRect ) - get_xmin( loadedTile.m_tileDeviceRect ) ) );
92  set_ymax( loadedTile.m_textureRect,
93  get_ymin( loadedTile.m_textureRect ) + hr * ( get_ymax( deviceSubRect ) - get_ymin( loadedTile.m_tileDeviceRect ) ) );
94  set_xmin( loadedTile.m_textureRect,
95  get_xmin( loadedTile.m_textureRect ) + wr * ( get_xmin( deviceSubRect ) - get_xmin( loadedTile.m_tileDeviceRect ) ) );
96  set_ymin( loadedTile.m_textureRect,
97  get_ymin( loadedTile.m_textureRect ) + hr * ( get_ymin( deviceSubRect ) - get_ymin( loadedTile.m_tileDeviceRect ) ) );
98  loadedTile.m_tileDeviceRect = deviceSubRect;
99 }

140  /*
141  RequestTile/GetLoadedTile have some important constraints on permitted call sequences
142  which permit subclasses of ImageTileSet to stream bytes close to optimally. Specifically:
143 
144  1) For any two calls to RequestTile or GetLoadedTile, the value of the y coordinate in gridLoc
145  must monotonically increase.
146 
147  2) All calls to RequestTile with the same y coordinate must precede all calls to GetLoadedTile
148  with that y coordinate.
149 
150  3) Any call to GetLoadedTile with a specified coordinate must be preceded by a call to RequestTile
151  with the same x/y coordinates.
152  */
153  /*
154  A consumer of an ImageTileSet indicates that this tile's data will be required. A call
155  to RequestTile() must precede any subsequent call to GetLoadedTile() with the same coordinate.
156  */
157  void RequestTile( const gpuPointInt32& gridLoc )
158  {
159  RequestTileImpl( gridLoc );
160  }
161 
162  /*
163  Get a LoadedTile corresponding to the given location in the ImageTileSet. The
164  rectangles in the LoadedTile will be clipped to the rectangle corresponding to
165  m_pixelSubRect.
166 
167  If an unconstructed optional is returned, that means there weren't sufficient
168  resources to create the tile.
169  */
170  boost::optional<LoadedTile> GetLoadedTile( const gpuPointInt32& gridLoc )
171  {
172  return GetLoadedTileImpl( gridLoc );
173  }
174 
175  /*
176  Utility function that gets the tile set's PixelFormat from its key
177  */
178  PixelFormat GetPixelFormat( void ) { return m_imageKey.m_format; }
179 
180 private:
181  virtual void RequestTileImpl( const gpuPointInt32& gridLoc ) = 0;
182 
183  virtual boost::optional<LoadedTile> GetLoadedTileImpl( const gpuPointInt32& gridLoc ) = 0;
184 };

186 /*
187  Enumerate all the tiles overlapping an ImageTileSet's m_pixelSubRect.
188 
189  functor f must have the following signature:
190  void f( const gpuPointInt32& objGridLoc, const gpuRectInt32& objSubDeviceBounds, const gpuRectInt32& objTileBounds );
191 
192  Meaning of the parameters:
193 
194  objGridLoc: The grid location of the tile
195  objSubDeviceBounds: The intersection of m_pixelSubRect and the pixels covered by the tile. In image device space.
196  objTileBounds: The subrectangle of the image spanned by the complete tile.
197 */
198 template<typename functor>
199 void WalkImageTileSet( functor f, ImageTileSet* tileSet )
200 {
201  gpuPointInt32 objGridLoc;
202  gpuRectInt32 objSubDeviceBounds;
203  gpuRectInt32 objTileBounds;
204 
205  get_y( objGridLoc ) = get_ymin( tileSet->m_subGridRect );
206  set_ymin( objTileBounds, get_y( objGridLoc ) * kMaxTileSize );
207  set_ymin( objSubDeviceBounds, get_ymin( tileSet->m_pixelSubRect ) );
208 
209  for ( ; get_y( objGridLoc ) < get_ymax( tileSet->m_subGridRect ); ++get_y( objGridLoc ) )
210  {
211  set_ymax( objTileBounds, std::min( get_ymin( objTileBounds ) + kMaxTileSize, get_y( tileSet->m_pixelSize ) ) );
212  set_ymax( objSubDeviceBounds, std::min( get_ymax( tileSet->m_pixelSubRect ), get_ymax( objTileBounds ) ) );
213 
214  get_x( objGridLoc ) = get_xmin( tileSet->m_subGridRect );
215  set_xmin( objTileBounds, get_x( objGridLoc ) * kMaxTileSize );
216  set_xmin( objSubDeviceBounds, get_xmin( tileSet->m_pixelSubRect ) );
217 
218  for ( ; get_x( objGridLoc ) < get_xmax( tileSet->m_subGridRect ); ++get_x( objGridLoc ) )
219  {
220  set_xmax( objTileBounds, std::min( get_xmin( objTileBounds ) + kMaxTileSize, get_x( tileSet->m_pixelSize ) ) );
221  set_xmax( objSubDeviceBounds, std::min( get_xmax( tileSet->m_pixelSubRect ), get_xmax( objTileBounds ) ) );
222 
223  f( objGridLoc, objSubDeviceBounds, objTileBounds );
224 
225  set_xmin( objSubDeviceBounds, get_xmax( objSubDeviceBounds ) );
226  set_xmin( objTileBounds, get_xmax( objTileBounds ) );
227  }
228 
229  set_ymin( objSubDeviceBounds, get_ymax( objSubDeviceBounds ) );
230  set_ymin( objTileBounds, get_ymax( objTileBounds ) );
231  }
232 }

234 /*
235  Enumerate all tile intersections of two ImageTileSets.
236 
237  functor f must have the following signature:
238  void f( const gpuPointInt32& objGridLoc, const gpuRectFloat& objSubDeviceBounds, const gpuRectInt32& objTileBounds,
239  const gpuPointInt32& maskGridLoc, const gpuRectFloat& maskSubDeviceBounds, const gpuRectInt32& maskTileBounds );
240 
241  Meaning of the parameters:
242 
243  objGridLoc: The grid location of the object tile
244  objSubDeviceBounds: The intersection of the tile intersection and the pixels covered by the object tile.
245  In image device space.
246  objTileBounds: The subrectangle of the object image spanned by the complete object tile.
247 
248  maskGridLoc: The grid location of the mask tile
249  maskSubDeviceBounds: The intersection of tile intersection and the pixels covered by the mask tile. In
250  image device space.
251  maskTileBounds: The subrectangle of the mask image spanned by the complete mask tile.
252 
253 */
254 template<typename functor>
255 void WalkImageTileSets( functor f, ImageTileSet* objTileSet, ImageTileSet* maskTileSet )
256 {
257  auto objToMaskRatio = t4::make<gpuPointFloat>( static_cast<float>( get_x( objTileSet->m_pixelSize ) ) / get_x( maskTileSet->m_pixelSize ),
258  static_cast<float>( get_y( objTileSet->m_pixelSize ) ) / get_y( maskTileSet->m_pixelSize ) );
259 
260  gpuPointInt32 objGridLoc;
261  gpuRectFloat objSubDeviceBounds;
262  gpuRectInt32 objTileBounds;
263  gpuRectFloat maskSubDeviceBounds;
264  gpuRectInt32 maskTileBounds;
265  gpuPointInt32 maskGridLoc;
266 
267  set_ymin( objSubDeviceBounds, get_ymin( objTileSet->m_pixelSubRect ) );
268  get_y( objGridLoc ) = static_cast<int32_t>( get_ymin( objSubDeviceBounds ) ) / kMaxTileSize;
269  set_ymin( objTileBounds, get_y( objGridLoc ) * kMaxTileSize );
270  set_ymin( maskSubDeviceBounds, std::max( static_cast<float>( get_ymin( maskTileSet->m_pixelSubRect ) ), get_ymin( objSubDeviceBounds ) / get_y( objToMaskRatio ) ) );
271  get_y( maskGridLoc ) = static_cast<int32_t>( get_ymin( maskSubDeviceBounds ) / kMaxTileSize );
272  set_ymin( maskTileBounds, get_y( maskGridLoc ) * kMaxTileSize );
273 
274  while ( get_ymin( objSubDeviceBounds ) < get_ymax( objTileSet->m_pixelSubRect )
275  && get_ymin( maskSubDeviceBounds ) < get_ymax( maskTileSet->m_pixelSubRect ) )
276  {
277  float nextObjYBoundary = std::min( ( get_y( objGridLoc ) + 1 ) * kMaxTileSize, get_ymax( objTileSet->m_pixelSubRect ) );
278  float nextMaskYBoundaryMaskSpace = std::min( ( get_y( maskGridLoc ) + 1 ) * kMaxTileSize, get_ymax( maskTileSet->m_pixelSubRect ) );
279  float nextMaskYBoundaryObjSpace = nextMaskYBoundaryMaskSpace * get_y( objToMaskRatio ); // in obj device space
280 
281  set_ymax( objTileBounds, std::min( get_ymin( objTileBounds ) + kMaxTileSize, get_y( objTileSet->m_pixelSize ) ) );
282  set_ymax( maskTileBounds, std::min( get_ymin( maskTileBounds ) + kMaxTileSize, get_y( maskTileSet->m_pixelSize ) ) );
283 
284  set_ymax( objSubDeviceBounds, std::min( nextObjYBoundary, nextMaskYBoundaryObjSpace ) );
285  if ( nextMaskYBoundaryObjSpace > nextObjYBoundary )
286  nextMaskYBoundaryMaskSpace = nextObjYBoundary / get_y( objToMaskRatio );
287  set_ymax( maskSubDeviceBounds, nextMaskYBoundaryMaskSpace );
288 
289  set_xmin( objSubDeviceBounds, get_xmin( objTileSet->m_pixelSubRect ) );
290  get_x( objGridLoc ) = static_cast<int32_t>( std::floor( get_xmin( objSubDeviceBounds ) / kMaxTileSize ) );
291  set_xmin( objTileBounds, get_x( objGridLoc ) * kMaxTileSize );
292  set_xmin( maskSubDeviceBounds, std::max( static_cast<float>( get_xmin( maskTileSet->m_pixelSubRect ) ), get_xmin( objSubDeviceBounds ) / get_x( objToMaskRatio ) ) );
293  get_x( maskGridLoc ) = static_cast<int32_t>( get_xmin( maskSubDeviceBounds ) / kMaxTileSize );
294  set_xmin( maskTileBounds, get_x( maskGridLoc ) * kMaxTileSize );
295 
296  while ( get_xmin( objSubDeviceBounds ) < get_xmax( objTileSet->m_pixelSubRect )
297  && get_xmin( maskSubDeviceBounds ) < get_xmax( maskTileSet->m_pixelSubRect ) )
298  {
299  float nextObjXBoundary = std::min( ( get_x( objGridLoc ) + 1 ) * kMaxTileSize, get_xmax( objTileSet->m_pixelSubRect ) );
300  float nextMaskXBoundaryMaskSpace = std::min( ( get_x( maskGridLoc ) + 1 ) * kMaxTileSize, get_xmax( maskTileSet->m_pixelSubRect ) );
301  float nextMaskXBoundaryObjSpace = nextMaskXBoundaryMaskSpace * get_x( objToMaskRatio ); // in obj device space
302 
303  set_xmax( objTileBounds, std::min( get_xmin( objTileBounds ) + kMaxTileSize, get_x( objTileSet->m_pixelSize ) ) );
304  set_xmax( maskTileBounds, std::min( get_xmin( maskTileBounds ) + kMaxTileSize, get_x( maskTileSet->m_pixelSize ) ) );
305 
306  set_xmax( objSubDeviceBounds, std::min( nextObjXBoundary, nextMaskXBoundaryObjSpace ) );
307  if ( nextMaskXBoundaryObjSpace > nextObjXBoundary )
308  nextMaskXBoundaryMaskSpace = nextObjXBoundary / get_x( objToMaskRatio );
309  set_xmax( maskSubDeviceBounds, nextMaskXBoundaryMaskSpace );
310 
311  f( objGridLoc, objSubDeviceBounds, objTileBounds, maskGridLoc, maskSubDeviceBounds, maskTileBounds );
312 
313  get_x( objGridLoc ) = static_cast<int32_t>( ( get_xmin( objSubDeviceBounds ) + get_xmax( objSubDeviceBounds ) ) / ( 2 * kMaxTileSize ) );
314  set_xmin( objTileBounds, get_x( objGridLoc ) * kMaxTileSize );
315  set_xmin( objSubDeviceBounds, get_xmax( objSubDeviceBounds ) );
316  get_x( maskGridLoc ) = static_cast<int32_t>( ( get_xmin( maskSubDeviceBounds ) + get_xmax( maskSubDeviceBounds ) ) / ( 2 * kMaxTileSize ) );
317  set_xmin( maskSubDeviceBounds, get_xmax( maskSubDeviceBounds ) );
318  set_xmin( maskTileBounds, get_x( maskGridLoc ) * kMaxTileSize );
319  }
320 
321  get_y( objGridLoc ) = static_cast<int32_t>( ( get_ymin( objSubDeviceBounds ) + get_ymax( objSubDeviceBounds ) ) / ( 2 * kMaxTileSize ) );
322  set_ymin( objTileBounds, get_y( objGridLoc ) * kMaxTileSize );
323  set_ymin( objSubDeviceBounds, get_ymax( objSubDeviceBounds ) );
324  get_y( maskGridLoc ) = static_cast<int32_t>( ( get_ymin( maskSubDeviceBounds ) + get_ymax( maskSubDeviceBounds ) ) / ( 2 * kMaxTileSize ) );
325  set_ymin( maskSubDeviceBounds, get_ymax( maskSubDeviceBounds ) );
326  set_ymin( maskTileBounds, get_y( maskGridLoc ) * kMaxTileSize );
327  }

357  void PrepareStorage( size_t bitsPerPixel, gpu::int32_t pipeStartX )
358  {
359  m_bitsPerPixel = bitsPerPixel;
360  m_offsetIntoScanLineInBits = static_cast<size_t>( std::max<gpu::int32_t>( 0, get_xmin( m_pixelBounds ) ) - pipeStartX ) * m_bitsPerPixel;
361  m_bitsToCopy = bitsPerPixel * static_cast<size_t>( std::min( get_x( m_imageSize ), get_xmax( m_pixelBounds ) ) - std::max( 0, get_xmin( m_pixelBounds ) ) );
362  m_scanLineStride = ( static_cast<size_t>( get_width( m_pixelBounds ) ) * bitsPerPixel + 7 ) / 8;
363 
364  m_bytes.reset( new unsigned char[ static_cast<size_t>( get_height( m_pixelBounds ) ) * m_scanLineStride ] );
365  m_addr = m_bytes.get();
366 
367  #ifdef _DEBUG
368  for ( unsigned char* x = m_addr; x < ( m_addr + static_cast<size_t>( get_height( m_pixelBounds ) ) * m_scanLineStride ); )
369  {
370  if ( bitsPerPixel == 1 )
371  *x++ = 0xAA;
372  else if ( bitsPerPixel == 8 )
373  *x++ = 0xBB;
374  else
375  {
376  tetraphilia_assert( bitsPerPixel == 24 );
377  *x++ = 0xAA;
378  *x++ = 0xBB;
379  *x++ = 0xCC;
380  }
381  }
382  #endif
383  }

390  void CopyScanLine( const unsigned char* scanLine )
391  {
392  if ( m_addr )
393  {
394  if ( m_bitsPerPixel % 8 == 0 && m_bitsToCopy % 8 == 0 )
395  CopyScanLineFast( scanLine );
396  else
397  CopyScanLineBitwise( scanLine );
398  m_addr += m_scanLineStride;
399  }
400  }
401 
402  void CopyScanLineFast( const unsigned char* scanLine )
403  {
404  size_t bytesPerPixel = m_bitsPerPixel / 8;
405  size_t bytesToCopy = m_bitsToCopy / 8;
406  unsigned char* addr = m_addr;
407  for ( gpu::int32_t x = get_xmin( m_pixelBounds ); x < 0; ++x )
408  {
409  memcpy( addr, scanLine, bytesPerPixel );
410  addr += bytesPerPixel;
411  }
412  scanLine += ( m_offsetIntoScanLineInBits / 8 );
413  memcpy( addr, scanLine, bytesToCopy );
414  addr += bytesToCopy;
415  const unsigned char* lastPixel = scanLine + bytesToCopy - bytesPerPixel;
416  for ( gpu::int32_t x = get_x( m_imageSize ); x < get_xmax( m_pixelBounds ); ++x )
417  {
418  memcpy( addr, lastPixel, bytesPerPixel );
419  addr += bytesPerPixel;
420  }
421  }
422 
423  void CopyScanLineBitwise( const unsigned char* scanLine )
424  {
425  using namespace tetraphilia::imaging_model;
426  const_bitptr src( scanLine, m_offsetIntoScanLineInBits );
427  bitptr dst( m_addr, 0 );
428  for ( gpu::int32_t x = get_xmin( m_pixelBounds ); x < 0; ++x )
429  *dst++ = *src;
430  Copy( dst, src, m_bitsToCopy );
431  dst += static_cast<ptrdiff_t>( m_bitsToCopy );
432  src += static_cast<ptrdiff_t>( m_bitsToCopy - 1 );
433  for ( gpu::int32_t x = get_x( m_imageSize ); x < get_xmax( m_pixelBounds ); ++x )
434  *dst++ = *src;
435  }

437  boost::optional<LoadedTile> EnsureLoaded( ImageCache* gpuImageCache, PixelFormat format )
438  {
439  ImageTile* tile = gpuImageCache->getImageTile( m_tileKey );
440  GpuAssert( format == m_tileKey.m_imageKey.m_format );
441 
442  if ( !tile->hasTextureData() )
443  {
444  tile->upload( t4::make<gpuRectInt16>( 0, 0, static_cast<int16_t>( get_width( m_pixelBounds ) ),
445  static_cast<int16_t>( get_height( m_pixelBounds ) ) ),
446  format, m_bytes.get() );
447  }
448 
449  boost::optional<LoadedTile> result;
450 
451  if ( tile->hasTextureData() )
452  {
453  result = LoadedTile();
454  result->m_tile = tile;
455 
456  #ifdef GPU_SHOW_TILE_BOUNDARIES
457  result->m_tileDeviceRect = t4::convert<gpuRectFloat>( t4::inset( m_pixelBounds, 4*gpu::kTileFringeSize, 4*gpu::kTileFringeSize ) );
458  #else
459  result->m_tileDeviceRect = t4::convert<gpuRectFloat>( t4::inset( m_pixelBounds, gpu::kTileFringeSize, gpu::kTileFringeSize ) );
460  #endif
461 
462  result->m_textureRect = tile->GetAtlasItem()->Normalize( t4::make<gpu::gpuRectInt32>(
463  gpu::kTileFringeSize, gpu::kTileFringeSize,
464  get_width( m_pixelBounds ) - gpu::kTileFringeSize, get_height( m_pixelBounds ) - gpu::kTileFringeSize ) );
465  }
466 
467  return result;
468  }
