60 template<typename T, typename NativeObjectHolder, typename... TArgs>
61 void create_and_set_java_proxy_common( JNIEnv* env, jobject ths, t4::unique_ptr<NativeObjectHolder> h, TArgs&&... args )
62 {
63  auto native_proxy_clazz = find_class( env, "com/adobe/t4/NativeProxy" );
64  check_is_instance_of( env, ths, native_proxy_clazz );
65  auto fid = get_field_id( env, native_proxy_clazz, "proxy", "J" );
66 
67  auto old_holder = static_cast<NativeObjectHolder*>( jni_detail::get_native_object_holder( env, ths ) );
68 
69  // This shouldn't ever happen--only a duplicate call to createNative() should trigger this
70  // condition--but let's muddle through best we can
71  assert( old_holder == nullptr );
72 
73  /*
74  Exception handling is tricky here. We need to be absolutely sure the call succeeded before
75  we release the created object, otherwise we'll leak. So first query the JNIEnv and make sure
76  there isn't already an exception pending.
77  */
78  check_exception( env );
79  env->SetLongField( ths, fid, reinterpret_cast<jlong>( h.get() ) );
80  if ( env->ExceptionOccurred() )
81  {
82  /*
83  We don't really know what went wrong here. It could be the field doesn't exist. Or maybe
84  it does and something else went wrong. Maybe the write even succeeded! Let's null out
85  the field just to be absolutely sure.
86  */
87  env->SetLongField( ths, fid, 0 );
88  check_exception( env );
89  }
90 
91  // If we got here, we can be pretty sure the write succeeded. So now we should delete the old native object;
92  // we do this by capturing it in a unique_ptr.
93  t4::unique_ptr<NativeObjectHolder> h2(( old_holder ));
94 
95  h.release();
96 }

100 template<typename NativeType>
101 inline NativeType* get_native_object( JNIEnv* env, jobject ths ) BOOST_NOEXCEPT
102 {
103  jclass clazz = env->FindClass( "com/adobe/t4/NativeProxy" );
104  if ( !env->IsInstanceOf( ths, clazz ) )
105  {
106  assert( false ); // This shouldn't ever happen
107  return nullptr;
108  }
109  else
110  {
111  auto noh = static_cast<jni_detail::native_object_holder<NativeType>*>( jni_detail::get_native_object_holder( env, ths ) );
112  if ( noh )
113  return &noh->get();
114  assert( false ); // This too should not ever happen
115  return nullptr;
116  }
117 }
