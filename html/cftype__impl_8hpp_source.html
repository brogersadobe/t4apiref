88  {

91  }

95  {

103  template<typename Right>
104  static std::enable_if_t<std::is_convertible<Left, Right>::value || std::is_convertible<Right, Left>::value, bool>
105  compare( Left lhs, Right rhs )
106  {
107  return lhs == rhs;

137 struct convert_helper<ArithmeticType, std::enable_if_t<is_arithmetic<ArithmeticType>::value>>
138 {
139  static ArithmeticType convert( CFNumberRef rhs )
140  {
141  ArithmeticType result;
142  if ( !CFNumberGetValue( rhs, type_to_CFNumberType<ArithmeticType>::value, &result ) )
143  {
144  /*
145  From the Apple docs:
146 
147  "If the argument type differs from the return type, and the conversion is lossy or the return value
148  is out of range, then this function passes back an approximate value in valuePtr and returns false."

151  */
152  throw std::range_error( "" );

170 template<typename Traits, typename Allocator>
171 struct convert_helper<std::basic_string<char, Traits, Allocator>, void>
172 {
173  typedef std::basic_string<char, Traits, Allocator> result_type;
174 
175  static result_type convert( CFStringRef rhs )
176  {
177  auto ulen = CFStringGetLength( rhs );
178  auto max_len = CFStringGetMaximumSizeForEncoding( ulen, kCFStringEncodingUTF8 );
179  auto buff = t4::make_unique<UInt8[]>( static_cast<size_t>( max_len ) + 1 );
180  CFIndex used_len;
181  CFStringGetBytes( rhs, { 0, ulen }, kCFStringEncodingUTF8, 0, false, buff.get(), max_len, &used_len );
182  assert( used_len == 0 || buff.get()[ used_len - 1 ] != '\0' );
183  return result_type( reinterpret_cast<char*>( buff.get() ), static_cast<size_t>( used_len ) );
184  }

187

251 template<typename Functor>
252 class call_on_derived_cf_type_helper
253 {
254  CFTypeRef m_obj;
255  CFTypeID m_obj_type;
256  bool* m_found_one;
257  Functor const& m_f;
258 
259 public:
260  call_on_derived_cf_type_helper( CFTypeRef obj, Functor const& f, bool* found_one )
261  : m_obj( obj ),
262  m_obj_type( CFGetTypeID( obj ) ),
263  m_found_one( found_one ),
264  m_f( f )
265  {
266  *m_found_one = false;
267  }
268 
269  template<typename CFType, CFTypeID Getter(void)>
270  void operator()( boost::mp11::mp_list<CFType, cf_info<Getter>> )
271  {
272  if ( *m_found_one )
273  return;
274 
275  if ( get_CFTypeID<CFType>() == m_obj_type )
276  {
277  m_f( static_cast<CFType>( m_obj ) );
278  *m_found_one = true;
279  }
280  }

314 /*
315  For a given CFTypeRef, identify its CFTypeID, cast it to the corresponding type,
316  and invoke f.
317 */
318 template<typename Functor>
319 void call_on_derived_cf_type( CFTypeRef obj, Functor const& f )
320 {
321  bool found_one;
322  boost::mp11::mp_for_each<CFType_info_map>( apple_detail::call_on_derived_cf_type_helper<Functor>( obj, f, &found_one ) );
323  if ( !found_one )
324  f( obj );
325 }
