92  static t4::string get_props(const t4::pdf::font& pdf_font)
93  {
94  t4::ostringstream str;
95 
96  auto base_font = pdf_font.get_dictionary().GetName("BaseFont"); // shows full name; doesn't strip out subset prefix
97  t4::string fontName = base_font ? base_font->get() : "unknown";
98  str << "font=" << fontName;
99 
100  t4::pdf::font::subtype subtype = pdf_font.get_subtype();
101  str << ", subtype=" << to_string(subtype);
102  switch (subtype)
103  {
104  case t4::pdf::font::subtype::type0:
105  {
106  str << "[";
107  auto descendant_fonts = pdf_font.get_dictionary().GetRequiredArray("DescendantFonts");
108  t4::pdf::dictionary cid_font_dict = descendant_fonts.GetRequiredDictionary(0);
109  t4::string cid_subtype = cid_font_dict.GetRequiredName("Subtype").get();
110  if (cid_subtype == "CIDFontType0")
111  str << "CIDFontType0";
112  else if (cid_subtype == "CIDFontType2")
113  str << "CIDFontType2";
114  str << "]";
115  break;
116  }
117  default:
118  break;
119  }
120 
121  t4::optional<t4::pdf::program_type> program_type = pdf_font.get_program_type();
122  if (program_type)
123  {
124  str << ", embedded_font=" << to_string(*program_type);
125  }
126 
127  return str.str();
128  }
