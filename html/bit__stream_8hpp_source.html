87 UInteger> BitStream::GetNextNBitsAsInteger( size_t numBits )
88 {
89  tetraphilia_assert( numBits >= 0 && numBits <= ( sizeof(int) * 8) );
90 
91  UInteger result = 0;
92  while ( numBits )
93  {
94  if ( m_remainingBits == 0 )
95  {
96  m_stm->PeekAssert();
97  m_remainingBits = 8;
98  }
99  auto mn = std::min( numBits, m_remainingBits );
100  result <<= mn;
101  unsigned char tmp = static_cast<unsigned char>(m_stm->Peek());
102  unsigned char mask = static_cast<unsigned char>(~0);
103  tmp &= mask >> (8 - m_remainingBits);
104  result += tmp >> (m_remainingBits - mn);
105  numBits -= mn;
106  m_remainingBits -= mn;
107  if ( m_remainingBits == 0 )
108  m_stm->Advance();
109  }
110  return result;
111 }
