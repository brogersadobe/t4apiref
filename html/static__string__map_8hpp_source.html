36 /*
37  A string-keyed heterogeneous map where the keys are all compile-time hana strings and
38  the values have arbitrary types. This is the preferred data structure for global string-keyed
39  maps because of the extremely low runtime overhead: the keys are all stored as literals.
40 */
41 template<typename Impl>
42 class static_string_map
43 {
44  Impl m_impl;
45 
46 public:
47  /*
48  Construct a static_string_map from a hana::tuple. Every element in this tuple must meet
49  the following requirements:
50 
51  1) Each element must itself be a two-element key/value tuple
52  2) The first element must be a hana::string
53 
54  There are no real constraints on the type of the value element, it can be any type you
55  like and the values within the same map don't have to be the same type.
56 
57  Direct use of this constructor is not recommended. Please consider using make_static_string_set
58  or make_static_string_map instead.
59  */
60  constexpr static_string_map( Impl impl ) : m_impl( impl ) {}
61 
62  /*
63  Get the value corresponding to the provided key, call the provided functor on the
64  retrieved value, and return the value. Because the value types are heterogeneous, f
65  must implement operator() for all possible value types, _and_ the special static_string_map_missing_key
66  type (which is invoked when the key is not found).
67 
68  Key must be one of:
69 
70  - char const*
71  - A char-based specialization of the std::basic_string template (e.g., std::string or t4::string)
72  - A Boost.Hana string (typically the result of BOOST_HANA_STRING or T4_STR)
73 
74  If Key is not one of these types, a compiler error will ensue.
75  */
76  template<typename Key, typename F>
77  constexpr auto get( Key key, F&& f ) const
78  {
79  static_assert( sss_detail::is_valid_key_type<Key>::value, "Invalid Key type" );
80  return get_helper( sss_detail::get_key_impl( key ), std::forward<F>( f ), boost::hana::size_c<0>, boost::hana::size( m_impl ) );
81  }
82 
83  /*
84  Returns true if the key is present in the static_string_map. This function may be called at compile
85  time.
86  */
87  template<typename Key>
88  constexpr bool contains( Key key ) const
89  {
90  static_assert( sss_detail::is_valid_key_type<Key>::value, "Invalid Key type" );
91  return get( sss_detail::get_key_impl( key ), sss_detail::contains_func() );
92  }
93 
94  /*
95  Calls boost::hana::for_each directly on the tuples in the map. f should take a two-element key/value
96  pair, where the key is always a hana string and the value depends on the map (and may vary from element
97  to element).
98  */
99  template<typename F>
100  constexpr void for_each( F&& f ) const;
101 
102  /*
103  Calls boost::hana::any_of directly on the tuples in the map. f should have overloads for all possible
104  value types in the map. These overloads should take a char const* key first parameter, and a value type
105  for the second parameter. Returns true if a key/value pair was found satisfying the predicate.
106  */
107  template<typename F>
108  constexpr bool any_of( F&& f ) const;
109 
110 
111 
112 private:
113  template<typename Key, typename F, size_t lo_, size_t hi_, typename boost::disable_if_c<lo_ + 1 == hi_>::type* = nullptr>
114  constexpr auto get_helper( Key key, F&& f, boost::hana::size_t<lo_> lo, boost::hana::size_t<hi_> hi ) const;
115 
116  template<typename Key, typename F, size_t v_>
117  constexpr auto get_helper( Key key, F&& f, boost::hana::size_t<v_> lo, boost::hana::size_t<v_ + 1> hi ) const;
118 
119  template<typename Key, typename F, size_t v_>
120  constexpr char const* get_helper( Key key, F&& f, boost::hana::size_t<v_>, boost::hana::size_t<v_> ) const;
121 };

123 /*
124  Create a static_string_map which is really just a set: the value type is just boost::none_t, and the only interesting
125  question to ask is whether or not a key is in the set. Use static_string_map::contains for these queries.
126 
127  Example:
128 
129  auto set = make_static_string_set( T4_STR( "prophase" ), T4_STR( "metaphase" ), T4_STR( "anaphase" ), T4_STR( "telophase" ) );
130 
131  You do _not_ have to alphabetize these elements, or insure that they are unique.
132 */
133 template<typename... Elements>
134 constexpr auto make_static_string_set( Elements... es );
135 
136 /*
137  Create a static_string_map of compile-time strings to arbitrary objects.
138 
139  Example:
140 
141  auto map1 = make_static_string_map(
142  T4_KEY_VALUE( "a", make_static_string_set( T4_STR( "1" ), T4_STR( "one" ) ) ),
143  T4_KEY_VALUE( "b", make_static_string_set( T4_STR( "2" ), T4_STR( "two" ) ) ) );
144 
145  ...creates a map of a -> { "1", "one" }, b -> { "2", "two" }
146 
147  A common use is a simple string -> string map. For this use T4_STR_PAIR:
148 
149  auto map2 = make_static_string_map( T4_STR_PAIR( "a", "1" ), T4_STR_PAIR( "b", "2" ), T4_STR_PAIR( "c", "3" ) );
150 
151  ...creates a map of a -> 1, b -> 2, c -> 3. These are all boost::hana::strings, and thus different types, and queryable
152  at compile time.
153 */
154 template<typename... Pairs>
155 constexpr auto make_static_string_map( Pairs... kv );

158  Free function which looks up a key in a string-valued static_string_map. If this is used on a static_string_map

164 /*

172 /*
173  Invoke f() on each key in the provided static_string_map. f() should accept a single char const* argument.
174 
175  Example:
176 
177  auto s = make_static_string_set( T4_STR( "one" ), T4_STR( "two" ) );
178 
179  for_each( s, []( char const* key ){ std::cout << key << std::endl; } );
180 
181  Outputs:
182 
183  one
184  two
185 */
186 template<typename Impl, typename F>
187 constexpr void for_each_key( static_string_map<Impl> const& ssm, F&& f );
188 
189 /*
190  Invoke f() on each key/value pair in the provided static_string_map. f must take two arguments:
191  the first will be a char const* representing the element's key type; the second will be the value of
192  that element (which may vary from element to element).
193 
194  Example:
195 
196  auto s = make_static_string_map( T4_KEY_VALUE( "one", 23 ),
197  T4_KEY_VALUE( "two", "foobar" ) );
198 
199  for_each( s, []( char const* key, auto const& value )
200  {
201  std::cout << key << ": " << value << std::endl;
202  } );
203 
204  Outputs:
205 
206  one: 23
207  two: foobar
208 
209 */
210 template<typename Impl, typename F>
211 constexpr void for_each( static_string_map<Impl> const& ssm, F&& f );
212 
213 /*
214  Invoke f() on each key/value pair in the provided static_string_map. f must take two arguments:
215  the first will be a char const* representing the element's key type;
216  the second will be the value of that element (which may vary from element to element).
217  f() must return a bool to indicate where to stop iteration.
218 
219  Example:
220 
221  auto s = make_static_string_map( T4_KEY_VALUE( "one", 23 ),
222  T4_KEY_VALUE( "two", "foobar" ) );
223  t4::string looking_at = "two";
224 
225  any_of( s, [&looking_at]( char const* key, auto const& value )
226  {
227  if(key == looking_at)
228  {
229  std::cout << key << ": " << value << std::endl;
230  return true;
231  }
232  return false;
233  } );
234 
235  Outputs:
236  two: foobar
237 
238  */
239 template<typename Impl, typename F>
240 constexpr bool any_of( static_string_map<Impl> const& ssm, F&& f );
