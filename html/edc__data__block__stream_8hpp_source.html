116 template<typename app_traits>
117 t4::data_block EDCDataBlockStream<app_traits>::get_next_block_impl( void )
118 {
119  using namespace aescrypt;
120 
121  /*
122  Callers of this api depend upon dataBlock->len to be zero. In case of aes encrytption, the finalize call
123  returns the last block of data which is mostly non zero. Thus GetNextBlockImpl will thus in such cases will
124  never return block of data len == 0; Hence this check, if you reach end of file (maintained by bool m_EOF)
125  return an empty block.Please mark m_source->AtEOF() == true is valid case for getting the last block of data
126  */
127  if ( m_EOF )
128  return t4::data_block();
129 
130  const char* bytes;
131  size_t thisSize = m_bufferedSource.PeekBytes( bytes, AESCryptor::block_size );
132  size_t used = 0;
133 
134  // If there are more than zero but fewer than BLOCK_SIZE bytes left, we have an invalid AES stream.
135  if ( 0 < thisSize && thisSize < AESCryptor::block_size )
136  throw t4::corrupt_data();
137 
138  // We need multiples of AESCryptor::block_size
139  AlignSize<app_traits>(thisSize);
140 
141  auto hdbo = t4::create_heap_data_block_owner( TET_EDC_MAX_BLOCK_SIZE );
142  t4::data_block block = { hdbo->data(), hdbo->size(), hdbo };
143 
144  size_t numDecryptedBytes = 0;
145  if ( thisSize == 0 )
146  {
147  // No more data to decrypt, so finalize AES
148  // Will get a max of one more AES block of decrypted data
149  block.m_len = AESCryptor::block_size;
150  numDecryptedBytes = m_aes->Finish( (unsigned char *)hdbo->data(), block.m_len );
151  block.m_len = numDecryptedBytes;
152  m_EOF = true;
153  }
154  else
155  {
156  numDecryptedBytes = m_aes->Decrypt( (const unsigned char *)bytes, thisSize, (unsigned char *)hdbo->data(), TET_EDC_MAX_BLOCK_SIZE );
157  m_bufferedSource.Advance( thisSize );
158  block.m_len = numDecryptedBytes;
159 
160  /* This code has been added to take care of the scenario when the size of first input block to the AES decryptor is equal to AES Block
161  * Size. In this case, the output block returned to us is of size zero (numDecryptedBytes = 0). This is because the AES decryptor
162  * always retains the last block of output data for handling the padding properly. But if we return the zero sized output from here,
163  * then the zlib code in 'zlib_data_block_stream.hpp' thinks that it has encountered end of the stream which leads to failure in
164  * rendering.
165  *
166  * Fix: We make a recursive call to this method to get more output bytes before returning.
167  */
168 
169  if( 0 == numDecryptedBytes )
170  {
171  tetraphilia_assert(m_aes->BLOCK_SIZE == thisSize);
172  return get_next_block_impl();
173  }
174  }
175 
176  return block;
177 }
