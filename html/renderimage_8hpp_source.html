75 /*
76  Compute the image-pixel-space-to-raster-device matrix. Takes the width and height
77  of an image, and the current CTM, and prepends a scale by the image's width and height.
78  Naive implementations of this operation are subject to precision issues in fixed-point
79  builds; this implementation does not have those precision problems.
80 
81  "matrix" is both an in (ctm) and out (image-pixel-space-to-raster-device) parameter.
82 */
83 template<typename matrix_type, typename int_type>
84 void ComputeImageMatrix( matrix_type& matrix /*in/out*/, int_type width, int_type height )
85 {
86  using namespace imaging_model;
87  typedef typename MatrixTypeHelper<matrix_type>::element_type real_coord;
88 
89 
90  real_coord hght = 1; // to account for the flip (see real_coord(-1) / real_coord(height))
91  real_coord realWidth = real_coord(width);
92  real_coord realHeight = real_coord(height);
93 
94  if ( real_services::RealTypeHelper<real_coord>::isFloatingPointType == 0 ) {
95  /* compensate for precision problems introduced by the
96  terms "1/width" and "1/height" when either of those
97  is even a reasonably large number and must be represented
98  in 16.16 fixed point.
99  */
100  real_coord mx = realWidth > realHeight ? realWidth : realHeight;
101 
102  while ( mx > 1 && hght < (1<<14) ) {
103  realWidth /= 2;
104  realHeight /= 2;
105  mx /= 2;
106  get_a( matrix ) /= 2;
107  get_b( matrix ) /= 2;
108  get_c( matrix ) /= 2;
109  get_d( matrix ) /= 2;
110  hght *= 2;
111  }
112  }
113 
114  matrix_type lm;
115  MakeMatrix( lm, real_coord(1) / realWidth, 0, 0, real_coord(-1) / realHeight, 0, hght );
116  matrix = lm * matrix;
117 }

261 /*
262  The purpose of ApplyConstantAlpha is to take the two "clip" and "mask" inputs and modify
263  them as appropriate based on the values of the /ca and /CA keys in the ExtGState dictionary.
264  This is special-case code for images only because images are supposed to _ignore_ the
265  ExtGState dictionary's /SMask entry if they already have an /SMask.
266 
267  Given a gstate, clip and mask RP, look in the gstate for a constant alpha value of
268  something other than one and apply it to the clip or the mask, as appropriate.
269 
270  clip and mask are in-out parameters and may be modified if the gstate has interesting
271  constant alpha.
272 
273  About the "desc" parameter
274 
275  The desc parameter should be the GraphicLayoutDescriptor of the TransparencyGroup or
276  buffer into which the current object is being composited.
277 
278  A "channel" (in the sense used here) is either shape, opacity or color. An "input" is a
279  tuple of (shape, opacity, color) values, although not all of the channels have to have
280  associated values for all operations (channels in an input are permitted to be null, but
281  they must not be read or an assertion failure/null deref occurs).
282 
283  The clip operation takes a "clip" input whose shape is used to linearly interpolate all the
284  values of the other two parameters. The clip input's opacity and color channels are ignored
285  and may be null.
286 
287  The mask operation takes a "mask" input whose opacity is used to linearly interpolate the
288  opacity and color channels of the other two inputs. The "mask" input's _shape_ is used to
289  interpolate the shape channels of the other two inputs.
290 
291  So, this means the following:
292 
293  1) If AIS = true, we only need to produce shape.
294  2) If AIS = false, we need to produce opacity unconditionally, but we will also need to
295  produce shape _if_ we're computing result shape.
296 
297  So, ApplyConstantAlpha sometimes needs to be able to know whether or not the client needs to
298  compute the result shape. The "desc" parameter's m_shapeRep member indicates whether or not
299  shape needs to be computed.
300 */
301 template<typename signal_traits>
302 void ApplyConstantAlpha( const imaging_model::Constraints& constraints,
303  imaging_model::GraphicLayoutDescriptor desc,
304  const GState& gState,
305  t4::brief_ptr<imaging_model::RasterPainter<signal_traits>>& clip,
306  t4::brief_ptr<imaging_model::RasterPainter<signal_traits>>& mask )
307 {
308  using namespace imaging_model;
309 
310  typedef typename signal_traits::app_traits app_traits;
311  typedef typename app_traits::real_geometry_traits::coord_type real_type;
312 
313  if ( gState.GetConstantAlpha( false ) != real_type( 1.0 ) )
314  {
315  // We're modifying the first parameter to either the clip or the mask step, depending on whether
316  // alpha is shape right now. Mask out the color channel, because we don't use the color channel
317  // in either case.
318  desc.m_colorRep = GraphicLayoutDescriptor::kColorIgnored;
319 
320  /*
321  We have to create a temporary buffer to hold the computed result of MakeContantClip() or
322  MakeConstantMasker(). The correct layout depends on whether or not alpha is shape.
323  */
324  if ( gState.GetAlphaIsShape() )
325  {
326  // If AIS == true, then we will multiply the constant alpha into the shape input. The result
327  // shape is then used to linearly interpolate all the channels in the other parameters. But
328  // _only_ the shape is required for the clip parameter.
329  desc.m_shapeRep = GraphicLayoutDescriptor::kShapeReplicated;
330  desc.m_opacityRep = GraphicLayoutDescriptor::kOpacityIgnored;
331 
332  auto tmpSeg = MakeTempBuffer<signal_traits>( desc, constraints );
333 
334  // "MakeConstantClip" should really have been named "MakeConstantShape."
335  auto constantShapeRP = MakeConstantClip<signal_traits>( constraints,
336  gState.GetConstantAlpha( false ) );
337  clip = MakeMultiplier( constraints, std::move( tmpSeg ), std::move( clip ), std::move( constantShapeRP ) );
338  }
339  else
340  {
341  // If AIS == false, then we will multiply the constant alpha into the mask input. The result
342  // mask is then used to linearly interpolate all the channels in the other parameters. We also
343  // require the shape if and only if the caller is accumulating shape (as indicated by
344  // desc.m_shapeRep != kShapeIgnored). This happens in, for example, KO groups.
345  auto tmpSeg = MakeTempBuffer<signal_traits>( desc, constraints );
346 
347  // "MakeConstantMasker" should really have been named "MakeConstantOpacity."
348  auto constantOpacityRP = MakeConstantMasker<signal_traits>( constraints,
349  gState.GetConstantAlpha( false ) );
350  mask = MakeMultiplier( constraints, std::move( tmpSeg ), mask, std::move( constantOpacityRP ) );
351  }
352  }
353 }

355 /*
356  Given a gstate, clip and mask RP, apply the current soft mask to the clip or mask,
357  as appropriate.
358 */
359 template<typename signal_traits>
360 void ApplyGStateSoftMask( imaging_model::Constraints& constraints, /* in/out */
361  const RenderGState<signal_traits>& gState,
362  t4::brief_ptr<imaging_model::RasterPainter<signal_traits>>& clip,
363  t4::brief_ptr<imaging_model::RasterPainter<signal_traits>>& mask
364 )
365 {
366  using namespace imaging_model;
367 
368  if ( gState.GetSoftMask() )
369  {
370  constraints.ConstrainFurther( gState.GetSoftMask()->GetNonZeroBounds() );
371  auto softMask = gState.GetSoftMask()->MakeRasterPainter( constraints, gState.GetAlphaIsShape() );
372  if ( gState.GetAlphaIsShape() )
373  {
374  GraphicLayoutDescriptor desc = MakeGraphicLayoutDescriptor( PixelBufferLayout::kReplicatedSignal,
375  GraphicLayoutDescriptor::kShapeReplicated,
376  GraphicLayoutDescriptor::kOpacityIgnored,
377  GraphicLayoutDescriptor::kColorIgnored );
378  auto tmpSeg = MakeTempBuffer<signal_traits>( desc, constraints );
379  clip = MakeMultiplier( constraints, std::move( tmpSeg ), std::move( clip ), std::move( softMask ) );
380  }
381  else
382  {
383  GraphicLayoutDescriptor desc = MakeGraphicLayoutDescriptor( PixelBufferLayout::kReplicatedSignal,
384  GraphicLayoutDescriptor::kShapeReplicated,
385  GraphicLayoutDescriptor::kOpacityReplicated,
386  GraphicLayoutDescriptor::kColorIgnored );
387  auto tmpSeg = MakeTempBuffer<signal_traits>( desc, constraints );
388  mask = MakeMultiplier( constraints, std::move( tmpSeg ), std::move( mask ), std::move( softMask ) );
389  }
390  }

550  if (display_traits::kSupportColorKeyMasks && imageRec.m_maskKind == kColorKeyMask)
551  {
552  ImageStream* istrAlpha = nullptr;
553  auto filteredStreamColorKeyMask = store::GetFilteredStream( imageDict, false /*isInlineImage*/, &istrAlpha );
554 
555  imageRecMask = ImageRecord( rc.m_documentCaches.GetDocument(), imageDict, resources, false/*isInlineImage*/,
556  imageRec.m_JPEG2K ? istrAlpha : nullptr,
557  PDFImageKey( PDFImageKey::kImageXObject, GetStoreKey( imageDict ) ),
558  imageRec.m_numSignals );
559 
560  /*
561  If a file (see Watson #3529694, merged_document_3(1).pdf) has both an indexed color space and
562  a color key mask, the color key mask min/max values are treated as samples, not color values.
563  The color key mask will then simply convert sample values (i.e., indices) into 0's or 1's, which
564  is the desired behavior (for the mask pipe only!).
565  */
566  imageRecMask.m_isIndexedColor = false;
567  pipeMask = t4::make_transient<StreamImagePipe>( *t4::get_thread_heap(), imageRecMask, std::move( filteredStreamColorKeyMask ) );
568  }
569  else if ( (display_traits::kSupportSoftMasks && imageRec.m_maskKind == kSoftMask ) || imageRec.m_maskKind == kStencilMask )
570  {
571  typedef tetraphilia::pdf::store::StoreObjTraits<app_traits> obj_traits;
572  typedef tetraphilia::pdf::store::Dictionary<obj_traits> dict_type;
573  boost::optional<dict_type> maskDict = imageDict.GetDictionary( imageRec.m_maskKind == kSoftMask ? "SMask" : "Mask" );
574  bool useImageDict = false;
575 
576  if(!maskDict)
577  {
578  if ( !istr || !istr->IsJPXStream() )
579  throw t4::corrupt_data();
580  useImageDict = true;
581  }
582 
583  ImageStream* istrMask = nullptr;
584  auto filteredStreamMask = store::GetFilteredStream( (useImageDict ? imageDict : *maskDict), false /*isInlineImage*/, &istrMask );
585 
586  // If a JPX image is being used as an SMask, then we need to use its color channel,
587  // not the alpha channel (Watson bug #2775742).
588  if(istrMask && (useImageDict || !istrMask->IsJPXStream()))
589  istrMask->SetColorProducer(false);
590 
591  // TODO -- if this image is JPEG2000 compressed then it may have the soft mask data embedded
592  // in the image stream (this is the case where the pdf image dictionary has its SMaskInData member
593  // set to something other than 0). In that case we need to construct an ImageRecord using
594  // the ImageStream istr and load the bits from that.
595 
596  imageRecMask = ImageRecord( rc.m_documentCaches.GetDocument(),
597  (useImageDict ? imageDict : *maskDict),
598  resources, false/*isInlineImage*/,
599  (istr && istr->IsJPXStream()) ? istrMask : nullptr/*istr*/,
600  PDFImageKey( PDFImageKey::kImageXObject, GetStoreKey( useImageDict ? imageDict : *maskDict ) ) );
601  if ( imageRecMask.m_numSignals != 1 )
602  throw t4::corrupt_data();
603 
604  imageRecMask.m_maskKind = imageRec.m_maskKind;
605  imageRec.m_maskKind = kNoMask;
606  pipeMask = t4::make_transient<StreamImagePipe>( *t4::get_thread_heap(), imageRecMask, std::move( filteredStreamMask) );
607  }
608  }
609  return RenderImageRec( rc, gState, ctm, imageRec, imageRecMask, pipe, istr, pipeMask, hardClip, blender );

685 template<typename signal_traits>
686 t4::brief_ptr<imaging_model::RasterPainter<signal_traits>>
687 CreateImageMaskRasterPainterFromDict(
688  const imaging_model::Constraints& constraints,
689  RenderContext<signal_traits>& rc,
690  const GState& gState,
691  const typename signal_traits::app_traits::real_geometry_traits::matrix_type& ctm,
692  store::Dictionary<store::StoreObjTraits<typename signal_traits::app_traits> >& imageDict,
693  const store::Dictionary<store::StoreObjTraits<typename signal_traits::app_traits> >* resources
694  )
695 {
696  using namespace imaging_model;
697 
698 
699  // note that by passing 0 for the ImageStream parameter (istr) we are electing to never be
700  // notified when an image is compressed with either JPEG or JPEG2000. This means that we
701  // lose the ability to make the decompressor do certain optimizations and lose the ability to
702  // get certain extended information about the image. The optimizations are primarily related to
703  // downsampling the image, on the fly, which for these two filters is really an issue of not
704  // upsampling the data. So they go faster when we find it appropriate to do this. We are choosing to
705  // not implement this because high resolution, jpg compressed image masks are uncommon, and implementing
706  // it makes the image mask cache logic harder. The "extended" information is the color space, mask,
707  // and chroma-key information that can be in a jpeg2k image. None of these are legal for images being
708  // used as a mask, so can safely ignore them in this situation.
709  ImageRecord imageRec( rc.m_documentCaches.GetDocument(), imageDict, resources,
710  false/*isInlineImage*/,
711  nullptr/*istr*/,
712  PDFImageKey( PDFImageKey::kImageXObject, GetStoreKey( imageDict ) ) );
713 
714  ImagePipe* pipe;
715  //use the cache if the image is < 4K
716  // TODO: move this constant into the traits somewhere
717  size_t imageDataSize = static_cast<size_t>(imageRec.m_height) * ((static_cast<size_t>(imageRec.m_width)*imageRec.m_numSignals*imageRec.m_bitsPerSignal+7)/8);
718  if (imageDataSize < 4096)
719  {
720  auto mem = GetImageCache(rc.m_documentCaches.GetDocument()).GetImageData( imageDict, imageDataSize );
721  pipe = t4::make_transient<BufferImagePipe>( *t4::get_thread_heap(), imageRec, reinterpret_cast<char*>( mem->data() ) );
722  }
723  else {
724  // TODO: possible? to construct this stream *after* the constraints are fully computed so we can skip if the constraints become empty
725  auto filteredStream = store::GetFilteredStream( imageDict, false /*isInlineImage*/ );
726  pipe = t4::make_transient<StreamImagePipe>( *t4::get_thread_heap(), imageRec, std::move(filteredStream));
727  }
728 
729  return CreateImageMaskRasterPainterFromRec( constraints, rc, gState, ctm, imageRec, pipe );
730 }
