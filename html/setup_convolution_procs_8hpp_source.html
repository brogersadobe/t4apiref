156 template<typename dp, typename image_sampler_traits>
157 typename SpatialFilterInfo<dp, image_sampler_traits>::convolutionFilteringPlanarProcType
158 SpatialFilterInfo<dp, image_sampler_traits>::SetupPlanarProc8Bit(
159  ImageScalingInfo<typename image_sampler_traits::app_traits>& isi) const
160 {
161  const FilteringConvTable<app_traits>& coefficientsTablesShort = *GetCoefficientsTablesShort(isi);
162 
163  if(isi.filter==InterpFunctions<app_traits>::BoxFilter && isi.scaleAdjusted<1)
164  {
165  if(coefficientsTablesShort.countAllocated==4)
166  {
167  return dp::BoxFilterSampling4By4;
168  }
169  else if(coefficientsTablesShort.countAllocated==8)
170  {
171  return dp::BoxFilterSampling8By8;
172  }
173  else
174  {
175  if((coefficientsTablesShort.countAllocated%4)==0)
176  {
177  return dp::BoxFilterSampling8xBy8x;
178  }
179  else
180  {
181  // if we don't have allocated mod 4 then we must be invoking the generic "C" code
182  return dp::BoxFilterSamplingNByNAccurate;
183  }
184  }
185  }
186 
187  if(coefficientsTablesShort.key.tapCount==4 || coefficientsTablesShort.countAllocated==4)
188  {
189  return dp::FilterSampling4By4;
190  }
191  else if(coefficientsTablesShort.countAllocated==8)
192  {
193  return dp::FilterSampling8By8;
194  }
195  else
196  {
197  if(coefficientsTablesShort.key.tapCount==1)
198  {
199  return dp::BoxFilterSamplingNByNAccurate;
200  }
201  else if(coefficientsTablesShort.key.tapCount==2)
202  {
203  return dp::FilterSampling2By2;
204  }
205  else if(coefficientsTablesShort.key.tapCount==3)
206  {
207  return dp::FilterSampling3By3;
208  }
209  else if(coefficientsTablesShort.key.tapCount==5)
210  {
211  return dp::FilterSampling5By5;
212  }
213  else if(coefficientsTablesShort.key.tapCount==6)
214  {
215  return dp::FilterSampling6By6;
216  }
217  else if(coefficientsTablesShort.key.tapCount==7)
218  {
219  return dp::FilterSampling7By7;
220  }
221  else
222  {
223  return
224  FilterSamplingAnyByAnyHelper<app_traits, unsigned char, PinFix29ToChar<app_traits> >::Filter;
225  }
226  }
227 }
