469  t4::real_rect GetUserSpaceBBoxForSubstring( size_t i ) const
470  {
471  t4::real_rect return_val;
472 
473  set_all( return_val, limits<t4::real_coord>::max(), // xmin
474  limits<t4::real_coord>::max(), // ymin
475  limits<t4::real_coord>::min(), // xmax
476  limits<t4::real_coord>::min()); // ymax
477 
478  t4::real_point ascentVec, descentVec;
479  GetUserSpaceAscenderDescenderVectorsForDisplay( ascentVec, descentVec );
480 
481  auto num_glyphs = GetNumGlyphsInSubstring(i);
482  auto pos = m_substringStartPosStack[i];
483 
484  auto it = m_glyphGeomInfoStack.begin() + static_cast<ptrdiff_t>(pos);
485  auto end = it + static_cast<ptrdiff_t>(num_glyphs);
486  while (it != end)
487  {
488  t4::real_coord x[4], y[4];
489  x[0] = get_x( it->penStartPt ) + get_x( ascentVec );
490  y[0] = get_y( it->penStartPt ) + get_y( ascentVec );
491  x[1] = get_x( it->penEndPt ) + get_x( ascentVec );
492  y[1] = get_y( it->penEndPt ) + get_y( ascentVec );
493  x[2] = get_x( it->penEndPt ) + get_x( descentVec );
494  y[2] = get_y( it->penEndPt ) + get_y( descentVec );
495  x[3] = get_x( it->penStartPt ) + get_x( descentVec );
496  y[3] = get_y( it->penStartPt ) + get_y( descentVec );
497 
498  if (i > 0)
499  set_xmin( return_val, t4::min( get_xmin(return_val), x[0], x[1], x[2], x[3]));
500  else
501  set_xmin( return_val, t4::min(get_xmin(m_bBox), x[0], x[1], x[2], x[3]));
502  set_xmax(return_val, t4::max( x[0], x[1], x[2], x[3]));
503  set_ymin(return_val, t4::min( y[0], y[1], y[2], y[3]));
504  set_ymax(return_val, t4::max( y[0], y[1], y[2], y[3]));
505  ++it;
506  }
507 
508  return return_val;
509  }

576  // hopelessly confused when it comes to reflow_layout_iter_type
577  void AdjustForReflow( reflow_layout_iter_type reflowLayoutIter,
578  const reflow_layout_iter_type & reflowLayoutEnd,
579  const t4::pdf::content_point& curContentPoint )
580  {
581  using namespace imaging_model;
582 
583  if( !(*reflowLayoutIter)->GetContentRange().Contains( curContentPoint ) )
584  {
585  // If you hit this assert, then the gstate consumer's content
586  // range iterator and the reflow layout iterator have gotten out
587  // of sync:
588  // TODO: This assert has been disabled because InferredStructureBuilder
589  // does in fact skip certain text ranges, meaning that ReflowLayout
590  // never learns about them when computing the layout. This can
591  // happen when a line begins with whitespace. In order for us to
592  // distinguish here between that case and the case where we have a
593  // bug involving out-of-sync iterators, InferredStructureBuilder will
594  // need to be enhanced so that it creates a read order content range
595  // for a leading space while marking it as invisible.
596  //tetraphilia_assert( 0 );
597  return;
598  }
599 
600  size_t index = m_substringStartPosStack[curContentPoint.m_stringNum] + curContentPoint.m_glyphNum;
601  auto it = m_glyphGeomInfoStack.begin() + static_cast<ptrdiff_t>( index );
602  t4::affine_transform reflowTransform = (*reflowLayoutIter)->GetReflowTransform();
603 
604  while( 1 )
605  {
606  auto glyphRangeIter =
607  (*reflowLayoutIter)->GlyphRangeReflowLayoutBegin();
608  auto glyphRangeEnd =
609  (*reflowLayoutIter)->GlyphRangeReflowLayoutEnd();
610 
611  while( glyphRangeIter != glyphRangeEnd )
612  {
613  tetraphilia_assert( glyphRangeIter->IsValid() );
614  m_hasRanges = true;
615  assert( it != m_glyphGeomInfoStack.end() );
616  it->marksNewRange = true;
617  t4::real_coord dx = glyphRangeIter->m_dx;
618  t4::real_coord dy = glyphRangeIter->m_dy;
619  t4::affine_transform glyphRangeReflowTransform = glyphRangeIter->m_glyphRangeCTM;
620  if( glyphRangeReflowTransform != reflowTransform )
621  {
622  // we are in a situation in which a show is split between multiple
623  // paragraphs; when this show gets rendered, because we only call
624  // GetUserToDeviceMat once per show, the matrix corresponding to
625  // the first record that this show spans will get used for the
626  // entire show; the following compensates for this problem:
627  t4::affine_transform deviceSpaceToUserSpace;
628  if( !MatrixInvert( deviceSpaceToUserSpace, m_userSpaceToDeviceSpaceMat ) )
629  throw t4::corrupt_data();
630  t4::affine_transform mat =
631  m_userSpaceToDefaultUserSpaceMat *
632  glyphRangeReflowTransform *
633  m_unreflowedDefaultUserSpaceToDeviceSpaceMat *
634  deviceSpaceToUserSpace;
635  t4::real_point pt = MatrixTransform( mat, MakePoint<t4::real_point>( dx, dy ) );
636  dx = get_x( pt );
637  dy = get_y( pt );
638  }
639  if( dx == 0 && dy == 0 )
640  it += static_cast<ptrdiff_t>( glyphRangeIter->GetNumGlyphs() );
641  else
642  for( size_t i = 0; i < glyphRangeIter->GetNumGlyphs(); ++i )
643  {
644  get_x( it->glyphStartPt ) += dx;
645  get_y( it->glyphStartPt ) += dy;
646  get_x( it->glyphEndPt ) += dx;
647  get_y( it->glyphEndPt ) += dy;
648  get_x( it->penStartPt ) += dx;
649  get_y( it->penStartPt ) += dy;
650  get_x( it->penEndPt ) += dx;
651  get_y( it->penEndPt ) += dy;
652  ++it;
653  }
654  ++glyphRangeIter;
655  }
656 
657  if( it != m_glyphGeomInfoStack.end() )
658  it->marksNewRange = true;
659 
660  reflow_layout_iter_type tmpIter = reflowLayoutIter;
661  ++tmpIter;
662  if( tmpIter == reflowLayoutEnd ||
663  (*reflowLayoutIter)->GetContentRange().m_from.m_dlEntryNum !=
664  (*tmpIter)->GetContentRange().m_from.m_dlEntryNum )
665  break;
666  reflowLayoutIter = tmpIter;
667  } // while( 1 )
668 
669  // invalidate the bbox
670  MakeRectangle( m_bBox, limits<t4::real_coord>::max(),
671  limits<t4::real_coord>::max(), limits<t4::real_coord>::min(), limits<t4::real_coord>::min() ); // xmin, xmax, ymin, ymax
672  m_bBoxValid = false;
673  }
