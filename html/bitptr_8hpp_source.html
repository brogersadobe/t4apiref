37 /*
38  const pointer to a single bit at a machine address.
39 */
40 class const_bitptr : boost::addable<const_bitptr, ptrdiff_t,
41  boost::subtractable<const_bitptr, ptrdiff_t,
42  boost::equality_comparable<const_bitptr,
43  boost::incrementable<const_bitptr,
44  boost::decrementable<const_bitptr>>>>>
45 {
46  // The address of the byte containing the referenced bit
47  const unsigned char* m_addr;
48 
49  /*
50  Index of the referenced bit. The index increases left to right in a big-endian
51  fashion (i.e., bit index 0 is in the 2^7 position). This is a signed value so
52  that we can subtract from it and later correctly normalize.
53  */
54  ptrdiff_t m_bitIndex;
55 
56  // Adjust m_addr and m_bitIndex such that m_bitIndex is in [0,8)
57  void Normalize( void )
58  {
59  m_addr += ( m_bitIndex / 8 );
60  m_bitIndex = m_bitIndex % 8;
61  if ( m_bitIndex < 0 )
62  {
63  --m_addr;
64  m_bitIndex += 8;
65  }
66  }

69  friend class bitref;

72 public:
73  const_bitptr( void ) {}
74  const_bitptr( const void* addr, size_t bitIndex )
75  : m_addr( static_cast<const unsigned char*>( addr ) ),
76  m_bitIndex( static_cast<ptrdiff_t>( bitIndex ) )
77  {

205  /*
206  Free function that copies bitsToCopy bits from src to dst. This behaves
207  like std::copy, but is more efficient, copying bits in batches of 8 where
208  possible.
209  */
210  friend void Copy( bitptr dst, const_bitptr src, size_t bitsToCopy );
211 
212 private:
213  // Helper function for Copy(), handling the left and right edges.
214  static void CopyEdgeBits( bitptr& dst, const_bitptr& src, size_t& bitsToCopy )
215  {
216  while ( bitsToCopy > 0 )
217  {
218  size_t theseBits = std::min( bitsToCopy,
219  std::min( 8u - dst.getBitIndex(), 8u - src.getBitIndex() ) );
220 
221  unsigned char dstMask = ( 0xFF >> dst.m_bitIndex ) & ( 0xFF << ( 8 - theseBits - dst.getBitIndex() ) );
222  unsigned char srcMask = ( 0xFF >> src.m_bitIndex ) & ( 0xFF << ( 8 - theseBits - src.getBitIndex() ) );
223  unsigned char maskedVal = static_cast<unsigned char>( ( ( *src.m_addr & srcMask ) >> ( 8 - theseBits - src.getBitIndex() ) ) << ( 8 - theseBits - dst.getBitIndex() ) );
224  *dst.addr() = ( *dst.addr() & ~dstMask ) | maskedVal;
225  dst += static_cast<ptrdiff_t>( theseBits );
226  src += static_cast<ptrdiff_t>( theseBits );
227  bitsToCopy -= theseBits;
228  }
229  }
230 };

272 inline void Copy( bitptr dst, const_bitptr src, size_t bitsToCopy )
273 {
274  while ( dst.m_bitIndex != 0 && bitsToCopy > 0 )
275  bitptr::CopyEdgeBits( dst, src, bitsToCopy );
276 
277  if ( src.m_bitIndex == 0 )
278  {
279  size_t bytesToCopy = bitsToCopy / 8;
280  memcpy( dst.addr(), src.m_addr, bytesToCopy );
281  src.m_addr += bytesToCopy;
282  dst.m_addr += bytesToCopy;
283  bitsToCopy %= 8;
284  }
285  else while ( bitsToCopy > 8 )
286  {
287  *dst.addr() = static_cast<unsigned char>( ( src.m_addr[0] << src.getBitIndex() ) | ( src.m_addr[1] >> ( 8u - src.getBitIndex() ) ) );
288  bitsToCopy -= 8;
289  ++dst.m_addr;
290  ++src.m_addr;
291  }
292 
293  if ( bitsToCopy > 0 )
294  bitptr::CopyEdgeBits( dst, src, bitsToCopy );
295 }
