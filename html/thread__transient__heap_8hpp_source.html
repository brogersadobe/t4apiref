54 /*
55  Insure that there exists a thread-local transient heap for the current scope.
56 
57  If an instance of establish_thread_heap exists on the stack, that insures that
58  the result of get_thread_heap() will be the same non-null value for the lifetime
59  of that instance.
60 
61  establish_thread_heap is re-entrant. Nested establish_thread_heaps do _not_ create
62  a new transient_heap; instead they behave exactly like an instance of transient_snapshot.
63 */
64 class establish_thread_heap
65 {
66  transient_heap* m_old_heap;
67  boost::optional<transient_snapshot> m_snapshot;
68 
69 public:
70  establish_thread_heap( void )
71  : m_old_heap( tl_thread_heap.get() )
72  {
73  if ( !m_old_heap )
74  tl_thread_heap.reset( new (t4_new) transient_heap() );
75  else
76  m_snapshot.emplace( *m_old_heap );
77  }
78 
79  ~establish_thread_heap( void )
80  {
81  if ( m_old_heap != tl_thread_heap.get() )
82  {
83  assert( !m_snapshot );
84  unique_ptr<transient_heap> old_heap( tl_thread_heap.release() );
85  tl_thread_heap.reset( m_old_heap );
86  }
87  }
88 };
