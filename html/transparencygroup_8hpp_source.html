107 template<typename signal_traits>
108 bool RasterDestinationCompatible( const TransparencyTuple<const PixelBuffer<signal_traits>*>* dest, const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& cs )
109 {
110  bool result = true;
111  size_t csNumSignals = cs ? cs->GetNumSignals() : PixelBufferLayout::kReplicatedSignal;
112 
113  /*
114  A destination is compatible with a color space if, for each of the (shape, opacity, color) channels,
115  the channel is either missing, replicated or has the same number of components as the provided
116  color space.
117 
118  If the color space is null then _all_ components must be missing or replicated. And note that color
119  can't be replicated, so it must in this case be missing.
120  */
121  if ( dest->m_shape )
122  result = result && SignalCountCompatible( dest->m_shape->m_layout->m_numSignals, csNumSignals );
123  if ( dest->m_opacity )
124  result = result && SignalCountCompatible( dest->m_opacity->m_layout->m_numSignals, csNumSignals );
125  if ( dest->m_color )
126  result = result && ( cs && dest->m_color->m_layout->m_numSignals == csNumSignals );
127  return result;
128 }
129 
130 template<typename app_traits>
131 bool DescriptorCompatible( const GraphicLayoutDescriptor& desc, const typename color::ColorSpace<app_traits>::smart_ptr_type& cs )
132 {
133  if ( cs && desc.m_numComponents != PixelBufferLayout::kReplicatedSignal )
134  return cs->GetNumSignals() == desc.m_numComponents;
135  else
136  {
137  // If we don't have a real CS or num components == replicated, we can only
138  // be compatible if all of the channels are ignored or replicated
139  bool result = desc.m_numComponents == PixelBufferLayout::kReplicatedSignal;
140 
141  result = result && ( desc.m_shapeRep != GraphicLayoutDescriptor::kShapePerChannel );
142  result = result && ( desc.m_opacityRep != GraphicLayoutDescriptor::kOpacityPerChannel );
143  result = result && ( desc.m_colorRep == GraphicLayoutDescriptor::kColorIgnored );
144 
145  return result;
146  }
147 }

406 /*
407  Implements the backdrop removal operation:
408 
409  RemoveBackdrop( <fb, qb, Cb>, <fs, qs, Cs> ):
410 
411  fr = fs
412  qr = qb U qs
413  Cr = cs - ( 1 - as ) * cb
414 */
415 template<typename signal_traits>
416 class RemoveBackdropOperation
417 {
418  typedef typename signal_traits::signal_type signal_type;
419 
420 public:
421  enum { num_args = 2 };
422 
423  template<typename walker_cluster_type>
424  TETRAPHILIA_FORCE_INLINE signal_type operator()( walker_cluster_type walkers, Shape, size_t component )
425  {
426  return GetSignal( walkers, Shape(), 1, component );
427  }
428 
429  template<typename walker_cluster_type>
430  TETRAPHILIA_FORCE_INLINE signal_type operator()( walker_cluster_type walkers, Opacity, size_t component )
431  {
432  return signal_traits::union_( GetSignal( walkers, Opacity(), 0, component ), GetSignal( walkers, Opacity(), 1, component ) );
433  }
434 
435  template<typename walker_cluster_type>
436  TETRAPHILIA_FORCE_INLINE signal_type operator()( walker_cluster_type walkers, Color, size_t component )
437  {
438  signal_type cb = GetSignal( walkers, Color(), 0, component );
439  signal_type ai = GetDestSignal( walkers, Opacity(), component );
440  signal_type cs = GetSignal( walkers, Color(), 1, component );
441 
442  signal_type subtrahend = signal_traits::multiply( signal_traits::complement( ai ), cb );
443 
444  // Due to rounding or color conversion, it's possible for the backdrop component to be quite different from cs.
445  if ( subtrahend > cs )
446  return signal_traits::zero();
447 
448  return t4::min<signal_type>( ai, signal_traits::subtract( cs, subtrahend ) );
449  }
450 
451  #ifdef _DEBUG
452  const char* GetName( void )
453  {
454  return "RemoveBackdrop";
455  }
456  #endif
457 };
