56 template<typename C>
57 t4::vector<C> read_stream_bytes( t4::data_block_stream stream )
58 {
59  static_assert (sizeof(C) == 1, "Must be single byte type");
60  if (!stream)
61  throw std::runtime_error("");
62  t4::vector<C> bytes;
63  while (!stream.at_eof())
64  {
65  t4::data_block block = stream.get_next_block();
66  if (block.get_size() == 0)
67  break;
68 
69  size_t total_bytes_read = bytes.size();
70  bytes.resize(total_bytes_read + block.get_size());
71  C* pos = bytes.data() + total_bytes_read;
72  memcpy(pos, block.get_bytes(), block.get_size());
73  }
74 
75  return bytes;
76 }
