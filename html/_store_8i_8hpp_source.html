89 template<typename app_traits>
90 template<typename Credentials>
91 bool Store<app_traits>::Decrypt( Credentials&& credentials )
92 {
93  // We clear the IDO cache again, so that if any strings/streams parsed prior to decryption
94  // are discarded (since their strings won't be decrypted).
95  m_securityHandler.reset();
96  m_objCache.clear();
97 
98  auto byteRange = GetByteRangeForEncryption();
99  if (byteRange)
100  GetDataStore().prefetch_bytes( *byteRange );
101 
102  m_securityHandler = ComputeSecurityHandler( std::forward<Credentials>( credentials ) );
103 
104  // We clear the IDO cache again, so that if any strings/streams parsed in the above steps (e.g., the
105  // /Lang key in the root dictionary) are discarded (since their strings won't be decrypted).
106  m_objCache.clear();
107 
108  return m_securityHandler->is_decrypted();
109 }

280 template<typename obj_traits>
281 const FiltRec<obj_traits>* GetFilterRec( const Name<obj_traits>& name, bool allowAbbreviations, bool reverse = false )
282 {
283  // Filters can be applied in 2 ways
284  // forward == !reverse : used while reading a PDF
285  // reverse : used to create a PDF for writing
286 
287  const FiltRec<obj_traits>* flt = nullptr;
288  if(!reverse)
289  {
290  static const FiltRec<obj_traits> list[] = {
291  {"ASCIIHexDecode", FilterProcs<obj_traits>::ASCIIHex, false, false},
292  {"AHx", FilterProcs<obj_traits>::ASCIIHex, true, false},
293  {"ASCII85Decode", FilterProcs<obj_traits>::ASCII85, false, false},
294  {"A85", FilterProcs<obj_traits>::ASCII85, true, false},
295  {"DCTDecode", FilterProcs<obj_traits>::JPEG, false, false},
296  {"DCT", FilterProcs<obj_traits>::JPEG, true, false},
297  {"JPXDecode", FilterProcs<obj_traits>::JPEG2K, false, false},
298  {"CCITTFaxDecode", FilterProcs<obj_traits>::CCITT, false, true},
299  {"CCF", FilterProcs<obj_traits>::CCITT, true, true},
300  {"LZWDecode", FilterProcs<obj_traits>::LZW, false, true},
301  {"LZW", FilterProcs<obj_traits>::LZW, true, true},
302  {"FlateDecode", FilterProcs<obj_traits>::Zlib, false, true},
303  {"Fl", FilterProcs<obj_traits>::Zlib, true, true},
304  {"RunLengthDecode", FilterProcs<obj_traits>::RunLength, false, true},
305  {"RL", FilterProcs<obj_traits>::RunLength, true, true},
306  {"JBIG2Decode", FilterProcs<obj_traits>::JBIG2, false, true},
307  { nullptr, nullptr, false, false }
308  };
309 
310  flt = list;
311  }
312  else
313  {
314  // TODO_READER_MOBILE: sanath
315  // Support other filters
316  static const FiltRec<obj_traits> list[] = {
317  {"FlateDecode", FilterProcs<obj_traits>::ZlibDeflate, false, true},
318  {"Fl", FilterProcs<obj_traits>::ZlibDeflate, true, true},
319  { nullptr, nullptr, false, false }
320  };
321 
322  flt = list;
323  }
324 
325 
326  while ( flt->filterName ) {
327  if ( (name == flt->filterName) && (allowAbbreviations || !flt->isAbbreviation) )
328  return flt;
329  ++flt;
330  }
331  return nullptr;
332 }

382 template<typename app_traits>
383 t4::data_block_stream GetUnfilteredStreamFromUnsavedStream( const Dictionary<StoreObjTraits<app_traits>> & dict,
384  bool isInlineImage )
385 {
386  // apply any filters
387  typedef StoreObjTraits<app_traits> obj_traits;
388  typedef Object<obj_traits> obj_type;
389  typedef Array<obj_traits> array_type;
390 
391  auto const& us = dict.GetUnsavedStream();
392 
393  if ( us.AlreadyCompressed() )
394  return t4::make_data_block_stream_from_data_store( us.GetDataStore() );
395  else
396  {
397  t4::data_block_stream dbs = t4::make_data_block_stream_from_data_store( us.GetDataStore() );
398  obj_type filter = dict.Get( inline_image::kFilter, isInlineImage );
399  obj_type decodeParams = dict.Get( inline_image::kDecodeParms, isInlineImage );
400 
401  switch( filter.Type() )
402  {
403  case kObjectTypeName :
404  if ( store_detail::SupportsFilter( filter.NameValue(), isInlineImage, true /*isReverse*/ ) )
405  {
406  dbs = store_detail::ApplyFilter( std::move( dbs ),
407  filter.NameValue(),
408  store_detail::ExtractFirstElementIfPermissive( decodeParams ),
409  isInlineImage, /*allowAbbreviations*/
410  nullptr,
411  true /*isReverse */);
412  }
413  else
414  throw t4::unimplemented();
415  break;
416  case kObjectTypeNull:
417  //Do nothing
418  break;
419  case kObjectTypeArray:
420  {
421  array_type filterArray = filter.ArrayValue_NoTypeCheck();
422  if ( filterArray.Length() > 1 )
423  {
424  // TODO_READER_MOBILE: sanath
425  // Implement support for multiple filters
426  throw t4::unimplemented();
427  }
428  else if ( filterArray.Length() == 1 )
429  {
430  obj_type firstFilter = filterArray.Get(0);
431  if ( firstFilter.Type() == kObjectTypeName )
432  {
433  if ( store_detail::SupportsFilter( firstFilter.NameValue(), isInlineImage, true /*isReverse */ ) )
434  {
435  dbs = store_detail::ApplyFilter( std::move( dbs ),
436  firstFilter.NameValue(),
437  decodeParams,
438  isInlineImage, /*allowAbbreviations*/
439  nullptr,
440  true /*isReverse */);
441  }
442  else
443  throw t4::unimplemented();
444  }
445  else
446  throw t4::corrupt_data();
447  }
448  }
449  break;
450  default:
451  throw t4::corrupt_data();
452  }
453  return dbs ;
454  }

554  }

558  throw t4::corrupt_data();

579  len = bds.Mark() - 9; // strlen("endstream")

582  }

585  throw t4::corrupt_data();
586  }
587  }

600 template<typename app_traits>
601 t4::data_block_stream ApplyFilters( t4::data_block_stream result, const Dictionary<StoreObjTraits<app_traits>>& dict, bool isInlineImage,
602  render::ImageStream** istr )
603 {
604  typedef StoreObjTraits<app_traits> obj_traits;
605  typedef Name<obj_traits> name_type;
606  typedef Array<obj_traits> array_type;
607  typedef typename app_traits::pdf_traits pdf_traits;
608 
609  auto filter = dict.Get( inline_image::kFilter, isInlineImage );
610  auto decodeParams = dict.Get( inline_image::kDecodeParms, isInlineImage );
611  auto ido = dict.GetPDFObject().GetIDOPtr();
612  auto decrypter = ido->GetDecrypter();
613 
614  switch ( filter.Type() ) {
615 
616  case kObjectTypeNull:
617  if( decrypter )
618  result = decrypter->decrypt_default_stream( std::move( result ) );
619  break;
620 
621  case kObjectTypeName: {
622  name_type filterName = filter.NameValue_NoTypeCheck();
623  if ( decrypter && (filterName != "Crypt") )
624  result = decrypter->decrypt_default_stream( std::move( result ) );
625  if ( store_detail::SupportsFilter( filterName, isInlineImage ) )
626  result = store_detail::ApplyFilter( std::move( result ), filterName, store_detail::ExtractFirstElementIfPermissive( decodeParams ), isInlineImage, istr );
627  else if ( decrypter && filterName == "Crypt" )
628  result = store_detail::ApplyCryptFilter( std::move( result ), *ido, store_detail::ExtractFirstElementIfPermissive( decodeParams ) );
629  else
630  throw t4::unimplemented();
631  } break;
632 
633  case kObjectTypeArray: {
634  auto filterArray = filter.ArrayValue_NoTypeCheck();
635  boost::optional<array_type> decodeParamsArray;
636  typename array_type::const_iterator_type decodeParamsIt;
637 
638  if ( decodeParams.Type() != kObjectTypeNull )
639  {
640  decodeParamsArray = decodeParams.ArrayValue();
641  if ( pdf_traits::permissive && decodeParamsArray->Length() == 0 )
642  decodeParamsArray.reset();
643  else
644  decodeParamsIt = decodeParamsArray->begin();
645  }
646 
647  size_t totalNumFilters = filterArray.Length();
648  if ( decodeParamsArray && decodeParamsArray->Length() != totalNumFilters )
649  throw t4::corrupt_data();
650 
651  typename array_type::const_iterator_type filterIt = filterArray.begin();
652  typename array_type::const_iterator_type filterEnd = filterArray.end();
653 
654  bool needsDefaultCryptFilter = decrypter != nullptr;
655  if ( needsDefaultCryptFilter ) {
656  while ( filterIt != filterEnd )
657  {
658  if ( filterIt->NameValue() == "Crypt" )
659  {
660  needsDefaultCryptFilter = false;
661  break;
662  }
663  ++filterIt;
664  }
665 
666  if ( needsDefaultCryptFilter )
667  result = decrypter->decrypt_default_stream( std::move( result ) );
668  }
669 
670  filterIt = filterArray.begin();
671  while ( filterIt != filterEnd )
672  {
673  name_type fltName = filterIt->NameValue();
674  if ( (!decrypter || fltName != "Crypt") && !store_detail::SupportsFilter( fltName, isInlineImage ) )
675  throw t4::unimplemented();
676 
677  if( decodeParamsArray )
678  {
679  decodeParams = *decodeParamsIt;
680  ++decodeParamsIt;
681  }
682 
683  if ( fltName == "Crypt" )
684  result = store_detail::ApplyCryptFilter( std::move( result ), *ido, decodeParams );
685  else
686  result = store_detail::ApplyFilter( std::move( result ), fltName, decodeParams, isInlineImage, istr );
687  ++filterIt;
688  }
689  } break;
690 
691  default:
692  throw t4::corrupt_data();
693  }
694 
695  return result ;
696 }

857 template<typename app_traits>
858 permissions::PermReqStatus Store<app_traits>::CheckSecurityHandlerPermissions( permissions::PermReqObj permObj, permissions::PermReqOpr permOpr )
859 {
860  if ( m_securityHandler.get() == nullptr )
861  throw t4::unknown(); //must call IsDecrypted before GetPermissions
862  security::SecurityFlags securityPerms = m_securityHandler->get_permissions();
863 
864  /* Currently, we don't have support for encrypted incremental updates. Hence, turning off those bits of 'SecurityFlags' which allow document modification for
865  * files having any kind of encryption. */
866  /* Treat provided PDF.Next files read-only */
867  //TODO_READER_MOBILE<vmarda>: Move this logic to ARApp level when we have APIs in place for AR layer to figure out whether the PDF is encrypted or not.
868  if ( m_securityHandlerName ) //Unconstructed 'm_securityHandlerName' means the document doesn't have a security handler.
869  {
870  securityPerms &= ~t4::pdf::permitted_operations({ security::kSecurityPermitModification, security::kSecurityPermitAnnotations, security::kSecurityPermitFormEntry,
871  security::kSecurityPermitPageOperation });
872  }
873 
874  permissions::PermReqStatus reqStatus = permissions::PS_PERM_REQ_DENIED;
875  switch ( permObj )
876  {
877  case permissions::PO_PERM_REQ_OBJ_DOCUMENT:
878  {
879  switch ( permOpr )
880  {
881  case permissions::PO_PERM_REQ_OPR_MODIFY:
882  {
883  if ( securityPerms[ security::kSecurityPermitModification ] )
884  reqStatus = permissions::PS_PERM_REQ_GRANTED;
885  break;
886  }
887 
888  case permissions::PO_PERM_REQ_OPR_COPY:
889  {
890  if ( securityPerms[ security::kSecurityPermitContentExtraction ] )
891  reqStatus = permissions::PS_PERM_REQ_GRANTED;
892  break;
893  }
894 
895  case permissions::PO_PERM_REQ_OPR_PRINT_HIGH:
896  {
897  if ( securityPerms[ security::kSecurityPermitHighQualityPrinting ] )
898  reqStatus = permissions::PS_PERM_REQ_GRANTED;
899  break;
900  }
901 
902  case permissions::PO_PERM_REQ_OPR_PRINT_LOW:
903  {
904  if ( securityPerms[ security::kSecurityPermitPrinting ] )
905  reqStatus = permissions::PS_PERM_REQ_GRANTED;
906  break;
907  }
908 
909  default:
910  reqStatus = permissions::PS_PERM_REQ_INVALID_OPR;
911  }
912 
913  break;
914  }
915 
916 
917  case permissions::PO_PERM_REQ_OBJ_ANNOT:
918  {
919  switch ( permOpr )
920  {
921  case permissions::PO_PERM_REQ_OPR_CREATE:
922 
923  case permissions::PO_PERM_REQ_OPR_DELETE:
924 
925  case permissions::PO_PERM_REQ_OPR_MODIFY:
926  {
927  if ( securityPerms[ security::kSecurityPermitAnnotations ] )
928  reqStatus = permissions::PS_PERM_REQ_GRANTED;
929  break;
930  }
931 
932  default:
933  reqStatus = permissions::PS_PERM_REQ_INVALID_OPR;
934  }
935 
936  break;
937  }
938 
939  case permissions::PO_PERM_REQ_OBJ_FORM:
940  {
941  switch ( permOpr )
942  {
943  case permissions::PO_PERM_REQ_OPR_CREATE:
944 
945  case permissions::PO_PERM_REQ_OPR_DELETE:
946 
947  case permissions::PO_PERM_REQ_OPR_MODIFY:
948  {
949  if ( securityPerms[ security::kSecurityPermitAnnotations ] && securityPerms[ security::kSecurityPermitModification ] )
950  reqStatus = permissions::PS_PERM_REQ_GRANTED;
951  break;
952  }
953 
954  case permissions::PO_PERM_REQ_OPR_FILLIN:
955  {
956  if ( securityPerms[ security::kSecurityPermitAnnotations ] || securityPerms[ security::kSecurityPermitFormEntry ] )
957  reqStatus = permissions::PS_PERM_REQ_GRANTED;
958  break;
959  }
960 
961  default:
962  reqStatus = permissions::PS_PERM_REQ_INVALID_OPR;
963  }
964 
965  break;
966  }
967  }
968 
969  return reqStatus;
970 }

1483 
1484  /*
1485  * Creating a private transient heap used for this function only. As we are passing an allocator created on this heap to the FileOutputStream (FOS) object created
1486  * below, it is highly recommended to NOT pass this heap to any other function and keep it's scope limited to this function only. Doing any such thing might
1487  * lead to interleaving of allocations made by FOS object and other allocations done in the sub-function it is passed to, which might lead to heap corruption if
1488  * TransientSnapShot is used in the sub-function to destroy all the allocations made within it's scope when it ends. For eg:
1489  * .
1490  * .
1491  * Sub-Function Starts
1492  * .

1494  * FOS allocation()
1495  * Function Allocation()
1496  * .

1502  */

1515  both of which fall in this "forbidden zone." So an untimely cache purge can cause the save to fail.

1517  are not flushed.

1574  }

1578  writeCheckSum, temporaryChangesSessionID );
