182  static void Compare2Numbers( Type4FunctionParser& parser, const Comp& cmp, bool isEq = false, bool isNe = false )
183  {
184  typedef typename TetraphiliaPDFTraits::int_type int_type;
185  typedef typename TetraphiliaPDFTraits::real_type real_type;
186 
187  const auto& objTop = parser.TopElement();
188  const auto& objSecondTop = *parser.Element(1);
189 
190  if ( objTop.type == objSecondTop.type ) {
191  switch( objTop.type ) {
192 
193  case store::kObjectTypeInteger: {
194  int_type i1 = parser.PopInt();
195  int_type i2 = parser.PopInt();
196  parser.PushBool( cmp.Compare( i2, i1 ) );
197  } break;
198 
199  case store::kObjectTypeReal: {
200  real_type r1 = parser.PopReal();
201  real_type r2 = parser.PopReal();
202  parser.PushBool( cmp.Compare( r2, r1 ) );
203  } break;
204 
205  case store::kObjectTypeBool:
206  if ( isEq )
207  parser.PushBool( parser.PopBool() == parser.PopBool() );
208  else if ( isNe )
209  parser.PushBool( parser.PopBool() != parser.PopBool() );
210  else
211  throw t4::corrupt_data(); //kAPEUnexpectedType
212  break;
213 
214  default: // always including a default case quiets gcc compiler warning about enumeration types not being handled in a switch statement
215  #ifdef _DEBUG
216  throw t4::corrupt_data(); //kAPEUnexpectedType
217  #endif
218  break;
219  }
220 
221  } else {
222 
223  if ( (objTop.type == store::kObjectTypeInteger)
224  && (objSecondTop.type == store::kObjectTypeReal) ) {
225  int_type i = parser.PopInt();
226  real_type r = parser.PopReal();
227  parser.PushBool( cmp.Compare( r, i ) );
228  } else if ( (objSecondTop.type == store::kObjectTypeInteger)
229  && (objTop.type == store::kObjectTypeReal) ) {
230  real_type r = parser.PopReal();
231  int_type i = parser.PopInt();
232  parser.PushBool( cmp.Compare( i, r ) );
233  } else {
234  if ( isEq )
235  parser.PushBool( false );
236  else if ( isNe )
237  parser.PushBool( true );
238  else
239  throw t4::corrupt_data(); //kAPEUnexpectedType
240  }
241  }
242  }

516 /*
517  General comment about overflow handling
518 
519  The pdf/PostScript spec stipulates that when these arithmetic operators
520  are applied to integers and the result of the operator overflows the
521  integer range, that the result produces a real instead of an integer.
522  This strategy is based on the assumption that real's have a larger
523  range than integers. This assumption is true for IEEE-style reals,
524  but not for 16.16 fixed point reals. In the implementations below,
525  when integer overflows accur the code does one of two things depending
526  on whether it is compiled with IEEE or 16.16 reals. The IEEE case
527  follows the PostScript definitions. The 16.16 implementation generally
528  pins the result to the representable integer that is closest to the
529  correct result.
530 
531  However, in 16.16 fixed point builds the code below does *not* check
532  for overflows in its real operations other than whatever checks
533  are being provided by underlying real_type class (which just checks
534  for mul and div).
535 
536  A few deviations that are worth knowing:
537 
538  1. The PostScript definition for some of the conversion operators is
539  asymetric with respect to error handling. The PostScript implementation
540  uses 32 bit (single precision) float's for its real and 32 bit integers
541  for its integer type. When converting between these two types (with operators
542  like 'cvi' and 'cvr') when starting with a real that is too big to
543  be represented as an integer, the cvi operator executes a rangecheck
544  error. But when starting with an integer that is too big to be represented
545  accurately as a real (for example 0x7fffffff), PostScript does not
546  execute any error. It simply uses the closest real that can represent
547  the result. In the implementation below it does not throw errors in
548  either case, and both simply use the closest available number.
549 */
550 static void AbsOpDo( const Op* /*op*/, Type4FunctionParser& parser )
551 {
552  typedef TetraphiliaPDFTraits::int_type int_type;
553  typedef TetraphiliaPDFTraits::real_type real_type;
554 
555  const auto& obj = parser.TopElement();
556  switch( obj.type )
557  {
558  case store::kObjectTypeInteger:
559  {
560  int_type i = parser.PopInt();
561  if ( i < 0 )
562  {
563  i = -i;
564  if ( i < 0 ) // handle the case where 'i' was originally (and still is) 0x80000000
565  {
566  parser.Push( store::ObjectImpl<typename Type4FunctionParser::object_traits>( HandleNegateOverflow<int_type, real_type>::Adjust( i ) ) );
567  return;
568  }
569  }
570  parser.PushInt( i );
571  } break;
572 
573  case store::kObjectTypeReal:
574  parser.PushReal( AbsVal( parser.PopReal() ) );
575  break;
576 
577  default:
578  throw t4::corrupt_data(); //kAPEUnexpectedType
579  }
