706  get_x(sPoint) = si;
707  }
708  else
709  {
710  set_ymin( dstBounds, dst );
711  set_ymax( dstBounds, dst + 1 );
712  get_y(sPoint) = si;
713  }
714 
715  if ( s < dataEnd[j] )
716  {
717  if ( s + 1 < dataEnd[j] )
718  {
719  tetraphilia_assert( dataEnd[j] - si - 1 >= 1 );
720  int_type z = t4::min<int_type>( limit, dst + dataEnd[j] - si - 1);
721  if ( j == 0 )
722  {
723  set_xmax( dstBounds, z );
724  get_x(s2Point) = get_x(sPoint) + 1;
725  }
726  else
727  {
728  set_ymax( dstBounds, z );
729  get_y(s2Point) = get_y(sPoint) + 1;
730  }
731  // lerp pairs of data values
732  BlendDataPixels<LerpOp<signal_traits>, signal_traits>::Do( storage2D, dstBounds, sPoint, s2Point, a1, a2 );
733  }
734  else if ( s + 1 <= tileEnd[j] )
735  {
736  // lerp rightmost data value with zero
737  // (1-frac) * rightmost data pixel + (s+1-dataEnd[j]) * 0
738  // NOTE: 1-frac = dataEnd[j] - s
739  BlendDataPixels<BlendSrc1WithZeroOp<signal_traits>, signal_traits>::Do( storage2D, dstBounds, sPoint, sPoint, a2, a2 );
740  }
741  else
742  {
743  // lerp rightmost data value with leftmost data value and possibly with zero if there is spacing between tiles
744  // (1-frac) * rightmost data pixel + (tileEnd[j]-dataEnd[j]) * 0 + (s+1-tileEnd[j]) * leftmost data pixel
745  // NOTE: 1-frac = dataEnd[j] - s
746  if ( j == 0 )
747  get_x(s2Point) = dataStart[j];
748  else
749  get_y(s2Point) = dataStart[j];
750  a1 = signal_traits::real_to_signal( (s+1)-tileEnd[j] );
751  BlendDataPixels<BlendWithPartialWeightsOp<signal_traits>, signal_traits>::Do( storage2D, dstBounds, sPoint, s2Point, a2, a1 );
752  }
753  }
754  else if ( s + 1 <= tileEnd[j] )
755  {
756  int_type w = std::min( limit, dst + real_services::FloorToInt( ArgType<int_type>(), tileEnd[j] - s ) );
757  w = std::max( w, dst + 1 );
758  if ( j == 0 )
759  set_xmax( dstBounds, w );
760  else
761  set_ymax( dstBounds, w );
762  // set to w dst pixels to zero
763  BlendDataPixels<SetZeroOp<signal_traits>, signal_traits>::Do( storage2D, dstBounds, sPoint, sPoint, a2, a2 );
764  }
765  else
766  {
767  // lerp zero with leftmost data value
768  // (tileEnd[j]-s) * 0 + (s+1-tileEnd[j]) * leftmost data pixel
769  if ( j == 0 )
770  get_x(s2Point) = dataStart[j];
771  else
772  get_y(s2Point) = dataStart[j];
773  a1 = signal_traits::real_to_signal( (s+1)-tileEnd[j] );
774  BlendDataPixels<BlendSrc1WithZeroOp<signal_traits>, signal_traits>::Do( storage2D, dstBounds, s2Point, s2Point, a1, a1 );
775  }
776  dst = (j == 0) ? get_xmax(dstBounds) : get_ymax(dstBounds);
777  }
778  tetraphilia_assert( dst == limit );
779  }
780  }
