58 /*
59  Specialization of add where all operands are the same size.
60 
61  This is the real meat of the checked add implementation. All calls go through
62  the uniformly-sized implementation.
63 */
64 template<typename Result, typename T1, typename T2>
65 struct add_impl_helper<Result, T1, T2, std::enable_if_t<sizeof(T1) == sizeof(T2) && sizeof(T2) == sizeof(Result)
66  && std::is_integral<Result>::value && std::is_integral<T1>::value && std::is_integral<T2>::value>>
67 {
68  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
69  {
70  static_assert( static_cast<unsigned int>( -1 ) == (~static_cast<unsigned int>( 1 ) + 1u), "This implementation presumes two's complement arithmetic" );
71 
72  using up_t = typename sm_detail::make_unsigned<Result>::type;
73 
74  /*
75  Perform the computation using all unsigned types so as not to generate UB.
76  */
77  auto ux = static_cast<up_t>( x );
78  auto uy = static_cast<up_t>( y );
79  auto ur = static_cast<up_t>( ux + uy );
80 
81  // There is a carry if the unsigned result is less than either of its inputs. This does not necessarily
82  // indicate an error if any of the involved types are signed.
83  auto carry = ( ur < ux ) ? 1 : 0;
84 
85  /*
86  Compute the "sign extension value" of the two inputs and the computed (possibly wrong)
87  output. This is always 0 for unsigned values and nonnegative signed values, and -1 for negative
88  signed values.
89 
90  Conceptually, one could imagine sign-extending a binary number off into infinity; this value is
91  the value of that sequence of bits.
92  */
93  auto xs = ( std::is_signed<T1>::value && high_bit_set( ux ) ) ? -1 : 0;
94  auto ys = ( std::is_signed<T2>::value && high_bit_set( uy ) ) ? -1 : 0;
95  auto rs = ( std::is_signed<Result>::value && high_bit_set( ur ) ) ? -1 : 0;
96 
97  /*
98  The computed value is correct if the carry plus the sign extensions of the inputs add up to the
99  sign extension of the computed result. In essence, we're checking that we would get a correct result
100  if we had infinity more bits to work with in all inputs.
101  */
102  if ( ( carry + xs + ys ) == rs )
103  return static_cast<Result>( ur );
104  return {};
105  }
106 };
