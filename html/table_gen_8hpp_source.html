257 template<typename image_sampler_traits>
258 void WeightTableAccessor<image_sampler_traits>::AssertTapsValid(FilteringConvTableType& taps)
259 {
260  #define _KERN_SHIFT (11)
261 
262  const float thresholdHighF = 1+(32*TETRAPHILIA_FLT_EPSILON);
263  const float thresholdLowF = 1-(32*TETRAPHILIA_FLT_EPSILON);
264  const short thresholdHighS = (short)((1L<<_KERN_SHIFT)+(taps.countAllocated+1)/2);
265  const short thresholdLowS = (short)((1L<<_KERN_SHIFT)-(taps.countAllocated+1)/2);
266  size_t i,j;
267 
268 
269  size_t high = taps.subPixelAccur/2;
270  size_t offsetM = high*taps.setStride;
271 
272  unsigned char* weights = static_cast<unsigned char*>(taps.weights) - offsetM;
273 
274  size_t offset = 0;
275  for(i=0; i<=taps.subPixelAccur; i++)
276  {
277  // compute some filter calculation variables
278  size_t cur = offset;
279  float sF = 0;
280  short sI = 0;
281  if(taps.key.vectorElemSize==sizeof(float))
282  {
283  for(j = 0; j <taps.countAllocated; j++)
284  {
285  sF += *((float*)(weights+cur));
286  cur += taps.key.vectorElemSize;
287  }
288  if(taps.key.convertFromRealCoord==RealToReal)
289  {
290  if(sF < thresholdLowF || sF > thresholdHighF)
291  {
292  if(sF == 0)
293  {
294  cur = offset;
295  for(j = 0; j <taps.key.tapCount; j++)
296  {
297  *((float*)(weights+cur)) = float(1)/(int)taps.key.tapCount;
298  cur += taps.key.vectorElemSize;
299  }
300  }
301  }
302  }
303  else
304  {
305  if(sF < thresholdLowS || sF > thresholdHighS)
306  {
307  if(sF == 0)
308  {
309  cur = offset;
310  for(j = 0; j <taps.key.tapCount; j++)
311  {
312  InterpFunctions<app_traits>::RealToFix( weights+cur, float(1)/(int)taps.key.tapCount);
313  cur += taps.key.vectorElemSize;
314  }
315  }
316  }
317  }
318  }
319  else if(taps.key.vectorElemSize==2)
320  {
321  for(j = 0; j <taps.countAllocated; j++)
322  {
323  sI += *((short*)(weights+cur));
324  cur += taps.key.vectorElemSize;
325  }
326  if(sI < thresholdLowS || sI > thresholdHighS)
327  {
328  if(sI == 0)
329  {
330  cur = offset;
331  for(j = 0; j <taps.key.tapCount; j++)
332  {
333  InterpFunctions<app_traits>::RealToFix( weights+cur, float(1)/(int)taps.key.tapCount);
334  cur += taps.key.vectorElemSize;
335  }
336  }
337  }
338  }
339 
340  else
341  {
342  tetraphilia_assert_always();
343  }
344  offset += taps.setStride;
345  }
346  #undef _KERN_SHIFT
347 }

355 template<typename image_sampler_traits>
356 size_t WeightTableAccessor<image_sampler_traits>::GetSubPixelAccur(real_coord scaleFactor)
357 {
358  #define SUB_PIXEL_ACCUR_F (32.0f)
359  #define SUB_PIXEL_ACCUR (32)
360  #define SUB_PIXEL_ACCUR_FILTER_GEN (32)
361 
362  typedef typename image_sampler_traits::app_traits::int_geometry_traits::coord_type int_coord;
363 
364 #ifdef _OLD_SUB_PIXEL_ACCUR
365  return SUB_PIXEL_ACCUR;
366 #endif
367 
368  int_coord v = SUB_PIXEL_ACCUR;
369  if(scaleFactor < 1)
370  {
371  v = real_services::FloorToInt(ArgType<int_coord>(), scaleFactor*real_coord(SUB_PIXEL_ACCUR)+real_coord(.5));
372 
373  v = t4::max<int_coord>( v, int_coord(2) );
374 
375  // Pixel Accuracy must be even for the system to work
376  if(v&1)
377  v++;
378  }
379 
380  tetraphilia_assert( v >= 2 );
381 
382  tetraphilia_assert( v>=2 && v<=SUB_PIXEL_ACCUR);
383  tetraphilia_assert((v%2) == 0);
384 
385  return static_cast<size_t>( v );
386 }

389 inline void SwapBytes(void* p1, void* p2, size_t count)
390 {
391  unsigned char* a = static_cast<unsigned char*>(p1);
392  unsigned char* b = static_cast<unsigned char*>(p2);
393 
394  for(size_t i=0; i<count; i++)
395  {
396  unsigned char t = *a;
397  *a = *b;
398  *b = t;
399  a++;
400  b++;
401  }

405 void WeightTableAccessor<image_sampler_traits>::ReComputeWeights( filter_function_type /*filter*/,
406  real_coord* tempTapsBuf,
407  real_coord& sumOfWeights,
408  real_coord /*updatedScale*/,
409  real_coord /*subPixel*/,
410  int_coord iLeft,
411  int_coord iRight)
412 {
413  real_coord norm = 1;
414  if(sumOfWeights != 0)
415  norm /= sumOfWeights;
416 
417  if(norm != 1)
418  {
419  real_coord sum = 0;
420

422  {
423  tempTapsBuf[j-iLeft] = tempTapsBuf[j-iLeft] * norm;
424  sum +=tempTapsBuf[j-iLeft];
425  }
426  sumOfWeights = sum;
427  }
