83 template<typename app_traits>
84 FreeTypeSubpixelPositionPolicy<app_traits>::FreeTypeSubpixelPositionPolicy( const matrix_type& mat ) :
85  SubpixelPositionPolicy<app_traits>( mat )
86 {
87  // Set subpixel position policy for CFF fonts
88  // text_traits::sub_pixel_count sets the maximum number of phases to generate
89  // use a size threshold at 24 and 32 ppem to match CoolType
90  // apply subpixel in font's x-direction for all 90 degree rotations
91  // No subpixel for other rotations at this time
92 
93  real_type ppem = t4::max<real_type>( t4::max<real_type>( AbsVal( get_a(mat) ), AbsVal( get_d(mat) ) ),
94  t4::max<real_type>( AbsVal( get_b(mat) ), AbsVal( get_c(mat) ) ) );
95 
96  this->m_xPositionsPerPixel = 1; // set defaults
97  this->m_yPositionsPerPixel = 1;
98 
99  if ( get_b(mat) == 0 ){ // subpixel in device x direction
100  if ( ppem < 24 && kSubPixelCount >= 4 ){
101  this->m_xPositionsPerPixel = 4;
102  } else if ( ppem < 32 && kSubPixelCount >= 2 ){
103  this->m_xPositionsPerPixel = 2;
104  }
105  } else if ( get_a(mat) == 0 ){ // subpixel in device y direction
106  if ( ppem < 24 && kSubPixelCount >= 4 ){
107  this->m_yPositionsPerPixel = 4;
108  } else if ( ppem < 32 && kSubPixelCount >= 2 ){
109  this->m_yPositionsPerPixel = 2;
110  }
111  }
112 }

458 template<typename app_traits>
459 void FreeType<app_traits>::ParseGlyphOutline(bezier_path_type& path, client_matrix_type& drawMatrix, size_t glyphID, const client_matrix_type& fontMatrix)
460 {
461  FT_Error ftError = 0;
462 
463  /*
464  Freetype works in integral dimensions i.e. the drawing callbacks contain integer
465  coordinates. While any ftWidth/ftHeight values will work since they will eventually
466  be scaled again by the client using drawMatrix, there will be loss of accuracy
467  when callback functions operate at values lower than intended font size.
468 
469  To avoid this, we ask Freetype to operate at a higher resolution by scaling ftWidth &
470  ftHeight upto the font-size (derived from fontMatrix) and adjust drawMatrix so that
471  client doesn't need to perform much scaling (if at all). This implies that for zero
472  rotation, drawMatrix will be an indentity matrix.
473  */
474 
475  // Match the char size and transform matrix with RenderBitmap so that outlines and bitmaps
476  // are consistent in all char sizes.
477  client_real_type fontSize = imaging_model::SqrtOfMatrixDeterminant(fontMatrix);
478  int upscaleFactor = fontSize < 20.0 ? 4 : fontSize < 40.0 ? 2 : 1;
479  Fixed16_16 fs = Fixed16_16(fontSize * upscaleFactor);
480  FT_F26Dot6 ftFontSize = fs.fVal >> 10;
481  ftError = FT_Set_Char_Size(m_FreeTypeFace, ftFontSize, ftFontSize, 72, 72);
482  if (ftError != FT_Err_Ok) {
483  // TODO What to do on an error?
484  tetraphilia_assert(false);
485  return;
486  }
487  client_matrix_type mtx = t4::scale(fontMatrix, 1 / (fontSize * upscaleFactor));
488  FT_Matrix ftmtx;
489  ftmtx.xx = static_cast<FT_Fixed>(Fixed16_16(get_a(mtx)).fVal);
490  ftmtx.yx = static_cast<FT_Fixed>(Fixed16_16(get_b(mtx)).fVal);
491  ftmtx.xy = static_cast<FT_Fixed>(Fixed16_16(get_c(mtx)).fVal);
492  ftmtx.yy = static_cast<FT_Fixed>(Fixed16_16(get_d(mtx)).fVal);
493  FT_Set_Transform(m_FreeTypeFace, &ftmtx, nullptr);
494  if (ftError != FT_Err_Ok) {
495  // TODO What to do on an error?
496  tetraphilia_assert(false);
497  return;
498  }
499 
500  drawMatrix = t4::identity<client_matrix_type>();
501 
502  if (ftError == FT_Err_Ok)
503  {
504  // Load glyph
505  // make sure it's not retrieving bitmap cache by specifying FT_LOAD_NO_BITMAP option.
506  ftError = FT_Load_Glyph(m_FreeTypeFace, static_cast<FT_UInt>(glyphID), FT_LOAD_NO_BITMAP);
507  if (ftError != FT_Err_Ok) {
508  tetraphilia_assert(false);
509  return;
510  }
511  if (m_FreeTypeFace->glyph == nullptr) {
512  tetraphilia_assert(false);
513  return;
514  }
515  if (m_FreeTypeFace->glyph != nullptr && m_FreeTypeFace->glyph->outline.n_points > 0)
516  {
517  // Receive glyph outline callbacks
518  ftError = FT_Outline_Decompose(&m_FreeTypeFace->glyph->outline, &m_outlineFunctions, &path);
519 
520  if (ftError != FT_Err_Ok)
521  {
522  // embedded font must be corrupted.
523  throw t4::corrupt_data();
524  }
525  }
526  }
527 }

529 template<typename app_traits>
530 FT_GlyphSlot FreeType<app_traits>::RenderBitmap(const unsigned int glyphIndex,
531  const typename app_traits::int_geometry_traits::rect_type& bounds,
532  const client_matrix_type& matrix)
533 {
534  FT_Error ftError = FT_Err_Ok;
535 
536  // determine font size and transformation matrix using given font matrix.
537  // FreeType by default produces very low quality bitmaps when char size is small. To improve this quality
538  // at the acceptable level (equivalent to old TrueType library,) up-scale char size and down scale final
539  // bitmap using glyph transform matrix.
540  client_real_type fontSize = imaging_model::SqrtOfMatrixDeterminant(matrix);
541  int upscaleFactor = fontSize < 20.0 ? 4 : fontSize < 40.0 ? 2 : 1;
542  Fixed16_16 fs = Fixed16_16(fontSize * upscaleFactor);
543  FT_F26Dot6 ftFontSize = fs.fVal >> 10;
544  ftError = FT_Set_Char_Size(m_FreeTypeFace, ftFontSize, ftFontSize, 72, 72);
545  if (ftError != FT_Err_Ok) {
546  // TODO What to do on an error?
547  tetraphilia_assert(false);
548  return nullptr;
549  }
550  client_matrix_type mtx = t4::scale(matrix, 1 / (fontSize * upscaleFactor)) * t4::make_flip<t4::affine_transform>();
551  FT_Matrix ftmtx;
552  ftmtx.xx = static_cast<FT_Fixed>(Fixed16_16(get_a(mtx)).fVal);
553  ftmtx.yx = static_cast<FT_Fixed>(Fixed16_16(get_b(mtx)).fVal);
554  ftmtx.xy = static_cast<FT_Fixed>(Fixed16_16(get_c(mtx)).fVal);
555  ftmtx.yy = static_cast<FT_Fixed>(Fixed16_16(get_d(mtx)).fVal);
556  // matrix.e and f contain subpixel positioning
557  FT_Vector delta;

658 template<typename app_traits>

660 {

662  return nullptr;
663  }
664 
665  // When FT_Face doesn't have any cmaps, character codes or encodings are useless in this font.
666  // Just return an empty encoding, and make sure don't call FT_Set_Charmap or FT_Select_Charmap.
667  if(1 > m_FreeTypeFace->num_charmaps) {
668  auto& heap = *t4::get_thread_heap();

672  return names;

677  if(m_PureCFF) {
678  if(FT_ENCODING_UNICODE == m_FreeTypeFace->charmaps[0]->encoding) {
679  charmapIndex = 1;
680  }
681  } else if(m_FileFormat == fonts::Font<app_traits>::kType1) {

684  charmapIndex = 1;

690  if(FT_ENCODING_ADOBE_STANDARD == charMap->encoding) {

692  }
693  if(FT_ENCODING_ADOBE_EXPERT == charMap->encoding) {
694  return buildEncoding(ExpertEncode, NUM_EXPERT_ENCODE_GLYPHS);
695  }
696  if(FT_ENCODING_ADOBE_CUSTOM == charMap->encoding || FT_ENCODING_UNICODE == charMap->encoding) {

700  glyphIndex != 0;
701  charCode = FT_Get_Next_Char(m_FreeTypeFace, charCode, &glyphIndex)) {
702  if(glyphIndex < m_FreeTypeFace->num_glyphs && 256 > charCode) {
703  encodingVector[glyphIndex] = static_cast<unsigned char>(charCode);
704  }
705  }
706  return buildEncoding(&encodingVector[0], encodingVector.size());
707  }
708  }
709  return nullptr;

712 template<typename app_traits>
713 const std::uint16_t* FreeType<app_traits>::GetTrueTypeEncodingTable(typename Font<app_traits>::NamedEncoding requestedEncoding, bool isSymbolic,
714  const char* const* * glyphNames ) const
715 {
716  if(m_FileFormat != fonts::Font<app_traits>::kTrueType) {
717  tetraphilia_assert(false);
718  return nullptr;
719  }
720 
721  FT_Error ftError = FT_Err_Ok;
722  FT_ULong offset = 0;
723  switch (requestedEncoding) {
724  case Font<app_traits>::kNamedEncoding_MacRoman:
725  ftError = FT_Select_Charmap(m_FreeTypeFace, FT_ENCODING_APPLE_ROMAN);
726  break;
727  case Font<app_traits>::kNamedEncoding_None:
728  // FT_Select_Charmap can't be called with FT_ENCODING_NONE,
729  // so just don't set it at all.
730  // We want to get the default cmap. If the font has one cmap, we want the only one
731  // cmap. If the font has multiple cmaps, we don't know which cmap should be used.
732  if(1 == m_FreeTypeFace->num_charmaps) {
733  FT_CharMap charMap = m_FreeTypeFace->charmaps[0];
734  ftError = FT_Set_Charmap(m_FreeTypeFace, charMap);
735  } else {
736  // If the font has no cmap or multiple cmaps, we can't set the cmap.
737  ftError = 1;
738  // however, if font has FT_ENCODING_MS_SYMBOL cmap and isSymbolic is true then
739  // select FT_ENCODING_MS_SYMBOL.
740  if(isSymbolic) {
741  for(int i = 0; i < m_FreeTypeFace->num_charmaps; i ++) {
742  if(FT_ENCODING_MS_SYMBOL == m_FreeTypeFace->charmaps[i]->encoding) {
743  ftError = FT_Select_Charmap(m_FreeTypeFace, FT_ENCODING_MS_SYMBOL);
744  offset = 0x0f000L;
745  break;
746  }
747  }
748  // if not the above case, FT_ENCODING_APPLE_ROMAN may work if the sybmolic font has this cmap.
749  // if the symbolic font has FT_ENCODING_APPLE_ROMAN cmap, the default FT_ENCODING_UNICODE
750  // may not work.
751  if(1 == ftError) {
752  for(int i = 0; i < m_FreeTypeFace->num_charmaps; i ++) {
753  if(FT_ENCODING_APPLE_ROMAN == m_FreeTypeFace->charmaps[i]->encoding) {
754  ftError = FT_Select_Charmap(m_FreeTypeFace, FT_ENCODING_APPLE_ROMAN);
755  break;
756  }
757  }
758  }
759  }
760  }
761  break;
762  case Font<app_traits>::kNamedEncoding_WinANSI:
763  if (isSymbolic) {
764  ftError = FT_Select_Charmap(m_FreeTypeFace, FT_ENCODING_MS_SYMBOL);
765  offset = 0x0f000L;
766  } else {
767  ftError = FT_Select_Charmap(m_FreeTypeFace, FT_ENCODING_UNICODE);
768  }
769  break;
770  case Font<app_traits>::kNamedEncoding_MacExpert:
771  // TrueType fonts can't be encoded with MacExpertEncoding named encoding.
772  ftError = 1;
773  break;
774  }
775 
776  if (ftError != FT_Err_Ok) {
777  if (FT_HAS_GLYPH_NAMES(m_FreeTypeFace)) {
778  // TODO Construct glyphNames
779  tetraphilia_assert(false);
780  return nullptr;
781  } else {
782  // TODO Is there anything we can do?
783  tetraphilia_assert(false);
784  return nullptr;
785  }
786  }
787 
788  auto& heap = *t4::get_thread_heap();
789 
790  // allocate the table we will return, and then initialize it to all zeros
791  std::uint16_t* table = t4::make_transient_array<std::uint16_t>( heap, 256 );
792  ::memset( table, 0, 256 * sizeof(std::uint16_t) );
793 
794  FT_UInt glyphIndex = 0;
795  for ( FT_ULong charCode = FT_Get_First_Char(m_FreeTypeFace, &glyphIndex);
796  glyphIndex != 0;
797  charCode = FT_Get_Next_Char(m_FreeTypeFace, charCode, &glyphIndex))
798  {
799  if (charCode - offset <= 255) {
800  table[charCode - offset] = static_cast<std::uint16_t>(glyphIndex);
801  }else {
802  if(Font<app_traits>::kNamedEncoding_WinANSI == requestedEncoding && !isSymbolic) {
803  for(FT_ULong ch = 0; ch < 256; ch ++) {
804  std::uint16_t nonsymbolicUnicode = tetraphilia::fonts::PublicEncodings<app_traits>::PDFWinANSIUnicodeValues[ch];
805  if(charCode == static_cast<FT_ULong>(nonsymbolicUnicode)) {
806  table[ch] = static_cast<std::uint16_t>(glyphIndex);
807  break;
808  }
809  }
810  } else {
811  // TODO What happens if we're out of range?
812  //tetraphilia_assert(false);
813  }
814  }
815  }
816 
817  return table;
818 }
