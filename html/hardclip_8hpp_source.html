184 //*****************************************************************************
185 //
186 // HardClip
187 //
188 // Create a hard edged clip (no antialiasing) from the intersection or union of
189 // a *RasterPainter and a parent HardClip
190 //-----------------------------------------------------------------------------
191 template<typename signal_traits>
192 HardClip<signal_traits>::HardClip( HCConstructOp op, raster_painter_type* rp, HardClip<signal_traits>& hc)
193  : m_bounds(ComputeInitialNonZeroBounds(op,rp->GetNonZeroBounds(),hc.GetNonZeroBounds())),
194  m_isRectangle(false),
195  m_x(0)
196 {
197  m_xstop.reserve( 4u * static_cast<size_t>( get_height( m_bounds ) + 1 ) );
198 
199  // never a rectangle
200 
201  if (RectIsEmpty(m_bounds)) // no valid ys
202  return;
203 
204  int_type height = get_height(m_bounds);
205  m_rowStart.reserve(static_cast<size_t>(height+1));
206 
207  int_type y = get_ymin(m_bounds);
208 
209  SetY( &hc, y );
210  SetY( rp, y );
211 
212  for (; y < get_ymax(m_bounds); y++)
213  {
214  m_rowStart.push_back(m_xstop.size());
215 
216  int_type xStart = get_xmin(m_bounds);
217  bool inside = false;
218 
219  for (;;)
220  {
221  int_type xStop = SetX( &hc, xStart, get_xmax(m_bounds));
222  xStop = SetX( rp, xStart, xStop );
223  tetraphilia_assert( xStop > xStart );
224 
225  // examine hc
226  bool insideHC = hc.Inside();
227 
228  if (op == HC_INTERSECT && !insideHC)
229  {
230  if (inside)
231  {
232  inside = false;
233  m_xstop.push_back(xStart);
234  }
235  }
236  else if (op == HC_UNION && insideHC)
237  {
238  if (!inside)
239  {
240  inside = true;
241  m_xstop.push_back(xStart);
242  }
243  }
244  else
245  {
246  // need to examine the internals of a rasterpainter
247  SegmentID segId = rp->GetCurrentSegmentID();
248  const segment_type *segment = &rp->GetSegments()[segId];
249  tetraphilia_assert(segment->m_shape);
250  if (IsZeroEverywhere(t4::get_ptr(segment->m_shape)))
251  {
252  if (inside)
253  {
254  inside = false;
255  m_xstop.push_back(xStart);
256  }
257  }
258  else if (IsOneEverywhere(t4::get_ptr(segment->m_shape)))
259  {
260  if (!inside)
261  {
262  inside = true;
263  m_xstop.push_back(xStart);
264  }
265  }
266  else
267  {
268  // examine each pixel - below assertions make a fast implementation possible
269  tetraphilia_assert(segment->m_shape->m_pixelBuffer.m_layout->m_numSignals == PixelBufferLayout::kReplicatedSignal);
270  tetraphilia_assert(segment->m_shape->m_pixelBuffer.m_layout->m_signalStride == 0);
271  tetraphilia_assert(segment->m_shape->m_pixelBuffer.m_layout->m_pixelStride == 1);
272  tetraphilia_assert(segment->m_shape->m_pixelBuffer.m_layout->m_firstSignalOffset == 0);
273 
274  // SetX has properly offset this
275  const_SpecializedRasterXWalker<signal_type, 0, 0, 1, 1> walker( t4::get_ptr( segment->m_shape ), xStart );
276 
277  for (int_type x = xStart; x < xStop; x++)
278  {
279  if ( GetSignal( &walker, 0 ) != 0)
280  {
281  if (!inside)
282  {
283  m_xstop.push_back(x);
284  inside = true;
285  }
286  }
287  else if (inside)
288  {
289  m_xstop.push_back(x);
290  inside = false;
291  }
292  NextPixel( &walker );
293  }
294  }
295  }
296 
297  if (xStop >= get_xmax(m_bounds))
298  {
299  if (inside)
300  m_xstop.push_back(get_xmax(m_bounds));
301  break;
302  }
303  xStart = xStop;
304  }
305  IncrementY( rp );
306  IncrementY( &hc );
307  tetraphilia_assert((m_xstop.size()%2) == 0);
308  }
309  // push one past the top this is at ymax
310  m_rowStart.push_back(m_xstop.size());
311  // assert proper reserve above
312  tetraphilia_assert(static_cast<int_type>(m_rowStart.size()) == height+1);
313  tetraphilia_assert(static_cast<int_type>(m_rowStart.capacity()) == height+1);
314 }

316 //*****************************************************************************
317 //
318 // HardClip
319 //
320 // Create a hard edged clip (no antialiasing)
321 // from the intersection or union of two HardClips
322 //-----------------------------------------------------------------------------
323 template<typename signal_traits>
324 HardClip<signal_traits>::HardClip( HCConstructOp op, HardClip<signal_traits>& hc1, HardClip<signal_traits>& hc2)
325  : m_bounds(ComputeInitialNonZeroBounds(op,hc1.GetNonZeroBounds(),hc2.GetNonZeroBounds())),
326  m_isRectangle(op == HC_INTERSECT && hc1.GetIsRectangle() && hc2.GetIsRectangle()),
327  m_x(0)
328 {
329  m_xstop.reserve( m_isRectangle ? 3u : 4u*static_cast<size_t>(get_height(m_bounds)+1) );
330  if (RectIsEmpty(m_bounds)) // no valid ys
331  return;
332 
333  if (m_isRectangle) // use the bounds
334  {
335  m_xstop.push_back(get_xmin(m_bounds));
336  m_xstop.push_back(get_xmax(m_bounds));
337  return;
338  }
339 
340  int_type height = get_height(m_bounds);
341  m_rowStart.reserve(static_cast<size_t>(height+1));
342 
343  int_type y = get_ymin(m_bounds);
344 
345  SetY( &hc1, y );
346  SetY( &hc2, y );
347 
348  for (; y < get_ymax(m_bounds); y++)
349  {
350  m_rowStart.push_back(m_xstop.size());
351 
352  int_type xStart = get_xmin(m_bounds);
353  bool inside = false;
354 
355  for (;;)
356  {
357  int_type xStop = SetX( &hc1, xStart, get_xmax( m_bounds ) );
358  xStop = SetX( &hc2, xStart, xStop );
359 
360  bool insideHC1 = hc1.Inside();
361  bool insideHC2 = hc2.Inside();
362  bool newInside = op == HC_INTERSECT ? insideHC1 && insideHC2 : insideHC1 || insideHC2;
363  if (inside != newInside)
364  {
365  m_xstop.push_back(xStart);
366  inside = newInside;
367  }
368 
369  if (xStop >= get_xmax(m_bounds))
370  {
371  if (inside)
372  m_xstop.push_back(get_xmax(m_bounds));
373  break;
374  }
375  xStart = xStop;
376  }
377 
378  IncrementY( &hc1 );
379  IncrementY( &hc2 );
380  }
381  // push one past the top this is at ymax
382  m_rowStart.push_back(m_xstop.size());
383 
384  // assert proper reserve above
385  tetraphilia_assert(static_cast<int_type>(m_rowStart.size()) == height+1);
386  tetraphilia_assert(static_cast<int_type>(m_rowStart.capacity()) == height+1);
387 }

404 template<typename signal_traits>
405 typename signal_traits::app_traits::int_geometry_traits::coord_type HardClip<signal_traits>::SetX_( int_type x, int_type xstop )
406 {
407  // check the y bounds
408  if (!m_validY)
409  {
410  m_inside = false;
411  m_nextStopState = eStopStateOutOfRange; // way out of bounds
412  return xstop;
413  }
414 
415  if (x < m_x)
416  m_nextStopState = eStopStateOutOfRange;
417 
418  m_x = x;
419 
420  if (m_x >= get_xmax(m_bounds))
421  {
422  m_inside = false;
423  m_nextStopState = eStopStateOutOfRange;
424  }
425  else if (m_isRectangle)
426  {
427  m_nextXStop = m_xstop.begin();
428  m_nextStopState = eStopStateValid;
429  if (x < get_xmin(m_bounds))
430  m_inside = false;
431  else
432  {
433  m_inside = true;
434  ++m_nextXStop;
435  }
436  }
437  else
438  {
439  // set it up to answer Inside and GetXStop
440  int_stack_iterator_type begIter;
441  if (m_nextStopState != eStopStateValid)
442  {
443  // start from the beginning of the line
444  m_inside = false;
445  begIter = m_xstop.begin() + static_cast<ptrdiff_t>( m_rowStart[static_cast<size_t>(m_y-get_ymin(m_bounds))] );
446  }
447  else
448  begIter = m_nextXStop;
449  int_stack_iterator_type endIter = m_xstop.begin() + static_cast<ptrdiff_t>( m_rowStart[static_cast<size_t>(m_y-get_ymin(m_bounds)+1)] );
450  for (; begIter != endIter; ++begIter)
451  {
452  if (m_x < *begIter)
453  {
454  m_nextStopState = eStopStateValid;
455  m_nextXStop = begIter;
456  return t4::min<int_type>( xstop, *m_nextXStop ); // all done
457  }
458  m_inside = !m_inside;
459  }
460  m_nextStopState = eStopStateOutOfRange;
461  }
462  return m_nextStopState != eStopStateValid ? xstop : t4::min<int_type>( xstop, *m_nextXStop );
463 }
