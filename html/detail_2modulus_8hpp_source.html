28 template<typename Result, typename T1, typename T2, typename = void>
29 struct modulus_impl_helper
30 {
31  /*
32  This is the fully-general version of modulus that works for all combinations of integer types.
33  */
34  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
35  {
36  if ( y == 0 )
37  return {};
38 
39  if ( x == 0 )
40  return 0;
41 
42  using up_t1 = typename std::make_unsigned<T1>::type;
43  using up_t2 = typename std::make_unsigned<T2>::type;
44  using up_result = typename std::make_unsigned<Result>::type;
45 
46  // Until C++11, if one or both operands to binary operator % were negative, the sign of the remainder was implementation-defined, as it depends on the rounding direction of integer division.
47  // Since C++11, the quotient is truncated towards zero (fractional part is discarded).
48  // If the quotient a/b is representable in the result type, (a/b)*b + a%b == a. If the second operand is zero, the behavior is undefined.
49  const bool negative_result = is_negative( x );
50 
51  // If result is positive we can treat both operands as unsigned
52  if ( !negative_result )
53  return modulus_impl_helper<Result, up_t1, up_t2>()( sm_detail::abs( x ), sm_detail::abs( y ) );
54 
55  // In case of negative result we still can piggyback on unsigned version and negate result in the end
56  auto maybe_safe_result = modulus_impl_helper<up_result, up_t1, up_t2>()( sm_detail::abs( x ), sm_detail::abs( y ) );
57  if ( maybe_safe_result )
58  {
59  // If the result type is unsigned and the result is negative, the only valid case is when the result is 0
60  if ( is_unsigned<Result>::value && negative_result )
61  {
62  if ( *maybe_safe_result == 0 )
63  return 0;
64  return {};
65  }
66  else if ( !sm_detail::high_bit_set( *maybe_safe_result ) )
67  return static_cast<Result>( 0 - *maybe_safe_result );
68  // Corner case: the only way to not overflow if high bit set
69  // is if result is negative power of 2
70  else if ( static_cast<up_result>( *maybe_safe_result << 1 ) == 0 )
71  return std::numeric_limits<Result>::min();
72  }
73 
74  return {};
75  }
76 };
