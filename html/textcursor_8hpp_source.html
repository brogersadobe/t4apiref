67  real_type CalcBaselineDist(const point_type &p) const
68  {
69  return get_x(m_baselineDir)*(get_x(p)-get_x(m_origin)) + get_y(m_baselineDir)*(get_y(p)-get_y(m_origin));
70  }

82  bool IsParallel(const TextCoordInfo& tci) const
83  {
84  return UnitVecsAreParallel(m_baselineDir, tci.m_baselineDir) &&
85  UnitVecsAreParallel(m_ascentDir, tci.m_ascentDir);
86  }
87 
88  bool m_isValid;
89 
90  static bool UnitVecsAreParallel(const point_type& a, const point_type& b)
91  {
92  return AbsVal(get_x(a)*get_y(b) - get_y(a)*get_x(b)) < real_type(0.01f); // about 1/2 degree;

101 template <typename app_traits>
102 void TextCoordInfo<app_traits>::Set(const point_type& baselinestart, const point_type& baselineend, const point_type &ascentvec, bool originIsStart)
103 {
104  m_origin = originIsStart ? baselinestart : baselineend;
105  get_x(m_baselineDir) = get_x(baselineend)-get_x(baselinestart);
106  get_y(m_baselineDir) = get_y(baselineend)-get_y(baselinestart);
107  real_type blen = real_services::VectorLength(get_x(m_baselineDir), get_y(m_baselineDir));
108  get_x(m_baselineDir) /= blen;
109  get_y(m_baselineDir) /= blen;
110  m_ascentDir = ascentvec;
111  m_ascentLen = real_services::VectorLength(get_x(m_ascentDir), get_y(m_ascentDir));
112  get_x(m_ascentDir) /= m_ascentLen;
113  get_y(m_ascentDir) /= m_ascentLen;
114  m_isValid = true;
115 }

290 template<typename app_traits>
291 bool SimpleLineCountingMarkHandler<app_traits>::HandleShow_Begin(
292  const render::GState& /*gs*/,
293  const content_point_type& /*curContentPt*/,
294  const ro_content_range_type& /*roContentRange*/,
295  const attrib_type* /*attrib*/,
296  render::ShowInfo& showInfo,
297  bool hasValidUCM,
298  bool isVisible )
299 {
300  m_pShowInfo = &showInfo;
301  m_subclassIsHandlingShow = false;
302 
303  if( !isVisible || !hasValidUCM )
304  return false;
305 
306  point_type firstGlyphStart = showInfo.GetDeviceSpaceFirstBaselinePos();
307  point_type firstGlyphEnd = showInfo.GetDeviceSpaceBaselinePos( 0, 0, 1, 1 );
308 
309  // Only attempt line counting on "visible" glyphs
310  if ( get_x( firstGlyphStart ) >= 0 )
311  {
312  if (!m_lineStartCoordInfo.m_isValid)
313  m_lineStartCoordInfo.Set(firstGlyphStart, firstGlyphEnd, GetAscentVector(showInfo), false);
314  else
315  {
316  real_type ascentDelta = AbsVal(m_lineStartCoordInfo.CalcAscentDist(firstGlyphStart));
317  if (ascentDelta > t4::min<t4::real_coord>(GetAscentVectorLen(showInfo), m_lineStartCoordInfo.GetAscentLen()))
318  {
319  m_lineNumber++;
320  m_lineStartCoordInfo.Set(firstGlyphStart, firstGlyphEnd, GetAscentVector(showInfo), false);
321  }
322  }
323  }
324 
325  m_subclassIsHandlingShow = SubclassHandleShow_Begin();
326  return m_subclassIsHandlingShow || ( m_inReflowMode && showInfo.m_hasRanges );
327 }

373  point_type glyphStart = m_pShowInfo->GetDeviceSpaceBaselinePos( curContentPt.m_stringNum, curContentPt.m_glyphNum, 0, 1 );

388  this->m_lineNumber++;
389  this->m_lineStartCoordInfo.Set(glyphStart, glyphEnd, GetAscentVector(*m_pShowInfo), false);

392  }

460 template<typename app_traits>
461 bool CursorInfoMarkHandler<app_traits>::SubclassHandleString_Begin(
462  const content_point_type& curContentPt, bool isLastInShow )
463 {
464  // we want to recurse under three conditions:
465  // 1) the result coords are still invalid, this allows us to capture the coords of the
466  // the first character on the page, even if the cursor is before it
467  // 2) this is the last string in the show, allows us to capture the last characters
468  // coords when the cursor is after it
469  // 3) our point is in this string (obvious)
470  return !m_result.m_isValid || m_cursorPoint.m_stringNum == curContentPt.m_stringNum || isLastInShow;
471 }

481 template<typename app_traits>
482 void CursorInfoMarkHandler<app_traits>::HandleUnicodeChar(
483  const content_point_type& curContentPt,
484  std::uint32_t /*c*/,
485  bool isLastInShow )
486 {
487  // see comments above
488  if (!m_result.m_isValid || m_cursorPoint.m_unicodeNum == curContentPt.m_unicodeNum || isLastInShow)
489  {
490  content::ContentPoint p = curContentPt;
491  p.m_isAfter = m_cursorPoint.m_isAfter;
492 
493  // get pos at beginning and end of glyph chunk
494  point_type start = this->m_pShowInfo->GetDeviceSpaceBaselinePos( p.m_stringNum, p.m_glyphNum, p.m_unicodeNum, p.m_totalNumUnicode );
495  point_type end = this->m_pShowInfo->GetDeviceSpaceBaselinePos( p.m_stringNum, p.m_glyphNum, p.m_unicodeNum + 1, p.m_totalNumUnicode );
496 
497  // build a TextCoordInfo coordinate system
498  m_result.Set(start, end, GetAscentVector(*this->m_pShowInfo), !m_cursorPoint.m_isAfter );
499  m_result.m_lineNumber = this->m_lineNumber;
500 
501  // we are done if we have passed the cursor point (cases 1 and 2 of recursion comments above).
502  if( StructureContentPoint_LessThanOrEqual( this->m_structure, m_cursorPoint, p ) )
503  this->m_isStopping = true;
504  }

509  content::DisplayList const& dl,

511  reflow::ReflowLayout* reflowLayout,
512  const document::PageID& pageID,
513  const content::ContentPoint& p )
