34 // Linear interpolation.
35 template<typename _app_traits>
36 typename _app_traits::real_geometry_traits::coord_type LinearInterpolation(
37  typename _app_traits::real_geometry_traits::coord_type x0,
38  typename _app_traits::real_geometry_traits::coord_type y0,
39  typename _app_traits::real_geometry_traits::coord_type x1,
40  typename _app_traits::real_geometry_traits::coord_type y1,
41  typename _app_traits::real_geometry_traits::coord_type x)
42 {
43  auto kLinearInterpolationTolerance = typename _app_traits::real_geometry_traits::coord_type(1.0e-5f);
44  auto t = (x1 - x0) < 0 ? -(x1 - x0) : (x1 - x0);
45  if ( t < kLinearInterpolationTolerance)
46  {
47  return (y0 + y1) * typename _app_traits::real_geometry_traits::coord_type{0.5};
48  }
49 
50  return ((x - x0) / (x1 - x0)) * (y1 - y0) + y0;
51 }

53 template<typename _app_traits>
54 void SetupFilterWidthAndFunctionForMediumFamily( const ImageScalingInfoQuery<_app_traits>& isiq,
55  ImageScalingInfo<_app_traits>& isi,
56  int& mnc,
57  int maxMNC)
58 {
59  typedef typename _app_traits::real_geometry_traits::coord_type real_coord;
60 
61  real_coord cubicL = 0;
62  bool strictlyPositive = false;
63 
64  if(isiq.sf == kSSharpen)
65  {
66  cubicL = Cubic_A_SS;
67  }
68  else if(isiq.sf == kSSharpenMore)
69  {
70  cubicL = Cubic_A_SM;
71  }
72  else if(isiq.sf == kSSmoothSlightly)
73  {
74  cubicL = Cubic_A_SMTS;
75  }
76  else if(isiq.sf == kSSmoothMore)
77  {
78  cubicL = Cubic_A_SMTM;
79  }
80  else if(isiq.sf == kSNoSharpening)
81  {
82  strictlyPositive = true;
83  }
84 
85  if(isiq.measureOfMatScaleFactor > 4)
86  {
87  if(strictlyPositive)
88  {
89  isi.halfFilterWidth = BSplineHalfFilterWidth;
90  isi.filter = InterpFunctions<_app_traits>::BSplineFilter;
91  }
92  else
93  {
94  // Note: ONLY Mitchel uses both param
95  isi.filterParam[0] = Mitchell_B1;
96  isi.filterParam[1] = Mitchell_C1;
97  isi.halfFilterWidth = MitchellHalfFilterWidth;
98  isi.filter = InterpFunctions<_app_traits>::MitchellFilter;
99  }
100  mnc = 4;
101 
102  }
103  else if(isiq.measureOfMatScaleFactor >= real_coord(.95))
104  {
105  if(strictlyPositive)
106  {
107  isi.halfFilterWidth = BSplineHalfFilterWidth;
108  isi.filter = InterpFunctions<_app_traits>::BSplineFilter;
109  }
110  else
111  {
112  //
113  // Find an appropriate A parm for A
114  // with a valid range -3..0 with -1, -.75, & -.5 being the ones used in books
115  // We reduce the sharpening (move toward -.5) as scale factor increases (We use values 0f -.75 to -.5 for A)
116  //
117 
118  // Note: Cubic uses only param[0]
119  cubicL += Cubic_AL_U;
120  isi.filterParam[0] = LinearInterpolation<_app_traits>(real_coord(.95), cubicL, 4, Cubic_AH_U, isiq.measureOfMatScaleFactor);
121  isi.filterParam[0] = real_services::Round(isi.filterParam[0]*10)*real_coord(0.1); // quantize it to reduce the cache contentions
122  isi.halfFilterWidth = CubicHalfFilterWidth;
123  isi.filter = InterpFunctions<_app_traits>::CubicFilterA;
124  }
125  mnc = 4;
126  }
127  else
128  {
129  typename _app_traits::real_geometry_traits::coord_type halfFilterWidth;
130  typename FilteringGlobalTypes<_app_traits>::FilterFunction filter;
131 
132  if(strictlyPositive)
133  {
134  halfFilterWidth = BSplineHalfFilterWidth;
135  filter = InterpFunctions<_app_traits>::BSplineFilter;
136  }
137  else
138  {
139  halfFilterWidth = CubicHalfFilterWidth;
140  filter = InterpFunctions<_app_traits>::CubicFilterA;
141  }
142 
144  // Minificaton case
145  //
146  // Find an appropriate A parm for A
147  // with a valid range -3..0 with -1, -.75, & -.5 being the ones used in books
148  // We increase the sharpening (move toward -2) as minificaton factor increases (We use values 0f -2 to -1 for A)
149  //
151 
152  typename _app_traits::real_geometry_traits::coord_type s;
153 
154  if(isiq.origMeasureOfMatScaleFactor==DEFAULT_ORIG_MEASURE_OF_MAT_SCALE_FACTOR)
155  s = isiq.measureOfMatScaleFactor;
156  else
157  s = isiq.origMeasureOfMatScaleFactor;
158 
159  if(s < real_coord(1)/32)
160  s = real_coord(1)/32;
161 
162  // Note: Cubic uses only param[0]
163  cubicL += Cubic_AL_D;
164  isi.filterParam[0] = LinearInterpolation<_app_traits>(real_coord(1)/32, cubicL, real_coord(.67), Cubic_AH_D, s);
165  isi.filterParam[0] = real_services::Round(isi.filterParam[0]*10)*real_coord(0.1); // quantize it to reduce the cache contentions
166 
167  if(isiq.measureOfMatScaleFactor >= real_coord(.67))
168  {
169  isi.halfFilterWidth = halfFilterWidth;
170  isi.filter = filter;
171  mnc = 6;
172  }
173  else if(isiq.measureOfMatScaleFactor >= real_coord(.45))
174  {
175  isi.halfFilterWidth = halfFilterWidth;
176  isi.filter = filter;
177  mnc = 8;
178  }
179  else
180  {
181  isi.halfFilterWidth = real_coord(BoxHalfFilterWidth);
182  isi.filter = InterpFunctions<_app_traits>::BoxFilter;
183  mnc = maxMNC;
184  }
185  }

268 template<typename _app_traits>
269 void GetScalingInfoS(
270  const ImageScalingInfoQuery<_app_traits>& isiq,
271  ImageScalingInfo<_app_traits>& isi,
272  bool hasVector/*=false*/)
273 {
274  typedef typename _app_traits::int_geometry_traits::coord_type int_coord;
275  typedef typename _app_traits::real_geometry_traits::coord_type real_coord;
276 
277  isi.hasVector = hasVector;
278 
279  int mnc = MAX_SAMPLING_AREA;
280 
281  switch(isiq.iq)
282  {
283  case kIQExtremelyLow:
284  case kIQVeryLow:
285  isi.halfFilterWidth = real_coord(BoxHalfFilterWidth);
286  isi.filter = InterpFunctions<_app_traits>::BoxFilter;
287  break;
288 
289  default:
290  case kIQLow:
291  SetupFilterWidthAndFunctionForLow(isiq, isi, mnc, 16);
292  break;
293 
294  case kIQMedium:
295  SetupFilterWidthAndFunctionForMediumFamily(isiq, isi, mnc, 16);
296  break;
297 
298  case kIQMediumHigh:
299  SetupFilterWidthAndFunctionForMediumFamily(isiq, isi, mnc, 20);
300  break;
301 
302  case kIQMediumVeryHigh:
303  SetupFilterWidthAndFunctionForMediumFamily(isiq, isi, mnc, 24);
304  break;
305 
306  case kIQMediumExtremelyHigh:
307  SetupFilterWidthAndFunctionForMediumFamily(isiq, isi, mnc, 32);
308  break;
309 
310  case kIQHigh:
311  case kIQVeryHigh:
312  case kIQExtremelyHigh:
313  SetupFilterWidthAndFunctionForHighFamily(isiq, isi);
314  break;
315  }
316 
317  // compute some filter calculation variables
318  isi.filterWidth = isi.halfFilterWidth*2;
319 
320  if(isiq.measureOfMatScaleFactor < 1)
321  {
322  if(isi.filterWidth>mnc)
323  {
324  // guard against too small of maxNumContributorsAdjusted
325  mnc = real_services::FloorToInt(ArgType<int>(), isi.filterWidth);
326  }
327 
328  typename _app_traits::real_geometry_traits::coord_type halfFilterWidthNeeded;
329 
330  halfFilterWidthNeeded = isi.halfFilterWidth/isiq.measureOfMatScaleFactor;
331  isi.scaledHalfFilterWidth = mnc*real_coord{.5};
332 
333  if(isi.scaledHalfFilterWidth > halfFilterWidthNeeded)
334  {
335  // don't oversample when you don't need to
336  isi.scaledHalfFilterWidth = halfFilterWidthNeeded;
337  }
338 
339  if(isi.scaledHalfFilterWidth >= MAX_SAMPLING_AREA_HALF)
340  {
341  isi.scaledHalfFilterWidthAdjusted = MAX_SAMPLING_AREA_HALF;
342  mnc = MAX_SAMPLING_AREA;
343  }
344  else
345  {
346  isi.scaledHalfFilterWidthAdjusted = isi.scaledHalfFilterWidth;
347  }
348  }
349  else
350  {
351  isi.scaledHalfFilterWidth = isi.halfFilterWidth;
352  isi.scaledHalfFilterWidthAdjusted = isi.halfFilterWidth;
353  }
354 
355 
356  isi.scaledFilterWidthAdjusted = isi.scaledHalfFilterWidthAdjusted*2;
357 
358  tetraphilia_assert(isi.scaledFilterWidthAdjusted<=MAX_SAMPLING_AREA);
359  tetraphilia_assert(isi.scaledHalfFilterWidthAdjusted<=MAX_SAMPLING_AREA_HALF);
360 
361  isi.left = real_services::FloorToInt(ArgType<int_coord>(), real_coord{.5}+-isi.scaledHalfFilterWidthAdjusted); // leftmost pixel for source range
362  isi.right = real_services::FloorToInt(ArgType<int_coord>(), real_coord{.5}+isi.scaledHalfFilterWidthAdjusted)-1; // rightmost pixel for source range
363 
364  tetraphilia_assert(isi.left<=isi.right);
365 
366  // allocated mem for computational cache
367  isi.tapCount = static_cast<unsigned int>( isi.right - isi.left + 1 );
368 
369  // the actual isi.filterWidth being used and the scale factor must correspond.
370  // since we might clip the theoretical isi.filterWidth, we need to fabricate
371  // a scale factor based on the actual clipped isi.filterWidth.
372  isi.scaleAdjusted = isi.filterWidth/static_cast<int_coord>(isi.tapCount);
373 
374 }

376 template<typename _app_traits>
377 void GetScalingInfoE(
378  const ImageScalingInfoQuery<_app_traits>& /*isiq*/,
379  ImageScalingInfo<_app_traits>& isi)
380 {
381  tetraphilia_assert((isi.tapCount)<=MAX_SAMPLING_AREA);
382 
383  for(size_t i=0; i<isi.weightTableInfoCount; i++)
384  {
385  if(isi.hasVector && isi.filter!=InterpFunctions<_app_traits>::BoxFilter)
386  {
387  isi.wti[i].tapCountAllocated = ((isi.tapCount + isi.wti[i].vectorElemAlignment-1)/isi.wti[i].vectorElemAlignment)*isi.wti[i].vectorElemAlignment; // round it up to the next multiple of isi.wti[i].vectorElemAlignment
388  tetraphilia_assert((isi.wti[i].tapCountAllocated%isi.wti[i].vectorElemAlignment)==0);
389  }
390  else
391  {
392  isi.wti[i].tapCountAllocated = isi.tapCount;
393  }
394  }
395 
396  t4::int_coord top = isi.right;
397  t4::int_coord bot = isi.left;
398  t4::int_coord left = isi.right;
399  t4::int_coord right = isi.left;
400  bool willAccessOutsideFilterSupport = false;
401 
402  if(isi.hasVector)
403  {
404  if(isi.tapCount != isi.wti[0].tapCountAllocated)
405  {
406  if( isi.filter == InterpFunctions<_app_traits>::BoxFilter )
407  {
408  tetraphilia_assert((isi.wti[0].tapCountAllocated%2)==0); // tapCountAllocated is multiple of 4
409 
410  // box filter just avg things out, there are no 0 coeff involved
411  // hence, we have to clip to the precise rectangle
412  bot = -static_cast<t4::int_coord>(isi.wti[0].tapCountAllocated/2);
413  top = static_cast<t4::int_coord>(isi.wti[0].tapCountAllocated/2)-1;
414  right = -static_cast<t4::int_coord>(isi.wti[0].tapCountAllocated/2);
415  left = static_cast<t4::int_coord>(isi.wti[0].tapCountAllocated/2)-1;
416  }
417  else
418  {
419  if(isi.scaledHalfFilterWidthAdjusted < (MIN_SAMPLING_AREA_CONV_NATIVE/2))
420  {
421  // 1<= filter width <=3
422  bot = -(MIN_SAMPLING_AREA_CONV_NATIVE/2+1); // leftmost pixel for source range (-(4/2+1) = -3)
423  willAccessOutsideFilterSupport = true;
424  }
425  else if(isi.scaledHalfFilterWidthAdjusted > 2 && isi.scaledHalfFilterWidthAdjusted < (MAX_SAMPLING_AREA_CONV_NATIVE/2))
426  {
427  // 5<= filter width <=7
428  bot = -(MAX_SAMPLING_AREA_CONV_NATIVE/2+1); // leftmost pixel for source range (-(8/2+1) = -5)
429  willAccessOutsideFilterSupport = true;
430  }
431  }
432  }
433  }
434 
435  isi.fringe.l = -left;
436  isi.fringe.t = -top;
437 
438  typename _app_traits::int_geometry_traits::coord_type max_bin = 0;
439 
440  if(isi.filter != InterpFunctions<_app_traits>::BoxFilter)
441  {
442  max_bin = MAX_BIN;
443  }
444 
445  isi.fringe.r = max_bin-right;
446  isi.fringe.b = max_bin-bot;
447 
448  if(isi.filter != InterpFunctions<_app_traits>::BoxFilter && !willAccessOutsideFilterSupport)
449  {
450  if(!((isi.fringe.b-isi.fringe.t)&1))
451  { // does the vertical filter has an even width?
452  isi.fringe.t--;
453  isi.vfAdjusted = true;
454  }
455 
456  if(!((isi.fringe.r-isi.fringe.l)&1))
457  { // does the horizontal filter has an even width?
458  isi.fringe.l--;
459  isi.hfAdjusted = true;
460  }
461  }
462 
463 
464  tetraphilia_assert(isi.vfAdjusted==isi.hfAdjusted);
465  tetraphilia_assert(isi.fringe.l <=0);
466  tetraphilia_assert(isi.fringe.t <=0);
467  tetraphilia_assert(isi.fringe.r >=0);
468  tetraphilia_assert(isi.fringe.b >=0);
469 
470  if(isi.filter != InterpFunctions<_app_traits>::BoxFilter)
471  {
472  tetraphilia_assert(isi.fringe.l <= isi.fringe.r);
473  tetraphilia_assert(isi.fringe.t <= isi.fringe.b);
474  tetraphilia_assert((isi.fringe.r - isi.fringe.l)>0);
475  tetraphilia_assert((isi.fringe.b - isi.fringe.t)>0);
476  }
477 
478 }
