38 /*
39  Compile-time tests which assess whether or not a callable object of type F
40  "conforms" to the function signature ReturnType(Params...).
41 
42  All that is tested here is whether or not, given an instance of type F (which may be a reference
43  type) is 1) invocable with a parameter list matching Params..., and 2) returns a type that is
44  convertible to ReturnType.
45 
46  If ReturnType is void, F's return type may be anything.
47 
48  invoke_conforms is a function, and invoke_conforms_v is a bool-valued template variable; they do the same
49  thing:
50 
51  template<typename F>
52  int accumulate( stuff_array const& s, int initial_value, F&& f )
53  {
54  // These will write the same values to stdout
55  std::cout << invoke_conforms<int(stuff const&, int)>( std::forward<F>( f ) ) << "\n";
56  std::cout << invoke_conforms_v<int(stuff const&, int), F> << "\n";
57  return std::accumulate( s.begin(), s.end(), initial_value, std::forward<F>( f ) );
58  }
59 
60  invoke_conforms has a more straightforward syntax but is sometimes hard to use in compile-time contexts
61  (like static_assert)
62 
63  invoke_conforms_v/invoke_conforms does _not_ test whether or not any compiler warnings would be generated by
64  the invocation described above. If it's allowed by the language, that's both necessary and sufficient.
65 
66  Examples:
67 
68  class thingum_container
69  {
70  std::vector<thingum> m_thingums;
71 
72  template<typename F>
73  void for_each_thingum( F&& f ) const
74  {
75  // If the provided callable doesn't conform to void(thingum const&) (i.e., if one cannot invoke it passing a thingum const&),
76  // the compile fails here, rather than deep inside the bowels of some STL algorithm.
77  static_assert( invoke_conforms_v<void(thingum const&)>, "provided functor must conform to void(thingum const&)" );
78  for ( auto const& e : m_thingums )
79  f( e );
80  }
81  };
82 
83  extern void foo(thingum const& x);
84  extern int bar(thingum const& x);
85  extern void grue(thingum const& x, int y = 23);
86  auto lambda = [](auto const& x){ std::cout << x; };
87  extern void baz(thingum& x);
88  extern int blah(int x, char const* y, std::array<thingum, 5> const& r);
89 
90  int main(void)
91  {
92  auto tc = make_thingum_container();
93 
94  tc.for_each_thingum(foo); // compiles
95  tc.for_each_thingum(bar); // compiles, return value is silently dropped on the floor
96  tc.for_each_thingum(grue); // compiles, default value is used for y
97  tc.for_each_thingum(lambda); // compiles, since x is generic
98 
99  tc.for_each_thingum(baz); // does not compile; baz expects a non-const reference but for_each_thingum expects a callable taking a const reference
100  tc.for_each_thingum(blah); // does not compile; there's just nothing at all right about this
101 
102  return 0;
103  }
104 */

106 constexpr bool invoke_conforms_v = invoke_conforms_impl<Signature, F>::value;
