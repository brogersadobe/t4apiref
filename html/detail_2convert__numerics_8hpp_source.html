151 template<typename To, typename From>
152 struct too_high_impl<To, From, typename std::enable_if_t<is_unsigned<To>::value && is_signed<From>::value>>
153 {
154  constexpr BOOST_FORCEINLINE bool operator()( From const& x ) const
155  {
156  #ifdef __clang__
157  #pragma clang diagnostic push
158  #pragma clang diagnostic ignored "-Wsign-compare"
159  #endif
160  /*
161  If x is nonpositive, then it can't possibly be too _high_ to convert to the target type (it might still
162  be too low, but that's too_low_impl's job).
163 
164  If x is positive, then we lean on C++'s "usual arithmetic conversions" ([expr]) to give us a correct answer.
165  Having established that x is positive we don't have to worry about the counterintuitive results of comparing
166  a negative signed value to a positive unsigned value. But we do need to quiet a warning.
167  */
168  return x > 0 && x > std::numeric_limits<To>::max();
169  #ifdef __clang__
170  #pragma clang diagnostic pop
171  #endif
172  }
173 };

329 template<typename To, typename From>
330 struct convert_impl<To, From, std::enable_if_t<std::is_floating_point<To>::value && std::is_floating_point<From>::value>>
331 {
332  constexpr BOOST_FORCEINLINE boost::optional<To> operator()( From const& x ) const
333  {
334  #ifdef __clang__
335  #pragma clang diagnostic push
336  #pragma clang diagnostic ignored "-Wdouble-promotion"
337  #endif
338  /*
339  The conversion succeeds if x falls between the lowest/max range (note that it is impossible
340  for inf or NaN to pass both these tests), OR if x is already NaN or infinity.
341 
342  There are many representations of NaN and you can't compare-equal for them. We could use
343  fpclassify(), except that that function isn't constexpr. cppreference.com provides this
344  jewel of intuition:
345 
346  "Another way to test if a floating-point value is NaN is to compare it with itself"
347  */
348  if ( ( std::numeric_limits<To>::lowest() <= x && x <= std::numeric_limits<To>::max() )
349  || x == std::numeric_limits<From>::infinity()
350  || x == -std::numeric_limits<From>::infinity()
351  || x != x ) // from cppreference.com: "Another way to test if a floating-point value is NaN is to compare it with itself"
352  {
353  return static_cast<To>( x );
354  }
355  #ifdef __clang__
356  #pragma clang diagnostic pop
357  #endif
358 
359  return {};
360  }
361 
362  [[noreturn]] void fail( void ) const
363  {
364  throw std::overflow_error("");
365  }
366 };
