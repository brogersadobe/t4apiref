92 /*
93  Utility template class providing simple synchronized access to some contained state.
94 
95  An instance of synchronize contains a mutex and an instance of T. Access to
96  this instance is only permitted while holding the mutex.
97 */
98 template<typename T, typename Mutex>
99 class synchronized
100 {
101  mutable Mutex m_mutex;
102  T m_val;
103 
104  typedef synchronized_detail::synchronized_ref<T, Mutex> ref_proxy;
105  typedef synchronized_detail::synchronized_ref<T const, Mutex> const_ref_proxy;
106  friend ref_proxy;
107  friend const_ref_proxy;
108 
109 public:
110  /*
111  Copy ctor. Creates a new mutex.
112  */
113  synchronized( synchronized const& rhs )
114  : m_val( *rhs )
115  {}
116 
117  /*
118  Move ctor
119  */
120  synchronized( synchronized&& rhs )
121  : m_val( std::move( rhs.m_val ) )
122  {}
123 
124  /*
125  Forwarding constructor. Constructs the contained value with the specified arguments.
126  */
127  template<typename... Args>
128  synchronized( Args&&... args )
129  : m_val( std::forward<Args>( args )... )
130  {}
131 
132  /*
133  Get a direct reference to the instance without any kind of locking.
134  This is dangerous! Only do this if you know what you're doing. This is
135  typically most useful during an owner object's construction, or during
136  a move.
137  */
138  T const& get_without_lock( void ) const { return m_val; }
139  T& get_without_lock( void ) { return m_val; }
140 
141  /*
142  Execute a provided functor on the value while holding the lock. The
143  functor must take a single parameter convertible from a T&.
144  */
145  template<typename functor>
146  auto while_locked( functor const& f ) const -> decltype(f(std::declval<T const&>()))
147  {
148  return boost::with_lock_guard( m_mutex, [this,&f]() -> decltype(f(std::declval<T const&>())) { return f( this->m_val ); } );
149  }
150 
151  /*
152  Mutable variant of above.
153  */
154  template<typename functor>
155  auto while_locked( functor const& f ) -> decltype(f(std::declval<T&>()))
156  {
157  return boost::with_lock_guard( m_mutex, [this,&f]() -> decltype(f(std::declval<T&>())) { return f( this->m_val ); } );
158  }
159 
160  /*
161  Arrow operator. This creates a temporary proxy object which holds
162  the current lock until the next sequence point. This is useful for
163  synchronized method invocations, e.g.:
164 
165  synchronized<Foo> x;
166  x->bar(); // bar() is called while holding the mutex
167  */
168  const_ref_proxy operator->( void ) const { return const_ref_proxy( *this ); }
169  ref_proxy operator->( void ) { return ref_proxy( *this ); }
170 
171  /*
172  Dereference operator. This creates a temporary object which may be
173  used for assignments, e.g.:
174 
175  synchronized<CGPoint> x;
176  *x = CGPointMake( 10, 20 ); // Assigns ( 10, 20 ) to x while holding the mutex.
177  */
178  const_ref_proxy operator*( void ) const { return const_ref_proxy( *this ); }
179  ref_proxy operator*( void ) { return ref_proxy( *this ); }
180 
181  /*
182  Subscript operator. Forwards operator[] to the contained object
183  while holding the lock.
184  */
185  template<typename Index>
186  auto operator[]( Index&& index ) -> decltype(m_val[index]) { return ref_proxy( *this )[ std::forward<Index>( index ) ]; }
187 
188  // simple load and store for value types.
189  T load( void ) const
190  {
191  boost::lock_guard<Mutex> lock( m_mutex );
192  return m_val;
193  }
194 
195  void store( T val )
196  {
197  boost::lock_guard<Mutex> lock( m_mutex );
198  m_val = val;
199  }
200 };
