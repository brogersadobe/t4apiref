69  client_type& client,
70  document::DocumentCaches<typename client_type::signal_traits::app_traits>& documentCaches,
71  typename client_type::signal_traits::app_traits::int_geometry_traits::coord_type greekThreshold,

95  auto& doc = documentCaches.GetDocument();

97  t4::pdf::draw_page( group, doc, get_page_dictionary( doc, pageID ), basePageTransform, transform, bounds, params );

269 template<typename signal_traits>
270 typename imaging_model::RasterSourceType<signal_traits>::type MakeRasterSourceFromDestination(
271  typename signal_traits::app_traits::context_type& context,
272  const imaging_model::GraphicLayoutDescriptor& desc,
273  const imaging_model::TransparencyTuple<const imaging_model::PixelBuffer<signal_traits>*>& destination )
274 {
275  typedef typename imaging_model::PixelBufferTypes<signal_traits>::const_type const_pixel_buffer_type;
276  typedef typename imaging_model::RasterSourceType<signal_traits>::type raster_source_type;
277 
278  raster_source_type result;
279  auto& heap = *t4::get_thread_heap();
280 
281  switch( desc.m_shapeRep )
282  {
283  case imaging_model::GraphicLayoutDescriptor::kShapeIgnored:
284  result.m_shape = 0;
285  break;
286  case imaging_model::GraphicLayoutDescriptor::kShapeImplicitlyOne:
287  result.m_shape = TRANSIENT_NEW( const_pixel_buffer_type, heap, ( imaging_model::IdentityPixelBuffers<signal_traits>::OnesBuffer() ) );
288  break;
289  default:
290  // If this assertion fires the client claims to care about shape but did not
291  // provide a raster destination that stores any
292  tetraphilia_assert( destination.m_shape );
293  result.m_shape = TRANSIENT_NEW( const_pixel_buffer_type, heap, ( *destination.m_shape ) );
294  break;
295  }
296 
297  switch( desc.m_opacityRep )
298  {
299  case imaging_model::GraphicLayoutDescriptor::kOpacityIgnored:
300  result.m_opacity = 0;
301  break;
302  case imaging_model::GraphicLayoutDescriptor::kOpacityImplicitlyOne:
303  result.m_opacity = TRANSIENT_NEW( const_pixel_buffer_type, heap, ( imaging_model::IdentityPixelBuffers<signal_traits>::OnesBuffer() ) );
304  break;
305  case imaging_model::GraphicLayoutDescriptor::kOpacitySameAsShape:
306  result.m_opacity = result.m_shape;
307  break;
308  default:
309  // If this assertion fires the client claims to care about opacity but did not
310  // provide a raster destination that stores any
311  tetraphilia_assert( destination.m_opacity );
312  result.m_opacity = TRANSIENT_NEW( const_pixel_buffer_type, heap, ( *destination.m_opacity ) );
313  break;
314  }
315 
316  switch( desc.m_colorRep )
317  {
318  case imaging_model::GraphicLayoutDescriptor::kColorIgnored:
319  result.m_color = 0;
320  break;
321  default:
322  // If this assertion fires the client claims to care about color but did not
323  // provide a raster destination that stores any
324  tetraphilia_assert( destination.m_color );
325  result.m_color = TRANSIENT_NEW( const_pixel_buffer_type, heap, ( *destination.m_color ) );
326  break;
327  }
328 
329  return result;
330 }
