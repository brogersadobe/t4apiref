127 static inline t4::real_point RotateAboutOrigin( t4::real_point pt, t4::real rotation_radians )
128 {
129  return t4::make<t4::real_point>( get_x( pt ) * std::cos( rotation_radians ) - get_y( pt ) * std::sin( rotation_radians ),
130  get_x( pt ) * std::sin( rotation_radians ) + get_y( pt ) * std::cos( rotation_radians ) );
131 }

164 static inline bool StartsNewLine( const ExtractedGlyph& previousGlyph, const ExtractedGlyph& currentGlyph, t4::real rotation_radians )
165 {
166  using namespace t4::literals;
167 
168  // Get the leading edges
169  auto old_leading = previousGlyph.GetLeadingEdge();
170  auto new_leading = currentGlyph.GetLeadingEdge();
171 
172  // Create a vector from the top of old_leading to the top of new_leading, and then transform this vector,
173  // first by translating it so that it starts at the origin, and then rotate it in the
174  // opposite direction of the given, original rotation
175  t4::real_point v_normalized = RotateAboutOrigin( new_leading.first - old_leading.first, -1 * rotation_radians );
176 
177  // Transform the old_leading edge as well, so we can compare the y-components accurately.
178  t4::real_point old_leading_normalized = RotateAboutOrigin( old_leading.first - old_leading.second, -1 * rotation_radians );
179 
180  // Compare the y components of these two vectors
181  auto baseline_shift = std::fabs( get_y( v_normalized ) );
182  auto old_glyph_height = std::fabs( get_y( old_leading_normalized ) );
183 
184  return ( baseline_shift > 0.75_r * old_glyph_height );
185 }

313 template<typename copy_client_type>
314 void TextCopyMarkHandler<copy_client_type>::HandleUnicodeChar(
315  const content_enumeration_state& state,
316  std::uint32_t c,
317  bool isLastInShow )
318 {
319  auto const& curContentPt = state.get_content_point();
320  m_geo.mContentPoint = &curContentPt;
321 
322  if ( !m_unicodeCategorizer.IsControl( c ) )
323  {
324  if ( m_unicodeCategorizer.IsHyphenLike( c ) )
325  {
326  // If current character is a hyphen ..
327  if ( !m_hasDelayedHyphen && isLastInShow )
328  {
329  // .. which also happens to be last character in current text show and previous
330  // character wasn't a hyphen, then there is a chance that this could be a word-breaking
331  // hypehn at line end which we don't wish to expose to client.
332  m_hasDelayedHyphen = true;
333  m_delayedGlyph = ExtractedGlyph{m_geo};
334  }
335  else
336  {
337  // Previously marked hyphen didn't turn out to be a line-ending hyphen, so it
338  // shouldn't be disposed off and has to be reported to client
339  if( m_hasDelayedHyphen )
340  {
341  m_copyClient.HandleUnicodeChar( m_lastUnicodeChar, m_delayedGlyphFunc );
342  }
343 
344  // Now, report current character
345  m_copyClient.HandleUnicodeChar( c, m_currentGlyphFunc );
346  m_hasDelayedHyphen = false;
347  m_delayedGlyph = boost::none;
348  }
349  }
350  else
351  {
352  bool beginsNewLine = m_beginsUnreflowedLine;
353 
354  // Tagged structure does not detect whether a new line should be inserted,
355  // so we should
356  if ( m_previousGlyph )
357  {
358  // Compare current glyph baseline with previous
359  auto rotation = m_geo.mShowInfo->GetBaselineRotationInRadians();
360  beginsNewLine |= StartsNewLine( *m_previousGlyph, m_currentGlyphFunc(), rotation );
361  }
362 
363  // If current character is not a hyphen ..
364  if ( !beginsNewLine )
365  {
366  // and it doesn't start a new line ..
367  if ( m_hasDelayedHyphen )
368  {
369  // and if previous character was a hyphen, then it wasn't disposable either.
370  // So, report it to client.
371  m_copyClient.HandleUnicodeChar( m_lastUnicodeChar, m_delayedGlyphFunc );
372  }
373  }
374  else if ( !m_hasDelayedHyphen &&
375  m_unicodeCategorizer.IsVisible( m_lastUnicodeChar ) &&
376  !m_unicodeCategorizer.IsCJK( m_lastUnicodeChar ) &&
377  !m_unicodeCategorizer.IsCJK( c ) )
378  {
379  // and it does start a new line and previous character on previous line wasn't a hyphen,
380  // then we need to insert a space character else the word ending a sentence will not be
381  // separated from the word beginning on next sentence
382  m_copyClient.HandleUnicodeChar( 32 /*space*/ , m_currentGlyphFunc );
383  }
384 
385  // Update extracted geometry to reflect new line status when handling the current unicode character,
386  // resets to false at the end of this function.
387  m_geo.mStartsNewLine = beginsNewLine;
388 
389  // report the current character
390  m_copyClient.HandleUnicodeChar( c, m_currentGlyphFunc );
391 
392  content::ContentPoint possibleInferredSpace( curContentPt );
393  possibleInferredSpace.AdvanceToAfter();
394 
395  // .. but also check if there is an inferred space between current and next character
396  // If yes, report it to client.
397 
398  // if we are not at the last point in the selection
399  if ( m_structure && possibleInferredSpace != m_selection.m_to )
400  {
401  possibleInferredSpace.AdvanceUnicodeNum(); // if there exists an inferred space after current point, then this will be its content point
402 
403  if ( ( m_cachedStructureNode == nullptr ) ||
404  ( m_cachedStructureNode->GetRange() == nullptr ) ||
405  ( !m_cachedStructureNode->GetRange()->m_contentRange.Contains( curContentPt ) ) )
406  {
407  m_cachedStructureNode = m_structure->GetRoot()->SearchContentPointInSubtree( curContentPt );
408  }
409 
410  // check if there exists an inferred space after the current content point
411  tetraphilia_assert( m_cachedStructureNode );
412  if ( m_cachedStructureNode )
413  {
414  auto const& infSpaces = m_cachedStructureNode->GetInferredSpaces();
415  if ( std::find( infSpaces.begin(), infSpaces.end(), possibleInferredSpace ) != infSpaces.end() )
416  {
417  // we found an inferred space. add it to the stream being copied.
418  c = 32;
419 
420  // assume inferred space to sit right next to the current glyph with same glyph width
421  t4::function<ExtractedGlyph()> inferredSpaceGlyphFunc = [this]() {
422  auto currentGlyph = m_currentGlyphFunc();
423  auto glyphWidth = currentGlyph.GetTrailingEdge().first - currentGlyph.GetLeadingEdge().first;
424 
425  auto newLeading = currentGlyph.GetTrailingEdge();
426  auto newTrailing = std::make_pair( newLeading.first + glyphWidth, newLeading.second + glyphWidth );
427 
428  return ExtractedGlyph{*m_geo.mContentPoint, newLeading, newTrailing, false, m_geo.mStartsNewLine};
429  };
430 
431  m_copyClient.HandleUnicodeChar( c, inferredSpaceGlyphFunc );
432  }
433  }
434  }
435 
436  m_hasDelayedHyphen = false;
437  m_delayedGlyph = boost::none;
438  }
439  m_previousGlyph = m_currentGlyphFunc();
440  m_lastUnicodeChar = c;
441  }
442 
443  m_geo.mStartsNewLine = false;
444  m_geo.mStartsNewRange = isLastInShow;
445 }

477 template <typename copy_text_client_type>
478 void CopyContentRange(
479  copy_text_client_type& copyClient,
480  TextExtractContext& teContext,
481  const content::ContentRange& range,
482  t4::pdf::content_order order )
483 {
484  auto& dc = teContext.GetDocumentCaches();
485  auto& store = dc.GetStore();
486  auto& doc = dc.GetDocument();
487  auto& heap = *t4::get_thread_heap();
488 
489  auto firstPage = t4::pdf::get_page_num( doc, range.m_from.m_pageID );
490  auto lastPage = t4::pdf::get_page_num( doc, range.m_to.m_pageID );
491 
492  for( auto page = firstPage; page <= lastPage; ++page )
493  {
494  auto pageID = t4::pdf::get_page_id( doc, page );
495  t4::transient_snapshot ss( heap );
496  store.PrefetchPage(page);
497  auto dlHolder = document::GetPageDisplayList( doc, t4::pdf::get_page_dictionary( doc, pageID ) );
498 
499  auto s = [&]() -> t4::sptr<textextract::Structure>
500  {
501  using t4::pdf::content_order;
502  switch( order )
503  {
504  case content_order::z:
505  return {};
506  case content_order::inferred:
507  return t4::pdf::get_structure( doc, pageID, order );
508  case content_order::tagged:
509  auto result = t4::pdf::get_structure( doc, pageID, order );
510  if ( !result->IsValid() )
511  result.reset();
512  return result;
513  }
514  }();
515 
516  TextCopyMarkHandler<copy_text_client_type> markHandler(
517  teContext.GetUnicodeCategorizer(), copyClient, s.get(), order, range );
518 
519  namespace tparams = t4::pdf::text_extract_params;
520  auto params = t4::pdf::text_extract_parameters_desc();
521  params += t4::parameter<tparams::unicode_normalizer>( teContext.GetUnicodeNormalizer() );
522  params += t4::parameter<tparams::unicode_categorizer>( teContext.GetUnicodeCategorizer() );
523  t4::pdf::enumerate_reordered_content( doc, range, order, markHandler, params );
524  }
525 
526  copyClient.HandleDoneCopying();
527 }
