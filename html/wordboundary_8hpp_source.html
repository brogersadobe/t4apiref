185  content_point_type getStartWordBoundary()
186  {
187  //If we didn't even reach the selection beginning content point while enumerating the structure, we can't rely on the word
188  //boundary detected. So it's best to fallback to character-by-character selection in such cases. One known case when this
189  //happens is when the structure gets built only partially, due to bailing out on "IsTooComplexSoFar()" call.
190  if( t4::lt( m_lastSeenContentPoint, m_selection.m_from, m_comparator ) )
191  return m_selection.m_from;
192 
193  return m_startWordBoundary;
194  }

196  void setStartWordBoundary(const content_point_type& contentPt, bool isCJK)
197  {
198  m_startWordBoundary = contentPt;
199  m_unicodeCharDistanceFromLastWordStart = 0;
200  m_CJKCharEncounteredSinceLastWordStart = isCJK;
201  }

204  {
205  // TODO_READER_MOBILE<vmarda>: For Android 16.4 release in November 2016, we have decided to snap only at selection beginning,
206  // not at selection end. That's why returning selection ending here directly.
207  return m_selection.m_to;

243 template<typename app_traits, typename content_comparator>
244 void WordBoundaryMarkHandler<app_traits, content_comparator>::HandleUnicodeChar(
245  const content_point_type& curContentPt,
246  std::uint32_t c,
247  bool isLastInShow )
248 {
249  ++m_unicodeCharDistanceFromLastWordStart;
250  bool isCJK = m_unicodeCategorizer.IsCJK( c );
251  m_CJKCharEncounteredSinceLastWordStart = m_CJKCharEncounteredSinceLastWordStart || isCJK;
252 
253  if( m_nextContentPointIsWordStart )
254  {
255  setStartWordBoundary(curContentPt, isCJK);
256  m_nextContentPointIsWordStart = false;
257  }
258 
259  if ( t4::gt( curContentPt, m_selection.m_from, m_comparator ) )
260  {
261  m_lastSeenContentPoint = curContentPt;
262  this->m_isStopping = true;
263  return; // we are within the selection range
264  }
265 
266  auto curContentPtAfter = curContentPt;
267  curContentPtAfter.AdvanceToAfter();
268 
269  bool isUnicodeWordBreak = false, isInferredWordBreak = false, isLineBeginning = false, isLineEnding = false;
270  bool isCharInvisible = !m_unicodeCategorizer.IsVisible(c);
271 
272  if( m_unicodeCategorizer.IsPunctuation(c) || isCharInvisible )
273  {
274  isUnicodeWordBreak = true;
275  }
276  else
277  {
278  content::ContentPoint possibleInferredSpace( curContentPt );
279  possibleInferredSpace.AdvanceToAfter();
280  possibleInferredSpace.AdvanceUnicodeNum(); // if there exists an inferred space after current point, then this will be its content point
281 
282  if( ( m_cachedStructureNode == nullptr ) ||
283  ( m_cachedStructureNode->GetRange() == nullptr ) ||
284  ( !m_cachedStructureNode->GetRange()->m_contentRange.Contains( curContentPt ) ) )
285  {
286  m_cachedStructureNode = m_structure.GetRoot()->SearchContentPointInSubtree( curContentPt );
287 
288 #ifdef _DEBUG //Logging in debug mode
289  const t4::vector<content_point_type>& infSpaces = m_cachedStructureNode->GetInferredSpaces();
290  auto& contentRange = m_cachedStructureNode->GetRange()->m_contentRange;
291  auto linesExtremities = m_cachedStructureNode->GetLinesExtremities();
292 
293  auto size = infSpaces.size();
294  T3_LOG_FLOW("curContentPoint = (%i, %zu, %zu, %zu)\n", t4::pdf::get_page_num(m_document, curContentPt.m_pageID), curContentPt.m_dlEntryNum, curContentPt.m_stringNum, curContentPt.m_glyphNum);
295  T3_LOG_FLOW("contentRange = (%i, %zu, %zu, %zu - %i, %zu, %zu, %zu)\n", t4::pdf::get_page_num(m_document, contentRange.m_from.m_pageID), contentRange.m_from.m_dlEntryNum, contentRange.m_from.m_stringNum, contentRange.m_from.m_glyphNum, t4::pdf::get_page_num(m_document, contentRange.m_to.m_pageID), contentRange.m_to.m_dlEntryNum, contentRange.m_to.m_stringNum, contentRange.m_to.m_glyphNum);
296  T3_LOG_FLOW("infSpaces size = %zu\n\n", size);
297  for (size_t i = 0; i < size; ++i)
298  {
299  T3_LOG_FLOW("infSpaces[%zu] = (%i, %zu, %zu, %zu)\n", i, t4::pdf::get_page_num(m_document, infSpaces[i].m_pageID), infSpaces[i].m_dlEntryNum, infSpaces[i].m_stringNum, infSpaces[i].m_glyphNum);
300  }
301 
302  size = linesExtremities.size();
303  T3_LOG_FLOW("linesExtremities size = %zu\n\n", size);
304 
305  for (size_t i = 0; i < size; ++i)
306  {
307  T3_LOG_FLOW("linesExtremities[%zu] = (%i, %zu, %zu, %zu) - (%i, %zu, %zu, %zu)\n", i, t4::pdf::get_page_num(m_document, linesExtremities[i].m_from.m_pageID), linesExtremities[i].m_from.m_dlEntryNum, linesExtremities[i].m_from.m_stringNum, linesExtremities[i].m_from.m_glyphNum, t4::pdf::get_page_num(m_document, linesExtremities[i].m_to.m_pageID), linesExtremities[i].m_to.m_dlEntryNum, linesExtremities[i].m_to.m_stringNum, linesExtremities[i].m_to.m_glyphNum);
308  }
309 #endif
310 
311  }
312 
313  // check if there exists an inferred space after the current content point
314  tetraphilia_assert( m_cachedStructureNode );
315  if( m_cachedStructureNode )
316  {
317  const auto& infSpaces = m_cachedStructureNode->GetInferredSpaces();
318  if( std::find( infSpaces.begin(), infSpaces.end(), possibleInferredSpace ) != infSpaces.end() )
319  {
320  // we found an inferred space.
321  isInferredWordBreak = true;
322  }
323 
324  const auto& contentRange = m_cachedStructureNode->GetRange()->m_contentRange;
325  auto linesExtremities = m_cachedStructureNode->GetLinesExtremities();
326  for( auto itr = linesExtremities.begin(); itr != linesExtremities.end(); ++itr )
327  {
328  if( t4::lt( itr->m_from, contentRange.m_from, m_comparator ) )
329  itr->m_from = contentRange.m_from;
330 
331  if( t4::gt( itr->m_to, contentRange.m_to, m_comparator ) )
332  itr->m_to = contentRange.m_to;
333 
334  if( itr->m_from == curContentPt )
335  {
336  isLineBeginning = true;
337  break;
338  }
339  else if( itr->m_to == curContentPt )
340  {
341  isLineEnding = true;
342  break;
343  }
344  }
345  }
346  }
347 
348  if( t4::lt( curContentPt, m_selection.m_from, m_comparator ) )
349  {
350  if( isUnicodeWordBreak || isInferredWordBreak || isLineEnding )
351  m_nextContentPointIsWordStart = true;
352  else if( isLineBeginning )
353  setStartWordBoundary( curContentPt, isCJK );
354  }
355  else if( curContentPt == m_selection.m_from )
356  {
357  if( isUnicodeWordBreak || isLineBeginning || isLineEnding || m_unicodeCharDistanceFromLastWordStart > m_wordStartSnappingThreshold || m_CJKCharEncounteredSinceLastWordStart )
358  setStartWordBoundary( curContentPt, isCJK );
359  }
360 
361  m_lastSeenContentPoint = curContentPt;
362 }

364 template <typename app_traits>
365 void GetWordBoundaries( document::DocumentCaches<app_traits>& docCaches,
366  TextExtractContext& teContext,
367  const content::ContentRange& range,
368  content::ContentPoint& startWordBoundary,
369  content::ContentPoint& endWordBoundary )
370 {
371  auto& heap = *t4::get_thread_heap();
372  auto& dc = teContext.GetDocumentCaches();
373  auto& pageID = range.m_from.m_pageID;
374  auto& doc = dc.GetDocument();
375 
376  auto dlHolder = document::GetPageDisplayList( doc, t4::pdf::get_page_dictionary( doc, pageID ) );
377  auto structure = t4::pdf::get_structure( doc, pageID );
378 
379  //For word detection, we don't want to worry about page complexity. But, we don't want to cache redundant data either. Hence
380  //populating Word Detection Structure Cache only when we find that the page was reported too complex while creating Structure Cache.
381  if( structure->PageWasTooComplex() )
382  structure = t4::pdf::get_word_detection_structure( doc, pageID );
383 
384  auto comparator = StructureComparator( *structure );
385 
386  WordBoundaryMarkHandler<app_traits, decltype(comparator)> markHandler( doc, teContext.GetUnicodeCategorizer(), *structure, range, comparator );
387 
388  t4::transient_snapshot ss( heap );
389  auto structEnumerator = t4::make_transient<StructureEnumerator<app_traits>>( heap, teContext, *dlHolder, pageID, markHandler, *structure, nullptr /*reflowLayoutPtr*/ );
390  markHandler.ProcessNode( *structEnumerator, structure->GetRoot() );
391 
392  startWordBoundary = markHandler.getStartWordBoundary();
393  endWordBoundary = markHandler.getEndWordBoundary();
394 }
