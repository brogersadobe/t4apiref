26 /*
27  This is a utility class for debugging only which allows one to easily track how many instances of
28  a given class exist. Any time you're writing a system (e.g., a cache) where the number of instances
29  of objects has the potential to be of concern, you can make the class in question (say, MyClass) inherit
30  from countable<MyClass>. After that point you can call MyClass::get_instance_count() and it will return
31  the number of instances that current exist (i.e., their constructors have been called (and completed at
32  least to the construction of the countable base class) but their destructors have not (or have not yet
33  reached the destructor of the countable base class).
34 
35  Example:
36 
37  class my_thing : public debug::countable<my_thing>
38  {
39  ... member data ...
40 
41  public:
42  my_thing( int id );
43  };
44 
45  cache<int, my_thing> c;
46 
47  vector<sptr<my_thing>> ptrs_to_things;
48  for ( auto i = 0; i < 50; ++i )
49  ptrs_to_things.push_back( c.get( i, [&]( int i ){ return my_thing( i ); } ) );
50 
51  assert( my_thing::get_instance_count() == 50 ); // one for each entry in ptrs_to_things
52  purge_memory_to( 0 );
53  assert( my_thing::get_instance_count() == 50 ); // None of these could be purged, since we're holding on to them
54 
55  ptrs_to_things.erase( ptrs_to_things.begin() + 25 ); // erase the last 25 entries
56  assert( my_thing::get_instance_count() == 50 ); // There are still 50, since no purging has happened
57 
58  purge_memory_to( 0 );
59  assert( my_thing::get_instance_count() == 25 ); // There should now be 25, as we have purged 0-24
60 
61  countable<T> is threadsafe, though results reported during multithreaded execution will be very dependent on
62  timing, and thus useful only as approximations.
63 
64  Use of countable incurs some overhead, and should not be used in production code.
65 */
66 template<typename T>
67 class countable
68 {
69  debug_detail::counter<T> m_counter;
70 
71 public:
72  static size_t get_instance_count( void ) { return debug_detail::counter<T>::get(); }
73 };
