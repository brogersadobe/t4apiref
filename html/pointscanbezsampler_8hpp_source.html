45 template<typename sampler_def>
46 class BasePointscanSamplerEdge
47 {
48  typedef typename sampler_def::int_type int_type;
49  typedef typename sampler_def::real_type real_type;
50  typedef typename sampler_def::point_type point_type;
51 
52 public:
53 
54  BasePointscanSamplerEdge(const point_type& p1, const point_type& p2)
55  {
56  // point scanner DOES NOT support horizontal edges
57  tetraphilia_assert(real_services::FloorToInt(ArgType<int_type>(), get_y(p1)) != real_services::FloorToInt(ArgType<int_type>(), get_y(p2)));
58 
59  if (get_y(p1) < get_y(p2))
60  {
61  m_bot = p1;
62  m_top = p2;
63  m_delx = (get_x(m_bot) - get_x(m_top)) / (get_y(m_bot) - get_y(m_top));
64  }
65  else
66  {
67  m_bot = p2;
68  m_top = p1;
69  m_delx = (get_x(m_bot) - get_x(m_top)) / (get_y(m_bot) - get_y(m_top));
70  }
71  }
72 
73  // no horizontals - they are all discarded
74  bool isHorizontalEdge() const
75  { return false; }
76 
77  void extendHorizontalEdge(const point_type&, const point_type&)
78  {
79  // no horizontal edges
80  tetraphilia_assert_always();
81  }
82 
83  // set x values based on Y - only used on first add
84  void CalcXAtY(real_type y)
85  {
86  tetraphilia_assert(y >= get_y(m_bot));
87  m_currx = get_x(m_bot) + m_delx * (y-get_y(m_bot));
88  }
89 
90  // increment Y value
91  void IncrementXForY(Fixed16_16 y)
92  {
93  TETRAPHILIA_UNUSED(y)
94  m_currx += m_delx;
95  }
96 
97  // adding deltax is more prone to signficant error in float
98  void IncrementXForY(float y)
99  {
100  CalcXAtY(y);
101  }
102 
103  real_type getCurrX() const
104  { return m_currx; }
105 
106  real_type getRealMinX() const
107  { return get_x(m_bot); }
108  real_type getRealMinY() const
109  { return get_y(m_bot); }
110 
111  int_type getIntMinY() const
112  { return real_services::FloorToInt(ArgType<int_type>(), get_y(m_bot)); }
113 
114  real_type getRealMaxX() const
115  { return get_x(m_top); }
116  real_type getRealMaxY() const
117  { return get_y(m_top); }
118 
119  int_type getIntMaxY() const
120  { return real_services::FloorToInt(ArgType<int_type>(), get_y(m_top)); }
121 
122  const point_type& getBot() const
123  { return m_bot; }
124  const point_type& getTop() const
125  { return m_top; }
126 
127 private:
128  point_type m_bot;
129  point_type m_top;
130 
131  // x on the current scanline
132  real_type m_currx; // current minx value
133 
134  // calculated edge values - winding and slope
135  real_type m_delx; // slope
136 };

315  static Edge *ListRightmost(Edge *list, const Edge *right)
316  {
317  tetraphilia_assert(list);
318  tetraphilia_assert(list->getCurrX() <= right->getCurrX());
319  for (;;)
320  {
321  Edge * next = list->m_next;
322  if (!next || next->getCurrX() >= right->getCurrX())
323  break;
324  list = next;
325  }

329  static void ListMerge( Edge* left, Edge* right )
330  {
331  while( right )
332  {
333  // j'ai 2 listes, left et right
334  tetraphilia_assert( left );
335  tetraphilia_assert( right );
336  tetraphilia_assert( left->getCurrX() <= right->getCurrX() );
337 
338  left = ListRightmost( left, right );
339  Edge* leftNext = left->m_next;
340  left->m_next = right;
341  left = right;
342  right = leftNext;
343  }

346  // single linked list - m_prev is unused
347  void AddEdgeToSampler(Edge *pEdge)
348  {
349  int_type miny = pEdge->getIntMinY();
350 
351  // TODO: Client should be responsible for these checks
352  // clamp miny to m_rowYMin; discard edges that are out of range
353  if (miny < m_rowYMin)
354  {
355  miny = m_rowYMin;
356  // discard edges below miny
357  if (pEdge->getIntMaxY() <= m_rowYMin)
358  return;
359  }
360  else if (miny-m_rowYMin >= (int_type)m_rowHeight) // discard edges above maxy
361  return;
362 
363  AddEdgeToSampler(pEdge,miny);
364  }

366  // direct call for client to avoid unnecessary edge ymin/ymax filtering
367  void AddEdgeToSampler(Edge *pEdge, int_type miny)
368  {
369  m_edgeCount++;
370 
371  // top of the edge for topleft point scan
372  pEdge->InitializeEdgeForSampler((real_type)miny+1);
373 
374  // adjust miny to be index in m_rows
375  miny -= m_rowYMin;
376  tetraphilia_assert((size_t)miny < m_rowHeight);
377  Edge *first = m_rows[miny];
378 
379  if (!first || first->getCurrX() >= pEdge->getCurrX())
380  {
381  pEdge->m_next = first;
382  m_rows[miny] = pEdge;
383  }
384  else
385  {
386  first = ListRightmost(first,pEdge);
387  pEdge->m_next = first->m_next;
388  first->m_next = pEdge;
389  }
390  }

417 template<typename sampler_def>
418 void PointscanBezierSampler<sampler_def>::SetY( int_type y )
419 {
420  // Edge::m_prev is only transitionally correct
421 
422  while (m_nextRow <= y)
423  {
424  tetraphilia_assert( (!IntegerToRealConversion<int_type,real_type>::ConversionToRealWillOverflow(m_nextRow)) );
425  const real_type yReal = (real_type)m_nextRow; // never changes in scope of this function
426 
427  Edge *doneEdge = &m_backGroundEdge;
428  for (;;)
429  {
430  Edge *node = doneEdge->m_next;
431  if (!node)
432  break;
433 
434  m_yielder.Yield(100);
435 
436  if (node->getRealMaxY() < yReal+1) // edge ends before the top of this scanline
437  {
438  doneEdge->m_next = node->m_next;
439  }
440  else
441  {
442  // update X location on this scanline
443  node->IncrementXForY(yReal);
444  // node->CalcXAtY(yReal);
445 
446  // increasing slope
447  if (node->getCurrX() >= doneEdge->getCurrX())
448  {
449  node->m_prev = doneEdge;
450  doneEdge = node;
451  }
452  else
453  {
454  doneEdge->m_next = node->m_next;
455  // Cross the edges
456  Edge *right = doneEdge;
457  while(right)
458  {
459  Edge* left = right->m_prev;
460  if( node->getCurrX() >= left->getCurrX() )
461  {
462  left->m_next = node;
463  right->m_prev = node;
464  node->m_prev = left;
465  node->m_next = right;
466  break;
467  }
468  right = left;
469  }
470  }
471  }
472  }
473 
474  // and merge in any new edges that start at this row
475  if (m_nextRow-m_rowYMin < (int_type)m_rowHeight)
476  ListMerge(&m_backGroundEdge,m_rows[m_nextRow-m_rowYMin]);
477 
478  m_nextRow++;
479  }

502  cache_logger(TEXT("%d: %x b (%.3f,%.3f) t (%.3f,%.3f) w %d currx %.3f\n"),idx,pEdge,

517  for (Edge *pEdge = m_backGroundEdge.m_next; pEdge; pEdge = pEdge->m_next)

535  }
536  else
537  {
538  ts.push_back(edgex);
539  currmax = edgex;
540  }
541  }
542  }
