62 public:
63  ConvertedRasterPainter( const constraints_type& constraints,
64  t4::brief_ptr<raster_painter_type> src, const color_space_ptr_type& srcCS,
65  const color_space_ptr_type& dstCS )
66  : m_src( std::move( src ) ),
67  m_converter( srcCS->template CreateConverter<signal_traits>( *dstCS, color::kRelColorimetric, color::kConverterCachingTransient ) ),
68  m_tmpColorBuff( constraints.GetAlignedInterestBounds(),
69  MakePixelBufferLayoutDescriptor( dstCS->GetNumSignals() ),
70  false, *t4::get_thread_heap() ),
71  m_tmpColorBuffWalker( t4::make_brief<RasterYWalker<signal_traits>>( m_tmpColorBuff.GetPixelBuffer() ) ),
72  m_segs( t4::transient_allocator<segment_type>( get_TransientHeap<app_traits>() ) ),
73  m_bufferConverters( t4::transient_allocator<color_buffer_converter_type*>( get_TransientHeap<app_traits>() ) )
74  {
75  m_segs.reserve( m_src->GetNumSegments() );
76  m_bufferConverters.reserve( m_src->GetNumSegments() );
77 
78  const segment_type* srcSeg = SegmentsBegin( t4::get_ptr( m_src ) );
79  const segment_type* endSeg = SegmentsEnd( t4::get_ptr( m_src ) );
80  while ( srcSeg != endSeg )
81  {
82  m_bufferConverters.push_back( MakeBufferConverter( srcSeg->m_color, m_tmpColorBuffWalker ) );
83  m_segs.push_back( segment_type( srcSeg->m_shape, srcSeg->m_opacity, m_tmpColorBuffWalker ) );
84  ++srcSeg;
85  }
86 
87  this->Initialize( m_segs.size(), m_segs.data(), m_src->GetNonZeroBounds(), kSetY_Unneeded );
88 
89  #ifdef _DEBUG
90  this->SetInputs( m_src );
91  this->SetName( "Convert" );
92  #endif
93  }

102  static PixelBufferLayoutDescriptor MakePixelBufferLayoutDescriptor( size_t numSignals )
103  {
104  typedef typename app_traits::imaging_model_traits::preferred_pixel_layouts::Color preferred_color_layout;
105  size_t padSignals;
106  size_t pixelSize;
107  if ( numSignals != preferred_color_layout::num_signals )
108  {
109  padSignals = 0;
110  pixelSize = numSignals;
111  }
112  else
113  {
114  padSignals = preferred_color_layout::pad_signals;
115  pixelSize = padSignals + numSignals;
116  }
117 
118  return PixelBufferLayoutDescriptor( padSignals, numSignals, pixelSize,
119  preferred_color_layout::little_endian, preferred_color_layout::planar, false, true, false );
120  }

159 template<typename signal_traits>
160 t4::brief_ptr<RasterPainter<signal_traits>>
161 ConvertRasterPainter( t4::brief_ptr<RasterPainter<signal_traits>> source,
162  const Constraints& constraints,
163  const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& srcCS,
164  const typename color::ColorSpace<typename signal_traits::app_traits>::smart_ptr_type& dstCS )
165 {
166  if ( srcCS == dstCS || !source->GetSegments()[0].m_color )
167  return source;
168  else
169  {

171  return t4::make_brief<result_type>( constraints, std::move( source ), srcCS, dstCS );
172  }
