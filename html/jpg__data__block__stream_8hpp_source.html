88 /*
89  * ERROR HANDLING:
90  *
91  * The JPEG library's standard error handler (jerror.c) is divided into
92  * several "methods" which you can override individually. This lets you
93  * adjust the behavior without duplicating a lot of code, which you might
94  * have to update with each future release.
95  *
96  * Our example here shows how to override the "error_exit" method so that
97  * control is returned to the library's caller when a fatal error occurs,
98  * rather than calling exit() as the standard error_exit method does.
99  *
100  * We use C's setjmp/longjmp facility to return control. This means that the
101  * routine which calls the JPEG library must first execute a setjmp() call to
102  * establish the return point. We want the replacement error_exit to do a
103  * longjmp(). But we need to make the setjmp buffer accessible to the
104  * error_exit routine. To do this, we make a private extension of the
105  * standard JPEG error handler object. (If we were using C++, we'd say we
106  * were making a subclass of the regular error handler.)
107  *
108  * Here's the extended error handler struct:
109  */
110 
111 struct my_error_mgr
112 {
113  struct jpeg_error_mgr pub; /* "public" fields */
114  bool inDestructor;
115  jmp_buf m_env;
116  t4::exception_ptr m_error;
117 
118  void ThrowCachedError( void )
119  {
120  if ( m_error )
121  t4::rethrow_exception( m_error );
122  }
123 };

381 
382  JPGDataBlockStream( t4::data_block_stream source )
383  : ParentClass( std::move( source ) ),
384  fUserData(fDInfo.cinfo.client_data),
385  fIsPreferredMinificationFactorSet(false),
386  fFirstBlock(true)
387  {
388 
389  fUserData.client_data = this;
390 
391 #ifdef __GNUC__
392 #pragma GCC diagnostic push
393 #pragma GCC diagnostic ignored "-Wdisabled-macro-expansion"
394 #endif
395  if (!setjmp(fDInfo.jerr.m_env))
396 #ifdef __GNUC__
397 #pragma GCC diagnostic pop
398 #endif
399  {
400  /* Step 2: specify data source (eg, a file) */
401  SetupDataSource(&fDInfo.cinfo);
402 
403  /* Step 3: read file parameters with jpeg_read_header() */
404  jpeg_read_header(&fDInfo.cinfo, TRUE);
405  /* We can ignore the return value from jpeg_read_header since
406  * (a) suspension is not possible with the stdio data source, and
407  * (b) we passed TRUE to reject a tables-only JPEG file as an error.
408  * See libjpeg.doc for more info.
409  */
410 
411  fDInfo.jerr.ThrowCachedError();
412 
413  /*
414  libjpeg v9a introduced a change to default_decompress_params() in how it selects the decompression color space, in
415  a way that breaks the display of jpegs that report YCbCr in their comp_info but also have the "Adobe marker" in
416  their header. We detect and repair that condition here.
417 
418  One test file that triggers this difference is tai98.pdf. The right butterfly appeared dramatically different after
419  switching from libjpeg 8b to libjpeg 9b (although the critical difference appeared between 9 and 9a). The troublesome
420  image has the "Adobe marker" (APP14) in its header, but it also has comp_info values of 0x01 0x02 0x3 (which indicates
421  JCS_YCbCr).
422 
423  In all versions of libjpeg, the algorithm that selects the default value of jpeg_decompress_struct::jpeg_color_space
424  chooses JCS_RGB if the Adobe marker is present, and JCS_YCbCr if the comp_info values are 0x01 0x02 0x03. But what
425  value to choose if both conditions are met has changed.
426 
427  Before v9a, the function "default_decompress_params" libjpeg gave precedence to the Adobe APP14 marker, and would select
428  JCS_RGB as the jpeg_color_space. v9a introduced a change in which the component info fields were checked first, so
429  it would select JCS_YCbCr. jdapimin.c v9 and earlier contained a comment "(Wish JPEG committee had provided a real way
430  to specify this...)" which suggests that the original algorithm included an arbitrary decision in its implementation
431  (absent formal guidance). v9a _deleted_ this comment, suggesting that the JPEG committee had somehow, somewhere (I
432  couldn't find it) provided formal guidance which indicated the precedence of these various markers in the header.
433 
434  It appears that the existing client code did depend on the old behavior, at least in the case described above. So this
435  code detects the case where T4 depended upon giving precedence to the Adobe marker, and restores old (pre-v9a) behavior.
436  */
437  if ( fDInfo.cinfo.num_components == 3 && fDInfo.cinfo.saw_Adobe_marker )
438  fDInfo.cinfo.jpeg_color_space = fDInfo.cinfo.Adobe_transform ? JCS_YCbCr : JCS_RGB;
439  }
440  fDInfo.jerr.ThrowCachedError();
441  SetupIJGOutLayout();
442  }

496  {
497  if(!fIsPreferredMinificationFactorSet)
498  {
499  if ( preferredMinificationFactor == 1
500  || preferredMinificationFactor == 2
501  || preferredMinificationFactor == 4
502  || preferredMinificationFactor == 8 ) {
503 
504  fIsPreferredMinificationFactorSet = true;
505 
506  /* Adjust default decompression parameters since jpeg_read_header sets up default parameters based on header data*/
507  fDInfo.cinfo.scale_num = 1;
508  fDInfo.cinfo.scale_denom = static_cast<unsigned int>( preferredMinificationFactor );
509 
510 #ifdef __GNUC__
511 #pragma GCC diagnostic push
512 #pragma GCC diagnostic ignored "-Wdisabled-macro-expansion"
513 #endif
514  if (!setjmp(fDInfo.jerr.m_env))
515 #ifdef __GNUC__
516 #pragma GCC diagnostic pop
517 #endif

521  /* We can ignore the return value since suspension is not possible
522  * with the stdio data source.
523  */
524  }
525 
526  fDInfo.jerr.ThrowCachedError();
527  if(w)
528  *w = (GetImageWidth() + (preferredMinificationFactor - 1)) / preferredMinificationFactor;
529  if(h)

531

533  }
534  }
535 
536  return 0;

580  void DecompressAll( const pixel_buffer_base_type& dstBuf )
581  {
582  /* if the minification has not already been set, this will set it to 1;

587  tetraphilia_assert( fDInfo.cinfo.output_scanline == 0 );
588

590  * loop counter, so that we don't have to keep track ourselves.
591  */
592 
593  unsigned char* dst; // 0/*x*/, 0/*y*/;
594  unsigned char* p2 = static_cast<unsigned char*>(dstBuf->m_pixels );
595  dst = p2; // get 0/*x*/, 0/*y*/

597 #ifdef __GNUC__
598 #pragma GCC diagnostic push
599 #pragma GCC diagnostic ignored "-Wdisabled-macro-expansion"
600 #endif

602 #ifdef __GNUC__
603 #pragma GCC diagnostic pop

607  fDInfo.jerr.ThrowCachedError();

610  {

614  }
615  DestructJPEGDecompressor();

667  static size_t Read( JPGDataBlockStream<app_traits>* lThis, const unsigned char*& buffer )
668  {
669  // Since this function is called by libjpeg, we do not call setjmp here and instead catch exceptions (if any, thrown by our code)
670  // in the function body itself. Then, the error info is saved in the error structure and the call to ERREXIT calls our error handler.
671  // This way, we catch C++ exceptions thrown by calls made by this function and never throw from this into the libjpeg function that called Read.
672 
673  /* there are cases where error's in the jpeg data itself can cause
674  us to abort the read process. This causes the desctruction of
675  the IJGDecompState object which in turn causes a call to
676  jpeg_finish_decompress. This can call consume_markers,
677  which calls read_markers, which can call first_marker,
678  which can call fill_input_buff, which finally can call this
679  Read routine. We don't permit failable operations to occur
680  during a destructor. So if we are in the middle of the destructor
681  we just return 0 bytes.
682  */
683  if ( lThis->fDInfo.jerr.inDestructor || lThis->fDInfo.jerr.m_error )
684  return 0;
685 
686  try
687  {

699  a CR/LF pair, and only then does the image data begin. The PDF spec is explicit
700  that the ID token should be followed by exactly one byte of whitespace; this is
701  important because the image data might legitimately start with a byte that happens
702  to be whitespace.
703 
704  With JPEG this is impossible, though, since a JPEG stream has to start with an 0xFF
705  followed by the SOI marker (0xd8). Desktop Acrobat appears to gracefully detect and
706  eat an extra whitespace byte, so we do too.
707  */
708  if ( result > 0 && pdf::store::Parser::IsWhiteSpace( static_cast<char>( buffer[0] ) ) )
709  {
710  ++buffer;
711  --result;
712  }
713  lThis->fFirstBlock = false;
714  }

718  }

834 template <typename app_traits>
835 boolean JPGSourceMgr<app_traits>::FillInputBuffer (j_decompress_ptr cinfo)
836 {
837  // This is a callback registered with libjpeg - so, we do not call setjmp here (because we can't throw from here - we need to jump out of the libjpeg code)
838  // If an error occurs in this call (for instance the ERREXIT call below), we jump to the last location set before we called into libjpeg.
839  JPGSourceMgr<app_traits>* src = static_cast<JPGSourceMgr<app_traits>*>(cinfo->src);
840  size_t nbytes;
841  nbytes = JPGDataBlockStream<app_traits>::Read( static_cast<JPGDataBlockStream<app_traits>*>((static_cast<IJGUserData*>(cinfo->client_data))->client_data), src->bufferPtr);
842 
843  if (nbytes <= 0)
844  {
845 #ifdef __GNUC__
846 #pragma GCC diagnostic push
847 #pragma GCC diagnostic ignored "-Wold-style-cast"
848 #endif
849  if (src->start_of_file) /* Treat empty input file as fatal error */
850  {
851  ERREXIT(cinfo, JERR_INPUT_EMPTY);
852  }
853  WARNMS(cinfo, JWRN_JPEG_EOF);
854 #ifdef __GNUC__
855 #pragma GCC diagnostic pop
856 #endif
857 
858  /* Insert a fake EOI marker */
859  src->buffer[0] = static_cast<unsigned char>(0xFF);
860  src->buffer[1] = static_cast<unsigned char>(JPEG_EOI);
861  nbytes = 2;
862  src->next_input_byte = src->buffer; /* SA */
863  }
864  else
865  {
866  src->next_input_byte = src->bufferPtr; /* SA */
867  }
868 
869  src->bytes_in_buffer = nbytes;
870  src->start_of_file = FALSE;
871 
872  return TRUE;
873 }

876 template <typename app_traits>

878 {

880 
881  /* Just a dumb implementation for now. Could use fseek() except
882  * it doesn't work on pipes. Not clear that being smart is worth
883  * any trouble anyway --- large skips are infrequent.
884  */
885  if (num_bytes > 0) {
886  while (num_bytes > static_cast<long>(src->bytes_in_buffer)) {
887  num_bytes -= static_cast<long>(src->bytes_in_buffer);
888  FillInputBuffer(cinfo);
889  /* note we assume that fill_input_buffer will never return FALSE,
890  * so suspension need not be handled.
891  */
892  }
893  src->next_input_byte += static_cast<size_t>(num_bytes);
894  src->bytes_in_buffer -= static_cast<size_t>(num_bytes);
895  }
896 }
