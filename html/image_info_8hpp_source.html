90 class ImageRenderingRecord
91 {
92  typedef unsigned char* ptr_type;
93 
94 public:
95  ImageRenderingRecord(
96  const ImageRecord& ir,
97  ImageStream* istr,
98  const t4::int_rect& cnstrInterestBounds,
99  const t4::affine_transform& totalMatrix);
100 
101  // Produces an IRR that does no downscaling at all and simply produces the entire image
102  // at full resolution
103  ImageRenderingRecord( const ImageRecord& ir )
104  : m_imageSubsectionDataBounds( t4::make<t4::int_rect>( 0, 0, ir.m_width, ir.m_height ) ),
105  m_streamScaleDownPower( 0 ),
106  m_thumbnailScaleDownPower( 0 ),
107  m_totalMatrix( t4::identity<t4::affine_transform>() ),
108  m_width( ir.m_width ),
109  m_height( ir.m_height )
110  {}
111 
112  /* these are the values that are correct after image
113  downsampling -- that is, they are the values above scaled down by
114  fOrigScaleDownFactor (after rounding, etc...). Note that in
115  the jpeg case, low scale factor down sampling (up to factor 8)
116  is handled by the jpeg decompressor itself.
117  */
118  t4::int_rect m_imageSubsectionDataBounds;
119 
120  /*
121  During construction of an ImageRenderingRecord, we call istr->SetPreferredMinificationFactor()
122  which requests a total scaledown POT to be done to the source image. The stream performs
123  none, some or all of the scaledown. If there is any scaledown left to be performed it is
124  handled by the thumbnailer.
125 
126  This value is the POT scaledown performed by the stream. For example, if a JP2K stream can do
127  up to 3 POTs of scaledown, then this value will always in [0,3].
128  */
129  int m_streamScaleDownPower;
130 
131  /*
132  This is the POT scaledown performed by the thumbnailer. For example, if a total scaledown of
133  1/32 is required, and the stream scaledown could only do 1/4x, the remaining three powers of 2
134  will be done by the thumbnailer, and m_thumbnailScaleDownPower will be 3.
135  */
136  int m_thumbnailScaleDownPower;
137 
138  /*
139  This is simply the sum of the stream and thumbnail scaledowns. It is the total scaledown POT
140  from the original image source space to the output of the ImagePipe.
141  */
142  int GetTotalScaleDownPower( void ) const { return m_streamScaleDownPower + m_thumbnailScaleDownPower; }
143 
144  t4::affine_transform m_totalMatrix;
145 
146  t4::int_coord m_width;
147  t4::int_coord m_height;
148 
149 protected:
150  t4::int_coord GetScaleDownFactor(const t4::affine_transform& totalMatrix);
151 };

153 template<typename object_traits>
154 void ImageRecord::SetImageColorSpace( store::Object<object_traits> csObj,
155  t4::pdf::document& doc,
156  const t4::pdf::dictionary* resources,
157  bool isInlineImage,
158  ImageStream* istr )
159 {
160  const char * csName = nullptr;
161  switch ( csObj.Type() ) {
162  case store::kObjectTypeArray: {
163  auto csArray = csObj.ArrayValue_NoTypeCheck();
164  csName = csArray.Get(0).NameValue().get();
165  pdfcolor::CSArray<object_traits> csa( csArray );
166  m_pdfCS = pdfcolor::PDFColorSpace::CreateFromName(
167  csName, resources, pdfcolor::GetColorSpaceCache( doc.get_document_caches() ), true /*permitAbbreviations*/,
168  true/*permitSpacesWithAlternates*/, &csa, 0, m_bitsPerSignal == 16 );
169  m_isIndexedColor = pdfcolor::PDFColorSpace::IsIndexedColorSpace( csName, isInlineImage );
170  } break;
171  case store::kObjectTypeName:
172  csName = csObj.NameValue_NoTypeCheck().get();
173  m_pdfCS = pdfcolor::PDFColorSpace::CreateFromName(
174  csName, resources, pdfcolor::GetColorSpaceCache( doc.get_document_caches() ), true /*permitAbbreviations*/,
175  true /*permitSpacesWithAlternates*/, nullptr, 0, m_bitsPerSignal == 16 );
176  m_isIndexedColor = m_pdfCS.GetType() == pdfcolor::kPDFCSTIndexed;
177  break;
178  default: {
179  // if an image is compressed using JPEG2000, its (pdf) image dictionary
180  // is allowed to have a missing ColorSpace entry. In that case the viewer
181  // is required to use the color space information in the JPEG2000 data stream
182  // itself (which is more than just pixel data). Both JPEG and JPEG2000
183  // decoders end up setting the 'istr' parameter to non-zero. In the JPEG case
184  // the GetEmbeddedColorSpace method always returns an empty value. In JPEG2000
185  // it always returns a non-empty value. So this test for empty is essentially
186  // a test for whether the image is compressed with JPEG or not.
187  if ( istr ) {
188  tetraphilia_assert(m_JPEG2K);
189 
190  boost::optional<pdfcolor::PDFColorSpace> cs = istr->GetEmbeddedColorSpace();
191  if ( !cs )
192  throw t4::corrupt_data();
193  m_pdfCS = *cs;
194  m_isIndexedColor = m_pdfCS.GetType() == pdfcolor::kPDFCSTIndexed;
195  } else
196  throw t4::corrupt_data();
197  }
198  }
199 
200  if( (m_pdfCS.GetType() != pdfcolor::kPDFCSTRegular) && (m_pdfCS.GetType() != pdfcolor::kPDFCSTIndexed) )
201  throw t4::corrupt_data();
202 
203  m_numSignals = m_pdfCS.GetNumSignals();
204  if ( m_bitsPerSignal != 1 && m_bitsPerSignal != 2
205  && m_bitsPerSignal != 4 && m_bitsPerSignal != 8
206  && m_bitsPerSignal != 16 )
207  throw t4::corrupt_data();
208 }

211 template<typename object_traits>
212 ImageRecord::ImageRecord( t4::pdf::document& doc,
213  const store::Dictionary<object_traits>& imageDict,
214  const t4::pdf::dictionary* resources,
215  bool isInlineImage,
216  ImageStream* istr,
217  const PDFImageKey& imageKey,
218  size_t numParentSignals)
219  : m_width(std::max(0, imageDict.GetRequiredInteger( inline_image::kWidth, isInlineImage) )),
220  m_height(std::max(0, imageDict.GetRequiredInteger( inline_image::kHeight, isInlineImage ))),
221  m_interpolate( imageDict.GetBoolean( inline_image::kInterpolate, isInlineImage, false ) ),
222  m_maskKind(kNoMask),
223  m_isImageMask(imageDict.GetBoolean( inline_image::kImageMask, isInlineImage, false )),
224  m_isIndexedColor(false),
225  m_isInvertedDecodeArray(false),
226  m_JPEG2K(false),
227  m_imageKey( imageKey )
228 {
229 
230  if(istr)
231  m_JPEG2K = istr->IsJPXStream();
232 
233  if ( m_isImageMask ) {
234 
235  m_numSignals = 1;
236  m_bitsPerSignal = 1;
237 
238  } else {
239 
240  // figure out if there is any mask data that must be applied to this image
241  if ( istr && istr->SupportsEmbeddedMaskData() && imageDict.GetInteger( "SMaskInData", 0 ) != 0 ) {
242 
243  // if this is a JPEG2000 image and the image dictionary contains an SMaskInData entry,
244  // then we will use that data and ignore any other mask-related data
245  m_maskKind = kSoftMask;
246 
247  } else if ( imageDict.GetDictionary( "SMask" ) ) {
248 
249  // not JPEG2000 or doesn't have SMaskInData, but does have SMask
250  m_maskKind = kSoftMask;
251 
252  } else {
253 
254  auto objMask = imageDict.Get( "Mask" );
255 
256  if(objMask.Type() != store::kObjectTypeNull) // explicit or color masking
257  {
258  if( objMask.Type() == store::kObjectTypeArray ) // color key masking
259  {
260  m_maskKind = kColorKeyMask;
261  }
262  else // explicit masking
263  {
264  m_maskKind = kStencilMask;
265  }
266  }
267 
268  }
269  // Note: For images using the JPXDecode filter (see Section 3.3.8, JPXDecode Filter),
270  // the number of bits per component is determined from the image data and not specified
271  // in the image dictionary; the color space may or may not be specified in the dictionary.
272  if(istr && istr->SupportsEmbeddedBitsPerSignal())
273  {
274  m_bitsPerSignal = istr->GetBitsPerSignal();
275  }
276  else
277  {
278  m_bitsPerSignal = t4::convert<size_t>( imageDict.GetRequiredInteger( inline_image::kBitsPerComponent, isInlineImage) );
279  }
280 
281  // this also sets m_numSignals
282  SetImageColorSpace( imageDict.Get( inline_image::kColorSpace, isInlineImage ), doc, resources, isInlineImage, istr );
283  }
284 
285  /*
286  load the decode array if it exists
287  */
288  if(!numParentSignals)
289  {
290  auto objDecode = imageDict.Get( inline_image::kDecode, isInlineImage );
291  if( objDecode.Type() != store::kObjectTypeNull )
292  {
293  auto decodeArray = objDecode.ArrayValue();
294  size_t decodeArrayLen = decodeArray.Length();
295  if( decodeArrayLen < 2 * m_numSignals )
296  throw t4::corrupt_data();
297  decodeArrayLen = 2 * m_numSignals;
298 
299  m_decodeArray.emplace( decodeArrayLen );
300 
301  size_t i;
302  for( i = 0; i < decodeArrayLen; i += 2 )
303  {
304  (*m_decodeArray)[i] = decodeArray.GetRequiredReal(i);
305  (*m_decodeArray)[i+1] = decodeArray.GetRequiredReal(i+1);
306  }
307 
308  if ( m_isIndexedColor ) {
309  int div = (1 << m_bitsPerSignal) - 1;
310  for ( i = 0; i < decodeArrayLen; ++i )
311  (*m_decodeArray)[i] /= div;
312  }
313 
314  // check to see if it is an identity decode array
315  for ( i = 0; i < decodeArrayLen; i += 2 ) {
316  if( (*m_decodeArray)[i] != 0 || (*m_decodeArray)[i+1] != 1 )
317  break;
318  }
319  if ( i == decodeArrayLen )
320  m_decodeArray.reset(); // no-op decode array
321  }
322  }
323  else
324  {
325  typedef t4::app_traits::pdf_traits::display_traits display_traits;
326  // color key mask
327  if( display_traits::kSupportColorKeyMasks && m_maskKind == kColorKeyMask && !m_isImageMask )
328  {
329  size_t colorKeyMaskLen = 2 * numParentSignals;
330  m_colorKeyMaskArray.emplace( colorKeyMaskLen );
331 
332  // get the color key mask array
333  auto objMask = imageDict.Get( "Mask" );// has to exist if m_maskKind == kColorKeyMask
334  tetraphilia_assert( (objMask.Type() != store::kObjectTypeNull) && (objMask.Type() == store::kObjectTypeArray) );
335 
336  auto mask = objMask.ArrayValue();
337  if( mask.Length() != 2 * numParentSignals )
338  throw t4::corrupt_data();
339 
340  for( size_t i = 0; i < colorKeyMaskLen; i++ )
341  (*m_colorKeyMaskArray)[i] = mask.GetRequiredInteger(i);
342  }
343  }
344 
345  // the PDF notion of the "meaning" of 1 and 0 is reversed from our notion
346  // if the image is an imageMask. So we have to reverse the bits using
347  // a decodeArray
348  if ( m_isImageMask )
349  {
350  if ( !m_decodeArray )
351  {
352  m_decodeArray.emplace( 2 );
353  (*m_decodeArray)[0] = 1;
354  (*m_decodeArray)[1] = 0;
355  }
356  else
357  {
358  (*m_decodeArray)[0] = 1 - (*m_decodeArray)[0];
359  (*m_decodeArray)[1] = 1 - (*m_decodeArray)[1];
360  }
361  }
362 
363  // now check to see if we have an "inverted" decode array because we have an optimization for that
364  // case, but only if we aren't handling indexed color
365  if ( m_decodeArray && !m_isIndexedColor ) {
366  size_t i;
367  for ( i = 0; i < 2 * m_numSignals; i += 2 ) {
368  if( (*m_decodeArray)[i] != 1 || (*m_decodeArray)[i+1] != 0 )
369  break;
370  }
371  if ( i == 2 * m_numSignals )
372  m_isInvertedDecodeArray = true; // all ranges in the decode array are [1,0] -- we have an optimization for that
373  }
374 }

378 template<typename signal_traits>
379 ImageRenderingRecord<signal_traits>::ImageRenderingRecord( const ImageRecord& ir,
380  ImageStream* istr,
381  const t4::int_rect& cnstrInterestBounds,
382  const t4::affine_transform& _totalMatrix)
383 {
384  using namespace imaging_model;
385 
386 
387  // determine the largest integer power of two that is less than or equal to
388  // the "1/magnitude" of the given matrix. We will be handling that part of the
389  // image down scaling using thumbnailing logic on the image. The rest of the
390  // image transform is done using the remaining matrix.
391  m_thumbnailScaleDownPower = m_streamScaleDownPower = 0;
392 
393  t4::int_coord curScaleDownFactor;
394  curScaleDownFactor = GetScaleDownFactor(_totalMatrix);
395  int totalScaleDownPower = 0;
396 
397  t4::int_coord scaleDn = 1;
398  while ( scaleDn <= curScaleDownFactor ) {
399  ++totalScaleDownPower;
400  scaleDn <<= 1;
401  }
402  if ( totalScaleDownPower > 0 )
403  --totalScaleDownPower;
404 
405  m_thumbnailScaleDownPower = totalScaleDownPower;
406  m_width = ir.m_width;
407  m_height = ir.m_height;
408  t4::affine_transform totalMatrix(_totalMatrix);
409 
410  if ( istr && !(ir.m_JPEG2K && (ir.m_isIndexedColor || (istr->GetStreamBitsPerSignal(0)<=4)))) {
411  m_streamScaleDownPower = totalScaleDownPower;
412  if ( m_streamScaleDownPower > 3 )
413  m_streamScaleDownPower = 3;
414 
415  if ( m_streamScaleDownPower > 0 )
416  {
417  int w=m_width, h=m_height;
418  int istrScaleFactor = 1 << m_streamScaleDownPower;
419  istrScaleFactor = istr->SetPreferredMinificationFactor(istrScaleFactor, &w, &h);
420 
421  /* fint the new power by converting from the new factor to power */
422  int power = 0;
423  int factor = 1;
424  while ( factor < istrScaleFactor ) {
425  factor <<= 1;
426  power += 1;
427  }
428  m_streamScaleDownPower = power;
429  m_thumbnailScaleDownPower -= m_streamScaleDownPower;
430 
431  // SA: The following compution for totalMatrix is not exactly right
432  // since istrScaleFactor may be different than wsf and hsf; This different
433  // only manifest itself in case of shrinking since our reported width and
434  // height might be off by one pixel from the one from jp2k (we round by
435  // adding .5 while they dont - we favor pixels to right). Statisticaly,
436  // our computation is 1 pixel off 1/4 of times for minifications but in
437  // practice this problem has barely been observed.
438  //
439  //real_coord wsf = real_coord(m_width)/real_coord(w);
440  //real_coord hsf = real_coord(m_height)/real_coord(h);
441 
442  m_width = w;
443  m_height = h;
444 
445  get_a(totalMatrix) = get_a(totalMatrix) * istrScaleFactor;
446  get_b(totalMatrix) = get_b(totalMatrix) * istrScaleFactor;
447  get_c(totalMatrix) = get_c(totalMatrix) * istrScaleFactor;
448  get_d(totalMatrix) = get_d(totalMatrix) * istrScaleFactor;
449  }
450  }
451 
452  t4::affine_transform noTransMat = totalMatrix;
453  get_e(noTransMat) = 0;
454  get_f(noTransMat) = 0;
455  t4::affine_transform invNoTransMat;
456  if ( !MatrixInvert( invNoTransMat, noTransMat ) ) {
457  MakeEmptyRect(m_imageSubsectionDataBounds);
458  m_totalMatrix = t4::identity<t4::affine_transform>();
459  return;
460  }
461 
462  MakeRectangle( m_imageSubsectionDataBounds, 0, 0, m_width, m_height );
463 
464  // determine the bounds, in device space, of where this image will draw
465  t4::int_rect transformedBounds = TransformAndBoundIntRect( m_imageSubsectionDataBounds, totalMatrix );
466 
467  // does any of the destination bounds fall outside the constraints?
468  if ( !RectContainsRect( cnstrInterestBounds, transformedBounds ) ) {
469 
470  // clip this bounds to the bounds of the current clipping region
471  transformedBounds = RectIntersect(cnstrInterestBounds, transformedBounds);
472 
473  if ( RectIsEmpty(transformedBounds) ) {
474  MakeEmptyRect(m_imageSubsectionDataBounds);
475  } else {
476  // Allow a little slop space
477  // the algorithm below essentially attempts take the device bounds that this
478  // image will draw into, map that back into image space, and then add the number
479  // of source image pixels to that mapped rectangle so that when this larger source
480  // rectangle is mapped into device space it is about one pixel larger than the
481  // device rectangle we will draw into. This larger source rectangle is then clipped
482  // to the actual bounds of the image (in case it was larger). The idea is to allow
483  // the code to only use a subset of the source pixels, but to make sure that the subset
484  // isn't so small that it changes any of the device pixels that are rendered -- particularly
485  // along the edge of what will be rendered.
486  // this is all fine and good except that it is prone to overflow problems in a fixed point
487  // build if the source image is very large. Watson bug#1568368 contains an image that is
488  // 32767 pixels wide, drawn into a much smaller rectangle. We currently use a very simple
489  // rule that says if the source image is either wider or taller than 16384 in a fixed point
490  // build then it is too dangerous to attempt to trim the pixels this way.
491  bool tooBigToClip = false;
492  if ( real_services::RealTypeHelper<t4::real_coord>::isFloatingPointType == 0 )
493  tooBigToClip = m_width > 16384 || m_height > 16384;
494 
495  if ( !tooBigToClip ) {
496  t4::real_coord major, minor;
497  math_utils::MatrixUtils<t4::affine_transform>::CalcRadiiOfMappedUnitCircle(totalMatrix, &major, &minor);
498  t4::int_coord inset = major < 1 ? -1 : -t4::floor<t4::int_coord>(major);
499  transformedBounds = RectInset(transformedBounds, inset, inset );
500 
501  t4::real_rect interestBounds = CastToRealCoordRect<t4::real_rect>( transformedBounds );
502 
503  // next step is to map this transformedBounds by the inverse of totalMatrix to determine
504  // an upper bounds on the set of pixels in the source image that have any possibility of
505  // showing up after applying the transformation matrix and clip.
506  // this sequence of subtracting .e and .f from the values of the rect, then
507  // setting .e and .f of the matrix to zero, then mapping by the inverse produces the same
508  // thing as just mapping the unadjusted points by the inverse of the
509  // unadjusted matrix. However, this way is much less likely to overflow
510  // in a fixed point build (and probably produces slighly more accurate
511  // answers in a floating point build)
512  interestBounds = RectTranslate( interestBounds, -get_e(totalMatrix), -get_f(totalMatrix) );
513  t4::int_rect sBounds = CastToIntCoordRect<t4::int_rect>(TransformAndBoundRealRect( interestBounds, invNoTransMat ));
514 
515  // flip x and/or y if this transformation flipped either of them
516  if ( get_xmin(sBounds) > get_xmax(sBounds) )
517  {
518  t4::int_coord t = get_xmin(sBounds);
519  set_xmin(sBounds, get_xmax(sBounds));
520  set_xmax(sBounds, t);
521  }
522  if ( get_ymin(sBounds) > get_ymax(sBounds) )
523  {
524  t4::int_coord t = get_ymin(sBounds);
525  set_ymin(sBounds, get_ymax(sBounds));
526  set_ymax(sBounds, t);
527  }
528 
529  // finally, pin sBounds to the valid values of the original image. The transformations above
530  // may have caused it to be slightly outside of this range.
531  set_xmin(sBounds, t4::pin<t4::int_coord>(0, get_xmin(sBounds), m_width));
532  set_ymin(sBounds, t4::pin<t4::int_coord>(0, get_ymin(sBounds), m_height));
533  set_xmax(sBounds, t4::pin<t4::int_coord>(0, get_xmax(sBounds), m_width));
534  set_ymax(sBounds, t4::pin<t4::int_coord>(0, get_ymax(sBounds), m_height));
535 
536  // at this point, sBounds is some subset of the original image data that is nearly optimally small,
537  // while still containing all of the pixels that will contribute to some device pixel that will
538  // be painted. If this rectangle is significantly smaller than the entire image we are going to
539  // want to use a subsectioner to reduce the total amount of memory we need to allocate to
540  // draw this image. But before doing that we need adjust this rectangle for the downsampler,
541  // if we are also going to run a down sampler. The down sampler runs whenever m_scaleDownPower
542  // is greater than 1, but it requires that the xMin and yMin edges of the bounds it works on
543  // (which will be the bounds of the subsectioner) be block aligned, where a "block" is 1<<m_scaleDownPower
544  // pixels.
545 
546  // in the x direction we have two competing factors to deal with. One is that the subsectioner
547  // will be running prior to the decoder, which means it is potentially dealing with pixels that
548  // are less than one byte. But it requires that its xmin represent a pixel that starts on a
549  // byte boundary. So we need to find the greatest pixel that satifies two conditions:
550  // 1) it starts on a byte boundary
551  // 2) its x value is an integer multiple of 1 << m_thumbnailScaleDownPower
552  tetraphilia_assert( m_thumbnailScaleDownPower >= 0 );
553  t4::int_coord pixelsPerThumbnailBlock = static_cast<t4::int_coord>(1 << m_thumbnailScaleDownPower);
554  t4::int_coord bpp = static_cast<t4::int_coord>(ir.m_bitsPerSignal * ir.m_numSignals);
555  t4::int_coord xBlockSizeInBits = t4::lcm(t4::int_coord{8}, bpp*pixelsPerThumbnailBlock);
556  t4::int_coord xBlockSizeInPixels = xBlockSizeInBits/bpp;
557 
558  if ( get_xmin(sBounds) != 0 )
559  set_xmin( sBounds, get_xmin(sBounds) - (get_xmin(sBounds) % xBlockSizeInPixels) );
560  if ( get_xmax(sBounds) != m_width ) {
561  t4::int_coord newX = get_xmax(sBounds) + xBlockSizeInPixels - 1;
562  newX -= newX % xBlockSizeInPixels;
563  if ( newX < get_xmax(sBounds) )
564  set_xmax( sBounds, newX );
565  }
566  if ( get_ymin(sBounds) != 0 )
