401 template<typename stroker_traits>
402 void DelayedPathPoint<stroker_traits>::Commit(raw_output_path_type& path)
403 {
404  switch(m_pathOp)
405  {
406  case kOpNone:
407  break;
408  case kOpMoveTo:
409  path.moveto(m_pt[0].x, m_pt[0].y);
410  break;
411  case kOpLineTo:
412  path.lineto(m_pt[0].x, m_pt[0].y);
413  break;
414  case kOpCurveTo:
415  path.curveto(m_pt[0].x, m_pt[0].y, m_pt[1].x, m_pt[1].y, m_pt[2].x, m_pt[2].y);
416  break;
417  case kOpClosePath:
418  path.closepath();
419  break;
420  }
421  m_pathOp = kOpNone;
422 }

844 template<typename stroker_traits>
845 typename ParamsPlus<stroker_traits>::coord_type ParamsPlus<stroker_traits>::computeActualStrokeWidth(const raw_params_type& p, const matrix_type& toDevMatrix, typename ParamsPlus<stroker_traits>::coord_type& deviceSpaceWidth)
846 {
847  bool willStrokeAdjust = getDoStrokeAdjust(p);
848  auto width = std::abs(getLineWidth(p));
849 
850  // compute the semi-major and semi-minor axes of the transformed unit circle
851  coord_type major, minor;
852 
853  math_utils::MatrixUtils<matrix_type>::CalcRadiiOfMappedUnitCircle(toDevMatrix, &major, &minor);
854  // see if the transform is "orthogonal enough" for stroke adjust to make sense
855  // this measures whether the major and minor axes differ by less than 10%
856  if (major-minor > minor*coord_type(.3))
857  willStrokeAdjust = false;
858 
859  // At some line thicknesses, stroke adjust begins to be a meaningless idea; at another point further
860  // on, it ceases to be computationally possible (think strokes aroud 2**23). I have chosen
861  // a cutoff way bigger than meaningful and way smaller than unreliable.
862  if (major <= 0 || width * major > 1000)
863  willStrokeAdjust = false;
864 
865  // compute the pessimistic min stroke width in user space. If the matrix is highly anisotropic
866  // this is an impossible request, since making the line thick enough to be seen in one direction
867  // may make it far too thick in the other. Oh well, we use the major axis since we encountered overflow
868  // issues when we used minor (under extreme CTM's).
869  coord_type minStrokeWidthInUserSpace;
870  if ( major >= 1 )
871  minStrokeWidthInUserSpace = getMinStrokeWidthInRasterizerSpace(p) / major;
872  else {
873  coord_type mx = limits<coord_type>::max() * major;
874  if ( getMinStrokeWidthInRasterizerSpace(p) >= mx )
875  minStrokeWidthInUserSpace = limits<coord_type>::max();
876  else
877  minStrokeWidthInUserSpace = getMinStrokeWidthInRasterizerSpace(p) / major;
878  }
879  while (real_services::SquareWillBeDangerouslyLarge(minStrokeWidthInUserSpace))
880  minStrokeWidthInUserSpace /= 2; // stick with a usable number.
881 
882  // max is unknown max devicespace width
883  deviceSpaceWidth = major <= 0 ? limits<coord_type>::max() : width * major;
884 
885  if ( willStrokeAdjust )
886  {
887  // OK, we have determined that the matrix is pretty much orthogonal, so we
888  // don't have to worry about differences in x and y coords. The "DTransform" of the
889  // matrix can be fairly summed up as a scaling since rotation does not affect the
890  // stroke adjust. So we map to device space (by multiplying by major), compute
891  // the adjusted stroke width in device space, then map back (by dividing by major).
892  if (getWillAntiAlias(p))
893  // set by AGM4PathStrokerParams code only (not ARE)
894  deviceSpaceWidth = deviceSpaceWidth - coord_type( 0.25f ); // adjust width by 1 subpixel only
895  else
896  deviceSpaceWidth = real_services::Round(deviceSpaceWidth) - 1;
897  if ( deviceSpaceWidth < 0 )
898  deviceSpaceWidth = 0;
899  width = deviceSpaceWidth/major;
900  }
901 
902  // return max of width and minStrokeWidthInUserSpace
903  return width > minStrokeWidthInUserSpace ? width : minStrokeWidthInUserSpace;
904 }
