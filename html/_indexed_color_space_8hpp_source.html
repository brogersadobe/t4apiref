163 template <typename app_traits>
164 IndexedColorSpace<app_traits>::IndexedColorSpace(
165  smart_ptr_type & baseCS,
166  int_type hiVal,
167  const t4::pdf::dictionary& streamDict )
168  : m_baseCS( baseCS )
169 {
170  if( hiVal < 0 || hiVal > 255 )
171  throw t4::corrupt_data();
172  m_hiVal = static_cast<size_t>(hiVal);
173  size_t required = ( m_hiVal + 1 ) * baseCS->GetNumSignals();
174  m_table.resize( required );
175 
176  auto& heap = *t4::get_thread_heap();
177  t4::transient_snapshot ss( heap );
178 
179  t4::data_block_stream dbs = store::GetFilteredStream( streamDict, false /*isInlineImage*/ );
180  data_io::BufferedStream* bds = TRANSIENT_NEW( data_io::BufferedDataBlockStream, heap, ( std::move( dbs ) ) );
181  const char * ptr;
182  size_t available = bds->PeekBytes( ptr, required );
183  #ifdef PDF_SLANG
184  if( available >= required )
185  memcpy( m_table.data(), ptr, required );
186  else
187  {
188  memcpy( m_table.data(), ptr, available );
189  memset( m_table.data() + available, 0, required - available );
190  }
191  #else
192  if( available < required )
193  throw t4::corrupt_data();
194  memcpy( m_table.data(), ptr, required );
195  #endif
196 
197  this->SetNumSignals( 1 );
198 }

200 template <typename app_traits>
201 IndexedColorSpace<app_traits>::IndexedColorSpace(
202  smart_ptr_type & baseCS,
203  const JP2KPalette & pal )
204  : m_baseCS( baseCS )
205 {
206  int c;
207  if( pal.numEntries < 0 || pal.numEntries > 1024 )
208  throw t4::corrupt_data();
209  for(c=0; c<pal.numCompsProduced; c++)
210  {
211  if( pal.compBitDepths[c] > 8 )
212  throw t4::corrupt_data();
213  }
214 
215  if ( !t4::eq( baseCS->GetNumSignals(), pal.numCompsProduced ) )
216  throw t4::corrupt_data();
217 
218  m_hiVal = static_cast<size_t>(pal.numEntries-1);
219  size_t required = static_cast<size_t>( ( m_hiVal + 1 ) * baseCS->GetNumSignals() );
220  m_table.resize( required );
221 
222  unsigned char* p = m_table.data();
223  for(c=0; c<pal.numCompsProduced; c++)
224  {
225  const unsigned char* table = static_cast<const unsigned char*>( pal.LUTs[c] );
226  for (int i=0; i<pal.numEntries; i++)
227  {
228  p[i*pal.numCompsProduced+c] = table[i];
229  }
230  }
231 
232  this->SetNumSignals( 1 );
233 }
