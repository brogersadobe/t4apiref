77  struct versioner_wrapper
78  {
79  // The versioner_wrapper (insted of the versioner_type) is added in the versioner_vector since the IDOVersionCounter
80  // maintains a list of the versioner_type pointers and the corresponding transaction numbers.
81  // This list is used to perform operations such as getting the list of dirty IDOs, dropping edits that are no longer reachable, etc.
82  // Using the versioner_wrapper ensures that the underlying versioner_type* is not modified when items are inserted
83  // in the versioner_vector.
84  versioner_wrapper(t4::unique_ptr<versioner_type> versioner)
85  : m_versioner(std::move(versioner)) {}
86 
87  t4::unique_ptr<versioner_type> m_versioner;
88  };

149  // Push an ObjectImplVersioner onto the m_versioners stack, and append it to the version of the array
150  // corresponding to the current transaction number.
151  template<typename... TArgs>
152  void Push( TArgs&&... args )
153  {
154  assert( !m_is_dictionary || m_is_parsing_dictionary );
155 
156  if ( StaysUnrevised() )
157  {
158  auto& version = GetCurrentVersion();
159  m_versioners.emplace_back( t4::make_unique<versioner_type>( *this, m_ido, std::forward<TArgs>( args )... ) );
160  version.push_back( std::prev( std::end( m_versioners ) ) );
161  }
162  else
163  {
164  m_versioners.emplace_back( t4::make_unique<versioner_type>( *this, m_ido, std::forward<TArgs>( args )... ) );
165  InvalidateCurrentVersion();
166  }
167  }

238  template<typename... TArgs>
239  void Put( t4::string key, TArgs&&... args )
240  {
241  assert( m_is_dictionary && !m_is_parsing_dictionary );
242 
243  if ( StaysUnrevised() )
244  {
245  auto& version = GetCurrentVersion();
246  auto& versionForDictionary = GetCurrentVersionForDictionary();

252  version.push_back( std::prev( std::end( m_versioners ) ) );

292  template<typename... TArgs>
293  void Insert( const const_iterator& before, TArgs&&... args )
294  {
295  m_unrevised = false;
296 
297  // Creating an ObjectVersionerImpl potentially invalidates all iterators of index_vectors,
298  // so we need to deref to get the iterator into the m_versioner stack _before_ creating
299  // the new element.
300  auto emplace_target = ( before == end() ) ? m_versioners.end() : *before.m_index;
301  BOOST_ATTRIBUTE_UNUSED auto result = m_versioners.emplace( emplace_target, t4::make_unique<versioner_type>( *this, m_ido, std::forward<TArgs>( args )... ) );
302 
303  // index_vector is a vector of iterators into m_versioners. By inserting an element into the stack we
304  // have potentially invalidated _all_ outstanding iterators, including the ones in the current version's
305  // index_vector.
306  t4::lock_guard<t4::mutex> lg( m_version_map_mutex );
307  m_version_map.clear();
308  m_version_map_for_dictionary.clear();
309  }

311  void Delete( const const_iterator& start_, const const_iterator& stop_ )
312  {
313  m_unrevised = false;
314 
315  auto ctn = transactions::TransactionManager::GetCurrentTransactionNumber();
316 
317  // Deleting an ObjectVersionerImpl potentially invalidates all iterators of index_vectors,
318  // so we need to deref to get the iterators into the m_versioner stack _before_ destroying
319  // the element.
320  auto start = ( start_ == end() ) ? m_versioners.end() : *start_.m_index;
321  auto stop = ( stop_ == end() ) ? m_versioners.end() : *stop_.m_index;
322 
323  for ( auto it = start; it != stop; ++it )
324  {
325  // This is unfortunate, but it's a necessary artifact of the ObjectImplVersioner's hybrid reference/object semantics
326  auto& obj = const_cast<versioner_type&>( *it->m_versioner );
327  if ( obj.IsValid( ctn ) )
328  obj.Delete( m_ido );
329  }
330 
331  // Since Delete doesn't actually remove elements from m_versioners, we only
332  // invalidate iterators from the current version when we do this.
333  InvalidateVersion( ctn );
334  }
335 
336  void InvalidateVersion( size_t tn )
337  {
338  if ( !m_unrevised )
339  {
340  /*
341  We invalidate all cached index_vectors after this version as well, because they may be
342  exact duplicates (i.e., no deltas)
343  */
344  t4::lock_guard<t4::mutex> lg( m_version_map_mutex );
345  m_version_map.erase( m_version_map.lower_bound( tn ), std::end( m_version_map ) );
346  m_version_map_for_dictionary.erase( m_version_map_for_dictionary.lower_bound( tn ), std::end( m_version_map_for_dictionary ) );
347  }
348  }

351  /*
352  Tests if the VersionedStack is in the "unrevised" state (see documentation of m_unrevised member)
353  and if the VersionedStack will still be unrevised if a change occurs under the CTN.
354  */
355  bool StaysUnrevised( void )
356  {
357  t4::lock_guard<t4::mutex> lg( m_version_map_mutex );
358  if ( m_unrevised && !m_version_map.empty() )
359  {
360  auto ctn = transactions::TransactionManager::GetCurrentTransactionNumber();
361  if ( m_version_map.begin()->first != ctn )
362  m_unrevised = false;
363  }
364 
365  return m_unrevised;
366  }

421  index_map_for_dictionary ComputeIndexMapForDictionary( size_t tn ) const
422  {
423  index_map_for_dictionary index_map;
424  index_map.reserve( m_versioners.size() / 2 );
425  VersionResolver<obj_traits_type, versioner_type> resolver;
426  auto end = std::end( m_versioners );
427  for ( auto it = std::begin( m_versioners ); it != end; ++it )
428  {
429  if ( it->m_versioner->IsValid( tn ) )
430  {
431  const ObjectImpl<obj_traits_type>& objImpl = resolver( const_cast<versioner_type&>( *it->m_versioner ) );
432  if ( objImpl.type != kObjectTypeName )
433  throw t4::corrupt_data();
434 
435  t4::string key(objImpl.u.nameAndStringRep->data(), objImpl.u.nameAndStringRep->size() );
436 
437  ++it;
438  if ( it == end )
439  throw t4::corrupt_data();
440 
441  index_map.emplace( key, it );
442  }
443  }
444 
445  return index_map;
446  }
