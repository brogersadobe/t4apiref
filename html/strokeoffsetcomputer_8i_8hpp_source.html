29 // class OffsetComputer is used to compute the offset curve from a given path segment;
30 // the offset is to the "left" in the move from start to end of the path sense. We may
31 // fail to get a single offset curve which is accurate enough, and need several.
32 // In this case, you need multiple passes to build the offset.
33 // The calls then should be of the form:
34 // OffsetComputerParams ocparams(mat, 1, true, true, 10.0);
35 // ...
36 //
37 // OffsetComputerLoop ocloop(ocparams, seg, false);
38 // while (ocloop.GetNextOffset())
39 // {
40 // if (ocloop.CurOffsetIsValid())
41 // {
42 // pts = ocloop.GetCurOffset();
43 // ...
44 // }
45 // else
46 // {
47 // DealWithPathogenicCurve()
48 // }
49 // }
50 //
51 // The functions ocloop.GetCurPathStart() and ocloop.GetCurPathEnd()
52 // return the endpoints of the section of the original curve which was
53 // used to construct the offset curve (used for joins).

235 template <typename stroker_traits>
236 OffsetComputerLoop<stroker_traits>::OffsetComputerLoop(const params_plus_type& params, coord_type tanTooShortToScaleLength, const contour_seg_type& seg, bool reverse)
237  : m_params(params)
238  , m_tanTooShortToScaleLength(tanTooShortToScaleLength)
239  , m_isReversed(reverse)
240  , m_dontusethepointsinthisseg(seg)
241  , m_startT(reverse ? 1 - seg.GetEndT() : seg.GetStartT())
242  , m_endT(reverse ? 1 - seg.GetStartT() : seg.GetEndT())
243  , m_curT(reverse ? 1 - seg.GetEndT() : seg.GetStartT()) // same as m_startT
244  , m_numGetNextOffsetCalls(0)
245  , m_curOffsetIsValid(false)
246  , m_segOrigin(0,0)
247  , m_tangentIsComputationallyUnreliableLength(0)
248 {
249  // get the control points and jiggle as necessary
250  seg.CopyControlPoints(m_controlPoints, m_isReversed);
251 
252  // if its a bezier, init the cubic polynomial
253  if (seg.IsBezier())
254  {
255  m_segOrigin = (m_controlPoints[0] + m_controlPoints[3])/static_cast<int_type>(2);
256  for (int i = 0; i < 4; i++)
257  m_controlPoints[i] -= m_segOrigin;
258 
259  m_cubicX.Set(get_x(m_controlPoints[0]), get_x(m_controlPoints[1]), get_x(m_controlPoints[2]), get_x(m_controlPoints[3]));
260  m_cubicY.Set(get_y(m_controlPoints[0]), get_y(m_controlPoints[1]), get_y(m_controlPoints[2]), get_y(m_controlPoints[3]));
261  }
262  // see comments for computeUnreliableTangentLength
263  m_tangentIsComputationallyUnreliableLength = computeUnreliableTangentLength();
264 }

266 template <typename stroker_traits>
267 typename OffsetComputerLoop<stroker_traits>::coord_type OffsetComputerLoop<stroker_traits>::computeUnreliableTangentLength()
268 {
269  if (PathIsBezier())
270  {
271  // First we compute the "average coeff size" of the derivative polynomial
272  coord_type avgcoeff = 0;
273  avgcoeff += AbsVal(m_cubicX.c[2]);
274  avgcoeff += AbsVal(m_cubicY.c[2]);
275  avgcoeff += AbsVal(m_cubicX.c[1])*2;
276  avgcoeff += AbsVal(m_cubicY.c[1])*2;
277  avgcoeff += AbsVal(m_cubicX.c[0])*3;
278  avgcoeff += AbsVal(m_cubicY.c[0])*3;
279  avgcoeff *= coord_type(1)/6;
280 
281  // Then we observe (you'll have to play with fixed and float numbers yourself)
282  // That the "error bars on the derivative" are as big as the epsilon on the polynomial
283  // coeffs, not the epsilon on the result (they differ in float space)
284  //
285  // deriverr ~ epsilon(avgcoeff)
286  //
287  // This is because in order to get a small number by cancelling a bunch of large
288  // ones, you lose precision. Very little precision is lost if the big numbers mix
289  // and result in a big number.
290  // We are going to scale the tangents up to the offset width, so the error on the
291  // offset point is of order
292  //
293  // offseterror ~ offsetwidth*epsilon(avgcoeff)/tangent
294  //
295  // We want this to be smaller than the desired precision, otherwise the computations
296  // are all noise, so we get
297  //
298  // mintangent ~ epsilon(avgcoeff)*offsetwidth/precision
299  //
300  // NOTE: may need better computation in fixed space....
301  return real_services::Epsilon(avgcoeff)*m_params.GetOffsetWidth()/m_params.GetUserSpacePrecision();
302  }
303  else
304  {
305  // the tangent is always "computationally reliable" on a line, just
306  // maybe too short to scale properly. That is handled elsewhere
307  return 0;
308  }

310 template <typename stroker_traits>
311 bool OffsetComputerLoop<stroker_traits>::GetNextOffset()
312 {
313  if (++m_numGetNextOffsetCalls > 100)
314  {
315  if (m_numGetNextOffsetCalls >= 16384)
316  {
317  // tetraphilia_assert_always(); // more than 100 would be surprising, 16384 stunning....
318  return false; // just a safety catch so we can't infinitely loop when in release mode
319  }
320  }
321 
322  if (m_startT != m_endT)
323  {
324  // general case
325  if (m_curT == m_endT)
326  {
327  m_curOffsetIsValid = false;
328  return false;
329  }
330  else
331  {
332  getNextOffset();
333  return true;
334  }
335  }
336  else
337  {
338  // make sure we do "zero length offsets" exactly once
339  if (m_numGetNextOffsetCalls == 1)
340  {
341  getNextOffset();
342  return true;
343  }
344  else
345  {
346  m_curOffsetIsValid = false;
347  return false;
348  }
349  }

363 template <typename stroker_traits>
364 bool OffsetComputerLoop<stroker_traits>::nextLineOffset()
365 {
366  point_type tangentVec = m_controlPoints[1]-m_controlPoints[0];
367  point_type offsetVec;
368 
369  m_curOffsetPathStartT = m_curT;
370  m_curOffsetPathEndT = m_endT;
371  if (!getOffsetVector(&offsetVec, tangentVec))
372  {
373  return false;
374  }
375  else
376  {
377  m_curOffsetPathStartPoint = m_controlPoints[0] + m_curOffsetPathStartT*tangentVec;
378  m_curOffsetPathEndPoint = m_controlPoints[0] + m_curOffsetPathEndT*tangentVec;
379  m_curOffset[0] = m_curOffsetPathStartPoint + offsetVec;
380  m_curOffset[1] = m_curOffsetPathEndPoint + offsetVec;
381  return true;
382  }

385 template <typename stroker_traits>
386 bool OffsetComputerLoop<stroker_traits>::nextBezierOffset()
387 {
388  m_curOffsetPathStartT = m_curT;
389  m_curOffsetPathEndT = m_endT;
390 
391  // special case for zero length beziers (common for zero length dashes)
392  if (m_curOffsetPathStartT == m_curOffsetPathEndT)
393  {
394  m_curOffsetPathStartPoint = EvalPathBez(m_curOffsetPathStartT);
395  m_curOffsetPathEndPoint = m_curOffsetPathStartPoint;
396  if (!computeOffsetVector( &m_curOffset[0], m_curOffsetPathStartT))
397  return false; // curve is pathogenic case
398  m_curOffset[0] += m_curOffsetPathStartPoint;
399  m_curOffset[1] = m_curOffset[0];
400  m_curOffset[2] = m_curOffset[0];
401  m_curOffset[3] = m_curOffset[0];
402  return true;
403  }
404 
405  // special case, the math breaks down if the offset width is
406  // zero, plus the answer is trivial so this is faster.
407  if (m_params.GetOffsetWidth() == 0)
408  {
409  ComputeBezSection(m_controlPoints[0], m_controlPoints[1], m_controlPoints[2], m_controlPoints[3], m_curOffsetPathStartT, m_curOffsetPathEndT, m_curOffset[0], m_curOffset[1], m_curOffset[2], m_curOffset[3]);
410  m_curOffsetPathStartPoint = m_curOffset[0];
411  m_curOffsetPathEndPoint = m_curOffset[3];
412  return true;
413  }
414 
415  while (1) // this loops when we need to split and try again.
416  {
417  if (tryToComputeSingleOffsetBez(false))
418  return true; // yay, got a good offset!
419 
420  // At some point splitting more is likely to be fruitless. We give up after
421  // about 8 levels pof split in FIXED builds, 11 in Float. This is about half of
422  // the precision of each type (not too well thought out though).
423  if ( m_curOffsetPathEndT - m_curOffsetPathStartT < real_services::GetMinSplittableBezierTRange(ArgType<coord_type>()) )
424  return false;
425 
426  // Oh well, halve and try again.
427  m_curOffsetPathEndT = real_services::Average(m_curOffsetPathStartT, m_curOffsetPathEndT);
428 
429  } // loop
430 }

432 // Find the tangent to the polynomail at a point t, resorting
433 // to higher level derivatives if they are too short. Returns
434 // true if a valid one is found, false if not.
435 template <typename stroker_traits>
436 bool OffsetComputerLoop<stroker_traits>::computeTangent( point_type* pTangent, coord_type t, bool anyNonzeroIsOK ) const
437 {
438  coord_type tooShort = 0;
439  if (!anyNonzeroIsOK)
440  {
441  tooShort = m_tanTooShortToScaleLength;
442  if (m_tangentIsComputationallyUnreliableLength > tooShort)
443  tooShort = m_tangentIsComputationallyUnreliableLength;
444  }
445 
446  if (PathIsBezier())
447  {
448  // try first derivative
449  pTangent->x = Deriv(m_cubicX, t);
450  pTangent->y = Deriv(m_cubicY, t);
451  if (pTangent->TaxicabLength() > tooShort)
452  return true;
453 
454  // OK, so the first derivative is not OK; we need a tangent from somewhere else.
455  // Note that we are looking for the "geometric tangent", not the parametric (derivative) tangent.
456  // This is sloppy terminology, but what I mean is a tangent to the curve in the picture on paper
457  // sense, not the "direction of the derivatives" sense. A pathological curve example
458  // may help. Imagine the following two curves:
459  //
460  // CURVE 1: A curve that starts at the origin, goes straight up, and comes back down,
461  // to the origin, i.e. x = 0, y = t - t^2. Then y' = 1-2t, y'' = -2, y''' = 0.
462 
463  // At its peak the first derivative is zero, but its second derivatice is constant.
464  // The problem then is the "before tangent" at t=1/2 vs. the after tangent at t=1/2.
465  // We want the tangent on the up side to point straight up, and on the down side to point down.
466  // Therefore the second derivative needs to be "aligned" with the desired direction, and that alignment is
467  // is different just before the apex than after. The "geometric tangent" is discontinuous!!!! Note that
468  // the actual parametric (derivative) tangent is perfectly well behaved.
469  //
470  // CURVE 2: A curve which starts at the origin, and keeps moving up, but its derivative vanishes
471  // in the middle, i.e. x = 0, y = (t-1/2)^3 + 1/8. Then y' = 3*(t-1/2)^2, y'' = 6*(t-1/2), y''' = 6
472  //
473  // In the middle, the first and second derivatives are both zero. The second derivative actually flips direction
474  // in the middle ( I believe for a cubic this is always true, but not in higher degree polynomials).
475  // Anway, we need to resort to higher order derivatives at the midpoint again, but not the second, and moreover the
476  // geomentric tangent is continuous here unlike the case above.
477 
478  // From considerations above we know our curves are monotonic in x and y, although the first derivative
479  // may (and often does) vanish at the endpoint. As such, we know the "geometric tangent's" general direction, it
480  // is from the start of the curve towards the end. It may vary a bit from that, but it will always be roughly in this
481  // direction. In particular, it is never true for a monotonic curve that its geometric tangent will go against its
482  // first derivative. It IS the first derivative when the first derivative is nonzero. But when the first derivative
483  // vanishes, we need to resort to the second or third derivatives, which may differ in direction from what the first
484  // derivative was nearby (Note that this is related to curvature issues, which are related to the cross product
485  // of the first and second derivatives). My claim then is that you look at the higher order derivatives, and if their
486  // direction has a negative dot product with the curve's overall direction, then you want to negate them.
487 
488  // Lets look at the two surves above in this light:
489 
490  // Curve 1: This curve is not monotonic, so we would have broken it and been looking only at either the first
491  // or second half. On the first half, the general direction is up, while the second derivative is down. Therefore at
492  // the peak (end) of the first half, we would use the negative of the second derivative.
493  // On the second half, the general direction is down, and the second derivative is down as well. Therefore at
494  // the peak (start) of the second half, we would use the second derivative un-negated. This is where the
495  // discontinuity is introduced!!
496 
497  // Curve 2: This curve is monotonic, but has vanishing first derivative so we would break it at t=1/2 anyhow.
498  // On the first half, the general direction is up, the second derivative vanishes at the end, and the third
499  // derivative is up. Therefore the end of the first half, we would use the third derivative un-negated.
500  // On the second half, the general direction is still up, and the second derivative vanishes at the beginning,
501  // and the third derivative is still up as well... Therefore the end of the first half, we would use the third
502  // derivative un-negated again, and no discontinuity in the geometric tangent is introduced. Whew.
503 
504  // The key then is to resort to "aligning" the higher derivatives with the first derivative in the region, and since
505  // the curves are monotone this is the same as aligning in the direction from the start point to the end.
506 
507  // Figure out the "general direction" of the curve in this region, f(end) - f(start),
508  // this had better not be perpendicular to the second derivative otherwise "orienting in the same way"
509  // is hopelessly ambiguous. We already know the path points at the start and end of the current section
510  // of curve, we can use those to get a general direction.
511  point_type generalDirection = m_curOffsetPathEndPoint - m_curOffsetPathStartPoint;
512  if (generalDirection.TaxicabLength() == 0)
513  {
514  // OK, lets try one more thing, the general direction of the whole curve. Since the
515  // curve is monotonic (or at least almost always so but occaisionally barely off to
516  // to numerical imprecision) we can use the direction from the beginning of the curve to
517  // the end. On the surface this would seem easier and better than the "local general direction"
518  // I tried to use above, but it fails miserably on a case like B=[(0,0),(0,0),(0,-e),(e,0)] where
519  // e is a tiny epsilon. This non monotonic curve can arrive to due to imprecision despite my
520  // best efforts at ironing them out, and in this pathological cae the resultant curve it highly
521  // ot monotone (albeit a very tiny curve). As such, the "local general direction" is preferable
522  // since it behave better on non-monotone curves.
523  generalDirection = m_controlPoints[3] - m_controlPoints[0];
524  if (generalDirection.TaxicabLength() == 0)
525  return false; // give up, we don't know how to orient the 2nd deriv.
526  }
527 
528  // keep from overflowing, we only need the general direction (not the
529  // magnitude). In IEEE builds we never get these kinds of overflows
530  // so we don't bother scaling in that case, but 16.16 fixed is susceptible
531  // to overflow during the Dot and Cross operations below
532  real_services::ScaleVectorLengthToNearOneIfDotCanOverflowOrUnderflow(&generalDirection.x, &generalDirection.y);
533 
534  // try second derivative, making sure to align it with the general direction
535  pTangent->x = SecondDeriv(m_cubicX, t);
536  pTangent->y = SecondDeriv(m_cubicY, t);
537  if (pTangent->TaxicabLength() > tooShort)
538  {
539  // make sure the 2nd deriv and general direction are not perpendicular-ish, this
540  // calc below means tan(angle) is in [-2, 2], so it is not within about 30 degrees
541  // of perpendicular; this number was chosen at randon, but in practice the 2nd deriv
542  // and general direction shiould be quite close to parallel, so this is a comfy choice.
543  coord_type absdot = AbsVal(Dot(*pTangent, generalDirection));
544  coord_type abscross = AbsVal(Cross(*pTangent, generalDirection));
545  if (abscross > 2*absdot)
546  return false; // too perpendicular for reliable "alignment"
547 
548  if (Dot(*pTangent, generalDirection) < 0)
549  *pTangent = -(*pTangent);
550  return true;
551  }
552 
553  // try third derivative, making sure to align it with the general direction
554  pTangent->x = ThirdDeriv(m_cubicX, t);
555  pTangent->y = ThirdDeriv(m_cubicY, t);
556  if (pTangent->TaxicabLength() > tooShort)
557  {
558  coord_type absdot = AbsVal(Dot(*pTangent, generalDirection));
559  coord_type abscross = AbsVal(Cross(*pTangent, generalDirection));
560  if (abscross > 2*absdot)
561  return false; // too perpendicular for reliable "alignment"
562 
563  if (Dot(*pTangent, generalDirection) < 0)
564  *pTangent = -(*pTangent);
565  return true;
566  }
567 
568  // give up
569  return false;
570  }
571  else
572  {
573  *pTangent = m_controlPoints[1] - m_controlPoints[0];
574  return pTangent->TaxicabLength() > tooShort;
575  }
576 }

612 template <typename stroker_traits>
613 bool OffsetComputerLoop<stroker_traits>::tryToComputeSingleOffsetBez(bool forceOK)
614 {
615  point_type offsetVecStart, offsetVecEnd;
616  coord_type roughComputationalErrorBars = 0;
617 
618  m_curOffsetPathStartPoint = EvalPathBez(m_curOffsetPathStartT);
619  m_curOffsetPathEndPoint = EvalPathBez(m_curOffsetPathEndT);
620 
621  // get the offset curve start point
622  if (!computeOffsetVector(&offsetVecStart, m_curOffsetPathStartT))
623  {
624  if (forceOK)
625  justComputeAnOffsetVectorDammit(&offsetVecStart, m_curOffsetPathStartT);
626  else
627  return false;
628  }
629 
630  m_curOffset[0] = m_curOffsetPathStartPoint + offsetVecStart;
631  roughComputationalErrorBars += real_services::Epsilon(m_curOffset[0].x);
632  roughComputationalErrorBars += real_services::Epsilon(m_curOffset[0].y);
633 
634  // clear the two control points so we can incrementally add to them below
635  m_curOffset[1].x = m_curOffset[1].y = 0;
636  m_curOffset[2].x = m_curOffset[2].y = 0;
637 
638  // get the offset curve end point
639  if (!computeOffsetVector(&offsetVecEnd, m_curOffsetPathEndT))
640  {
641  if (forceOK)
642  justComputeAnOffsetVectorDammit(&offsetVecEnd, m_curOffsetPathEndT);
643  else
644  return false;
645  }
646 
647  m_curOffset[3] = m_curOffsetPathEndPoint+offsetVecEnd;
648 
649  // we know the whole bezier does intersect the visible region, ohterwise we would have thrown it out.
650  // However, that does not mean this piece of it does. We can replace precise offsets with simple lineto's
651  // when the are outside the visible region, saving us computation now and later on in the reducer as well.
652  if ( m_params.InOneOctantOutsideOfVisible(m_curOffsetPathStartPoint + m_segOrigin, m_curOffsetPathEndPoint + m_segOrigin))
653  {
654  m_curOffset[1] = m_curOffset[0];
655  m_curOffset[2] = m_curOffset[3];
656  return true;
657  }
658 
659  roughComputationalErrorBars += real_services::Epsilon(m_curOffset[3].x);
660  roughComputationalErrorBars += real_services::Epsilon(m_curOffset[3].y);
661 
662  // Now we have to decide whether we are close enough; there are two factors here: one,
663  // the tolerances we were asked to achieve. The other, the precision that this computation was done with.
664  // If the precision is not good enough to hope for the tolerances requested, there is no point in
665  // continuing.
666  coord_type tolerance = m_params.GetUserSpacePrecision() + roughComputationalErrorBars;
667 
668  // if the offset curve is very short, there is no point wasting time in computing
669  // an accurate offset curve. Note that since the path is monotone, the offsets
670  // can't be too ill-behaved so checking the distance between endpoints is a decent
671  // measure of the offset curve's scale.
672  if (TaxicabDist(m_curOffset[0], m_curOffset[3]) < tolerance)
673  {
674  // may as well slap in a line and be done with it.
675  m_curOffset[1] = m_curOffset[0];
676  m_curOffset[2] = m_curOffset[3];
677  return true; // good enough.
678  }
679 
680  // load the constants for the interpolation
681  point_type interpolant[TETRAPHILIA_NUM_STROKE_OFFSET_INTERPOLANTS];
682  size_t numInterpolants = m_params.GetNumBezOffsetInterpolants();
683  const coord_type* interpConst = static_cast<const coord_type*>( InterpTableHelper<coord_type, static_coord_type>::gInterpConst[numInterpolants] );
684  const coord_type* tCubed = static_cast<const coord_type*>( InterpTableHelper<coord_type, static_coord_type>::gInterpTCubed[numInterpolants] );
685  const coord_type* oneMinusTCubed = static_cast<const coord_type*>( InterpTableHelper<coord_type, static_coord_type>::gInterpOneMinusTCubed[numInterpolants] );
686 
687  // do the interpolation
688  coord_type dt = (m_curOffsetPathEndT-m_curOffsetPathStartT)/int(numInterpolants+1);
689  coord_type t = m_curOffsetPathStartT+dt;
690  for (size_t i = 0; i < numInterpolants; i++, t += dt)
691  {
692  point_type offsetVec;
693  if (!computeOffsetVector(&offsetVec, t))
694  {
695  if (forceOK)
696  justComputeAnOffsetVectorDammit(&offsetVec, t);
697  else
698  return false;
699  }
700  interpolant[i] = EvalPathBez(t)+offsetVec;
701  roughComputationalErrorBars += real_services::Epsilon(interpolant[i].x);
702  roughComputationalErrorBars += real_services::Epsilon(interpolant[i].y);
703 
704  // we need to "subtract off the contributions of the begining and end points"
705  // to make this interpolation work. Therfore we subtract offset[0]*(1-t)^3
706  // and offset[3]*t^3. Note that these t's are in the [0,1] range of the offset curve
707  // not the [m_curOffsetPathStartT,m_curOffsetPathEndT] range of the original.
708  point_type adjustedInterpolant = interpolant[i] - m_curOffset[0]*oneMinusTCubed[i] - m_curOffset[3]*tCubed[i];
709 
710  m_curOffset[1] += interpConst[i] * adjustedInterpolant;
711  m_curOffset[2] += interpConst[numInterpolants-1-i] * adjustedInterpolant;
712  }
713 
714  if (forceOK)
715  return true;
716 
717  // make sure the curve is monotonically increasing or decreasing
718  // in the directions perpendicular the the offset edges (which perp
719  // CW or CCW doesn't matter).
720  bool forcedToMonotone = false;
721  if (!BezIsMonotonicInDirection(m_curOffset[0], m_curOffset[1], m_curOffset[2], m_curOffset[3], offsetVecStart.Rotate90CCW()))
722  {
723  if (offsetCannotBeMadeMonotonicInDirection(offsetVecStart.Rotate90CCW(), m_curOffset))
724  return false; // cannot be made monotonic, split and continue
725  else
726  forcedToMonotone = true;
727  }
728  if (!BezIsMonotonicInDirection(m_curOffset[0], m_curOffset[1], m_curOffset[2], m_curOffset[3], offsetVecEnd.Rotate90CCW()))
729  {
730  if (offsetCannotBeMadeMonotonicInDirection(offsetVecEnd.Rotate90CCW(), m_curOffset))
731  return false; // cannot be made monotonic, split and continue
732  else
733  forcedToMonotone = true;
734  }
735 
736  // see how the interpolation turned out, but don't bother if there were only two interpolants
737  // since we know it hit exactly (-ish :)). Note that we also check if the curve was forced to
738  // be monotonic as that distorts the interpolation
739  if (numInterpolants > 2 || forcedToMonotone)
740  {
741  // see how we did, return failure if any interpolant point is too far away from the
742  // offset bezier
743  dt = coord_type(1)/int(numInterpolants+1);
744  t = dt;
745  for (size_t i = 0; i < numInterpolants; i++, t += dt)
746  {
747  // see if the curve is near enough to the interpolant
748  if (TaxicabDist(interpolant[i], BezEval(m_curOffset[0], m_curOffset[1], m_curOffset[2], m_curOffset[3], t)) > tolerance)
749  return false;
750  }
751  }
752 
753  // if we had two interpolants, we fit exactly so we need to sanity check another point,
754  // this is the old algorithm pretty much exactly...
755  if (numInterpolants == 2)
756  {
757  // No need to check points, we hit them exactly. Check midpoint instead.
758  coord_type midPathT = (m_curOffsetPathStartT+m_curOffsetPathEndT)/2;
759  point_type midOffsetVec;
760  if (!computeOffsetVector(&midOffsetVec, midPathT))
761  return false;
762  if (TaxicabDist(BezEval(m_curOffset[0], m_curOffset[1], m_curOffset[2], m_curOffset[3], coord_type(0.5f)), EvalPathBez(midPathT)+midOffsetVec) > tolerance)
763  return false;
764  }
765 
766  return true;
767 }

769 template <typename stroker_traits>
770 bool OffsetComputerLoop<stroker_traits>::offsetCannotBeMadeMonotonicInDirection( const point_type& dir, point_type offset[4] ) const
771 {
772  coord_type mag = dir.Mag();
773  if (mag == 0)
774  return false; // we're OK in pathogenic zero width curve case....
775 
776  point_type unitdir = dir / mag;
777 
778  // We are going to try to resolve the following sort of non-monotonicity:
779  //
780  // C C
781  // dir-> E E
782  //
783  // C(ontrol) point is "outside" of the E(nd) points. We are going to push
784  // the left control point over so that that it is between the E's in the
785  // direction specified
786  snapPointToBetweenInDir(&offset[1], unitdir, offset[0], offset[3]);
787  snapPointToBetweenInDir(&offset[2], unitdir, offset[0], offset[3]);
788 
789  // now see how we did...
790  return !BezIsMonotonicInDirection(offset[0], offset[1], offset[2], offset[3], unitdir);
791 }

803 template <typename stroker_traits>
804 void OffsetComputerLoop<stroker_traits>::snapPointToBetweenInDir( point_type* p, const point_type& dir, const point_type& a, const point_type& b ) const
805 {
806  coord_type pz = Dot(*p, dir);
807  coord_type az = Dot(a, dir);
808  coord_type bz = Dot(b, dir);
809  if (between(pz, az, bz))
810  return; // nothing to do
811 
812  // OK, need to slide it. Figure out how much to slide
813  coord_type slide = (AbsVal(az - pz) < AbsVal(bz - pz)) ? az - pz : bz - pz;
814  // Now make sure to slide a little more than absolutely required, so we are really inside.
815  // scootching by 1/1000 of the dist from a to b is a nice choice...
816  if (slide > 0)
817  slide += AbsVal(az-bz) / 1000;
818  else
819  slide -= AbsVal(az-bz) / 1000;
820 
821  // Now move the point
822  *p += slide*dir;
823 }
