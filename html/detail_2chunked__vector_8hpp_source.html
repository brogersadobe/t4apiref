44 /*
45  Instances of chunk store a single, contiguous subsequence of a chunked_vector's
46  elements.
47 
48  chunks are allocated using the chunked_vector's allocator, and the storage for
49  their elements starts "just past the end" of the chunk object. As a consequence
50  of this chunks must be allocated carefully, see chunked_vector::add_chunk.
51 */
52 template<typename T, bool StableEndIterator>
53 class chunk : public boost::intrusive::list_base_hook<>
54 {
55  friend chunked_vector_iterator_base<T const, StableEndIterator>;
56  friend chunked_vector_iterator_base<T, StableEndIterator>;
57 
58  using chunked_vector_type = chunked_vector_base<T, StableEndIterator>;
59  chunked_vector_type* m_owner;
60  size_t m_index;
61  T* m_top; // pointer to first unused location
62  T* m_stop; // end of storage
63 
64 public:
65  chunk( chunked_vector_type* owner, size_t index, size_t chunk_size )
66  : m_owner( owner ),
67  m_index( index )
68  {
69  m_top = get_start();
70  m_stop = m_top + chunk_size;
71  }
72 
73  ~chunk( void )
74  {
75  // Destroy the elments in [get_start(), m_top)
76  auto start = get_start();
77  while ( m_top != start )
78  {
79  --m_top;
80  m_top->~T();
81  }
82  }
83 
84  // Because of chunk's tricky allocation requirements, these copy/move ctor/assign and swap aren't allowed
85  chunk( chunk const& rhs ) = delete;
86  chunk( chunk&& rhs ) = delete;
87  chunk& operator=( chunk const& rhs ) = delete;
88  chunk& operator=( chunk&& rhs ) = delete;
89 
90  bool empty( void ) const { return m_top == get_start(); }
91  size_t size( void ) const { return static_cast<size_t>( m_top - get_start() ); }
92  size_t avail( void ) const { return static_cast<size_t>( m_stop - m_top ); }
93  size_t capacity( void ) const { return static_cast<size_t>( m_stop - get_start() ); }
94  bool full( void ) const { return m_top == m_stop; }
95 
96  T* storage_start( void ) const { return get_start(); }
97  T* first_unused( void ) const { return m_top; }
98  T* storage_end( void ) const { return m_stop; }
99 
100  void pop( size_t count = 1 )
101  {
102  assert( count <= size() );
103  while ( count-- )
104  {
105  --m_top;
106  m_top->~T();
107  }
108  }
109 
110  template<typename... Targs>
111  T& emplace_push( Targs&&... args )
112  {
113  assert( !full() );
114  ::new ( m_top ) T( std::forward<Targs>( args )... );
115  return *m_top++;
116  }
117 
118  void push( T val )
119  {
120  assert( !full() );
121  ::new ( m_top ) T( std::move( val ) );
122  ++m_top;
123  }
124 
125 private:
126  T* get_start( void ) const { return align_up( const_cast<T*>( static_cast<T const*>( static_cast<void const*>( this + 1 ) ) ), alignof( T ) ); }
127 };

129 /*
130  chunked_vector iterators derive from chunked_vector_iterator_base. This is done to keep EBCO possible
131  in the reverse iterator (chunked_vector_iterator and reverse_chunked_vector_iterator have some
132  ancestor classes in common, which means that by C++ rules the m_base member can't have the same
133  memory location as the reverse_chunked_vector_iterator class.)
134 
135  Using a base class that doesn't have boost::random_access_iterator_helper or std::iterator as
136  a base class solves this problem.
137 */
138 template<typename T, bool StableEndIterator>
139 class chunked_vector_iterator_base

144  using chunk_iterator_type = typename chunked_vector_type::cchunk_iterator_type;

152  using mutable_chunked_vector_iterator_base = chunked_vector_iterator_base<mutable_T, StableEndIterator>;
153  friend const_chunked_vector_iterator_base;

193  void validate( void ) const
194  {
195  #if T4_VALIDATE_CHUNKED_VECTORS
196  if ( m_element != nullptr )
197  {
198  if ( StableEndIterator )
199  {
200  assert( m_chunk_iterator->m_index <= m_owner->m_end_chunk->m_index );
201 
202  // If we're in a "full" chunk, iterators must point to a constructed element. "One past the end"
203  // iterators are not allowed in full chunks if StableEndIterator == true.
204  if ( m_chunk_iterator->full() )
205  assert( m_chunk_iterator->storage_start() <= m_element && m_element < m_chunk_iterator->first_unused() );
206  else
207  assert( m_chunk_iterator->storage_start() <= m_element && m_element <= m_chunk_iterator->first_unused() );
208  }
209  else
210  {
211  assert( m_chunk_iterator->m_index <= m_owner->m_end_chunk->m_index );
212  if ( m_chunk_iterator == m_owner->m_end_chunk )
213  assert( m_chunk_iterator->storage_start() <= m_element && m_element <= m_chunk_iterator->first_unused() );
214  else
215  assert( m_chunk_iterator->storage_start() <= m_element && m_element < m_chunk_iterator->first_unused() );
216  }
217  }
218  #endif
219  }

221  /*
222  The operators provided here (*, <, ==, +=, -=, -, ++, --) are what is required by boost::random_access_iterator_helper.
223  All other operators are provided for free by that template.
224  */
225  BOOST_FORCEINLINE T& operator*( void ) const
226  {
227  validate();
228  return *m_element;
229  }

231  bool operator<( const_chunked_vector_iterator_base const& rhs ) const
232  {
233  validate();
234  rhs.validate();
235  assert( m_owner == rhs.m_owner );
236 
237  if ( StableEndIterator )
238  {
239  assert( m_element != nullptr );
240  assert( rhs.m_element != nullptr );
241  }
242  else
243  {
244  // The only valid iterator with a nullptr m_element is one coming from an end() call on an empty, non-StableEndIterator
245  // chunked_vector. Since the vector is empty, and you can only compare iterators from the same container, the other
246  // argument must also be nullptr.
247  if ( m_element == nullptr )
248  {
249  assert( rhs.m_element == nullptr );
250 
251  // And, since they are both "null" iterators (they must be the result of a call to end(), they must be equal.
252  return false;
253  }
254 
255  assert( rhs.m_element != nullptr );
256  }
257 
258  if ( m_chunk_iterator == rhs.m_chunk_iterator )
259  return m_element < rhs.m_element;
260  return m_chunk_iterator->m_index < rhs.m_chunk_iterator->m_index;
261  }
262 
263  bool operator==( const_chunked_vector_iterator_base const& rhs ) const
264  {
265 #ifdef TETRAPHILIA_T3_COMPAT
266  /*
267  There is at least one place in the AR code base (PDFFieldManager::hasCurrentFieldBeenReset)
268  where we depend on a default-initialized iterator to be comparable (as opposed to simply unusable).
269  We should probably revisit that code using boost::optional (there's a comment there already saying as
270  much), but until then we have to support the comparisons gracefully.
271 
272  At the moment the only guarantees are that default-initialized iterators have a nullptr m_owner field.
273  */
274  if ( m_owner == nullptr )
275  return rhs.m_owner == nullptr;
276  else if ( rhs.m_owner == nullptr )
277  return false;
278 #endif
279 
280  validate();
281  rhs.validate();
282  assert( m_owner == rhs.m_owner );
283  // No need to compare chunk iterators here, m_element is a unique address in memory (also if nullptr,
284  // m_chunk_iterator is ignored)
285  return m_element == rhs.m_element;
286  }
287 
288  chunked_vector_iterator_base& operator+=( ptrdiff_t p )
289  {
290  validate();
291  if ( p < 0 )
292  return operator-=( -p );
293  else while ( p > 0 )
294  {
295  assert( m_owner->m_size > 0 );
296  if ( p < m_chunk_iterator->first_unused() - m_element )
297  {
298  m_element += p;
299  break;
300  }
301  else
302  {
303  p -= ( m_chunk_iterator->first_unused() - m_element );
304  if ( m_chunk_iterator != m_owner->m_end_chunk )
305  {
306  ++m_chunk_iterator;
307  m_element = m_chunk_iterator->storage_start();
308  }
309  else
310  {
311  assert( p == 0 );
312  m_element = m_chunk_iterator->first_unused();
313  }
314  }
315  }
316 
317  if ( StableEndIterator && m_chunk_iterator->full() && m_element == m_chunk_iterator->storage_end() )
318  {
319  assert( m_chunk_iterator != m_owner->m_chunks.end() );
320  ++m_chunk_iterator;
321  m_element = m_chunk_iterator->storage_start();
322  }
323 
324  validate();
325  return *this;
326  }
327 
328  chunked_vector_iterator_base& operator-=( ptrdiff_t p )
329  {
330  validate();
331  if ( p < 0 )
332  return operator+=( -p );
333  else while ( p > 0 )
334  {
335  assert( m_owner->m_size > 0 );
336  if ( p <= m_element - m_chunk_iterator->storage_start() )
337  {
338  m_element -= p;
339  break;
340  }
341  else
342  {
343  p -= ( m_element - m_chunk_iterator->storage_start() );
344  assert( m_chunk_iterator != std::begin( m_owner->m_chunks ) );
345  --m_chunk_iterator;
346  m_element = m_chunk_iterator->first_unused();
347  }
348  }
349 
350  if ( StableEndIterator && m_chunk_iterator->full() && m_element == m_chunk_iterator->storage_end() )
351  {
352  assert( m_chunk_iterator != m_owner->m_chunks.end() );
353  ++m_chunk_iterator;
354  m_element = m_chunk_iterator->storage_start();
355  }
356 
357  validate();
358  return *this;
359  }
360 
361  ptrdiff_t operator-( const_chunked_vector_iterator_base rhs ) const
362  {
363  validate();
364 
365  if ( *this < rhs )
366  return -rhs.operator-( *this );
367 
368  if ( m_element == nullptr )
369  {
370  assert( rhs.m_element == nullptr );
371  return 0;
372  }
373  assert( rhs.m_element != nullptr );
374 
375  ptrdiff_t result = 0;
376  while ( m_chunk_iterator != rhs.m_chunk_iterator )
377  {
378  assert( rhs.m_chunk_iterator != m_owner->m_end_chunk );
379  result += ( rhs.m_chunk_iterator->first_unused() - rhs.m_element );
380  ++rhs.m_chunk_iterator;
381  rhs.m_element = rhs.m_chunk_iterator->storage_start();
382  }
383 
384  result += ( m_element - rhs.m_element );
385  return result;
386  }
387 
388  chunked_vector_iterator_base& operator++( void )
389  {
390  validate();
391 
392  assert( m_owner->m_size > 0 );
393  assert( m_element != nullptr );
394 
395  ++m_element;
396  if ( m_element == m_chunk_iterator->first_unused() )
397  {
398  if ( m_chunk_iterator != m_owner->m_end_chunk )
399  {
400  ++m_chunk_iterator;
401  m_element = m_chunk_iterator->storage_start();
402  }
403  }
404  validate();
405  return *this;
406  }
407 
408  chunked_vector_iterator_base& operator--( void )
409  {
410  validate();
411  assert( m_element != nullptr );
412 
413  if ( m_element == m_chunk_iterator->storage_start() )
414  {
415  assert( m_chunk_iterator != m_owner->m_chunks.begin() );
416  --m_chunk_iterator;
417  m_element = m_chunk_iterator->first_unused();
418  }
419  --m_element;
420 
421  validate();
422  return *this;
423  }

568 /*

571  iterators.

578  using reference = value_type&;
579  using const_reference = value_type const&;

593  using cchunk_iterator_type = typename chunk_list_type::const_iterator;
594 
595  chunk_list_type m_chunks;
596 
597  // If StableEndIterator == true, this is the last nonempty chunk.
598  // If StableEndIterator == false, this is the first chunk with any empty slots.
599  chunk_iterator_type m_end_chunk;
600  size_t m_size;
601 
602  template<typename T2, typename Allocator, typename Traits>

609 
610  friend iterator begin( chunked_vector_base& ths ) { return ths.begin(); }
