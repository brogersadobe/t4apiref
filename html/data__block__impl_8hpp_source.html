38 /*
39  Truncate the data_block at the front. After this call get_bytes() will return a pointer
40  to the location in memory len bytes after its original value, and the data_block's length
41  will be reduced by the same amount. It is undefined behavior to provide a len larger than
42  the block's current length.
43 */
44 inline void data_block::truncate_front( size_t len )
45 {
46  assert( m_len >= len );
47  m_addr = add_bytes( m_addr, len );
48  m_len -= len;
49 }

51 /*
52  Truncate the data_block at the front. After this call get_bytes() return the same pointer as
53  before, but the data_block's length will be reduced by len. It is undefined behavior to provide
54  a len larger than the block's current length.
55 */
56 inline void data_block::truncate_back( size_t len )
57 {
58  assert( m_len >= len );
59  m_len -= len;
60 }
