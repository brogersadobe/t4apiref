216 template<typename element_type>
217 const_StackIterator<element_type>& const_StackIterator<element_type>::operator+=( ptrdiff_t distance )
218 {
219  if ( distance > 0 )
220  {
221  while ( distance >= m_chunk->m_chunkEnd - m_element )
222  {
223  tetraphilia_assert( m_chunk->m_nextChunk );
224  distance -= ( m_chunk->m_chunkEnd - m_element );
225  m_chunk = m_chunk->m_nextChunk;
226  m_element = m_chunk->m_chunkStart;
227  }
228  }
229  else
230  {
231  while ( -distance > m_element - m_chunk->m_chunkStart )
232  {
233  tetraphilia_assert( m_chunk->m_prevChunk != nullptr );
234  distance += m_element - m_chunk->m_chunkStart;
235  m_chunk = m_chunk->m_prevChunk;
236 
237  // note that this temporarily sets m_element pointing to an invalid
238  // memory location, but it will always be adjusted to a valid element
239  // by the time we return from this function (eventually by the
240  // += distance statement below)
241  m_element = m_chunk->m_chunkEnd;
242  }
243  }
244 
245  m_element += distance;
246 
247  return *this;
248 }

250 template<typename element_type>
251 ptrdiff_t const_StackIterator<element_type>::operator-( const const_StackIterator& rhs ) const
252 {
253  ptrdiff_t result = 0;
254  chunk_ptr_type chunk = m_chunk;
255  element_ptr_type element = m_element;
256  while ( chunk != rhs.m_chunk )
257  {
258  element_ptr_type cs = chunk->m_chunkStart; // work around VC7 bug
259  result += ( element - cs );
260  chunk = chunk->m_prevChunk;
261  element = chunk->m_chunkEnd;
262 
263  // see note above
264  tetraphilia_assert( chunk != NULL );
265  }
266 
267  result += element - rhs.m_element;
268  tetraphilia_assert( result >= 0 );
269  return result;
270 }

576  size_t GetCachedSize() const
577  {
578  size_t sz = sizeof (*this);
579 
580  if (!std::is_trivially_destructible<element_type>::value)
581  {
582  const_iterator it = begin();
583  while (it != end())
584  {
585  sz += t4::member_size(*it);
586  ++it;
587  }
588  }
589 
590  if (allocator_type::free_is_trivial)
591  return sz;
592 
593  // loop through the chunks
594  for (chunk_ptr_type chunk = m_firstChunk; chunk; chunk = chunk->m_nextChunk)
595  sz += GetChunkSize( *chunk );

597  }

747 void Stack<allocator_type, element_type>::Delete( const const_iterator& start, const const_iterator& end )
748 {
