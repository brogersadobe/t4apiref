79 // supporting class for holding a colorspace
80 template<typename client_type>
81 class GStateColorSpace
82 {
83  typedef content::DLByteStack::const_iterator byte_stack_iter_type;
84  typedef content::DLDataStack::const_iterator data_stack_iter_type;
85  typedef content::ContentRecord content_rec_type;
86  typedef pdfcolor::PDFColorSpace color_space_type;
87 public:
88  GStateColorSpace( void ) = default;
89 
90  void SetColorSpace(client_type& client, content::DLColorSpaceType csType, const byte_stack_iter_type *pName, content_rec_type *contentRecord)
91  {
92  tetraphilia_assert(!m_contentRecord || IsCSChanged(csType,pName,contentRecord));
93  // clear the previous CS and save the parameters
94  m_CS = {};
95  m_CSType = csType;
96  m_CSNamePtr = pName ? *pName : byte_stack_iter_type();
97  m_contentRecord = contentRecord;
98  }
99 
100  const color_space_type& GetColorSpace(const client_type& client) const
101  {
102  tetraphilia_assert(m_contentRecord);
103  if (!m_CS.HasColorSpace())
104  {
105  t4::string name;
106 
107  switch(m_CSType)
108  {
109  case content::kDLDeviceGray: name = "DeviceGray"; break;
110  case content::kDLDeviceRGB: name = "DeviceRGB"; break;
111  case content::kDLDeviceCMYK: name = "DeviceCMYK"; break;
112  case content::kDLNamedCS:
113  {
114  byte_stack_iter_type pName = m_CSNamePtr;
115  size_t len = *pName; ++pName;
116 
117  name.assign( pName, pName + static_cast<ptrdiff_t>( len ) );
118  break;
119  }
120  default:
121  break;
122  }
123 
124  m_CS = color_space_type::CreateFromName( name.c_str(),
125  m_contentRecord->GetResources().get_ptr(),
126  client.GetColorSpaceCache(),
127  false/*permitAbbreviations*/, true/*permitSpacesWithAlternates*/ );
128  }
129 
130  return m_CS;
131  }
132 
133  bool IsCSChanged(content::DLColorSpaceType csType, const byte_stack_iter_type *pName, content_rec_type *contentRecord) const
134  {
135  // if the name or content record changed force a lookup even if the name is the same
136  if (csType != m_CSType || m_contentRecord != contentRecord)
137  return true;
138 
139  if (csType == content::kDLNamedCS)
140  {
141  tetraphilia_assert(pName);
142  // types are both kDLNamedCS - check the names
143  byte_stack_iter_type i1 = *pName;
144  byte_stack_iter_type i2 = m_CSNamePtr;
145 
146  size_t lenToCheck = *i1+1;
147  while (lenToCheck--)
148  {
149  if (*i1 != *i2)
150  return true;
151  ++i1; ++i2;
152  }
153  }
154  // its the same
155  return false;
156  }
157 
158 private:
159 
160  // lazy evaluation of CS
161  mutable color_space_type m_CS;
162  content::DLColorSpaceType m_CSType;
163  byte_stack_iter_type m_CSNamePtr;
164  content_rec_type* m_contentRecord = nullptr;
165 };

201  void SetColor(const client_type& client, size_t nColorChannels, data_stack_iter_type& pColorValues, boost::optional<byte_stack_iter_type> const& pName)
202  {
203  // must set colorvalues or a name or both
204  if (nColorChannels == 0 && !pName)
205  throw t4::corrupt_data();
206 
207  m_HasColorValues = nColorChannels > 0;
208  m_useBlackInsteadOfCSDefault = false;
209 
210  m_nColorChannels = nColorChannels;
211  m_pColorValues = pColorValues;
212  m_pColorName = pName;
213  }
214 
215  void GetColor(const color_space_type& cs, const client_type& client, GStateExtractedColor& rslt) const
216  {
217  tetraphilia_assert(rslt.GetColor().empty());
218  tetraphilia_assert(rslt.GetColorName().empty());
219 
220  // WHEN SETTING A COLOR, name should be nonzero if and only if the current color space is a Pattern color space
221  // (however, keep in mind that a pattern color space, like all color spaces gets an initial color with name = 0
222  // which we use to indicate that the pattern should not mark per PDF-1.5 section 4.5.7 Table 4.21).
223  if ((cs.GetType() == pdfcolor::kPDFCSTPattern) != bool(m_pColorName))
224  throw t4::corrupt_data();
225 
226  if ( m_nColorChannels && m_nColorChannels < cs.GetNumSignals() )
227  throw t4::corrupt_data();
228 
229  if (m_nColorChannels)
230  {
231  if ( !m_pColorValues )
232  throw t4::corrupt_data();
233 
234  data_stack_iter_type pColorVals = *m_pColorValues;
235  // too many colors --> skip the trailing colors
236  if (m_nColorChannels > cs.GetNumSignals())
237  pColorVals += static_cast<ptrdiff_t>( m_nColorChannels-cs.GetNumSignals() );
238  // optimization - supply pointer into the DL if all color values are contiguous
239  data_stack_iter_type pLastColorVal = pColorVals + static_cast<ptrdiff_t>( cs.GetNumSignals() ) - 1;
240  if ((sizeof(t4::real_coord) == sizeof(data_stack_type)) &&
241  (pColorVals.operator ->()+cs.GetNumSignals()-1 == pLastColorVal.operator ->()))
242  {
243  auto it = reinterpret_cast<const t4::real_coord*>( pColorVals.operator->() );
244  rslt.m_color.assign( it, it + cs.GetNumSignals() );
245  }
246  else
247  {
248  rslt.m_color.resize( cs.GetNumSignals() );
249  for ( auto& e : rslt.m_color )
250  {
251  e = pColorVals->template Get<t4::real_coord>();
252  ++pColorVals;
253  }
254  }
255  }
256  else if (cs.GetNumSignals())
257  {
258  const t4::real_coord black = 0;
259  const t4::real_coord *initialColorVals = m_useBlackInsteadOfCSDefault ? &black: cs.GetInitialColor();
260 
261  // m_useBlackInsteadOfCSDefault --> cs.GetNumSignals() == 1
262  tetraphilia_assert(!m_useBlackInsteadOfCSDefault || cs.GetNumSignals() == 1);
263  rslt.m_color.resize( cs.GetNumSignals() );
264  ranges::copy( ranges::make_span( initialColorVals, initialColorVals + cs.GetNumSignals() ), rslt.m_color.begin() );
265  }
266 
267  if (m_pColorName)
268  {
269  byte_stack_iter_type pName = *m_pColorName;
270  size_t numBytesNotIncludingTerminatingNulCharacter = *pName; ++pName;
271  rslt.m_colorName.assign( pName, pName + static_cast<ptrdiff_t>( numBytesNotIncludingTerminatingNulCharacter ) );
272  }
273  }
274 
275 private:
276  bool m_HasColorValues; // only for the client
277  bool m_useBlackInsteadOfCSDefault;
278  size_t m_nColorChannels;
279  boost::optional<data_stack_iter_type> m_pColorValues;
280  boost::optional<byte_stack_iter_type> m_pColorName;
281 };
