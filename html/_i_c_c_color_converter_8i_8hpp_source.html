106 template <typename app_traits>
107 ICCStep1DTable<app_traits>::ICCStep1DTable( bool isInputStage,
108  const table_type & table,
109  std::uint8_t channel )
110  : m_channel( channel )
111 {
112  using namespace t4::literals;
113 
114  if( isInputStage )
115  m_table = table;
116  else
117  {
118  // Make a copy of the table:
119  m_table = t4::make_shared<ColorTable<app_traits>>( *table );
120 
121  // Invert the copy:
122  const size_t numEntries = m_table->GetNumEntries();
123  if( numEntries )
124  {
125  // Reverse the table:
126  if( m_table->operator[]( numEntries - 1 ) < m_table->operator[]( 0 ) )
127  {
128  size_t aIndex = 0;
129  size_t bIndex = numEntries - 1;
130  while( aIndex < numEntries / 2 )
131  {
132  boost::swap( m_table->operator[]( aIndex ), m_table->operator[]( bIndex ) );
133  ++aIndex;
134  --bIndex;
135  }
136  }
137 
138  // Force the table to be non-decreasing:
139  color_conv_type prevValue = m_table->operator[]( 0 );
140  std::uint32_t index;
141  for( index = 1; index < numEntries; ++index )
142  {
143  auto y = t4::max<color_conv_type>( prevValue, m_table->operator[]( index ) );
144  prevValue = m_table->operator[]( index ) = y;
145  }
146 
147  // Invert each entry:
148  std::uint32_t segment = 0;
149  color_conv_type y1 = m_table->operator[]( 0 );
150  color_conv_type y2 = m_table->operator[]( 1 );
151  color_conv_type x1 = 0;
152  color_conv_type xIncrement = 1_r / ( numEntries - 1 );
153  color_conv_type x2 = xIncrement;
154  const color_conv_type minValue = 1_r / 65536;
155  for( index = 0; index < numEntries; ++index )
156  {
157  color_conv_type y = index * xIncrement;
158  // Figure out which segment to use:
159  while( segment + 2 < numEntries && y >= m_table->operator[]( segment + 1 ) )
160  {
161  ++segment;
162  y1 = y2;
163  y2 = m_table->operator[]( segment + 1 );
164  x1 = x2;
165  x2 += xIncrement;
166  }
167  // Determine interpolated or extrapolated value:
168  color_conv_type dy = color_conv_type( numEntries ) *
169  t4::max<color_conv_type>( y2 - y1, minValue );
170  m_table->operator[]( index ) = x1 + ( ( y - y1 ) / dy );
171  }
172  }
173  }
174 }

181 {

556 {

567  // the color range and see which produces the darker result.
568

570  color_conv_type buf2[ 4 ];

573  {

575  case icSigRgbData:
576  case icSigXYZData:

578  buf1[ 1 ] = 0;

582  buf2[ 1 ] = 1;
583  buf2[ 2 ] = 1;
584  buf2[ 3 ] = 0;
585  break;

588  case icSigYCbCrData:
589  buf1[ 0 ] = 0;
590  buf1[ 1 ] = color_conv_type( 0.5 );
591  buf1[ 2 ] = color_conv_type( 0.5 );
592  buf1[ 3 ] = 0;
593  buf2[ 0 ] = 1;
594  buf2[ 1 ] = color_conv_type( 0.5 );
595  buf2[ 2 ] = color_conv_type( 0.5 );
596  buf2[ 3 ] = 0;
597  break;
598

600  buf1[ 0 ] = 1;
601  buf1[ 1 ] = 1;
602  buf1[ 2 ] = 1;
603  buf1[ 3 ] = 1;
604  buf2[ 0 ] = 0;
605  buf2[ 1 ] = 0;
606  buf2[ 2 ] = 0;
607  buf2[ 3 ] = 0;
608  break;
609 
610  default:
611  tetraphilia_assert_always(); // Shouldn't happen
612  return 0;
613  }

615  transform1->Convert( buf1 );
616  transform1->Convert( buf2 );
617  if( transform2 )
618  {
619  transform2->Convert( buf1 );
620  transform2->Convert( buf2 );
621  }

626  estimate = buf1;
627  else
628  estimate = buf2;

710 template <typename app_traits>
711 typename app_traits::color_traits::color_conv_type
712 ICCBlackPointHelper<app_traits,true>::EstimateDstBlack(
713  const conv_smart_ptr_type & transform1,
714  const conv_smart_ptr_type & transform2,
715  color_conv_type * blackLab,
716  RenderIntent renderIntent )
717 {
718  tetraphilia_assert( transform1->GetSrcMode() == icSigLabData );
719  tetraphilia_assert( transform1->GetDstMode() == transform2->GetSrcMode() );
720  tetraphilia_assert( transform2->GetDstMode() == icSigLabData );
721  tetraphilia_assert( blackLab[ 0 ] != 1 );
722 
723  // Estimate the color cast at L* of zero:
724  color_conv_type castA = ( blackLab[ 1 ] - color_conv_type( 0.5 ) ) / ( color_conv_type( 1 ) - blackLab[ 0 ] );
725  color_conv_type castB = ( blackLab[ 2 ] - color_conv_type( 0.5 ) ) / ( color_conv_type( 1 ) - blackLab[ 0 ] );
726  castA = Clamp<color_conv_type>( color_conv_type( -0.5 ), castA, color_conv_type( 0.5 ) );
727  castB = Clamp<color_conv_type>( color_conv_type( -0.5 ), castB, color_conv_type( 0.5 ) );
728 
729  // Convert a gray ramp from Lab to the target color space, and back to Lab:
730  t4::transient_snapshot ss( *t4::get_thread_heap() );
731  t4::transient_vector<color_conv_type> buffer( 4 * 256, *t4::get_thread_heap() );
732  color_conv_type * bufPtr = buffer.data();
733  unsigned int index;
734  for( index = 0; index < 256; ++index, bufPtr += 4 )
735  {
736  color_conv_type L = color_conv_type( index ) / 255;
737  *bufPtr = L;
738  bufPtr[ 1 ] = color_conv_type( 0.5 ) + ( castA * ( 1 - L ) );
739  bufPtr[ 2 ] = color_conv_type( 0.5 ) + ( castB * ( 1 - L ) );
740  bufPtr[ 3 ] = 0;
741  transform1->Convert( bufPtr );
742  transform2->Convert( bufPtr );
743  }
744 
745  // See how black and white round-tripped:
746  color_conv_type rtBlack = buffer[ 0 ];
747  color_conv_type rtWhite = buffer[ 4 * 255 ];
748 
749  // Handle the degenerate case:
750  if( rtWhite <= rtBlack )
751  return 0;
752 
753  // Some profiles may actually tip up after the black point, so we force
754  // the curve to be monotonically decreasing:
755  for( size_t index2 = 255; index2-- > 0; )
756  if( buffer[ 4 * index2 ] > buffer[ 4 * ( index2 + 1 ) ] )
757  buffer[ 4 * index2 ] = buffer[ 4 * ( index2 + 1 ) ];
758  rtBlack = buffer[ 0 ];
759 
760  // If the intent is relative colorimetric, see if most of the gray range
761  // is roundtripping well. If so, use the black point estimated from the
762  // source side of the profile:
763  if( renderIntent == kRelColorimetric )
764  {
765  // Don't test the bottom 20% of the gray ramp:
766  color_conv_type testLimit = rtBlack + ( color_conv_type( 0.2 ) * ( rtWhite - rtBlack ) );
767  bool goodRoundTrip = true;
768  for( index = 0; index < 256; ++index )
769  {
770  const color_conv_type L = static_cast<color_conv_type>( index ) / 255;
771  if( L > testLimit && AbsVal( L - buffer[ 4 * index ] ) > color_conv_type( 0.04 ) /*4 delta-Es*/ )
772  {
773  goodRoundTrip = false;
774  break;
775  }
776  }
777  if( goodRoundTrip )
778  return blackLab[ 0 ];
779  }
780 
781  // The round trip curve should look like a nearly constant section at the
782  // black point, with a corner and a nearly straight line to the white
783  // point. We need to estimate the length of the constant section to
784  // estimate the destination black point.
785  //
786  // The problem is that there is often noise in the constant section, and
787  // the corner is often rounded.
788  //
789  // We need to ignore any "toe" around the black point and estimate exactly
790  // where the extrapolated round trip curve would intersect the nearly
791  // constant black limited area.
792  //
793  // So we fit a least squares error quadradric curve though the shadow
794  // section of the curve, and use the point where this curve intersects the
795  // L*=0 round trip value as the destination black point.
796  //
797  // The code below works as follows:
798  //
799  // Y is a scaled value such that Y = 0 -> loopK (L* at X = 0)
800  // Y = 1 -> loopW (L* at X = 255)
801  //
802  // The shadow section is the part of the L* curve such that
803  // fitLowerLimit <= Y < fitUpperLimit.
804  //
805  // For points on the curve in the shadow section, a quadratic curve is
806  // fitted: Y = (A * X * X) + (B * X) + C.
807  //
808  // Then, we solve for X when Y = 0, i.e., the quadratic formula:
809  //
810  // 0 = (A * X * X) + (B * X) + C
811  // X = ((-B) +/- sqrt ((B * B) - (4 * A * C)) / (2 * A)
812  //
813  // Then we use X as the estimated Destination Black Point (scaling X to
814  // the proper range, i.e., 0 .. 100 in L*, or 0 .. 1 in floating point).
815 
816  color_conv_type fitLowerLimit; // Ignore toe near corner.
817  color_conv_type fitUpperLimit; // Only the shadow part of curve.
818 
819  // Perceptual and saturation intents often curve more by design, so we
820  // need to fit a lower part of the curve:
821  if( renderIntent == kRelColorimetric )
822  {
823  fitLowerLimit = color_conv_type(0.10);
824  fitUpperLimit = color_conv_type(0.50);
825  }
826  else
827  {
828  fitLowerLimit = color_conv_type(0.03);
829  fitUpperLimit = color_conv_type(0.25);
830  }
831 
832  unsigned int nPoints = 0; // Number of points for the curve fit
833 
834  color_conv_type sumX = 0; // Sum of X
835  color_conv_type sumX2 = 0; // Sum of X squared
836  color_conv_type sumX3 = 0; // Sum of X cubed
837  color_conv_type sumX4 = 0; // Sum of X to the 4th power
838  color_conv_type sumY = 0; // Sum of Y
839  color_conv_type sumYX = 0; // Sum of Y * X
840  color_conv_type sumYX2 = 0; // Sum of Y * X * X
841 
842  // This scale factor makes Y vary from 0..1 as L varies from black to
843  // white:
844  color_conv_type scaleY = 1 / ( rtWhite - rtBlack );
845 
846  for( index = 1; index <= 254; ++index )
847  {
848  const color_conv_type L = buffer[ 4 * index ];
849  const color_conv_type Y = scaleY * ( L - rtBlack );
850  if( Y >= fitLowerLimit && Y < fitUpperLimit )
851  {
852  ++nPoints;
853  const color_conv_type X = index * ( color_conv_type(1) / 255 );
854  const color_conv_type X2 = X * X;
855  const color_conv_type X3 = X * X2;
856  const color_conv_type YX = Y * X;
857  sumX += X;
858  sumX2 += X2;
859  sumX3 += X3;
860  sumX4 += X3 * X;
861  sumY += Y;
862  sumYX += YX;
863  sumYX2 += YX * X;
864  }
865  }
866 
867  // We need at least three points to fit the curve:
868  if( nPoints < 3 )
869  return 0;
870 
871  Matrix3by4<color_conv_type> matrix;
872  matrix( 0, 0 ) = sumX4;
873  matrix( 0, 1 ) = sumX3;
874  matrix( 0, 2 ) = sumX2;
875  matrix( 0, 3 ) = 0;
876  matrix( 1, 0 ) = sumX3;
877  matrix( 1, 1 ) = sumX2;
878  matrix( 1, 2 ) = sumX;
879  matrix( 1, 3 ) = 0;
880  matrix( 2, 0 ) = sumX2;
881  matrix( 2, 1 ) = sumX;
882  matrix( 2, 2 ) = static_cast<color_conv_type>(nPoints);
883  matrix( 2, 3 ) = 0;
884  if( !matrix.Invert() )
885  return 0;
886 
887  const color_conv_type A = matrix( 0, 0 ) * sumYX2 +
888  matrix( 0, 1 ) * sumYX +
889  matrix( 0, 2 ) * sumY;
890  const color_conv_type B = matrix( 1, 0 ) * sumYX2 +
891  matrix( 1, 1 ) * sumYX +
892  matrix( 1, 2 ) * sumY;
893  const color_conv_type C = matrix( 2, 0 ) * sumYX2 +
894  matrix( 2, 1 ) * sumYX +
895  matrix( 2, 2 ) * sumY;
896  color_conv_type result;
897 
898  if( AbsVal( A ) <= limits<color_conv_type>::epsilon() )
899  {
900  if( B <= limits<color_conv_type>::epsilon() )
901  return 0;
902  result = -C / B;
903  }
904  else
905  {
906  const color_conv_type D = B * B - 4 * A * C;
907  if( D <= 0 )
908  return 0;
909  result = ( -B + real_services::Sqrt( D ) ) / ( 2 * A );
910  }
911 
912  if( result <= 0 )
913  result = 0;
914  else if( result > color_conv_type(0.5) )
915  result = color_conv_type(0.5);
916 
917  return result;
918 }

974 template <typename app_traits>
975 ICCColorConverterImpl<app_traits>::ICCColorConverterImpl(
976  const ICCColorSpace<app_traits> & srcCS,
977  const ICCColorSpace<app_traits> & dstCS,
978  RenderIntent renderIntent,
979  ConverterCachingType cachingType,
980  bool linear,
981  bool mapBlack)
982  : m_srcMode( srcCS.m_colorSpace ),
983  m_dstMode( dstCS.m_colorSpace )
984 {
985  typedef typename app_traits::color_traits color_traits;
986  tetraphilia_assert( srcCS.GetNumSignals() <= 4 && dstCS.GetNumSignals() <= 4 );
987 
988  UpdateCachingType(cachingType,srcCS.GetNumSignals(), dstCS.GetNumSignals() );
989 
990  // Determine whether we need to do white point mapping:
991  XYZColor<color_conv_type> srcWhite; // Default ctor zeroes color values
992  XYZColor<color_conv_type> dstWhite = srcWhite;
993  bool mapWhite = ( renderIntent == kAbsColorimetric );
994  if( mapWhite )
995  {
996  srcCS.GetMediaWhitePoint( srcWhite );
997  dstCS.GetMediaWhitePoint( dstWhite );
998  mapWhite = srcWhite != dstWhite;
999  }
1000 
1001  // Determine whether we need to do black point mapping:
1002  color_conv_type srcBlack = 0;
1003  color_conv_type dstBlack = 0;
1004  if( mapBlack )
1005  {
1006  typedef ICCBlackPointHelper<app_traits,color_traits::mapBlackPoint> bp_helper;
1007  try
1008  {
1009  srcBlack = bp_helper::EstimateSrcBlack( srcCS, renderIntent );
1010  dstBlack = bp_helper::EstimateDstBlack( dstCS, renderIntent );
1011  mapBlack = srcBlack != dstBlack;
1012  }
1013  catch( const std::bad_alloc& )
1014  {
1015  mapBlack = false;
1016  }
1017  }
1018 
1019  // Create the stage that converts the source color space to the PCS:
1020  AppendIOStage( true /* =input */, srcCS, renderIntent, linear );
1021 
1022  // Handle absolute colorimetric connections:
1023  if( mapWhite )
1024  {
1025  if( srcCS.m_pcs == icSigLabData )
1026  AppendStep( t4::make_shared<ICCStepLabToXYZ<app_traits>>( srcCS.m_illuminant ) );
1027  if( srcCS.m_illuminant != srcWhite )
1028  AppendWhitePointStage( srcCS.m_illuminant, srcWhite );
1029  if( dstCS.m_illuminant != dstWhite )
1030  AppendWhitePointStage( dstWhite, dstCS.m_illuminant );
1031  if( dstCS.m_pcs == icSigLabData )
1032  AppendStep( t4::make_shared<ICCStepXYZToLab<app_traits>>( dstCS.m_illuminant ) );
1033  }
1034  else // Handle non-absolute cases:
1035  {
1036  // The first test is redundant, but allows code-stripping when the feature is disabled:
1037  if( color_traits::mapBlackPoint && mapBlack )
1038  {
1039  if( srcCS.m_pcs == icSigLabData )
1040  AppendStep( t4::make_shared<ICCStepLabToXYZ<app_traits>>( srcCS.m_illuminant ) );
1041  color_conv_type k1 = 1 - DecodeLabGamma( srcBlack * 100 );
1042  color_conv_type k2 = 1 - DecodeLabGamma( dstBlack * 100 );
1043  if( real_services::DivWillOverflow( k2, k1 ) )
1044  throw t4::corrupt_data();
1045  color_conv_type scale = k2 / k1;
1046  Matrix3by4<color_conv_type> matrix;
1047  matrix.Zero();
1048  for( unsigned int index = 0; index < 3; ++index )
1049  {
1050  matrix( index, index ) = scale;
1051  matrix( index, 3 ) = 1 - scale;
1052  }
1053  AppendStep( t4::make_shared<ICCStepMatrix3by4<app_traits>>( matrix ) );
1054  if( srcCS.m_pcs == icSigLabData )
1055  AppendStep( t4::make_shared<ICCStepXYZToLab<app_traits>>( dstCS.m_illuminant ) );
1056  }
1057 
1058  // To connect, XYZ connection spaces need to have exactly the same
1059  // white point:
1060  if( srcCS.m_pcs == icSigXYZData &&
1061  dstCS.m_pcs == icSigXYZData &&
1062  srcCS.m_illuminant != dstCS.m_illuminant )
1063  AppendWhitePointStage( srcCS.m_illuminant, dstCS.m_illuminant );
1064  else if( srcCS.m_pcs == icSigXYZData &&
1065  dstCS.m_pcs == icSigLabData )
1066  AppendStep( t4::make_shared<ICCStepXYZToLab<app_traits>>( srcCS.m_illuminant ) );
1067  else if( srcCS.m_pcs == icSigLabData &&
1068  dstCS.m_pcs == icSigXYZData )
1069  AppendStep( t4::make_shared<ICCStepLabToXYZ<app_traits>>( dstCS.m_illuminant ) );
1070  }
1071 
1072  // Optimize the steps we've created so far:
1073  OptimizeSteps( m_firstStep, m_lastStep );
1074 
1075  // Create the stage that converts the PCS to the destination color space,
1076  // or fetch if from the cache on the destination color space:
1077  if( dstCS.HasCachedOutputStage( renderIntent ) )
1078  {
1079  AppendStep( dstCS.GetCachedOutputStage() );
1080  m_unlinearize = dstCS.GetCachedOutputUnlinearizeStage();
1081  m_linearize = dstCS.GetCachedOutputLinearizeStage();
1082  }
1083  else
1084  {
1085  engine_step_ptr bookmark = m_lastStep;
1086  AppendIOStage( false /* =output */, dstCS, renderIntent, linear );
1087  OptimizeSteps( bookmark->m_next, m_lastStep );
1088  dstCS.CacheOutputStage( renderIntent, bookmark->m_next, m_unlinearize, m_linearize );
1089  }
1090 }

1097 template <typename app_traits>
1098 void
1099 ICCColorConverterImpl<app_traits>::OptimizeSteps(
1100  engine_step_ptr & start,
1101  engine_step_ptr & end )
1102 {
1103  engine_step_ptr step = start;
1104  while( step.get() && step->DoesNothing() )
1105  step = step->m_next;
1106  start = step;
1107  if ( !step.get() )
1108  return;
1109  while( 1 )
1110  {
1111  // Remove steps that do nothing:
1112  while( step->m_next.get() && step->m_next->DoesNothing() )
1113  step->m_next = step->m_next->m_next;
1114 
1115  // Compose adjacent matrices:
1116  while( step->GetType() == kICCStepMatrix3by4 &&
1117  step->m_next.get() && step->m_next->GetType() == kICCStepMatrix3by4 )
1118  {
1119  ICCStepMatrix3by4<app_traits> & stepMatrix =
1120  static_cast<ICCStepMatrix3by4<app_traits>&>( *step );
1121  ICCStepMatrix3by4<app_traits> & stepNextMatrix =
1122  static_cast<ICCStepMatrix3by4<app_traits>&>( *step->m_next );
1123  if( stepMatrix.CombineWith( stepNextMatrix ) )
1124  step->m_next = step->m_next->m_next;
1125  else
1126  break;
1127  }
1128 
1129  // Eliminate symmetric pairs:
1130  if( step->m_next.get() && step->m_next->m_next.get() &&
1131  ( ( step->m_next->GetType() == kICCStepLabToXYZ &&
1132  step->m_next->m_next->GetType() == kICCStepXYZToLab ) ||
1133  ( step->m_next->GetType() == kICCStepXYZToLab &&
1134  step->m_next->m_next->GetType() == kICCStepLabToXYZ ) ) )
1135  step->m_next = step->m_next->m_next->m_next;
1136 
1137  if( step->m_next.get() )
1138  step = step->m_next;
1139  else
1140  {
1141  end = step;
1142  return;
1143  }
1144  }
1145 }
