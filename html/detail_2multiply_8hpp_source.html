30 template<typename Result, typename T1, typename T2, typename = void>
31 struct multiply_impl_helper
32 {
33  /*
34  This is the fully-general version of multiply that works for all combinations of integer types.
35  */
36  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
37  {
38  constexpr size_t sbr = significant_bit_count_in_type<Result>;
39 
40  // Pure compile-time test: if the result is signed and the most significant possible result
41  // still fits inside the result type, there is nothing to check. The size check is complicated
42  // by the fact that the product of two negative powers of 2 require sbx + sby + 1 bits.
43  if ( ( is_signed<Result>::value )
44  && ( sbr >= ( significant_bit_count_in_type<T1> + significant_bit_count_in_type<T2> + ( ( is_signed<T1>::value && is_signed<T2>::value ) ? 1 : 0 ) ) ) )
45  {
46  return static_cast<Result>( x ) * static_cast<Result>( y );
47  }
48 
49  if ( x == 0 || y == 0 )
50  return 0;
51 
52  // Compilers should also be able to simplify most tests below at compile time for unsigned result
53  const bool is_negative_x = is_negative( x );
54  const bool negative_result = is_negative_x != is_negative( y );
55  if ( is_unsigned<Result>::value && negative_result )
56  return {};
57 
58  using up_t1 = typename std::make_unsigned<T1>::type;
59  using up_t2 = typename std::make_unsigned<T2>::type;
60  using up_result = typename std::make_unsigned<Result>::type;
61 
62  // If result is positive we can treat both operands as unsigned
63  if ( !negative_result )
64  return multiply_impl_helper<Result, up_t1, up_t2>()( sm_detail::abs( x ), sm_detail::abs( y ) );
65 
66  // In case of negative result we still can piggyback on unsigned version and negate result in the end
67  auto maybe_safe_result = multiply<boost::optional<up_result>>( static_cast<up_t1>( abs( x ) ), static_cast<up_t2>( abs( y ) ) );
68  if ( maybe_safe_result )
69  {
70  if ( !sm_detail::high_bit_set( *maybe_safe_result ) )
71  return static_cast<Result>( 0 - *maybe_safe_result );
72 
73  // Corner case: the only way to not overflow if high bit set
74  // is if result is negative power of 2
75  else if ( static_cast<up_result>( *maybe_safe_result << 1 ) == 0 )
76  return std::numeric_limits<Result>::min();
77  }
78 
79  return {};
80  }
81 };

83 template<typename Result, typename T1, typename T2>
84 struct multiply_impl_helper<Result, T1, T2, enable_if_unsigned<T1, T2>>
85 {
86  /*
87  This is the real meat of the implementation of overflow-checked multiply. It specialized for
88  both operands being unsigned, which simplifies checks a lot since we don't need to worry about
89  sign bit. Calls to complicated cases of signed types also go through this implementation after
90  operands casted to unsigned types.
91 
92  Additional specializations can be written pretty easily; simply write an explicit or partial
93  specialization on the multiply template class. For an explicit (full) specialization,
94  simply use the desired parameter types directly; for partial specialization use enable_if
95  as the fourth template parameter.
96 
97  Note that the C++ spec explicitly says that the result of any integral arithmetic operation
98  whose mathematically correct result is not expressible in the type selected by integral type
99  promotion is undefined. This means that we cannot look at the result of an operation and make any
100  useful assessment as to whether or not it overflowed.
101  */
102  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
103  {
104  // Pure compile-time test: if most significant possible result still fits inside the result type,
105  // there is nothing to check
106  constexpr size_t sbr = significant_bit_count_in_type<Result>;
107 
108  if ( sbr >= ( significant_bit_count_in_type<T1> + significant_bit_count_in_type<T2> ) )
109  return static_cast<Result>( x ) * static_cast<Result>( y );
110 
111  if ( x == 0 || y == 0 )
112  return 0;
113 
114  size_t sbx = significant_bit_count( x );
115  size_t sby = significant_bit_count( y );
116 
117  // There are 2 cases left to consider, either:
118  // 1. Most common, sbr clearly indicates that result type can fit the product.
119  // 2. Product is on "the edge" of overflowing, in such case either of following must hold true
120  // - one of operands is 1, it means that result will be equal to the other operand
121  // - the second operand is smaller than the value that would give max result
122  // if multiplied by first operand
123  if ( ( sbr >= ( sbx + sby ) )
124  || ( ( sbr == ( sbx + sby - 1 ) )
125  && ( ( x == 1 ) || ( y == 1 )
126  || ( std::numeric_limits<Result>::max() / x ) >= y ) ) )
127  {
128  return static_cast<Result>( x ) * static_cast<Result>( y );
129  }
130 
131  return {};
132  }
133 };
