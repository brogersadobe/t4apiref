496  static unsigned char subtract( unsigned char p, unsigned char q )
497  {

508  static unsigned char multiply( unsigned char p, unsigned char q )
509  {
510  int temp = static_cast<int>( p ) * static_cast<int>( q ) + 128;
511  return static_cast<unsigned char>((temp + (temp >> 8)) >> 8);
512  }
513 
514  static unsigned char divide( unsigned char p, unsigned char q )
515  {
516  if ( q == 0 )
517  {
518  tetraphilia_assert( p == 0 );
519  return 0;
520  }
521 
522  int temp = ( 255 * p + ( q >> 1 ) ) / q;
523  return static_cast<unsigned char>( temp );
524  }
525 
526  static unsigned char union_( unsigned char p, unsigned char q )
527  {
528  return 0xFF - multiply( 0xFF - p, 0xFF - q );
529  }

531  static unsigned char inverted_premultiply( unsigned char a, unsigned char b, unsigned char c, unsigned char d )
532  {
533  tetraphilia_assert( a >= b && c >= d );
534  int temp = a * d + b * ( c - d ) + 128;
535  return subtractOneAndDivideBy255( temp );
536  }

538  TETRAPHILIA_FORCE_INLINE static unsigned char lerp( unsigned char a, unsigned char b, unsigned char c )
539  {
540  int temp = times255( b ) + a * ( c - b ) + 128;
541  return subtractOneAndDivideBy255( temp );
542  }
543 
544  TETRAPHILIA_FORCE_INLINE static unsigned char over( unsigned char a, unsigned char b, unsigned char ac )
545  {
546  int temp = ( 0xFF - a ) * b + times255( ac ) + 128;
547  return subtractOneAndDivideBy255( temp );
548  }

550  static unsigned char ko_over( unsigned char f, unsigned char ciminus1, unsigned char cb, unsigned char c )
551  {

553  return subtractOneAndDivideBy255_pinned( temp );
554  }

556  static unsigned char remove_backdrop( unsigned char a, unsigned char b, unsigned char c )

566  static unsigned char subtractOneAndDivideBy255( int x )
567  {
568  int tmp = ( x + ( x >> 8 ) ) >> 8;
