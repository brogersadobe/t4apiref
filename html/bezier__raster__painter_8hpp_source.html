172 template<typename _sampler_traits>
173 void BezierRasterPainter<_sampler_traits>::SetYImpl( int_type y )
174 {
175  m_y = y;
176 
177  const int_rect_type& nzb = m_bezierPathSampler.GetNonZeroBounds();
178  m_insideSamplersNonzeroYBounds = ( y >= get_ymin(nzb) && y < get_ymax(nzb) );
179  if ( m_insideSamplersNonzeroYBounds ) {
180  m_bezierPathSampler.NextScanLine( m_y );
181 
182  #ifdef PROFILE_BEZIER_REDUCER
183  sampler_traits::app_traits::context_type::high_precision_time_type start_conversion_time = sampler_traits::app_traits::context_type::current_time();
184  #endif
185 
186  // convert pixel values from real_type to signal_type
187  int_type xMax = get_xmax(m_constraints.m_interestBounds);
188  int_type xSegStart = m_bezierPathSampler.GetSegmentStop(); // initial segment stop should be end of initial run of zeros
189  int_type xSegEnd;
190  const real_type* srcOrigin = m_samplerPixelBuffer;
191  signal_type* dstOrigin = m_pixelStore.GetPixelBuffer().GetPixels() - get_xmin(m_pixelStore.GetBounds());
192  while (xSegStart < xMax) {
193  m_bezierPathSampler.SetX( xSegStart );
194  xSegEnd = m_bezierPathSampler.GetSegmentStop();
195  tetraphilia::imaging_model::ReducerSegmentType type = m_bezierPathSampler.GetCurrentSegmentType();
196  if ( type == kPixels ) {
197  auto start = t4::max<int_type>( xSegStart, get_xmin(m_constraints.m_interestBounds) );
198  auto end = t4::min<int_type>( xSegEnd, get_xmax(m_constraints.m_interestBounds) );
199  const real_type* src = &srcOrigin[ start ];
200  signal_type* dst = &dstOrigin[ start ];
201  for ( ; start < end; ++start ) {
202  real_type r = *src++;
203  tetraphilia_assert( (start >= get_xmin(nzb) && start < get_xmax(nzb)) || r < real_type(0.001) );
204  signal_type sig = signal_traits::real_to_signal( r );
205  *dst++ = sig;
206  }
207  }
208  xSegStart = xSegEnd;
209  }
210  #ifdef PROFILE_BEZIER_REDUCER
211  sampler_traits::app_traits::context_type::high_precision_time_type end_conversion_time = sampler_traits::app_traits::context_type::current_time();
212  m_time_spent_converting_to_char_data += end_conversion_time - start_conversion_time;
213  ++m_num_conversions_to_char_data;
214  #endif
215  }
216 }
