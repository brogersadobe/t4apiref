43 template <typename app_traits, typename search_client_type>
44 void FindAllOccurrenceOnPage(
45  store::Store<app_traits>& store,
46  document::DocumentCaches<app_traits>& doccaches,
47  search_client_type& searchClient,
48  const TextMatcher& textMatcher,
49  const document::PageID& pageID )
50 {
51  auto& teContext = textMatcher.GetTextExtractContext();
52  auto& doc = store.GetDocument();
53  using t4::pdf::content_range;
54 
55  t4::transient_heap transientHeap( 4096, 1024 );
56  TextMatcherState textMatcherState(transientHeap);
57 
58  t4::transient_snapshot ss( transientHeap );
59  if (searchClient.PageIsAuthorized(pageID))
60  {
61  SearchMarkHandler<search_client_type> markHandler(
62  teContext.GetUnicodeCategorizer(), searchClient, textMatcher, textMatcherState );
63  namespace tparams = t4::pdf::text_extract_params;
64  auto params = t4::pdf::text_extract_parameters_desc();
65  params += t4::parameter<tparams::unicode_normalizer>( teContext.GetUnicodeNormalizer() );
66  params += t4::parameter<tparams::unicode_categorizer>( teContext.GetUnicodeCategorizer() );
67  t4::pdf::enumerate_reordered_content( doc, content_range::MakeContentRangeForPage( pageID ), t4::pdf::content_order::inferred, markHandler, params );
68 
69  // See if we are at the end of the doc, if so special case
70  if (t4::pdf::get_page_num(doc, pageID) == t4::pdf::count_num_pages(store) - 1)
71  {
72  // need to add a word break at the end of the doc, and give the matcher a chance to find it
73  textMatcher.SubmitWordBreak(textMatcherState);
74  textMatcher.RequestStopSoon_CheckStopped(textMatcherState, searchClient);
75  }
76  else
77  {
78  // tell the text matcher we would like to stop at the end of this page, see if we can
79  textMatcher.RequestStopSoon_CheckStopped(textMatcherState, searchClient);
80  }
81  }
82 
83  searchClient.HandleSearchDone();
84 }

141 template <typename app_traits, typename search_client_type>
142 void FindAllOccurrenceInRange(
143  store::Store<app_traits>& store,
144  document::DocumentCaches<app_traits>& doccaches,
145  search_client_type& searchClient,
146  const TextMatcher& textMatcher,
147  const content::ContentRange& range)
148 {
149  auto firstPage = t4::pdf::get_page_num( store.GetDocument(), range.m_from.m_pageID );
150  auto lastPage = t4::pdf::get_page_num( store.GetDocument(), range.m_to.m_pageID );
151 
152  // Adjust last page if it's pointing to the beginning of the page range.
153  if ( range.m_to == content::ContentPoint::MakeContentPointAtStartOfPage( range.m_to.m_pageID ) )
154  --lastPage;
155 
156  const bool multiThreading = TETRAPHILIA_IsMultiThreadingEnabled();
157  boost::optional<t4::thread_pool> tp;
158  if (multiThreading)
159  tp.emplace();
160 
161  t4::synchronized<t4::exception_ptr> exc;
162 
163  // Queue search for all pages
164  auto futures = t4::make_transient_vector<t4::future<void>>( *t4::get_thread_heap() );
165  for (auto page = firstPage; page <= lastPage; ++page)
166  {
167  auto pageID = t4::pdf::get_page_id( store.GetDocument(), page );
168  auto task = [pageID, &store, &searchClient, range, &doccaches, &textMatcher, &exc]()
169  {
170  FPUControl<float> fpuControl;
171  t4::establish_thread_heap eth;
172 
173  try
174  {
175  typedef te_detail::FindAllOnPageSearchClient<app_traits, search_client_type> fop_search_client_type;
176 
177  fop_search_client_type fopsc(searchClient, pageID, range);
178  te_detail::FindAllOccurrenceOnPage<app_traits, fop_search_client_type> (store, doccaches, fopsc, textMatcher, pageID);
179  }
180  catch( ... )
181  {
182  // If multiple pages fail, we might overwrite this, but it doesn't really matter
183  // as long as _something_ is thrown to signal failure
184  *exc = t4::current_exception();
185  }
186  };
187 
188  if ( multiThreading )
189  futures.push_back( tp->schedule( task ) );
190  else
191  task();
192  }
193 
194  if ( multiThreading )
195  {
196  t4::context_unlock unlock;
197  for ( auto& f : futures )
198  f.wait();
199  }
200  // Refer Bug #3792529: Following code snippet has been disabled to enable search occurrences to be displayed from the pages where exception did not happen
201 // if ( *exc != t4::exception_ptr() )
202 // t4::rethrow_exception( *exc );
203 };
