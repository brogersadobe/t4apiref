163 template<typename app_traits>
164 PageNodeType GetTypeFromPageNodeDict( const store::Dictionary<store::StoreObjTraits<app_traits> >& dict )
165 {
166  typedef store::StoreObjTraits<app_traits> object_traits;
167  PageNodeType node_type = PageNodeType::invalid;
168 
169  /*
170  PDF Slang
171  PDFRM section 3.6.2 describes the structure of a pdf file's page tree.
172  At the level this code is working on, "dict" must either be a Page Tree Node
173  or a Page Object. Both of these are required to have a /Type entry that is
174  set to either "Pages" for a Page Tree Node or "Page" for a Page Object.
175  Watson bug#1243485 contains a file with the /Type entry missing in (at least)
176  the root Page Tree Node. In permissive mode we attempt to distinguish the
177  two cases by looking for a /Count entry in the dictionary if the /Type is missing.
178  /Count is required for Page Tree Nodes, and is not meaningful for Page Objects.
179  */
180  #ifdef TETRAPHILIA_PDF_PERMISSIVE
181  {
182  store::Object<object_traits> typeObj = dict.Get( "Type" );
183  if ( typeObj.Type() == store::kObjectTypeNull ) {
184  // Make an intelligent guess based on the presence or absence of
185  // the /Count key
186  if ( dict.Get( "Count" ).Type() == store::kObjectTypeNull )
187  node_type = PageNodeType::page;
188  else
189  node_type = PageNodeType::pages;
190  } else {
191  auto type = typeObj.NameValue();
192  if ( type == "Pages" )
193  node_type = PageNodeType::pages;
194  else if ( type == "Page" )
195  node_type = PageNodeType::page;
196  }
197  }
198  #else
199  {
200  auto type = dict.GetRequiredName( "Type" );
201  if ( type == "Pages" )
202  node_type = PageNodeType::pages;
203  else if ( type == "Page" )
204  node_type = PageNodeType::page;
205  }
206  #endif /* TETRAPHILIA_PDF_PERMISSIVE */
207 
208  return node_type;
209 }

218 template<typename F>
219 bool ForEachPageHelper( t4::pdf::dictionary parent, t4::pdf::page_num start, t4::pdf::page_num stop, t4::pdf::page_num num_visited, F&& f )
220 {
221  using namespace t3::pdf::store;
222 
223  t4::check_stack_depth();
224 
225  bool do_tail_recursion;
226  do
227  {
228  do_tail_recursion = false;
229  auto cur_page = start;
230  auto array = parent.Get( "Kids" ).ArrayValue();
231  auto it = array.begin_noresolve();
232  auto end = array.end_noresolve();
233  while ( it != end && ( cur_page <= stop ) )
234  {
235  auto ref = it->ReferenceValue();
236  auto dict = GetStore( parent ).ResolveReference( ref ).DictionaryValue();
237  auto node_type = GetTypeFromPageNodeDict(dict);
238 
239  if ( node_type == PageNodeType::pages )
240  {
241  auto count = dict.GetRequiredInteger( "Count" );
242  auto last_page = num_visited + count - 1;
243 
244  // Walk down the subtree only if the start page lies within its page range.
245  if ( num_visited <= cur_page && cur_page <= last_page )
246  {
247  if ( stop < last_page )
248  last_page = stop;
249  /*
250  Watson #3587126 contains a file with a pathological page "tree"--every node in the page tree
251  consists of two nodes: a page node followed by a page tree containing all the subsequent
252  pages. This causes our code to throw a stack-too-deep exception after about 330 pages or
253  so.
254 
255  Unfortunately this is real-world PDF (produced by something called "ImagePDF"), but fortunately
256  there's a pretty easy workaround; as long as the page tree node is the last one in the array
257  we can do tail recursion.
258  */
259  if ( it + 1 == end )
260  {
261  do_tail_recursion = true;
262  parent = dict;
263  start = cur_page;
264  stop = last_page;
265  }
266  else
267  {
268  auto continue_enumeration = ForEachPageHelper( dict, cur_page, last_page, num_visited, std::forward<F>( f ) );
269  if ( continue_enumeration )
270  {
271  num_visited = last_page + 1;
272  cur_page = num_visited;
273  }
274  else
275  {
276  return false;
277  }
278  }
279  }
280  else
281  {
282  num_visited += count;
283  }
284  }
285  else if ( node_type == PageNodeType::page )
286  {
287  if ( cur_page == num_visited )
288  {
289  auto continue_enumeration = f( cur_page, dict );
290  if ( continue_enumeration )
291  ++cur_page;
292  else
293  return false;
294  }
295  ++num_visited;
296  }
297  ++it;
298  }
299  }
300  while ( do_tail_recursion );
301  return true;
302 }

305 template<typename F>
306 void ForEachPage( t4::pdf::document& doc, t4::pdf::page_num start, t4::pdf::page_num stop, F&& f )
307 {
308  auto& store = doc.get_store();
309  auto linearization_dict = store.GetLinearizationDict();
310  t4::pdf::page_num cur_page = start;
311  if ( linearization_dict )
312  {
313  for ( ; cur_page <= stop; ++cur_page )
314  {
315  auto ref = GetPageRefFromHintTable( store, cur_page );
316 
317  // Watson bug#1297254
318  // in permissive builds -1 is returned if the hint_table is damaged
319  #ifdef TETRAPHILIA_PDF_PERMISSIVE
320  if ( ref.m_objNum >= 0 )
321  #endif /* TETRAPHILIA_PDF_PERMISSIVE */
322  {
323  auto dict = store.ResolveReference( ref ).DictionaryValue();
324  f( cur_page, dict );
325  }
326  #ifdef TETRAPHILIA_PDF_PERMISSIVE
327  else
328  break;
329  #endif /* TETRAPHILIA_PDF_PERMISSIVE */
330  }
331  if ( cur_page == stop + 1 )
332  return;
333  }
334  ForEachPageHelper( store.GetRoot().GetRequiredDictionary( "Pages" ), cur_page, stop, 0,
335  [&](t4::pdf::page_num pg, t4::pdf::dictionary page_dict)
336  {
337  f( pg, page_dict );
338  return true;
339  });
340 }

342 template<typename app_traits>
343 store::Reference GetPageRefFromPageTree( const store::Dictionary<store::StoreObjTraits<app_traits> >& parent,
344  PageIndex pageIndex )
345 {
346  typedef store::StoreObjTraits<app_traits> object_traits;
347  typedef store::Array<object_traits> array_type;
348  typedef store::Reference reference_type;
349  typedef store::Dictionary<object_traits> dict_type;
350 
351  typedef typename array_type::const_iterator_noresolve_type iterator_type;
352 
353  t4::check_stack_depth();
354 
355  array_type array = parent.Get( "Kids" ).ArrayValue();
356  iterator_type it = array.begin_noresolve();
357  iterator_type end = array.end_noresolve();
358 
359  while ( it != end )
360  {
361  reference_type ref = it->ReferenceValue();
362  dict_type dict = GetStore( parent ).ResolveReference( ref ).DictionaryValue();
363  auto node_type = GetTypeFromPageNodeDict(dict);
364 
365  if ( node_type == PageNodeType::pages )
366  {
367  std::int32_t count = dict.GetRequiredInteger( "Count" );
368  if ( count > pageIndex )
369  return GetPageRefFromPageTree( dict, pageIndex );
370  else
371  pageIndex -= count;
372  }
373  else if ( node_type == PageNodeType::page )
374  {
375  if ( pageIndex == 0 )
376  return ref;
377  else
378  pageIndex--;
379  }
380  ++it;
381  }
382 
383  throw t4::corrupt_data();
384 }

396 template<typename app_traits>
397 t4::app_traits::pdf_traits::int_type CountSiblingsBefore( store::Store<app_traits>& store,
398  const store::Dictionary<store::StoreObjTraits<app_traits> >& parent,
399  const store::Reference& childRef )
400 {
401  typedef t4::app_traits::pdf_traits::int_type page_index_type;
402 
403  auto arr = parent.GetRequiredArray( "Kids" );
404  auto it = arr.begin_noresolve();
405  auto end = arr.end_noresolve();
406 
407  page_index_type result = 0;
408 
409  while ( it != end )
410  {
411  if ( childRef == it->ReferenceValue() )
412  return result;
413 
414  auto dict = store.ResolveReference( it->ReferenceValue_NoTypeCheck() ).DictionaryValue();
415 
416  auto node_type = GetTypeFromPageNodeDict(dict);
417 
418  if( node_type != PageNodeType::invalid )
419  result += dict.GetInteger( "Count", 1 );
420  ++it;
421  }
422 
423  throw t4::corrupt_data();
424 }

426 template<typename app_traits>
427 PageIndex GetPageIndexFromPageTree( store::Store<app_traits>& store,
428  const t4::pdf::reference& ref )
429 {
430  typedef t4::pdf::dictionary dict_type;
431  typedef t4::pdf::reference reference_type;
432  typedef t4::app_traits::pdf_traits::int_type page_index_type;
433 
434  reference_type childRef = ref;
435  page_index_type result = 0;
436  boost::optional<reference_type> parentRef = store.ResolveReference( ref ).DictionaryValue().GetReference( "Parent" );
437 
438  IterativeRecursionChecker checker( 1000 );
439  while ( parentRef )
440  {
441  checker.Bump();
442  dict_type parentDict = store.ResolveReference( *parentRef ).DictionaryValue();
443  result += CountSiblingsBefore( store, parentDict, childRef );
444  childRef = *parentRef;
445  parentRef = parentDict.GetReference( "Parent" );
446  }
447 
448  return result;
449 }

632 template<typename object_traits>
633 t4::data_block_stream GetPageContentStream( const store::Dictionary<object_traits>& pageDict )
634 {
635  /*
636  PDF slang
637 
638  Section 3.3 of the PDFRM discusses the legal names of the supported
639  compression filters. Section 4.8.6 discusses inline images and in that
640  discussion lists some abbreviations for a variety of things (including
641  compression filter names) that are allowed for inline images. But the
642  spec says that these abbreviations are only allowed for inline images.
643 
644  Watson bug#1256307 contains a file where the content stream of the first
645  page is compressed with Flate, but the name given is "Fl", which is the
646  abbreviation allowed for inline images. The name should have been "FlateDecode".
647  Desktop acrobat does not complain about this file, so in permissive builds
648  we also do not complain. That is accomplished by telling GetFilteredStream
649  that we are looking for inline image data, which tells it that it is ok
650  to use the abbreviated names.
651  */
652  #ifdef TETRAPHILIA_PDF_PERMISSIVE
653  bool isInlineImage = true;
654  #else
655  bool isInlineImage = false;
656  #endif
657 
658  store::Object<object_traits> contents = pageDict.Get( "Contents" );
659 
660  store::ObjectType type = contents.Type();
661  if ( type == store::kObjectTypeDictionary )
662  return store::GetFilteredStream( contents.DictionaryValue_NoTypeCheck(), isInlineImage );
663  if ( type == store::kObjectTypeNull ) {
664  // The PDF spec explicitly states that the absence of a /Contents
665  // key is legal and should be treated as an empty page. Generate
666  // an empty stream here.
667  auto raw = t4::make_memory_data_store_no_copy( nullptr, 0 );
668  return t4::make_data_block_stream_from_data_store( std::move( raw ) );
669  }
670  else if ( type != store::kObjectTypeArray )
671  throw t4::corrupt_data();
672 
673  // array case
674  store::Array<object_traits> arr = contents.ArrayValue();
675  size_t len = arr.Length();
676  if ( len == 1 )
677  return store::GetFilteredStream( arr.Get(0).DictionaryValue(), isInlineImage );
678 
679  /*
680  PDF slang
681 
682  The PDF spec is somewhat ambiguous on the rules for Contents streams that are an array.
683  In table 3.27 in the discussion of the Page object it discusses the fact that a Page's
684  Contents entry can be either a single stream, or an array of streams. One statement
685  in this description is:
686 
687  "If the value is an array, the effect is as if all of the streams in the array were
688  concatenated, in order, to form a single stream."
689 
690  A second statement says:
691 
692  "The division between streams may occur only at the boundaries between lexical tokens"
693 
694  A literal reading of this would imply that if one stream were to end with:
695 
696  " q" (that is, some white space, followed by a 'q')
697 
698  and the next stream were to begin with:
699 
700  "q "
701 
702  then it would be invalid PDF because after concatenating the streams together you would
703  see that the stream had been split in the middle of the single token 'qq'. Desktop Acrobat
704  treats the above as two separate 'q' tokens, not a single 'qq' token. What is unclear is
705  whether the spec meant to say that a PDF parser must ensure that streams are always split
706  at token boundaries (and so would have to insert a token boundary character between the streams
707  before viewing the concatenated streams as a single stream), or whether the producer is
708  required to only split the streams at token boundaries (and so required to leave in the
709  token boundary character at either the end of the first stream or start of the second stream).
710  This code considers the literal reading of the spec to be correct and considers the Acrobat
711  behavior incorrect. So it only accepts the above as two separate 'q' tokens in a permissive
712  build. In non-permissive builds we do not validate the requirement that streams only be
713  split at token boundaries, and simply concatenate the streams.
714 
715  See Watson bug#1274492 for an example of a pdf file with this issue
716  */
717  #ifdef TETRAPHILIA_PDF_PERMISSIVE
718  bool intersperseWhiteSpaceBlocks = true;
719  #else
720  bool intersperseWhiteSpaceBlocks = false;
721  #endif
722  pdf::store::UnionDataBlockStream<object_traits> unionStream( intersperseWhiteSpaceBlocks );
723  typename store::Array<object_traits>::const_iterator_type it = arr.begin();
724  for ( size_t index = 0; index < len; index++, ++it )
725  {
726  store::Dictionary<object_traits> dict = it->DictionaryValue();
727  unionStream.Append( dict, isInlineImage );
728  }
729 
730  return t4::make_data_block_stream( std::move( unionStream ) );
731 }

895 template<typename app_traits>
896 void MovePages( store::Store<app_traits>& store, const t4::vector<PageID>& pageList, const PageIndex& insertAtPageIndex )
897 {
898  typedef store::StoreObjTraits<app_traits> object_traits;
899  typedef store::Dictionary<object_traits> dict_type;
900  typedef store::Array<object_traits> array_type;
901  typedef store::Reference ref_type;
902  typedef store::Object<object_traits> obj_type;
903  typedef boost::optional<ref_type> optional_ref_type;
904  typedef t4::app_traits::pdf_traits::int_type int_type;
905 
906  PageIndex numPages = t4::pdf::count_num_pages( store );
907  ref_type insertAtPageRef, parentRef;
908 
909  if( insertAtPageIndex < numPages )
910  {
911  insertAtPageRef = GetPageRefFromPageIndex( store, insertAtPageIndex );
912  parentRef = store.ResolveReference( insertAtPageRef ).DictionaryValue().GetRequiredReference( "Parent" );
913  }
914  else
915  {
916  ref_type lastPageRef = GetPageRefFromPageIndex( store, numPages - 1 );
917  parentRef = store.ResolveReference( lastPageRef ).DictionaryValue().GetRequiredReference( "Parent" );
918  }
919 
920  //Populate a vector of page refs because page indices will change while moving pages
921  std::vector<ref_type> pageRefList;
922  for( auto itr = pageList.begin(); itr != pageList.end(); ++itr )
923  {
924  pageRefList.push_back( t4::pdf::get_page_ref( store.GetDocument(), (*itr) ) );
925  }
926 
927  size_t sameParentKids = 0;
928  for( auto iter = pageRefList.begin(); iter != pageRefList.end(); ++iter )
929  {
930  ref_type pageRef = *iter;
931  dict_type pageDict = store.ResolveReference( pageRef ).DictionaryValue();
932  ref_type pageParentRef = pageDict.GetRequiredReference("Parent");
933  if( pageParentRef != parentRef ) //Parent for this page is going to change
934  {
935  optional_ref_type optPageIDORef = pageDict.GetObject().GetContainerIDOReference();
936  if( !optPageIDORef )
937  throw t4::corrupt_data();
938 
939  //Get all inherited attributes and put them in the page's dictionary
940 
941  //Resources
942  obj_type resourcesObj = GetInheritablePageAttribute(pageDict, "Resources", true /*noResolve*/);
943  if( resourcesObj.GetContainerIDOReference() != optPageIDORef )
944  {
945  //Means this attribute doesn't reside in the page dictionary and is inherited, so we need to put it in the page's dictionary
946  if( resourcesObj.Type() == store::kObjectTypeReference )
947  {
948  pageDict.PutReference( "Resources", resourcesObj.ReferenceValue_NoTypeCheck() );
949  }
950  else if( resourcesObj.Type() == store::kObjectTypeDictionary )
951  {
952  pageDict.CreateDictionaryFrom( "Resources", resourcesObj.DictionaryValue_NoTypeCheck() );
953  }
954  else if( resourcesObj.Type() == store::kObjectTypeNull )
955  {
956  pageDict.CreateDictionary( "Resources" ); //Create empty dictionary
957  }
958  else
959  throw t4::corrupt_data();
960  }
961 
962  //MediaBox
963  obj_type mediaBoxObj = GetInheritablePageAttribute(pageDict, "MediaBox", true /*noResolve*/);
964  if( mediaBoxObj.GetContainerIDOReference() != optPageIDORef )
965  {
966  //Means this attribute doesn't reside in the page dictionary and is inherited, so we need to put it in the page's dictionary
967  if( mediaBoxObj.Type() == store::kObjectTypeReference )
968  {
969  pageDict.PutReference( "MediaBox", mediaBoxObj.ReferenceValue_NoTypeCheck() );
970  }
971  else if( mediaBoxObj.Type() == store::kObjectTypeArray )
972  {
973  pageDict.CreateArrayFrom( "MediaBox", mediaBoxObj.ArrayValue_NoTypeCheck() );
974  }
975  else if( mediaBoxObj.Type() == store::kObjectTypeNull )
976  {
977  array_type mediaBox = pageDict.CreateArray( "MediaBox" ); //Creates empty array
978  //Using MediaBox [0,0,612,792] as default (as suggested by PDF spec)
979  mediaBox.PutInteger( 0, 0 );
980  mediaBox.PutInteger( 1, 0 );
981  mediaBox.PutInteger( 2, 612 );
982  mediaBox.PutInteger( 3, 792 );
983 
984  }
985  else
986  throw t4::corrupt_data();
987  }
988 
989  //CropBox
990  obj_type cropBoxObj = GetInheritablePageAttribute( pageDict, "CropBox", true /*noResolve*/ );
991  if( cropBoxObj.GetContainerIDOReference() != optPageIDORef )
992  {
993  //Means this attribute doesn't reside in the page dictionary and is inherited, so we need to put it in the page's dictionary
994  if( cropBoxObj.Type() == store::kObjectTypeReference )
995  {
996  pageDict.PutReference( "CropBox", cropBoxObj.ReferenceValue_NoTypeCheck() );
997  }
998  else if( cropBoxObj.Type() == store::kObjectTypeArray )
999  {
1000  pageDict.CreateArrayFrom( "CropBox", cropBoxObj.ArrayValue_NoTypeCheck() );
1001  }
1002  else if( cropBoxObj.Type() == store::kObjectTypeNull )
1003  {
1004  //Default value (as per PDF Spec): the value of MediaBox
1005  obj_type mediaBox = pageDict.Get_NoResolve( "MediaBox" );
1006  if( mediaBox.Type() == store::kObjectTypeReference )
1007  pageDict.PutReference( "CropBox", mediaBox.ReferenceValue_NoTypeCheck() );
1008  else if( mediaBox.Type() == store::kObjectTypeArray )
1009  pageDict.CreateArrayFrom( "CropBox", mediaBox.ArrayValue_NoTypeCheck() );
1010  else
1011  throw t4::corrupt_data();
1012  }
1013  else
1014  throw t4::corrupt_data();
1015  }
1016 
1017  //Rotate
1018  obj_type rotateObj = GetInheritablePageAttribute( pageDict, "Rotate" );
1019  if( rotateObj.GetContainerIDOReference() != optPageIDORef )
1020  {
1021  //Means this attribute doesn't reside in the page dictionary and is inherited, so we need to put it in the page's dictionary
1022  if( rotateObj.Type() == store::kObjectTypeInteger )
1023  pageDict.PutInteger( "Rotate", rotateObj.IntegerValue_NoTypeCheck() );
1024  else if( rotateObj.Type() == store::kObjectTypeNull )
1025  pageDict.PutInteger( "Rotate", 0 );
1026  else
1027  throw t4::corrupt_data();
1028  }
1029 
1030  //Now detach this page node from it's existing location in the page tree
1031  DeletePage( store, pageRef );
1032  }
1033  else //(pageParentRef == parentRef)
1034  {
1035  dict_type pageParentDict = store.ResolveReference( pageParentRef ).DictionaryValue();
1036  DeleteReferenceFromPagesKidsArray( pageParentDict, pageRef );
1037  sameParentKids++;
1038  }
1039  }
1040 
1041  array_type kidsArray = store.ResolveReference( parentRef ).DictionaryValue().GetRequiredArray( "Kids" );
1042  size_t index = 0;
1043  std::vector<ref_type> remainingKids;
1044 
1045  if( insertAtPageIndex < numPages )
1046  {
1047  auto it = kidsArray.begin_noresolve();
1048  for( ; it != kidsArray.end_noresolve(); ++it )
1049  {
1050  if( insertAtPageRef == it->ReferenceValue() )
1051  break;
1052  index++;
1053  }
1054 
1055  while( it != kidsArray.end_noresolve() )
1056  {
1057  remainingKids.push_back( it->ReferenceValue() );
1058  ++it;
1059  }
1060  }
1061  else
1062  index = kidsArray.Length();
1063 
1064  //Attach these page nodes at their new position in the page tree
1065  for( auto iter = pageRefList.begin(); iter != pageRefList.end(); ++iter )
1066  {
1067  ref_type pageRef = *iter;
1068  dict_type pageDict = store.ResolveReference( pageRef ).DictionaryValue();
1069  kidsArray.PutReference( index++, pageRef );
1070  pageDict.PutReference( "Parent", parentRef );
1071  }
1072 
1073  if( insertAtPageIndex < numPages )
1074  {
1075  //Populate the remaining kids in the kids array
1076  for( auto iter = remainingKids.begin(); iter != remainingKids.end(); ++iter )
1077  {
1078  kidsArray.PutReference( index++, *iter );
1079  }
1080  }
1081 
1082  //Update "Count" for all ancestors
1083  ptrdiff_t delta = static_cast<ptrdiff_t>( pageRefList.size() ) - static_cast<ptrdiff_t>( sameParentKids );
1084  obj_type parentObj = store.ResolveReference( parentRef );
1085  while( parentObj.Type() != store::kObjectTypeNull )
1086  {
1087  dict_type parentDict = parentObj.DictionaryValue();
1088  int_type count = parentDict.GetInteger( "Count", -1 );
1089 
1090  if( count != -1 )
1091  {
1092  parentDict.PutInteger( "Count", (count + static_cast<int_type>( delta ) ) );
1093  }
1094 
1095  parentObj = parentDict.Get( "Parent" );
1096  }
1097 }
