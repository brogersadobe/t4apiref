175  static t4::unique_ptr<color::ColorConverter<signal_traits>> Create( const color::ColorSpace<app_traits> & srcCS,
176  const color::ColorSpace<app_traits> & dstCS,
177  color::RenderIntent renderIntent,
178  color::ConverterCachingType cachingType,
179  bool linear )
180  {
181  const DeviceNColorSpace<app_traits> & deviceNSrcCS = static_cast<const DeviceNColorSpace<app_traits>&>( srcCS );
182  if( deviceNSrcCS.IsSepAll() )
183  {
184  if( dstCS.IsSubtractive() )
185  {
186  typedef SeparationAllColorConverter<signal_traits, true> sep_conv_type;
187  return t4::make_unique<sep_conv_type>( dstCS.GetNumSignals() );
188  }
189  else
190  {
191  typedef SeparationAllColorConverter<signal_traits, false> sep_conv_type;
192  return t4::make_unique<sep_conv_type>( dstCS.GetNumSignals() );
193  }
194  }
195 
196  return t4::make_unique<DeviceNColorConverter<signal_traits>>( GetGlobalContext(), srcCS, dstCS, renderIntent, cachingType, linear, hide_this_public_ctor() );
197  }
