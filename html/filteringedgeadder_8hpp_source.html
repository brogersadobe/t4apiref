19 /*
20  tetraphilia/imaging_model/filteringedgeadder.hpp
21 
22  Preprocess edges and add to the simple bezier sampler.
23 
24  The idea is help the sampler avoid overflowing fixed point precision.
25 
26  The filteringedgeadder is input a path in user space, a ctm and a device space constraints. The
27  constraints are mapped to user space constraints (if possible). The input path is converted into an
28  equivalent path in user space with edges snapped to the user space mapped constraints. The goal is that
29  this new equivalent path will not over flow fixed point precision when mapped to device space and will
30  reduce to the same pixels.
31 
32 */

832 void FilteringBezierSamplerEdgeAdder<sampler_def>:: InitializeBoundsAndOffsets(const constraints_type& constraints, const edge_adder_params& params, bool zeroConstraints)
833 {
834  using namespace t4;
835 
836  int_type dx;
837  int_type dy;
838 
839  m_userToDeviceMatrix = params.GetMatrix();
840 
841  if (zeroConstraints)
842  {
843  dx = -get_xmin(constraints.m_interestBounds);
844  dy = -get_ymin(constraints.m_interestBounds);
845 
846  // adjust with dx and dy
847  set_xmin(m_origConstraintsRect,int_type(0));
848  set_ymin(m_origConstraintsRect,int_type(0));
849  set_xmax(m_origConstraintsRect,get_xmax(constraints.m_interestBounds) + dx);
850  set_ymax(m_origConstraintsRect,get_ymax(constraints.m_interestBounds) + dy);
851  get_e(m_userToDeviceMatrix) += dx;
852  get_f(m_userToDeviceMatrix) += dy;
853  }
854  else
855  {
856  m_origConstraintsRect = constraints.m_interestBounds;
857  dx = dy = 0;
858  }
859 
860 #ifdef BEZSAMPLER_DEBUG
861  if (detail::debugging)
862  {
863  const matrix_type& matrix = params.GetMatrix();
864  cache_logger(TEXT("INPUT Matrix %.3f,%.3f,%.3f,%.3f,%.3f,%.3f\n"),
865  real_services::GetFloat(get_a(matrix)),real_services::GetFloat(get_b(matrix)),real_services::GetFloat(get_c(matrix)),
866  real_services::GetFloat(get_d(matrix)),real_services::GetFloat(get_e(matrix)),real_services::GetFloat(get_f(matrix)));
867  cache_logger(TEXT("INPUT constraints (%d,%d) (%d,%d)\n"),
868  get_xmin(constraints.m_interestBounds),get_ymin(constraints.m_interestBounds),
869  get_xmax(constraints.m_interestBounds),get_ymax(constraints.m_interestBounds));
870  cache_logger(TEXT("INPUT delx %d dely %d\n"),dx,dy);
871  }
872 #endif
873 
874  m_userToDeviceMatrixClass = ClassifyMatrix(m_userToDeviceMatrix);
875 
876  {
877  matrix_type scratchDeviceToUserMatrix;
878  const matrix_type *deviceToUserMatrix;
879 
880  if (!zeroConstraints && params.GetInverseMatrix())
881  {
882  tetraphilia_assert(MatrixInvert(scratchDeviceToUserMatrix, m_userToDeviceMatrix));
883  tetraphilia_assert(scratchDeviceToUserMatrix == *params.GetInverseMatrix());
884  deviceToUserMatrix = params.GetInverseMatrix();
885  m_userSpaceBoundsInfinite = false;
886  }
887  else
888  {
889  m_userSpaceBoundsInfinite = !MatrixInvert(scratchDeviceToUserMatrix, m_userToDeviceMatrix); // userspacebounds infinite if matrix won't invert
890  deviceToUserMatrix = &scratchDeviceToUserMatrix;
891  }
892 
893  if (!m_userSpaceBoundsInfinite)
894  {
895  // convert constraints to user space by inverting the matrix
896 
897  tetraphilia_assert(!RectIsEmpty(m_origConstraintsRect));
898  int_rect_type constraintsRect = m_origConstraintsRect;
899  if (m_thinStroke)
900  {
901  // for thinkstrokes expand y by one pixel in each direction to prevent false local minimas
902  set_ymin(constraintsRect,PinnedAdd(get_ymin(constraintsRect),int_type(-1)));
903  set_ymax(constraintsRect,PinnedAdd(get_ymax(constraintsRect),int_type(+1)));
904  }
905 
906  // Watch out -- some times we just can't transform the interestBounds back into user space
907  if (DetectOverflowOrTransformAndBoundIntRectToIntRect(m_intUserSpaceBounds,constraintsRect,*deviceToUserMatrix))
908  {
909  // expand by one pixel in all directions
910  set_xmin(m_intUserSpaceBounds, PinnedAdd(get_xmin(m_intUserSpaceBounds),int_type(-1)));
911  set_ymin(m_intUserSpaceBounds, PinnedAdd(get_ymin(m_intUserSpaceBounds),int_type(-1)));
912  set_xmax(m_intUserSpaceBounds, PinnedAdd(get_xmax(m_intUserSpaceBounds),int_type(+1)));
913  set_ymax(m_intUserSpaceBounds, PinnedAdd(get_ymax(m_intUserSpaceBounds),int_type(+1)));
914 
915  // infinite bounds case if m_intUserSpaceBounds is not representable as reals
916  if (!CanCastToRealCoordRectWithoutOverflow<real_rect_type>(m_intUserSpaceBounds))
917  m_userSpaceBoundsInfinite = true;
918  else
919  {
920  // compute the constraints center in user space. We translate user space points by this
921  // value to minimize the chance of overflow.
922  m_deviceSpacePointAdjustmentX = average( get_xmin( m_origConstraintsRect ), get_xmax( m_origConstraintsRect ) );
923  m_deviceSpacePointAdjustmentY = average( get_ymin( m_origConstraintsRect ), get_ymax( m_origConstraintsRect ) );
924  point_type deviceSpacePointAdjustment = MakePoint<point_type>(static_cast<real_type>(m_deviceSpacePointAdjustmentX),static_cast<real_type>(m_deviceSpacePointAdjustmentY));
925 
926  m_realUserSpaceBounds = CastToRealCoordRect<real_rect_type>(m_intUserSpaceBounds);
927  m_userSpacePointAdjustment = MatrixTransform(*deviceToUserMatrix,deviceSpacePointAdjustment);
928  // m_userSpacePointAdjustment and setDeviceMatrixOffsets takes care of e and f adjustment
929  get_e(m_userToDeviceMatrix) = 0;
930  get_f(m_userToDeviceMatrix) = 0;
931 
932  // adjust the sampler to account for userSpacePointAdjustment
933  m_sampler.setDeviceMatrixOffsets(
934  multiply<int_coord>(subtract<int_coord>(dx, m_deviceSpacePointAdjustmentX),m_aaScanlines),
935  multiply<int_coord>(subtract<int_coord>(dy, m_deviceSpacePointAdjustmentY), m_aaScanlines));
936  }
937  }
938  else
939  m_userSpaceBoundsInfinite = true;
940  }
941  }
942 
943 #ifdef BEZSAMPLER_DEBUG
944  if (detail::debugging)
945  {
946  if (m_userSpaceBoundsInfinite)
947  cache_logger(TEXT("RSLT m_userSpaceBoundsInfinite TRUE\n"));
948  else
949  {
950  cache_logger(TEXT("RSLT m_userToDeviceMatrix %.3f,%.3f,%.3f,%.3f,%.3f,%.3f\n"),
951  real_services::GetFloat(get_a(m_userToDeviceMatrix)),real_services::GetFloat(get_b(m_userToDeviceMatrix)),real_services::GetFloat(get_c(m_userToDeviceMatrix)),
952  real_services::GetFloat(get_d(m_userToDeviceMatrix)),real_services::GetFloat(get_e(m_userToDeviceMatrix)),real_services::GetFloat(get_f(m_userToDeviceMatrix)));
953  cache_logger(TEXT("RSLT userSpaceBounds (%d,%d) (%d,%d)\n"),
954  get_xmin(m_intUserSpaceBounds),get_ymin(m_intUserSpaceBounds),
955  get_xmax(m_intUserSpaceBounds),get_ymax(m_intUserSpaceBounds));
956  cache_logger(TEXT("RSLT m_deviceSpacePointAdjustmentX %d m_deviceSpacePointAdjustmentY %d\n"),m_deviceSpacePointAdjustmentX,m_deviceSpacePointAdjustmentY);
957  cache_logger(TEXT("RSLT m_userSpacePointAdjustmentX %.3f m_userSpacePointAdjustmentY %.3f\n"),real_services::GetFloat(get_x(m_userSpacePointAdjustment)),real_services::GetFloat(get_y(m_userSpacePointAdjustment)));

964 template<typename sampler_def>
965 template<typename path_point_iterator_type>
966 void FilteringBezierSamplerEdgeAdder<sampler_def>::AddEdges(path_point_iterator_type& curPoint, const path_point_iterator_type& end)
967 {
968  if (!m_usable)
969  return; // matrix was not invertible - this path will be empty
970 
971  // Watson bug#1297644
972  // permissive builds allow invalid paths to be constructed. If we detect one here we act as if the path were empty
973  if ( sampler_def::permissive ) {
974  if ( !(curPoint == end || (*curPoint).type() == t4::path_op::moveto || (*curPoint).type() == t4::path_op::closepath ) )
975  curPoint = end;
976  } else {
977  tetraphilia_assert( curPoint == end || (*curPoint).type() == t4::path_op::moveto || (*curPoint).type() == t4::path_op::closepath );
978  }
979 
980  point_type lastmovetopoint;
981  bool lastmovetovalid = false;
982 
983  auto moveto = [&]( t4::real_point const& p )
984  {
985  lastmovetopoint = p;
986  lastmovetovalid = true;
987  MoveTo(lastmovetopoint);
988  ++curPoint;
989  };
990 
991  auto closepath = [&]( void )
992  {
993  ClosePath();
994  ++curPoint;
995  };
996 
997  // see bug 1446807
998  // treat ... closepath lineto or ... closepath curveto as a moveto
999  // of the previous subpaths moveto followed by lineto/closepath. PDFRef seems to imply this.
1000  // note: in permissive builds closepath is permitted at the beginning of a path
1001  // closepath lineto is an invalid path and just end it
1002  auto check_valid = [&]()
1003  {
1004  if (!m_curPointValid)
1005  {
1006  if (lastmovetovalid)
1007  MoveTo(lastmovetopoint);
1008  else
1009  {
1010  // this is an error closepath lineto at the beginning of the path
1011  // has no possible current point
1012  if (!sampler_def::permissive)
1013  throw t4::corrupt_data();
1014  curPoint = end;
1015  return false;
1016  }
1017  }
1018 
1019  return true;
1020  };
1021 
1022  auto curveto = [&]( t4::real_point const& p0, t4::real_point const& p1, t4::real_point const& p2 )
1023  {
1024  if ( check_valid() )
1025  {
1026  CurveTo( p0, p1, p2 );
1027  ++curPoint;
1028  }
1029  };
1030 
1031  auto lineto = [&]( t4::real_point const& p )
1032  {
1033  if ( check_valid() )
1034  {
1035  LineTo( p );
1036  ++curPoint;
1037  }
1038  };
1039 
1040  while (curPoint != end)
1041  (*curPoint).visit( moveto, lineto, curveto, closepath );
1042 
1043  // TODO: caller's responsibility? Only if they we support more than one FilteringBezierSamplerEdgeAdder call
1044  // close the remaining open path
1045  ImplicitClosePath();
1046 
1047  // push the edges into the sampler
1048  // if (!m_thinStroke && RectIsEmpty(m_pathBounds))
1049  if (get_ymin(m_pathBounds) > get_ymax(m_pathBounds))
1050  return;
1051 
1052  // ymin/ymax rows
1053  int_type yMin = real_services::FloorToIntPinned(ArgType<int_type>(), get_ymin(m_pathBounds));
1054  int_type yMax = real_services::CeilToIntPinned(ArgType<int_type>(), get_ymax(m_pathBounds));
1055  int_type xMin = real_services::FloorToIntPinned(ArgType<int_type>(), get_xmin(m_pathBounds));
1056  int_type xMax = real_services::CeilToIntPinned(ArgType<int_type>(), get_xmax(m_pathBounds));
1057  if (yMin == yMax) {
1058  // If we're a thin horizontal line on an integer boundary, make sure we're not discarded
1059  real_type yAdjust = tetraphilia::real_services::Epsilon(get_ymin(m_pathBounds));
1060  addPointToPathBounds(get_xmin(m_pathBounds), get_ymin(m_pathBounds) - yAdjust);
1061  addPointToPathBounds(get_xmin(m_pathBounds), get_ymax(m_pathBounds) + yAdjust);
1062  }
1063  if (xMin == xMax) {
1064  // If we're a thin vertical line on an integer boundary, make sure we're not discarded
1065  real_type xAdjust = tetraphilia::real_services::Epsilon(get_xmin(m_pathBounds));
1066  addPointToPathBounds(get_xmin(m_pathBounds) - xAdjust, get_ymin(m_pathBounds));
1067  addPointToPathBounds(get_xmax(m_pathBounds) + xAdjust, get_ymin(m_pathBounds));
1068  }
1069 
1070  // constrain height of the rowTable to the original constraints rect
1071  yMin = std::max(yMin,static_cast<int_type>((get_ymin(m_origConstraintsRect)-m_deviceSpacePointAdjustmentY)*static_cast<int_type>(m_aaScanlines)));
1072  yMax = std::min(yMax,static_cast<int_type>((get_ymax(m_origConstraintsRect)-m_deviceSpacePointAdjustmentY)*static_cast<int_type>(m_aaScanlines)+1));
1073 
1074  // the rect can be zero height (ymin==ymax) so make sure we have at least one scanlines worth
1075  if (yMax <= yMin)
1076  yMax = yMin+1;
1077 
1078  if (IntegerToRealConversion<int_type,real_type>::ConversionToRealWillOverflow(yMin)
1079  || IntegerToRealConversion<int_type,real_type>::ConversionToRealWillOverflow(yMax))
1080  {
1081 
1082  // The matrix couldn't be inverted, the page was huge and the original constraints rect
1083  // is not usable
1084  tetraphilia_assert(m_userSpaceBoundsInfinite);
1085  m_usable = false;
1086  return;
1087  }
1088 
1089  m_sampler.InitializeRows(yMin,static_cast<size_t>(yMax-yMin));
1090 
1091  sampler_edge *next;
1092  for (sampler_edge *pEdge = m_rootEdge; pEdge; pEdge = next)
1093  {
1094  next = pEdge->m_next;
1095  m_sampler.AddEdgeToSampler(pEdge); // modifies m_next
1096  }
1097 }
