68 template<typename signal_traits>
69 ImagePipe* GetImagePipeLine(
70  ImagePipe* srcPipe,
71  const ImageRecord& imageRecord,
72  const ImageRenderingRecord<signal_traits>& imageRendRecord,
73  const color::ColorSpace<t4::app_traits>::smart_ptr_type& dstCS,
74  color::RenderIntent intent )
75 {
76  auto& heap = *t4::get_thread_heap();
77 
78  if(!(get_width(imageRendRecord.m_imageSubsectionDataBounds) == imageRendRecord.m_width && get_height(imageRendRecord.m_imageSubsectionDataBounds) == imageRendRecord.m_height))
79  {
80  srcPipe = t4::make_transient<SubsectionImagePipe<signal_traits>>( heap, srcPipe, imageRecord.m_numSignals * imageRecord.m_bitsPerSignal,
81  imageRendRecord.m_imageSubsectionDataBounds );
82  }
83 
84  size_t numDstSignals = imageRecord.m_numSignals;
85 
86  /*
87  This has to come before the DecoderImagePipe because that one can truncate 16 bit samples
88  to 8 bit (if the signal type is unsigned char). If a 16 bit per component image has any
89  color key mask data, it might specify the ranges using a precision greater than 8 bits.
90  So even if we tried to apply the color key masking after the DecoderImagePipe by "appropriately"
91  adjusting the values in the Mask array, we might lose precision in doing so, and then end up
92  masking the wrong pixels. The only pipe that can go before the color key mask pipe is the
93  subsectioning pipe.
94  */
95  if( imageRecord.m_colorKeyMaskArray ) {
96  srcPipe = t4::make_transient<ColorKeyMaskImagePipe<signal_traits>>( heap, imageRecord, srcPipe );
97  numDstSignals = 1;
98  }
99 
100  if(imageRecord.m_bitsPerSignal != sizeof(typename signal_traits::signal_type) * 8)
101  srcPipe = t4::make_transient<DecoderImagePipe<signal_traits>>( heap, imageRecord, numDstSignals, srcPipe);
102 
103  // note that there is an optimization for the case where the image has an "inverted" decode array
104  // -- that is, a decode array where all ranges in the decode array are [1,0]. In that case both
105  // the decoding (i.e. conversion to "signals") and the decode array are processed in a single step
106  // by the DecoderImagePipe (assuming the code above did any decoding).
107  // And note that if we are being asked to construct the alpha mask from a Masked image
108  // (imageRecord.m_colorKeyMaskArray != 0), then we do not want to execute the DecodeArrray logic on that data
109  // (which would be the data in srcPipe)
110  if ( imageRecord.m_decodeArray &&
111  (imageRecord.m_bitsPerSignal == sizeof(typename signal_traits::signal_type) * 8 || !imageRecord.m_isInvertedDecodeArray)
112  && !imageRecord.m_colorKeyMaskArray )
113  srcPipe = t4::make_transient<DecodeArrayImagePipe<signal_traits>>( heap, imageRecord, srcPipe);
114 
115  bool preScaleColorConversion = true; // If true, convert colors before scaling image
116  if( imageRecord.m_colorKeyMaskArray || imageRecord.m_maskKind == kSoftMask ) {
117  // numDstSignals = 1; already done above
118  } else if( imageRecord.m_pdfCS.HasColorSpace() )
119  {
120  const auto& srcCS = imageRecord.m_pdfCS.GetColorSpace();
121  preScaleColorConversion = srcCS->IsIndexed() || srcCS->IsDeviceN();
122  // Note that we can't skip the color conversion even when the source
123  // and destination color spaces are the same because we may be
124  // converting to a linearized destination space:
125  if( preScaleColorConversion )
126  {
127  numDstSignals = dstCS->GetNumSignals();
128  srcPipe = t4::make_transient<ColorConverterImagePipe<signal_traits>>( heap, srcPipe, *srcCS, *dstCS, intent );
129  }
130  }
131 
132  // Tn
133  if( imageRendRecord.m_thumbnailScaleDownPower > 0 )
134  {
135  srcPipe = t4::make_transient<HorizThumbnailImagePipe<signal_traits>>( heap, numDstSignals, srcPipe, imageRendRecord.m_thumbnailScaleDownPower );
136 
137  srcPipe = t4::make_transient<VertThumbnailImagePipe<signal_traits>>( heap, numDstSignals, srcPipe, imageRendRecord.m_thumbnailScaleDownPower );
138  }
139 
140  if( !preScaleColorConversion )
141  srcPipe = t4::make_transient<ColorConverterImagePipe<signal_traits>>( heap, srcPipe, *imageRecord.m_pdfCS.GetColorSpace(), *dstCS, intent );
142 
143  return srcPipe;
144 }

146 template<typename signal_traits>
147 ImagePipe* GetOneBitImagePipeLine(
148  ImagePipe* srcPipe,
149  const ImageRecord& imageRecord,
150  const ImageRenderingRecord<signal_traits>& imageRendRecord,
151  const color::ColorSpace<t4::app_traits>::smart_ptr_type& dstCS,
152  color::RenderIntent intent,
153  t4::small_vector<typename signal_traits::signal_type, 8>& zeroColor,
154  t4::small_vector<typename signal_traits::signal_type, 8>& oneColor )
155 {
156  using signal_type = typename signal_traits::signal_type;
157  using color_val_type = t4::small_vector<signal_type, 8>;
158 
159  auto& heap = *t4::get_thread_heap();
160 
161  if(!(get_width(imageRendRecord.m_imageSubsectionDataBounds) == imageRendRecord.m_width && get_height(imageRendRecord.m_imageSubsectionDataBounds) == imageRendRecord.m_height))
162  {
163  srcPipe = t4::make_transient<SubsectionImagePipe<signal_traits>>( heap, srcPipe,
164  imageRecord.m_numSignals * imageRecord.m_bitsPerSignal,
165  imageRendRecord.m_imageSubsectionDataBounds );
166  }
167 
168  size_t numDstSignals = imageRecord.m_numSignals;
169 
170  /*
171  This has to come before the DecoderImagePipe because that one can truncate 16 bit samples
172  to 8 bit (if the signal type is unsigned char). If a 16 bit per component image has any
173  color key mask data, it might specify the ranges using a precision greater than 8 bits.
174  So even if we tried to apply the color key masking after the DecoderImagePipe by "appropriately"
175  adjusting the values in the Mask array, we might lose precision in doing so, and then end up
176  masking the wrong pixels. The only pipe that can go before the color key mask pipe is the
177  subsectioning pipe.
178  */
179  if( imageRecord.m_colorKeyMaskArray ) {
180  srcPipe = t4::make_transient<ColorKeyMaskImagePipe<signal_traits>>( heap, imageRecord, srcPipe );
181  numDstSignals = 1;
182  }
183 
184  // 8 bytes worth of destination data is the most any Decoder could produce from a single bit of source data
185  // (it will decode all 8 bits in the decode step)
186  zeroColor.SetNumElements( 8 );
187  oneColor.SetNumElements( 8 );
188  for ( int i = 0; i < 4; ++i )
189  {
190  zeroColor[i] = signal_traits::zero();
191  oneColor[i] = signal_traits::one();
192  }
193 
194  if(imageRecord.m_bitsPerSignal != sizeof(typename signal_traits::signal_type) * 8)
195  {
196  imaging_model::Decoder<signal_traits> decoder( imageRecord.m_bitsPerSignal, imageRecord.m_isIndexedColor/*zeroExtend*/,
197  imageRecord.m_isInvertedDecodeArray );
198  color_val_type src = zeroColor;
199  decoder.Convert( zeroColor.GetBuffer(), src.GetBuffer(), sizeof( signal_type ) );
200  src = oneColor;
201  decoder.Convert( oneColor.GetBuffer(), src.GetBuffer(), sizeof( signal_type ) );
202  }
203 
204  // note that there is an optimization for the case where the image has an "inverted" decode array
205  // -- that is, a decode array where all ranges in the decode array are [1,0]. In that case both
206  // the decoding (i.e. conversion to "signals") and the decode array are processed in a single step
207  // by the DecoderImagePipe (assuming the code above did any decoding).
208  // And note that if we are being asked to construct the alpha mask from a Masked image
209  // (imageRecord.m_colorKeyMaskArray != 0), then we do not want to execute the DecodeArrray logic on that data
210  // (which would be the data in srcPipe)
211  if ( imageRecord.m_decodeArray &&
212  (imageRecord.m_bitsPerSignal == sizeof(typename signal_traits::signal_type) * 8 || !imageRecord.m_isInvertedDecodeArray)
213  && !imageRecord.m_colorKeyMaskArray )
214  {
215  imaging_model::ArrayDecoder arrayDecoder( imageRecord.m_decodeArray->data(), imageRecord.m_numSignals );
216  color_val_type src = zeroColor;
217  arrayDecoder.Convert( zeroColor.GetBuffer(), src.GetBuffer(), sizeof( signal_type ) );
218  src = oneColor;
219  arrayDecoder.Convert( oneColor.GetBuffer(), src.GetBuffer(), sizeof( signal_type ) );
220  }
221 
222  if( imageRecord.m_colorKeyMaskArray || imageRecord.m_maskKind == kSoftMask ) {
223  // numDstSignals = 1; already done above
224  }
225  else if( imageRecord.m_pdfCS.HasColorSpace() )
226  {
227  const auto& srcCS = imageRecord.m_pdfCS.GetColorSpace();
228  numDstSignals = dstCS->GetNumSignals();
229  typedef typename color::ColorConverter<signal_traits>::ptr_type color_converter_type;
230  color_converter_type converter( srcCS->template CreateConverter<signal_traits>( *dstCS, intent, color::kConverterCachingTransient ) );
231  color_val_type src = zeroColor;
232  converter->Convert( zeroColor.GetBuffer(), sizeof( signal_type ), src.GetBuffer(), sizeof( signal_type ) );
233  src = oneColor;
234  converter->Convert( oneColor.GetBuffer(), sizeof( signal_type ), src.GetBuffer(), sizeof( signal_type ) );
235  }
236 
237  return srcPipe;
238 }

241 template<typename signal_traits>
242 void FillPixelBufferFromPipe( ImagePipe* pipe,
243  const typename imaging_model::PixelBufferTypes<signal_traits>::type& pixelBuffer,
244  bool planar)
245 {
246  using signal_type = typename signal_traits::signal_type;
247 
248  CostBasedConditionalYielder yielder;
249 
250  size_t totalPlaneCount = pixelBuffer.m_layout->m_numSignals;
251  size_t srcPixelStride = totalPlaneCount*sizeof(signal_type); // pixelSize;
252  const t4::int_rect& dataRect = pipe->GetBounds();
253  t4::int_coord numScanLines = get_height( pipe->GetBounds() );
254  t4::int_coord numPixels = get_width( pipe->GetBounds() );
255  size_t bytesToCopy = pipe->GetBytesPerScanLine();
256 
257  tetraphilia_assert( imaging_model::RectContainsRect( *pixelBuffer.m_bounds, pipe->GetBounds() ) );
258  unsigned char* pp = pixelBuffer.GetPixels();
259 
260  unsigned char* pDstForThisPlane = pp +
261  (pixelBuffer.m_layout->m_firstSignalOffset) +
262  static_cast<size_t>(get_ymin(dataRect)-get_ymin(*pixelBuffer.m_bounds))*pixelBuffer.m_layout->m_scanLineStride + /* get to right row */
263  static_cast<size_t>(get_xmin(dataRect)-get_xmin(*pixelBuffer.m_bounds))*pixelBuffer.m_layout->m_pixelStride;
264 
265  // An approximate per-scanline cost to insure that we yield a reasonable number of times
266  // during image sampling.
267  auto cost = static_cast<int>( pixelBuffer.m_layout->m_scanLineStride * 20 );
268 
269  if( !planar)
270  {
271  unsigned char* dst = pDstForThisPlane;
272  for( t4::int_coord i = 0; i < numScanLines; i++ )
273  {
274  yielder.Yield( cost );
275  memcpy( dst, pipe->GetNextScanLine(), bytesToCopy );
276  dst += pixelBuffer.m_layout->m_scanLineStride;
277  }
278  }
279  else
280  {
281  for( t4::int_coord r = 0; r < numScanLines; r++ )
282  {
283  yielder.Yield( cost );
284  const unsigned char* srcScanLine = pipe->GetNextScanLine();
285  unsigned char* pDst = pDstForThisPlane;
286 
287  for (size_t p=0; p<totalPlaneCount; p++)
288  {
289  unsigned char* dest = pDst;
290  const unsigned char* src = srcScanLine;
291  for( t4::int_coord c = 0; c < numPixels; c++ )
292  {
293  memcpy(dest, src, sizeof(signal_type));
294  src += srcPixelStride; // next pixel of chunky src
295  dest += sizeof(signal_type); // next signal of planar src
296  }
297  srcScanLine += sizeof(signal_type); // next signal of chunky src
298  pDst += pixelBuffer.m_layout->m_signalStride; // next dst plane
299 
300  }
301  pDstForThisPlane += pixelBuffer.m_layout->m_scanLineStride; // next dst scanline
302  }
303  }
304 }

391 inline const unsigned char* StreamImagePipe::GetNextScanLine()
392 {
393  if ( m_firstCall )
394  m_firstCall = false;
395  else
396  m_bufferedDataBlockStream->Advance(m_bytesPerScanLine);
397 
398  /*
399  PDF_SLANG
400 
401  Both desktop Acrobat and Tetraphilia have a general design to not attempt
402  to read data that is not being used for the current rendering. This means that
403  if the unread data happens to be damaged neither code base notices this damage
404  or informs the user of it. But there can be subtle differences in how much data
405  the two code bases think they need to read. For example, if an image is clipped
406  at the very bottom then the last scan line may or may not be read depending on
407  the algorithm. Tetraphilia will frequently read blocks of scan lines at a time
408  if any of those scan lines will be used, and if the image is going to be downsampled.
409  This can cause it to attempt to read a few scan lines that Acrobat does not attempt
410  to read. In permissive builds we try to not display errors on files that Acrobat
411  does not display them. So here there is a little bit of special code that allows
412  up to 1 truncated scan line. This will cause us to display a few files that
413  Acrobat does not. But that is considered better than not displaying a few that
414  Acrobat does.
415 
416  See Watson bug#1302874 for a set of such files.
417  */
418 
419  const char* ptr;
420 
421  #ifdef TETRAPHILIA_PDF_PERMISSIVE
422  {
423  size_t nb = m_bufferedDataBlockStream->PeekBytes(ptr, m_bytesPerScanLine);
424  if ( nb < m_bytesPerScanLine ) {
425  if ( m_allocPtr )
426  throw t4::corrupt_data();
427  m_allocPtr = t4::make_transient_array<char>( *t4::get_thread_heap(), m_bytesPerScanLine );
428  memset( &m_allocPtr[nb], 0, m_bytesPerScanLine - nb );
429  memcpy( m_allocPtr, ptr, nb );
430  ptr = m_allocPtr;
431  }
432  }
433  #else
434  m_bufferedDataBlockStream->PeekBytesAssert(ptr, m_bytesPerScanLine);
435  #endif
436 
437  this->m_yielder.Yield(100000);
438  return reinterpret_cast<const unsigned char*>( ptr );
439 }

485 template<typename signal_traits>
486 DecoderImagePipe<signal_traits>::DecoderImagePipe (
487  const ImageRecord& imageRec,
488  size_t numSignals,
489  ImagePipe* pipe )
490  : m_pipe( pipe ),
491  m_Decoder(imageRec.m_bitsPerSignal, imageRec.m_isIndexedColor/*zeroExtend*/, imageRec.m_isInvertedDecodeArray ),
492  m_bounds( pipe->GetBounds() )
493 {
494  // decoders are simple beasts that always process a whole number of source pixels. This
495  // means that they might write slightly past the end of the destination rectangle if we
496  // make it exactly the size of the source. Here we adjust the right side of the rectangle
497  // to make sure that the decoder doesn't write past the end of the buffer
498  size_t numSrcBytes = pipe->GetBytesPerScanLine();
499  size_t numSrcBits = numSrcBytes * 8;
500  size_t numSrcSignals = numSrcBits / imageRec.m_bitsPerSignal;
501 
502  // out destination buffer has to be big enough to hold numSrcSignals signals, since that
503  // is how many the decoder will write
504 
505  m_buff = t4::make_transient_array<signal_type>( *t4::get_thread_heap(), numSrcSignals );
506 
507  // for this computation we don't want to report any of the extra bytes that the decoder may
508  // have written, because those would have garbage values in them. So we use the actual
509  // bounds of our parent pipe. Note that if this image uses a color key mask, then the
510  // ColorKeyImagePipe will be in the pipeline before our code, and it will have changed
511  // the number of components to 1 instead of the number in imageRec.m_numSignals. For this
512  // reason we take an additional parameter "numSignals" that tells us how many we are supposed
513  // to process instead of relying on the value in imageRec.m_numSignals.
514  size_t numSrcPixels = t4::convert<size_t>( get_width( m_bounds ) );
515  m_bitsPerPixel = t4::multiply<size_t>( t4::multiply<size_t>( numSignals, sizeof( signal_type ) ), 8 );
516  m_bytesPerScanLine = t4::multiply<size_t>( t4::multiply<size_t>( numSrcPixels, numSignals ), sizeof( signal_type ) );
517 
518  tetraphilia_assert( m_bytesPerScanLine <= numSrcSignals * sizeof(signal_type) );
519 }

796 class ColorKeyMaskImagePipe : public ImagePipe
797 {
798  using signal_type = typename signal_traits::signal_type;
799 
800 public:
801  ColorKeyMaskImagePipe (
802  const ImageRecord& imageRec,
803  ImagePipe* pipe );
804 
805 
806  const t4::int_rect& GetBounds() const override
807  {
808  return m_bounds;
809  }
810

812  {
813  return m_bytesPerDstScanLine;
814  }
815 
816  size_t GetBitsPerPixel() const override
817  {
818  // The color key mask produces a single alpha signal of the same size as
819  // the input color components
820  return m_bitsPerSignal;
821  }
822 
823 protected:
824  const unsigned char* GetNextScanLine() override
825  {
826  const unsigned char* src = m_pipe->GetNextScanLine();
827  const unsigned char* srcEnd = src + m_bytesPerSrcScanLine;
828  memset(m_buff, 0, m_bytesPerDstScanLine);
829  unsigned char* dst = m_buff;
830 
831  size_t index = 0;
832  bool paintIt = false;
833  int numSrcBitsRemaining = 8;
834  int numDstBitsRemaining = 8;
835  t4::int_coord srcVal = 0;
836  while ( src < srcEnd )
837  {
838  // read the next component
839  switch ( m_bitsPerSignal ) {
840  case 1:
841  srcVal = (*src >> (numSrcBitsRemaining - 1)) & 1;
842  break;
843  case 2:
844  srcVal = (*src >> (numSrcBitsRemaining - 2)) & 3;
845  break;
846  case 4:
847  srcVal = (*src >> (numSrcBitsRemaining - 4)) & 15;
848  break;
849  case 8:
850  srcVal = *src;
851  break;
852  case 16:
853  srcVal = *src;
854  srcVal <<= 8;
855  ++src;
856  srcVal += *src;
857  break;
858  }
859  numSrcBitsRemaining -= static_cast<int>( m_bitsPerSignal );
860  if ( numSrcBitsRemaining <= 0 ) {
861  ++src;
862  numSrcBitsRemaining = 8;
863  }
864 
865  // we paint this pixel if any component is outside of the range specified by the mask
866  // (the mask array specifies colors that are not painted)
867  t4::int_coord minVal = m_colorKeyMaskArray[2*index];
868  t4::int_coord maxVal = m_colorKeyMaskArray[2*index+1];
869  if ( srcVal < minVal || srcVal > maxVal )
870  paintIt = true;
871 
872  ++index;
873 
874  // write our alpha value if we have stepped across as many components are there are
875  // in one source pixel.
876  if ( index == m_numComponentsSrc )
877  {
878  if ( paintIt ) {
879  switch ( m_bitsPerSignal ) {
880  case 1:
881  *dst |= 1 << (numDstBitsRemaining - 1);
882  break;
883  case 2:
884  *dst |= 3 << (numDstBitsRemaining - 2);
885  break;
886  case 4:
887  *dst |= 15 << (numDstBitsRemaining - 4);
888  break;
889  case 8:
890  *dst = 0xff;
891  break;
892  case 16:
893  *dst = 0xff;
894  ++dst;
895  *dst = 0xff;
896  break;
897  }
898  paintIt = false;
899  }
900  numDstBitsRemaining -= static_cast<int>( m_bitsPerSignal );
901  if ( numDstBitsRemaining <= 0 ) {
902  ++dst;
903  numDstBitsRemaining = 8;
904  tetraphilia_assert( (dst < m_buff + m_bytesPerDstScanLine) || ((dst == m_buff + m_bytesPerDstScanLine) && (src == srcEnd)) );
905  }
906  index = 0;
907  }
908  }
909 
910  tetraphilia_assert( (dst == m_buff + m_bytesPerDstScanLine) || (dst == m_buff + m_bytesPerDstScanLine - 1) );
911 
912  return m_buff;
913  }
914 
915  ImagePipe* m_pipe;
916  t4::int_rect m_bounds;
917  unsigned char* m_buff;
918  size_t m_bytesPerSrcScanLine;
919  size_t m_bytesPerDstScanLine;
920  size_t m_numComponentsSrc;
921  size_t m_bitsPerSignal;
922  const t4::int_coord* m_colorKeyMaskArray;
923 };
