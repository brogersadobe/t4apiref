45 template<typename app_traits>
46 class ICCBasedCSCreator
47 {
48 public:

50  typedef typename color_traits::color_conv_type color_conv_type;

54  static color::ColorSpace<app_traits> * create( context_type& context, const char* profile, size_t length )
55  {
56  auto ds = t4::make_memory_data_store_no_copy( profile, length );
57  t4::unique_ptr<color::ColorSpace<app_traits>> cs(color::ICCColorSpace<app_traits>::CreateFromProfile( context, ds ));
58  return cs.release();
59  }

61  static typename cache_type::cache_entry_ptr create(
62  cache_type const& cache,
63  const store::Dictionary<store::StoreObjTraits<app_traits> > & dict )
64  {
65  t4::small_vector<color_conv_type,8> ranges;
66  boost::optional<store::Array<store::StoreObjTraits<app_traits> > > rangeArray =
67  dict.GetArray( "Range" );
68  if( rangeArray )
69  {
70  ranges.SetNumElements( rangeArray->Length() );
71  for( unsigned int index = 0; index < rangeArray->Length(); ++index )
72  ranges[ index ] = rangeArray->Get( index ).RealValue();
73  }
74 
75  t4::data_block_stream dbs =
76  store::GetFilteredStream( dict, false /*isInlineImage*/ );
77  auto store = t4::make_data_store_from_data_block_stream( std::move( dbs ) );
78  auto key = GetStoreKey( dict );
79  auto cs = color::ICCColorSpace<app_traits>::CreateFromProfile( cache, key, std::move( store ) );
80  if( t4::ne( dict.GetRequiredInteger( "N" ), cs->get_value().GetNumSignals() ) )
81  throw t4::corrupt_data();
82  if( ranges.GetNumElements() > 0 )
83  cs->get_value().SetRanges( ranges.begin(), ranges.GetNumElements() );
84 
85  return cs;
86  }
87 };

126 template<typename app_traits>
127 class CalCSCreator
128 {
129 public:
130  typedef typename app_traits::color_traits color_traits;
131  typedef typename color_traits::color_conv_type color_conv_type;
132  typedef t4::cache<store::StoreKey, color::ColorSpace<app_traits>> cache_type;
133  typedef typename cache_type::cache_entry_ptr cache_entry_type;
134 
135  static cache_entry_type create( cache_type const& cache,
136  const store::Dictionary<store::StoreObjTraits<app_traits> > & dict,
137  color::CSFamily family,
138  bool halveRangeValues )
139  {
140  tetraphilia_assert( family == color::kCalGray || family == color::kCalRGB || family == color::kLab );
141  auto key = GetStoreKey( dict );
142 
143  // Get media white point (required):
144  color_conv_type whitePt[ 3 ];
145  store::Array<store::StoreObjTraits<app_traits> > whitePtArray = dict.GetRequiredArray( "WhitePoint" );
146  unsigned int index;
147  for( index = 0; index < 3; ++index )
148  whitePt[ index ] = whitePtArray.Get( index ).RealValue();
149 
150  // Get black point (optional):
151  color_conv_type blackPtData[ 3 ];
152  color_conv_type * blackPt = nullptr;
153  boost::optional<store::Array<store::StoreObjTraits<app_traits> > > blackPtArray = dict.GetArray( "BlackPoint" );
154  if( blackPtArray )
155  {
156  for( index = 0; index < 3; ++index )
157  blackPtData[ index ] = blackPtArray->Get( index ).RealValue();
158  blackPt = blackPtData;
159  }
160 
161  if( family == color::kCalGray )
162  {
163  color_conv_type gamma = dict.GetReal( "Gamma", 1 );
164  return color::ICCColorSpace<app_traits>::CreateGraySpace( cache, key, whitePt, blackPt, gamma );
165  }
166  if( family == color::kCalRGB )
167  {
168  color_conv_type gammas[ 3 ] = { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ),
169  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ),
170  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ) };
171  boost::optional<store::Array<store::StoreObjTraits<app_traits> > > gammaArray = dict.GetArray( "Gamma" );
172  if ( gammaArray )
173  for( index = 0; index < 3; ++index )
174  gammas[ index ] = gammaArray->Get( index ).RealValue();
175 
176  color_conv_type rColorant[ 3 ] = { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ),
177  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ),
178  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ) };
179  color_conv_type gColorant[ 3 ] = { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ),
180  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ),
181  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ) };
182  color_conv_type bColorant[ 3 ] = { TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ),
183  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0 ),
184  TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 1 ) };
185  boost::optional<store::Array<store::StoreObjTraits<app_traits> > > matrixArray = dict.GetArray( "Matrix" );
186  if ( matrixArray )
187  {
188  for( index = 0; index < 3; ++index )
189  rColorant[ index ] = matrixArray->Get( index ).RealValue();
190  for( index = 0; index < 3; ++index )
191  gColorant[ index ] = matrixArray->Get( 3 + index ).RealValue();
192  for( index = 0; index < 3; ++index )
193  bColorant[ index ] = matrixArray->Get( 6 + index ).RealValue();
194  }
195 
196  return color::ICCColorSpace<app_traits>::CreateRGBSpace( cache, key, whitePt, blackPt, rColorant, gColorant, bColorant, gammas );
197  }
198  // family == kLab
199 
200  color_conv_type ranges[ 6 ];
201  ranges[ 0 ] = 0;
202  ranges[ 1 ] = 100;
203  boost::optional<store::Array<store::StoreObjTraits<app_traits> > > rangeArray = dict.GetArray( "Range" );
204  if( rangeArray )
205  for( index = 2; index < 6; ++index )
206  ranges[ index ] = rangeArray->Get( index - 2 ).RealValue();
207  else if( color_traits::emulateAcrobat )
208  {
209  // PDF Slang bug
210  //
211  // See Watson bug #1274058. Although the PDF spec says that the
212  // default range values should be 0..1, for Lab profiles Acrobat
213  // defaults to L* = [0..100], and a* and b* = [-128..127] when
214  // dealing with real value colors.
215  ranges[ 2 ] = ranges[ 4 ] = -128;
216  ranges[ 3 ] = ranges[ 5 ] = 127;
217  }
218  else
219  {
220  ranges[ 2 ] = ranges[ 4 ] = 0;
221  ranges[ 3 ] = ranges[ 5 ] = 1;
222  }
223  if( halveRangeValues )
224  for( index = 2; index < 6; ++index )
225  ranges[ index ] /= 2;
226  return color::ICCColorSpace<app_traits>::CreateLabSpace( cache, key, whitePt, blackPt, ranges );
227  }
228 };
