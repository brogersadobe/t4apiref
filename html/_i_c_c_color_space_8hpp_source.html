304 template <typename scalar_type>
305 bool
306 Matrix3by4<scalar_type>::PostMultiplyBy( const Matrix3by4<scalar_type> & rhs )
307 {
308  size_t overflow = false;
309  Matrix3by4<scalar_type> result;
310  result.Zero();
311  for(unsigned int row = 0; row < 3; ++row )
312  {
313  for(unsigned int col = 0; col < 3; ++col )
314  for(unsigned int index = 0; index < 3; ++index )
315  result( row, col ) = real_services::Add( result( row, col ),
316  real_services::Multiply( m_data[ row ][ index ],
317  rhs( index, col ),
318  &overflow ),
319  &overflow );
320  result( row, 3 ) = real_services::Add( m_data[ row ][ 3 ],
321  rhs( row, 3 ),
322  &overflow );
323  }
324  memcpy( m_data, result.m_data, 3 * 4 * sizeof( scalar_type ) );
325  return !overflow;
326 }

385 template <typename scalar_type>
386 bool
387 Matrix3by4<scalar_type>::SnapWhite( const XYZColor<scalar_type> & white,
388  scalar_type threshold )
389 {
390  scalar_type minScale = 1 - threshold;
391  scalar_type maxScale = 1 + threshold;
392 
393  // We only adjust the 3 by 3 part of the matrix, so adjust the white point
394  // to exclude the offset terms:
395  scalar_type whiteX = white.m_x - m_data[ 0 ][ 3 ];
396  scalar_type whiteY = white.m_y - m_data[ 1 ][ 3 ];
397  scalar_type whiteZ = white.m_z - m_data[ 2 ][ 3 ];
398 
399  scalar_type totalX = m_data[ 0 ][ 0 ] + m_data[ 0 ][ 1 ] + m_data[ 0 ][ 2 ];
400  scalar_type totalY = m_data[ 1 ][ 0 ] + m_data[ 1 ][ 1 ] + m_data[ 1 ][ 2 ];
401  scalar_type totalZ = m_data[ 2 ][ 0 ] + m_data[ 2 ][ 1 ] + m_data[ 2 ][ 2 ];
402 
403  if( totalX > scalar_type( 0.1 ) &&
404  totalY > scalar_type( 0.1 ) &&
405  totalZ > scalar_type( 0.1 ) )
406  {
407  scalar_type scaleX = whiteX / totalX;
408  scalar_type scaleY = whiteY / totalY;
409  scalar_type scaleZ = whiteZ / totalZ;
410 
411  if( scaleX >= minScale && scaleX <= maxScale &&
412  scaleY >= minScale && scaleY <= maxScale &&
413  scaleZ >= minScale && scaleZ <= maxScale )
414  {
415  for( unsigned int col = 0; col < 3; ++col )
416  {
417  m_data[ 0 ][ col ] *= scaleX;
418  m_data[ 1 ][ col ] *= scaleY;
419  m_data[ 2 ][ col ] *= scaleZ;
420  }
421  return true;
422  }
423  }
424  return false;
425 }

433 template <typename scalar_type>
434 bool
435 Matrix3by4<scalar_type>::Invert( void )
436 {
437  // Find the determinants of the minors of the 3 by 3 part:
438  scalar_type detMin[3][3];
439  detMin[0][0] = m_data[1][1] * m_data[2][2] - m_data[1][2] * m_data[2][1];
440  detMin[0][1] = m_data[0][2] * m_data[2][1] - m_data[0][1] * m_data[2][2];
441  detMin[0][2] = m_data[0][1] * m_data[1][2] - m_data[0][2] * m_data[1][1];
442  detMin[1][0] = m_data[1][2] * m_data[2][0] - m_data[1][0] * m_data[2][2];
443  detMin[1][1] = m_data[0][0] * m_data[2][2] - m_data[0][2] * m_data[2][0];
444  detMin[1][2] = m_data[0][2] * m_data[1][0] - m_data[0][0] * m_data[1][2];
445  detMin[2][0] = m_data[1][0] * m_data[2][1] - m_data[1][1] * m_data[2][0];
446  detMin[2][1] = m_data[0][1] * m_data[2][0] - m_data[0][0] * m_data[2][1];
447  detMin[2][2] = m_data[0][0] * m_data[1][1] - m_data[0][1] * m_data[1][0];
448 
449  // Note that if the determinant is zero, this will result in mx being
450  // zero below, which will cause the code to return false eventually.
451  // Therefore we don't need to explicitly test for singulatiry here:
452  scalar_type det = GetDeterminant();
453 
454  // Test for invertibility. If |det(M)| >= 1 then dividing by the determinant
455  // won't overflow, so the matrix is invertible. Otherwise, if det(M) is
456  // between zero and one, and mx = det(M) * max_real, then mx <= max_real and
457  // therefore if detMin[][] < mx, then detMin[][] / det(M) < max_real:
458  int row, col;
459  if( AbsVal( det ) < 1 )
460  {
461  scalar_type mx = det * limits<scalar_type>::max();
462  for( row = 0; row < 3; ++row )
463  for( col = 0; col < 3; ++col )
464  if( AbsVal( detMin[ row ][ col ] ) >= mx )
465  return false;
466  }
467 
468  // Invert 3 by 3 part:
469  for( row = 0; row < 3; ++row )
470  for( col = 0; col < 3; ++col )
471  m_data[ row ][ col ] = detMin[ row ][ col ] / det;
472 
473  // Invert offset terms:
474  scalar_type b0 = -m_data[ 0 ][ 3 ];
475  scalar_type b1 = -m_data[ 1 ][ 3 ];
476  scalar_type b2 = -m_data[ 2 ][ 3 ];
477  for( row = 0; row < 3; ++row )
478  m_data[ row ][ 3 ] = m_data[ row ][ 0 ] * b0 +
479  m_data[ row ][ 1 ] * b1 +
480  m_data[ row ][ 2 ] * b2;
481 
482  return true;
483 }

487 template <typename scalar_type>
488 scalar_type
489 Matrix3by4<scalar_type>::GetDeterminant( void )
490 {
491 #ifdef _DEBUG
492  size_t overflow = false;
493  scalar_type dbg00 = real_services::Multiply( m_data[ 1 ][ 1 ], m_data[ 2 ][ 2 ], &overflow );
494  scalar_type tmp = real_services::Multiply( m_data[ 1 ][ 2 ], m_data[ 2 ][ 1 ], &overflow );
495  dbg00 = real_services::Add( dbg00, -tmp, &overflow );
496 
497  scalar_type dbg10 = real_services::Multiply( m_data[ 1 ][ 2 ], m_data[ 2 ][ 0 ], &overflow );
498  tmp = real_services::Multiply( m_data[ 1 ][ 0 ], m_data[ 2 ][ 2 ], &overflow );
499  dbg10 = real_services::Add( dbg10, -tmp, &overflow );
500 
501  scalar_type dbg20 = real_services::Multiply( m_data[ 1 ][ 0 ], m_data[ 2 ][ 1 ], &overflow );
502  tmp = real_services::Multiply( m_data[ 1 ][ 1 ], m_data[ 2 ][ 0 ], &overflow );
503  dbg20 = real_services::Add( dbg20, -tmp, &overflow );
504 
505  tmp = real_services::Multiply( m_data[ 0 ][ 0 ], dbg00, &overflow );
506  scalar_type tmp2 = real_services::Multiply( m_data[ 0 ][ 1 ], dbg10, &overflow );
507  tmp = real_services::Add( tmp, tmp2, &overflow );
508  tmp2 = real_services::Multiply( m_data[ 0 ][ 2 ], dbg20, &overflow );
509  tmp = real_services::Add( tmp, tmp2, &overflow );
510 
511  tetraphilia_assert( !overflow );
512 #endif
513 
514  // First calculate the determinates of the column zero minors:
515  scalar_type detMin00 = m_data[ 1 ][ 1 ] * m_data[ 2 ][ 2 ] - m_data[ 1 ][ 2 ] * m_data[ 2 ][ 1 ];
516  scalar_type detMin10 = m_data[ 1 ][ 2 ] * m_data[ 2 ][ 0 ] - m_data[ 1 ][ 0 ] * m_data[ 2 ][ 2 ];
517  scalar_type detMin20 = m_data[ 1 ][ 0 ] * m_data[ 2 ][ 1 ] - m_data[ 1 ][ 1 ] * m_data[ 2 ][ 0 ];
518  return m_data[ 0 ][ 0 ] * detMin00 + m_data[ 0 ][ 1 ] * detMin10 + m_data[ 0 ][ 2 ] * detMin20;
519 }

1373 // Called by sub-class constructors:
1374 template <typename app_traits>
1375 void
1376 ICCTagLut5Stage<app_traits>::Parse(
1377  data_io::BufferedDataStore& bds,
1378  const unsigned int nACurveChannels,
1379  const unsigned int nBCurveChannels,
1380  const XYZColor<typename ICCTagLut<app_traits>::color_conv_type> & pcsWhite )
1381 {
1382  typedef typename app_traits::color_traits::color_conv_type color_conv_type;
1383  typedef typename app_traits::file_system_traits::file_size_type marker_type;
1384 
1385  // Read the "B" curves:
1386  unsigned int offset = data_io::BytesToInteger<unsigned int>( bds, 1 );
1387  marker_type marker;
1388  unsigned int index;
1389  if( offset )
1390  {
1391  marker = bds.Mark();
1392  bds.Seek( this->m_tagOffset + offset );
1393 
1394  m_bCurves.reserve( nBCurveChannels );
1395  for( index = 0; index < nBCurveChannels; ++index )
1396  {
1397  marker_type curveStart = bds.Mark();
1398 
1399  m_bCurves.push_back( t4::make_shared<ICCTagTRC<app_traits>>( bds ) );
1400 
1401  // The curves are aligned to four byte boundaries:
1402  marker_type overflow = ( bds.Mark() - curveStart ) % 4;
1403  if( overflow > 0 )
1404  bds.Advance( 4 - overflow );
1405  }
1406 
1407  bds.Seek( marker );
1408  }
1409 
1410  // Read the matrix:
1411  offset = data_io::BytesToInteger<unsigned int>( bds, 1 );
1412  unsigned int row, col;
1413  if( offset )
1414  {
1415  if( nBCurveChannels != 3 )
1416  throw t4::corrupt_data();
1417  marker = bds.Mark();
1418 
1419  // Read the matrix data:
1420  bds.Seek( this->m_tagOffset + offset );
1421  for( row = 0; row < 3; ++row )
1422  for( col = 0; col < 3; ++col )
1423  this->m_matrix( row, col ) = RealFrom15Dot16<color_conv_type>( data_io::BytesToInteger<unsigned int>( bds, 4 ) );
1424  for( row = 0; row < 3; ++row )
1425  this->m_matrix( row, 3 ) = RealFrom15Dot16<color_conv_type>( data_io::BytesToInteger<unsigned int>( bds, 4 ) );
1426 
1427  // Shap white, if necessary:
1428  Matrix3by4<color_conv_type> temp( this->m_matrix );
1429  if( temp.Invert() &&
1430  temp.SnapWhite( pcsWhite, TETRAPHILIA_STATIC_TYPENAME_REAL( color_conv_type, 0.001 /* 0.1% error */ ) ) &&
1431  temp.Invert() )
1432  this->m_matrix = temp;
1433 
1434  bds.Seek( marker );
1435  }
1436  else // Initialize identity matrix
1437  for( row = 0; row < 3; ++row )
1438  for( col = 0; col < 4; ++col )
1439  this->m_matrix( row, col ) = ( row == col ) ? color_conv_type(1) : color_conv_type(0);
1440 
1441  // Read the "M" curves:
1442  offset = data_io::BytesToInteger<unsigned int>( bds, 1 );
1443  if( offset )
1444  {
1445  marker = bds.Mark();
1446  bds.Seek( this->m_tagOffset + offset );
1447 
1448  m_mCurves.reserve( nBCurveChannels ); // # M curves == # B curves
1449  for( index = 0; index < nBCurveChannels; ++index )
1450  {
1451  marker_type curveStart = bds.Mark();
1452 
1453  m_mCurves.push_back( t4::make_shared<ICCTagTRC<app_traits>>( bds ) );
1454 
1455  // The curves are aligned to four byte boundaries:
1456  marker_type overflow = ( bds.Mark() - curveStart ) % 4;
1457  if( overflow > 0 )
1458  bds.Advance( 4 - overflow );
1459  }
1460 
1461  bds.Seek( marker );
1462  }
1463 
1464  // Read the CLUT:
1465  offset = data_io::BytesToInteger<unsigned int>( bds, 1 );
1466  if( offset )
1467  {
1468  marker = bds.Mark();
1469 
1470  // Read the grid sizes:
1471  bds.Seek( this->m_tagOffset + offset );
1472  m_gridSizes.resize( this->m_nInputSignals );
1473  for( std::uint8_t channel = 0; channel < this->m_nInputSignals; ++channel )
1474  {
1475  m_gridSizes[ channel ] = data_io::BytesToInteger<unsigned int>( bds, 1 );
1476  if( m_gridSizes[ channel ] < 2 )
1477  throw t4::corrupt_data();
1478  }
1479  bds.Seek( marker + 16 );
1480  unsigned int precision = data_io::BytesToInteger<unsigned int>( bds, 1 );
1481  if( precision != 1 && precision != 2 )
1482  throw t4::corrupt_data();
1483  bds.Advance( 3 );
1484 
1485  // Determine the number of entries in the CLUT:
1486  size_t lutEntries = this->m_nOutputSignals;
1487  for( std::uint8_t channel = 0; channel < this->m_nInputSignals; ++channel )
1488  lutEntries *= m_gridSizes[ channel ];
1489  m_CLUT->SetNumEntries( lutEntries );
1490 
1491  // Read the CLUT values:
1492  this->m_CLUT->LoadData( bds, precision );
1493 
1494  bds.Seek( marker );
1495  }
1496  else if( this->m_nInputSignals != this->m_nOutputSignals )
1497  // There must be a CLUT if the number of input and output signals
1498  // don't match:
1499  throw t4::corrupt_data();
1500 
1501  // Read the "A" curves:
1502  offset = data_io::BytesToInteger<unsigned int>( bds, 1 );
1503  if( offset )
1504  {
1505  marker = bds.Mark();
1506  bds.Seek( this->m_tagOffset + offset );
1507 
1508  m_aCurves.reserve( nACurveChannels );
1509  for( index = 0; index < nACurveChannels; ++index )
1510  {
1511  marker_type curveStart = bds.Mark();
1512 
1513  m_aCurves.push_back( t4::make_shared<ICCTagTRC<app_traits>>( bds ) );
1514 
1515  // The curves are aligned to four byte boundaries:
1516  marker_type overflow = ( bds.Mark() - curveStart ) % 4;
1517  if( overflow > 0 )
1518  bds.Advance( 4 - overflow );
1519  }
1520 
1521  bds.Seek( marker );
1522  }
1523 }
