31 struct left_shift_impl
32 {
33  template<typename T1, typename T2>
34  using promotion_type = decltype(std::declval<T1>() << std::declval<T2>());
35 
36  /*
37  This is the fully-general version of left_shift that works for all combinations of integer types.
38  */
39  template<typename Result, typename T1, typename T2>
40  constexpr boost::optional<Result> operator()( T1 const& x, T2 const& y ) const
41  {
42  // left shift is undefined if the shift distance is negative
43  if ( is_negative( y ) )
44  return {};
45 
46  if ( y == 0 )
47  return convert_optional<Result>( x );
48 
49  if ( x == 0 )
50  return 0;
51 
52  if ( is_unsigned<Result>::value )
53  {
54  // shifting a negative value is always negative; we won't be able to store the result in an unsigned value
55  if ( is_negative( x ) )
56  return {};
57  using promotion_type = typename make_unsigned<extension_type<std::max( { sizeof(T1), sizeof(Result) } ), Result>>::type;
58  if ( lt( significant_bit_count_in_type<promotion_type> - significant_bit_count( x ), y ) )
59  return {};
60  return convert_optional<Result>( static_cast<promotion_type>( x ) << y );
61  }
62  else
63  {
64  #if __cplusplus >= 202002L
65  using promotion_type = extension_type<std::max( { sizeof(T1), sizeof(Result) } ), T1>;
66  if ( lt( significant_bit_count_in_type<promotion_type> - significant_bit_count( x ), y ) )
67  return {};
68  return convert_optional( static_cast<promotion_type>( x ) << y );
69  #else
70  using sp = extension_type<std::max( { sizeof(T1), sizeof(Result) } ), T1>;
71  using up = typename make_unsigned<sp>::type;
72  if ( lt( significant_bit_count_in_type<sp> - significant_bit_count( x ), y ) )
73  return {};
74  auto tmp = static_cast<sp>( static_cast<up>( x ) << y );
75  if ( is_negative( x ) != is_negative( tmp ) )
76  return {};
77  return convert_optional<Result>( tmp );
78  #endif
79  }
80  }
81 };
