122  #define POS_INF 0x7fffffff
123  #define NEG_INF 0x80000000
124 #if defined(_MSC_VER)
125  inline std::int32_t __declspec(naked) FixedDiv( std::int32_t a, std::int32_t b )
126  {
127  /*
128  This routine is derived from the version in PostScript done for the
129  PharLap compiler on the Pentium (see 3016.xxx.packages/fp/sources/I80386).
130  Basically, we do the divide here with only *positive* values, so that
131  we can have 31 bits of value and a rounding bit. If we did this signed,
132  we would have to keep the top bit. So we compute the sign of the result
133  up front with an xor. ebx holds the sign of the result as it's top bit.
134  We then convert the operands to positive values, if need be, and multiply
135  the divisor by 128k. We then do the divide, and shift the result down one,
136  dropping the rounding "guard bit" into the carry flag. The adc (add with
137  carry) of zero will add in the bottom bit (we could have simply added one
138  and then shifted). We check for overflows and (possibly) negate the
139  result and exit.
140 
141  */
142  _asm {
143  // uses eax, ebx, ecx, edx -- div itself uses eax, ecx and edx.
144  // we'll be safe and save ebx, but the others should be safe to
145  // trash. ebx is assumed to be preserved across calls (also esi, edi)
146  mov eax, [esp+4]
147  mov ecx, [esp+8]
148  push ebx
149  mov ebx, ecx
150  // the xor will hold the sign bit as the top bit of ebx
151  xor ebx, eax
152 
153  // check for a "divide by zero", and return the properly signed infinity
154  cmp ecx, 0
155  jz fxd_oflow
156 
157  // check for a negative operand and "fix it"
158  jns fxd_next_op
159  neg ecx
160 
161  // now let's do the same for the next operand
162  fxd_next_op:
163  cmp eax, 0
164  jns fxd_do_div
165  neg eax
166 
167  // here we're going to make the 32-bit eax value into a 64-bit
168  // one spread across edx:eax and multiplied by 128k
169  fxd_do_div:
170  mov edx, eax
171  shr edx, 15
172  shl eax, 17
173 
174  // check for a possible overflow first
175  cmp edx, ecx
176  jae fxd_oflow
177 
178  // now do the divide. The result is 2x too big, but has the
179  // rounding bit at the bottom
180  div ecx
181 
182  // divide by 2, and get the rounding bit into the carry flag
183  shr eax, 1
184 
185  // add in the rounding bit so that we round up. Check for
186  // overflow when we round up...
187  adc eax, 0
188  jo fxd_oflow
189 
190  // now correct the result if it was supposed to be negative
191  cmp ebx, 0
192  jns fxd_out
193  neg eax
194 
195  // if the the top bit is not set after this neg, then we overflowed
196  // in the divide. exit if the sign bit is set
197  js fxd_out
198 
199  // special case -- zero result does not show sign after negation
200  // but is not an overflow, either. (zero divided by negative number)
201  jz fxd_out
202 
203  // deal with choosing the right version of infinity for overflows
204  // top bit of ebx is sign bit for result. Adding it to 7fffffff
205  // will yield 80000000 if it was negative
206  fxd_oflow:
207  mov eax, POS_INF
208  shl ebx, 1
209  adc eax, 0
210 
211  fxd_out:
212  pop ebx
213  ret
214  }
215  }
