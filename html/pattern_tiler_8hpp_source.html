457 template<typename signal_traits, typename pattern_tile_painter_type>
458 t4::brief_ptr<RasterPainter<signal_traits>> MakePatternTiler(
459  pattern_tile_painter_type& tilePainter,
460  const Constraints& constraints,
461  TransparencyGroup<signal_traits>* parentGroup,
462  t4::brief_ptr<RasterPainter<signal_traits>> solidColor )
463 {
464  typedef RasterPainter<signal_traits> raster_painter_type;
465  typedef t4::brief_ptr<raster_painter_type> raster_painter_ptr_type;
466  typedef PatternTiler<signal_traits, pattern_tile_painter_type> pattern_tiler_type;
467 
468  auto result = t4::make_brief<pattern_tiler_type>( tilePainter, constraints, parentGroup->GetColorSpace(), parentGroup );
469 
470  if ( result->NeedsBackdropRemoval() )
471  {
472  // In the extremely rare case that a pattern uses a blend mode, the resulting PatterTiler will produce
473  // nonisolated results. This step removes the original parent backdrop, producing a result that can
474  // be composited against that same backdrop (if we really cared about performance in this case, we'd
475  // skip the backdrop removal and the composition steps, which cancel each other out, but we don't. This
476  // never happens in practice.)
477  raster_painter_ptr_type backdrop = parentGroup->GetRemovalBackdrop( constraints, parentGroup->GetColorSpace(), true );
478 
479  return MakeOperationRasterPainter( constraints, MakeSeparableOperation( RemoveBackdropOperation<signal_traits>() ),
480  MakeTempBuffer<signal_traits>( parentGroup->GetLayout(), constraints ),
481  std::move( backdrop ), raster_painter_ptr_type( std::move( result ) ) );
482  }
483 
484  // if pattern is colored, we can return the raster painter now. Note that the tiler might be
485  // colored even if the pattern is uncolored! This is to support that oh-so-important case
486  // where an uncolored pattern cell might contain a blend mode!
487  if ( !result->IsUncolored() )
488  return result;
489 
490  tetraphilia_assert( solidColor );
491 
492  // multiply tiled "shape" RP by solid color RP. We always care about opacity, even
493  // if the parent group does not.
494  GraphicLayoutDescriptor desc = parentGroup->GetLayout();
495  if ( desc.m_opacityRep == GraphicLayoutDescriptor::kOpacityIgnored )
496  {
497  desc = MakeGraphicLayoutDescriptor( desc.m_numComponents, desc.m_shapeRep,
498  GraphicLayoutDescriptor::kOpacityReplicated, desc.m_colorRep, false, true );
499  }
500 
501  return MakeMultiplier( constraints,
502  MakeTempBuffer<signal_traits>( desc, constraints ),
503  raster_painter_ptr_type( std::move( result ) ), std::move( solidColor ) );
504 }

506 template<typename _signal_traits, typename _pattern_tile_painter_type>
507 PatternTiler<_signal_traits, _pattern_tile_painter_type>::PatternTiler( pattern_tile_painter_type& tilePainter,
508  const constraints_type& constraints,
509  const color_space_ptr_type& cs,
510  transparency_group_type* parentBackdrop,
511  real_type superSamplingFactor,
512  real_type blurSpeed )
513  : m_heap( *t4::get_thread_heap() )
514  , m_memcpyDst( nullptr )
515  , m_memcpySrc( nullptr )
516  , m_graphicStore1( m_heap )
517  , m_usingSampler( false )
518  , m_sampler( nullptr )
519  , m_zeroWalker{}
520  , m_isUncolored( tilePainter.GetTilingPatternInfo().isUncolored )
521  , m_needsBackdropRemoval( false )
522 {
523 #ifdef _DEBUG
524  this->SetName( "Pattern" );
525 #endif
526 
527  const TilingPatternInfo<app_traits>& tilingPatternInfo = tilePainter.GetTilingPatternInfo();
528  const real_rect_type& bbox = tilingPatternInfo.tileCellBBox;
529 
530  // return zero (shapeless) raster painters if
531  // 1. constraints are empty
532  // 2. matrix is noninvertible
533  // 3. pattern tile bbox is empty
534  if ( RectIsEmpty( constraints.m_interestBounds ) ||
535  MatrixDeterminant( tilingPatternInfo.patternSpaceToCurrentDeviceSpaceMatrix ) == 0 ||
536  RectIsEmpty( bbox ) )
537  {
538  SetRasterPainterToAllZeros();
539  }
540  else
541  {
542  // The construction of the PatternTilerHelper below does all the
543  // work to set up the data used for the raster painter.
544  // The subsequent destruction of the PatternTilerHelper frees up all
545  // the state which was used to construct this data leaving only the
546  // necessary state required for the RP to work.
547  //
548  // We declare the PatternTilerHelper on the heap rather than the
549  // stack to accomodate environments with minimal stack space.
550  //
551  // We try to create one without a backdrop. If the pattern tiler hits
552  // a blend mode this results in a "failed" pattern tiler; we create a
553  // new (slower) one with the real parent backdrop. In practice this
554  // never happens.
555  typedef PatternTilerHelper<signal_traits, pattern_tile_painter_type> pattern_tiler_helper_type;
556  pattern_tiler_helper_type* pattern_tiler_helper = nullptr;
557 
558  pattern_tiler_helper = t4::make_transient<pattern_tiler_helper_type>( m_heap, this, tilePainter, constraints, cs, superSamplingFactor, blurSpeed, nullptr );
559  if ( pattern_tiler_helper->Failed() )
560  {
561  pattern_tiler_helper = t4::make_transient<pattern_tiler_helper_type>( m_heap, this, tilePainter, constraints, cs, superSamplingFactor, blurSpeed, parentBackdrop );
562 
563  m_needsBackdropRemoval = true;
564  }
565 
566  tetraphilia_assert( pattern_tiler_helper && !pattern_tiler_helper->Failed() );
567 
568  if ( pattern_tiler_helper->IsBadPattern() )
569  SetRasterPainterToAllZeros();
570  }
571 
572  this->Initialize( 1, &m_segment, InfiniteRect<int_rect_type>(), kEverythingNeeded );
573  this->SetCurrentSegmentID( 0 );
574 }

752 void PatternTilerHelper<_signal_traits, _pattern_tile_painter_type>::InitializePatternSpaceSteppingAndBBoxInfo()
753 {
754  // Set up stepping values
755  const real_point_type& stepping = m_tilingPatternInfo.tileStepping;
756  m_patternSpaceAbsXStep = AbsVal(get_x(stepping));
757  tetraphilia_assert( m_patternSpaceAbsXStep != 0 );
758  m_xSteppingIndexDelta = (get_x(stepping) > 0) ? 1 : -1;
759  m_patternSpaceAbsYStep = AbsVal(get_y(stepping));
760  tetraphilia_assert( m_patternSpaceAbsYStep != 0 );
761  m_ySteppingIndexDelta = (get_y(stepping) > 0) ? 1 : -1;
762  tetraphilia_assert(m_patternSpaceAbsXStep > 0 && m_patternSpaceAbsYStep > 0); // i.e., not zero
763 
764  // Set up pattern space bbox values
765  m_patternSpaceCellBBox = m_tilingPatternInfo.tileCellBBox;
766  if ( RectIsEmpty( m_patternSpaceCellBBox ) )
767  {
768  MakeRectangle(m_patternSpaceCellBBox, 0, 0, 1, 1);
769  m_patternIsBad = true;
770  }
771 
772  set_xmin(m_patternSpaceSteppingBBox, get_xmin(m_patternSpaceCellBBox));

783  m_patternSpaceToCurrentDeviceSpaceMatrix = m_tilingPatternInfo.patternSpaceToCurrentDeviceSpaceMatrix;

787  MakeRectangle( m_patternSpaceDrawingBBox, 0, 0, 1, 1 );
788  m_patternSpaceToCurrentDeviceSpaceMatrix = IdentityMatrix<matrix_type>();

790  }

793 template <typename _signal_traits, typename _pattern_tile_painter_type>
794 void PatternTilerHelper<_signal_traits, _pattern_tile_painter_type>::InitializeTileSpaceInfo( bool forceNoTile )
795 {
796  // We define a new coordinate space which we will call TILE SPACE.
797  // Tile space will be represented by physical pixels in an 2D pixel buffer
798  // and the mapping between pattern space and tile space will be given by a
799  // matrix of the form
800  //
801  // [ sx 0 ]
802  // [ 0 sy ]
803  // [ -m_patternSpaceCellBBox.xMin -m_patternSpaceCellBBox.yMin ]
804  //
805  // which maps the origin of m_patternSpaceCellBBox to (0,0) and the other corners
806  // of m_patternSpaceCellBBox to integer lattice points (modulo potential real-valued round-off error)
807  //
808  // First we compute the desired rendered tile dimensions, m_tileWidth and m_tileHeight.
809  // sx and sy are then computed directly from these values and stored in the appropriate matrix below.
810 
811  real_type patternDrawingWidth = get_xmax(m_patternSpaceDrawingBBox) - get_xmin(m_patternSpaceDrawingBBox);
812  real_type patternDrawingHeight = get_ymax(m_patternSpaceDrawingBBox) - get_ymin(m_patternSpaceDrawingBBox);
813  if ( patternDrawingWidth <= 0 || patternDrawingHeight <= 0 )
814  {
815  MakeRectangle( m_patternSpaceDrawingBBox, 0, 0, 1, 1 );
816  patternDrawingWidth = patternDrawingHeight = 1;
817  m_patternIsBad = true;
818  }
819 
820  const matrix_type& m = m_patternSpaceToCurrentDeviceSpaceMatrix;
821  real_type xDirectionGrowth = real_services::Sqrt(get_a(m)*get_a(m)+get_b(m)*get_b(m));
822  real_type yDirectionGrowth = real_services::Sqrt(get_c(m)*get_c(m)+get_d(m)*get_d(m));
823  real_type absDeterminant = AbsVal( MatrixDeterminant(m) ); // abs value of determinant is the factor by which the area of the tile grows or shrinks under the transformation
824  tetraphilia_assert( absDeterminant > 0 ); // the determinant was verified to be nonzero much earlier
825  if ( xDirectionGrowth * yDirectionGrowth <= 0 || absDeterminant == 0 )
826  {
827  xDirectionGrowth = 1;
828  yDirectionGrowth = 1;
829  absDeterminant = 1;
830  m_patternIsBad = true;
831  }
832 
833  // When shear is present in the matrix from pattern to device space, we can shrink the tile size by multiplying by what we'll call the "shearAreaAdjustmentFactor".
834  // In the typical case when no shear is present the "shearAreaAdjustmentFactor" will be 1.0
835  real_type shearAreaAdjustmentFactor = real_services::Sqrt( absDeterminant / (xDirectionGrowth * yDirectionGrowth) );
836  tetraphilia_assert( shearAreaAdjustmentFactor <= real_type( 1.01 ) ); // should have shearAreaAdjustmentFactor <= 1 but we compare against 1.01 to allow for minor rounding errors (previously we compared against xxx but that still sometimes hit)
837  if ( shearAreaAdjustmentFactor > 1 )
838  shearAreaAdjustmentFactor = 1;
839 
840  m_currentDeviceSpaceLengthOfDrawingWidth = xDirectionGrowth * patternDrawingWidth * shearAreaAdjustmentFactor;
841  m_currentDeviceSpaceLengthOfDrawingHeight = yDirectionGrowth * patternDrawingHeight * shearAreaAdjustmentFactor;
842 
843  real_type realValuedTileWidth = m_superSamplingFactor * m_currentDeviceSpaceLengthOfDrawingWidth;
844  real_type realValuedTileHeight = m_superSamplingFactor * m_currentDeviceSpaceLengthOfDrawingHeight;
845  tetraphilia_assert( realValuedTileWidth > 0 );
846  tetraphilia_assert( realValuedTileHeight > 0 );
847 
848  tetraphilia_assert( PATTERN_TILE_MEMORY_USAGE_LIMIT > 200 * m_numTotalComponents * sizeof(signal_type) * MIN_PATTERN_TILE_WIDTH * MIN_PATTERN_TILE_HEIGHT ); // max tile size should be much much bigger than min tile size
849  tetraphilia_assert( MIN_PATTERN_TILE_WIDTH > 0 );
850  tetraphilia_assert( MIN_PATTERN_TILE_HEIGHT > 0 );
851 
852  // compute tile width and height, constraining to at least minimum size
853  m_tileWidth = (realValuedTileWidth < MIN_PATTERN_TILE_WIDTH) ? MIN_PATTERN_TILE_WIDTH : real_services::CeilToInt(ArgType<int_type>(), realValuedTileWidth);
854  m_tileHeight = (realValuedTileHeight < MIN_PATTERN_TILE_HEIGHT) ? MIN_PATTERN_TILE_HEIGHT : real_services::CeilToInt(ArgType<int_type>(), realValuedTileHeight);
855 
856  // constrain width and height based on tile area
857  // Check for the possibility of overflow
858  m_usingSampler = (!forceNoTile && m_tileWidth < 65536 && m_tileHeight < 65536 &&
859  static_cast<size_t>(m_tileWidth * m_tileHeight) <= ( PATTERN_TILE_MEMORY_USAGE_LIMIT + m_numTotalComponents * sizeof(signal_type) - 1 ) / ( m_numTotalComponents * sizeof(signal_type) ));
860 
861 #ifdef TILING_PATTERNS_NEVER_USE_SAMPLER
862  // Intended primarily for debugging. Activating this assignment treats all patterns as if they exceed the tile size limit.
863  m_usingSampler = false;
864 #endif
865 
866  // If the tile is larger than the specified limit, we will not sample a single tile but instead render into a pixel buffer the size of the constraints
867  if ( !m_usingSampler )
868  {
869  m_tilePixelBufferBounds = m_constraints.m_interestBounds;
870  return;
871  }
872 
873  // define pixel buffer bounds
874  set_xmin(m_tilePixelBufferBounds, int_type(0));
875  set_ymin(m_tilePixelBufferBounds, int_type(0));
876  set_xmax(m_tilePixelBufferBounds, m_tileWidth);
877  set_ymax(m_tilePixelBufferBounds, m_tileHeight);
878 
879  // define pattern space --> tile space matrix
880  get_a(m_patternSpaceToTileSpaceMatrix) = m_tileWidth / patternDrawingWidth;
881  get_b(m_patternSpaceToTileSpaceMatrix) = 0;
882  get_c(m_patternSpaceToTileSpaceMatrix) = 0;
883  get_d(m_patternSpaceToTileSpaceMatrix) = m_tileHeight / patternDrawingHeight;
884  get_e(m_patternSpaceToTileSpaceMatrix) = -(get_a(m_patternSpaceToTileSpaceMatrix) * get_xmin(m_patternSpaceCellBBox));
885  get_f(m_patternSpaceToTileSpaceMatrix) = -(get_d(m_patternSpaceToTileSpaceMatrix) * get_ymin(m_patternSpaceCellBBox));
886 
887  // compute the stepping rect in tile space
888  if ( RectContainsRect( m_patternSpaceDrawingBBox, m_patternSpaceSteppingBBox ) )
889  {
890  get_x(m_tileSpaceStepping) = real_type(m_tileWidth);
891  get_y(m_tileSpaceStepping) = real_type(m_tileHeight);
892  }
893  else
894  {
895  // Applying matrix can be slightly lossy due to imprecision in the above division, so instead we do it directly.
896  // The logic/math above dictates that the results are at least as big as m_tileWidth and m_tileHeight so we ensure that here as we've seen cases where floating point rounding causes this to not be true.
897  get_x(m_tileSpaceStepping) = t4::max<real_type>( real_services::MultiplyAndPin( real_services::SubtractAndPin(get_xmax(m_patternSpaceSteppingBBox), get_xmin(m_patternSpaceCellBBox)), (m_tileWidth / patternDrawingWidth)), real_type(m_tileWidth) );
898  get_y(m_tileSpaceStepping) = t4::max<real_type>( real_services::MultiplyAndPin( real_services::SubtractAndPin(get_ymax(m_patternSpaceSteppingBBox), get_ymin(m_patternSpaceCellBBox)), (m_tileHeight / patternDrawingHeight)), real_type(m_tileHeight) );
899  }
900 
901  // set up the Tile Space --> Current Device Space matrix
902  if ( !MatrixInvert( m_tileSpaceToCurrentDeviceSpaceMatrix, m_patternSpaceToTileSpaceMatrix ) )
903  throw t4::corrupt_data();
904  m_tileSpaceToCurrentDeviceSpaceMatrix = m_tileSpaceToCurrentDeviceSpaceMatrix * m_patternSpaceToCurrentDeviceSpaceMatrix;
905 }

1023 void GetIndexStartAndEnd( int& start, int& end, int steppingDelta,
1024  real_type targetBoundsMin, real_type targetBoundsMax,
1025  real_type cellBBoxMin, real_type cellBBoxMax,
1026  real_type absStepSize )
1027 {
1028  // The computation here of start and end is more subtle than it appears.
1029  // It may seem strange or even wrong at first, but here is the explanation (I keep forgetting it myself).
1030  // Below we discuss the x direction case.
1031  //
1032  // The rough idea is that we are measuring the number of STEPS to bring the right hand side (RHS) of
1033  // the tile to the LHS of the bounds, and similarly, the number of STEPS to bring the LHS of
1034  // the tile to the RHS of the bounds (don't forget that the stepping can be much smaller than the tile).
1035  // The delicate part is how we should handle the "edge" (integral) cases.
1036  // By way of example, the thinking is as follows:
1037  // fStartX: Consider the number of xsteps between the RHS of the tile and the LHS of the bounds.
1038  // Remember that fStartX is first value at which we will render.
1039  // For 2.99 xsteps we would set fStartX=3 since with 2 steps the RHS of the tile would still be to the left of the bounds, but with 3 it would not.
1040  // For 3.01 xsteps we would set fStartX=4 since with 3 steps the RHS of the tile would still be to the left of the bounds, but with 4 it would not.
1041  // For 3 xsteps we will set fStartX=4 since with 3 steps the RHS of the tile will be flush with the LHS of the bounds (intersection has zero area) and we don't need to render that case.
1042  // Thus, we compute fStartX using the formula:
1043  // 1 + FLOOR(#xsteps from RHS of tile to LHS of bounds)
1044  // fEndX: Consider the number of xsteps between the LHS of the tile and the RHS of the bounds.
1045  // Remember that fEndX is first value at which we will STOP (and not render).
1046  // For 7.99 xsteps we would set fEndX=8 since with 8 steps the LHS of the tile would be to the right of the bounds, but with 7 it would not.
1047  // For 8.01 xsteps we would set fEndX=9 since with 9 steps the LHS of the tile would be to the right of the bounds, but with 8 it would not.
1048  // For 8 xsteps we will set fEndX=8 since with 8 steps the LHS of the tile will be flush with the RHS of the bounds (intersection has zero area) and we don't need to render that case.
1049  // Thus, we compute fEndX using the formula:
1050  // CEIL(#xsteps from LHS of tile to RHS of bounds)
1051  //
1052  // NOTE: It is possible for fStartX == fEndX when nothing needs to be drawn. For example, when no tile will ever intersect the bounds, perhaps because the bounds is entirely covered by spaces between the tiles.
1053  start = 1 + real_services::FloorToInt( ArgType<int>(), (targetBoundsMin - cellBBoxMax) / absStepSize );
1054  end = real_services::CeilToInt( ArgType<int>(), (targetBoundsMax - cellBBoxMin) / absStepSize );
1055  if ( steppingDelta < 0 )
1056  {
1057  int tmp = start;
1058  start = end - 1;
1059  end = tmp - 1;
1060  }
1061 }
