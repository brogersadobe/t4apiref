87  bool HandleTextInfo( const textextract::TextInfoElement& textInfo, const t4::pdf::content_range& range ) override
88  {
89  /* not checking for HasValidUCM here - refer bug 3746486 */
90  if (!textInfo.IsVisible())
91  {
92  // We don't care about invisible or invalid text elements
93  return true;
94  }
95 
96  size_t numGlyphs = textInfo.GetGlyphLength();
97  tetraphilia_assert( numGlyphs > 0 );
98  point_type start;
99  point_type end;
100  point_type ascentVec;
101  point_type descentVec;
102  textInfo.GetGlyphPosition( start, 0, true, m_deviceMatrix );
103  textInfo.GetGlyphPosition( end, numGlyphs-1, false, m_deviceMatrix );
104  textInfo.GetAscenderDescenderVectors( ascentVec, descentVec, m_deviceMatrix );
105 
106  float distanceSq = 0;
107 
108  if ( m_force )
109  {
110  // If m_force is set, don't bail out if this isn't a direct hit,
111  // since we're looking for the closest possible
112  if ( !isInside( start, end, ascentVec, descentVec, &distanceSq ) )
113  {
114  if ( distanceSq > m_distanceSq )
115  {
116  // We've already found a closer hit, so don't bother checking the glyphs
117  return true;
118  }
119  }
120  }
121  else if ( !isInside( start, end, ascentVec, descentVec ) )
122  {
123  // m_force isn't set, and we don't have a hit, so bail out.
124  return true;
125  }
126 
127  size_t unicodeLength = 0;
128  const unicode_type* unicodeValues = textInfo.GetUnicodeValues( unicodeLength );
129  size_t currValue = 0;
130  size_t currUnicodeGlyph = 0;
131 
132  // These are values to identify word breaks
133  // Start by assuming the entire show is a word
134  size_t startBreak = 0; // The furthest word break encountered that we know precedes our best hit
135  size_t endBreak = numGlyphs - 1; // The closest word break encountered that we know follows our best hit
136  size_t lastBreak = numGlyphs; // The most recent word break encountered, initialized to numGlyphs so we know it's out of range
137 
138  // Find the glyph that's closest to our hit point
139  bool foundBetterHit = false;
140  size_t glyphHit = 0;
141 
142  for ( size_t currGlyph = 0; currGlyph < numGlyphs; ++currGlyph )
143  {
144  bool currGlyphIsBetter = false;
145 
146  textInfo.GetGlyphPosition( start, currGlyph, true, m_deviceMatrix );
147  textInfo.GetGlyphPosition( end, currGlyph, false, m_deviceMatrix );
148 
149  bool directHit = isInside( start, end, ascentVec, descentVec, &distanceSq );
150  if ( directHit || ( distanceSq < m_distanceSq ) )
151  {
152  // Only count this as a hit if it's not on a word break
153  while ( currValue < unicodeLength )
154  {
155  currUnicodeGlyph = textInfo.GetGlyphNum( currValue );
156 
157  if ( currUnicodeGlyph > currGlyph )
158  {
159  // Don't let our unicode enumeration exceed our current glyph
160  break;
161  }
162  else
163  {
164  unicode_type unicodeValue = unicodeValues[currValue];
165 
166  // Treating puncutations and invisible characters as word break
167  // doing unicode categorization needs valid UCM - refer bug 3746486
168  if ( m_snapToWords && textInfo.HasValidUCM() && ( m_teContext.GetUnicodeCategorizer().IsPunctuation( unicodeValue ) || !m_teContext.GetUnicodeCategorizer().IsVisible( unicodeValue ) ) )
169  {
170  // We found a word break; keep track of it
171  lastBreak = currUnicodeGlyph;
172  if ( lastBreak < glyphHit )
173  {
174  // This actually occurs before our current hit, so update our starting break
175  startBreak = lastBreak+1;
176  }
177  else if ( lastBreak < endBreak )
178  {
179  // This is the first break we've seen since our last hit, so assume
180  // it's our ending break
181  endBreak = lastBreak-1;
182  }
183  }
184  else if ( currUnicodeGlyph == currGlyph )
185  {
186  // Only count this as a hit if we found a real glyph
187  currGlyphIsBetter = true;
188  break;
189  }
190  }
191  ++currValue;
192  }
193  }
194  if ( currGlyphIsBetter )
195  {
196  foundBetterHit = true;
197  glyphHit = currGlyph;
198  if ( auto p = textInfo.GetContentPointForGlyph( glyphHit, true ) )
199  {
200  m_closestPoint = *p;
201  }
202 
203  real_type dStart = t3::real_services::VectorLength( get_x(m_hitPoint) - get_x(start), get_y(m_hitPoint) - get_y(start) );
204  real_type dEnd = t3::real_services::VectorLength( get_x(m_hitPoint) - get_x(end), get_y(m_hitPoint) - get_y(end) );
205 
206  if ( dStart < dEnd )
207  {
208  m_charHitPoint = start;
209  }
210  else
211  {
212  m_charHitPoint = end;
213  }
214 
215  if ( lastBreak < numGlyphs )
216  {
217  // Update startBreak just past the last word break that we saw
218  startBreak = lastBreak+1;
219  }
220 
221  // Any previous word breaks must not be our ending breaks, so reset it to the end
222  endBreak = numGlyphs-1;
223 
224  if ( directHit )
225  {
226  // If we have a direct hit, updated m_hit and m_distanceSq accordingly,
227  // and stop looking for a closer hit
228  m_haveHit = true;
229  m_distanceSq = 0;
230  m_hitStart = m_closestPoint;
231  if ( auto p = textInfo.GetContentPointForGlyph( glyphHit, false ) )
232  {
233  m_hitEnd = *p;
234  }
235 
236  break;
237  }
238  else
239  {
240  // If we don't have a direct hit, update m_distanceSq accordingly,
241  // but keep looking for a closer hit
242  m_distanceSq = distanceSq;
243  }
244  }
245  }
246 
247  if ( foundBetterHit )
248  {
249  // We've got a hit/candidate, so find the word boundaries around it
250  if ( endBreak >= ( numGlyphs-1 ) )
251  {
252  // Continue looking for word breaks within the unicode values
253  // currValue should still be where we left off, with startBreak set accordingly
254  for ( ; currValue < unicodeLength; ++currValue )
255  {
256 
257  unicode_type unicodeValue = unicodeValues[currValue];
258 
259  // Treating puncutations and invisible characters as word break
260  // doing unicode categorization needs valid UCM - refer bug 3746486
261  bool isWordBreak = m_snapToWords && textInfo.HasValidUCM() && ( m_teContext.GetUnicodeCategorizer().IsPunctuation( unicodeValue ) || !m_teContext.GetUnicodeCategorizer().IsVisible( unicodeValue ) );
262 
263  if ( !isWordBreak )
264  {
265  continue;
266  }
267  currUnicodeGlyph = textInfo.GetGlyphNum( currValue );
268  if ( currUnicodeGlyph < glyphHit )
269  {
270  // We found a word break before our hit, so the next glyph
271  // is a better match for the start of a word
272  startBreak = currUnicodeGlyph+1;
273  }
274  else
275  {
276  // We found a word break after our hit, so the previous glyph is
277  // the end of our word.
278  endBreak = currUnicodeGlyph-1;
279  break;
280  }
281  }
282  }
283 
284  // Fill in the word settings with the boundaries that we found
285  if ( auto p = textInfo.GetContentPointForGlyph( startBreak, true ) )
286  {
287  m_wordStartPoint = *p;
288  }
289  textInfo.GetGlyphPosition(m_startPoint, startBreak, true, m_deviceMatrix);
290 
291  if ( auto p = textInfo.GetContentPointForGlyph( endBreak, false ) )
292  {
293  m_wordEndPoint = *p;
294  }
295  textInfo.GetGlyphPosition(m_endPoint, endBreak, false, m_deviceMatrix);
296  }
297 
298  // We're enumerating our text content in reverse display list order, so if we found
299  // a direct hit then it's top-most in the z-order, and we can stop enumerating.
300  return !m_haveHit;
301  }

340  bool isInside(const point_type& baselineStart,
341  const point_type& baselineEnd,
342  const point_type& ascentVec,
343  const point_type& descentVec,
344  float *distanceSq = nullptr) const
345  {
346  real_type x[4], y[4];
347  x[0] = get_x(baselineStart) + get_x(ascentVec);
348  y[0] = get_y(baselineStart) + get_y(ascentVec);
349  x[1] = get_x(baselineEnd) + get_x(ascentVec);
350  y[1] = get_y(baselineEnd) + get_y(ascentVec);
351  x[2] = get_x(baselineEnd) + get_x(descentVec);
352  y[2] = get_y(baselineEnd) + get_y(descentVec);
353  x[3] = get_x(baselineStart) + get_x(descentVec);
354  y[3] = get_y(baselineStart) + get_y(descentVec);
355 
356  // TODO for now assume no rotation
357  auto xmin = t4::min<real_type>(x[0], x[1], x[2], x[3]);
358  auto xmax = t4::max<real_type>(x[0], x[1], x[2], x[3]);
359  auto ymin = t4::min<real_type>(y[0], y[1], y[2], y[3]);
360  auto ymax = t4::max<real_type>(y[0], y[1], y[2], y[3]);
361 
362  float hitPointDistSq = 0.0f;
363 
364  if (xmin <= get_x(m_hitPoint) && get_x(m_hitPoint) <= xmax && ymin <= get_y(m_hitPoint) && get_y(m_hitPoint) <= ymax)
365  {
366  // touch point lies on text, so return true
367  return true; // TODO for now assume no rotation, so bbok is path
368  }
369  else
370  {
371  real_type xDist;
372  if ( xmin > get_x(m_hitPoint) )
373  xDist = xmin - get_x(m_hitPoint);
374  else if ( get_x(m_hitPoint) > xmax )
375  xDist = get_x(m_hitPoint) - xmax;
376  else
377  xDist = 0;
378  real_type yDist;
379  if ( ymin > get_y(m_hitPoint) )
380  yDist = ymin - get_y(m_hitPoint);
381  else if ( get_y(m_hitPoint) > ymax )
382  yDist = get_y(m_hitPoint) - ymax;
383  else
384  yDist = 0;
385  float xDistFlt = t3::real_services::GetFloat( xDist );
386  float yDistFlt = t3::real_services::GetFloat( yDist );
387  xDistFlt *= static_cast<float>(HIT_TEST_WEIGHT_X); // favor y direction
388 
389  hitPointDistSq = (xDistFlt * xDistFlt) + (yDistFlt * yDistFlt);
390 
391  if ( hitPointDistSq <= (m_hitTestDimension * m_hitTestDimension) )
392  {
393  return true; // we found a hit 'near' the touch point
394  }
395  }
396 
397  if ( distanceSq != nullptr )
398  {
399  *distanceSq = hitPointDistSq;
400  }
401 
402  return false;
403  }

435  void HandleUnicodeChar( std::uint32_t c, bool isLastInShow ) override
436  {
437  const t4::pdf::content_point& textPoint = this->m_curContentPoint;
438 
439  point_type start = m_pShowInfo->GetDeviceSpaceBaselinePos( textPoint.m_stringNum, textPoint.m_glyphNum, textPoint.m_unicodeNum, textPoint.m_totalNumUnicode );
440  point_type end = m_pShowInfo->GetDeviceSpaceBaselinePos( textPoint.m_stringNum, textPoint.m_glyphNum, 1, 1 );
441  point_type ascentVec;
442  point_type descentVec;
443  m_pShowInfo->GetUserSpaceAscenderDescenderVectorsForDisplay( ascentVec, descentVec );
444 
445  float distanceSq = 0;
446  if ( isInside(start, end, ascentVec, descentVec, &distanceSq ) )
447  {
448  real_type dStart = t3::real_services::VectorLength(get_x(m_hitPoint) - get_x(start), get_y(m_hitPoint) - get_y(start) );
449  real_type dEnd = t3::real_services::VectorLength(get_x(m_hitPoint) - get_x(end), get_y(m_hitPoint) - get_y(end) );
450  // I beieve all text points coming into here are set to before the glyph, so I don't work too hard here...
451  tetraphilia_assert(textPoint.m_isAfter == false);
452  m_haveHit = true;
453  if (dStart < dEnd)
454  {
455  m_hitStart = textPoint;
456  m_hitStart.m_isAfter = false;
457  m_charHitPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_hitStart.m_stringNum, m_hitStart.m_glyphNum, m_hitStart.m_unicodeNum, m_hitStart.m_totalNumUnicode );
458  }
459  else
460  {
461  m_hitStart = textPoint;
462  m_hitStart.m_isAfter = true;
463  m_charHitPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_hitStart.m_stringNum, m_hitStart.m_glyphNum, 1, 1 );
464  }
465  m_hitEnd = m_hitStart;
466  m_hitEnd.m_isAfter = true;
467  // do NOT stop consumption, there may be more than one glyph under the cursor.

469  else

471  if ( distanceSq < m_distanceSq )
472  {
473  m_distanceSq = distanceSq;
474  real_type dStart = t3::real_services::VectorLength( get_x( m_hitPoint ) - get_x( start ), get_y( m_hitPoint ) - get_y( start ) );
475  real_type dEnd = t3::real_services::VectorLength( get_x( m_hitPoint ) - get_x( end ), get_y( m_hitPoint ) - get_y( end ) );
476  // I beieve all text points coming into here are set to before the glyph, so I don't work too hard here...
477  tetraphilia_assert( textPoint.m_isAfter == false );
478  if ( dStart < dEnd )
479  {
480  //m_client.HandleTextClickFound(textPoint, start, m_pShowInfo->m_ascentVec, m_pShowInfo->m_descentVec);
481  m_closestPoint = textPoint;
482  m_closestPoint.m_isAfter = false;
483  m_charHitPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_closestPoint.m_stringNum, m_closestPoint.m_glyphNum, m_closestPoint.m_unicodeNum, m_closestPoint.m_totalNumUnicode );
484  }
485  else
486  {
487  m_closestPoint = textPoint;
488  m_closestPoint.m_isAfter = true;
489  m_charHitPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_closestPoint.m_stringNum, m_closestPoint.m_glyphNum, 1, 1 );
490  }
491  }
492  }
493 
494  bool isWordBreak = m_teContext.GetUnicodeCategorizer().IsWordBreak( c );
495 
496  if ( !isWordBreak )
497  {
498  m_prevTextPoint = textPoint;
499  m_prevTextPoint.m_isAfter = true;
500  m_prevPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_prevTextPoint.m_stringNum, m_prevTextPoint.m_glyphNum, 1, 1 );
501  }
502  if ( !m_firstTextPointSet )
503  {
504  m_wordStartPoint = textPoint;
505  m_wordStartPoint.m_isAfter = false;
506  m_startPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_wordStartPoint.m_stringNum, m_wordStartPoint.m_glyphNum, m_wordStartPoint.m_unicodeNum, m_wordStartPoint.m_totalNumUnicode );
507  m_firstTextPointSet = true;
508  }
509 
510  if ( !m_haveHit )
511  {
512  m_wordEndPoint = textPoint;
513  m_wordEndPoint.m_isAfter = true;
514  m_endPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_wordEndPoint.m_stringNum,m_wordEndPoint.m_glyphNum, 1, 1 );
515  }
516  else
517  {
518  if ( !m_anyWordBreakObserved )
519  {
520  m_wordEndPoint = m_prevTextPoint;
521  m_endPoint = m_prevPoint;
522  }
523  if (isWordBreak || isLastInShow)
524  {
525  m_anyWordBreakObserved = true;
526  }
527  }
528 
529  if ( m_prevWasAWordBreakAndNoHit && !isWordBreak )
530  {
531  m_wordStartPoint = textPoint;
532  m_wordStartPoint.m_isAfter = false;
533  m_startPoint = m_pShowInfo->GetDeviceSpaceBaselinePos( m_wordStartPoint.m_stringNum, m_wordStartPoint.m_glyphNum, m_wordStartPoint.m_unicodeNum, m_wordStartPoint.m_totalNumUnicode );
534  m_prevWasAWordBreakAndNoHit = false;
535  }
536  if ( ( isWordBreak || isLastInShow ) && !m_haveHit )
537  {
538  m_prevWasAWordBreakAndNoHit = true;
539  }
540  }
