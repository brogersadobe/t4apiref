682 struct rect_ordering : public std::binary_function<gpuRectFloat, gpuRectFloat, bool>
683 {
684  bool operator()(const gpuRectFloat& r1, const gpuRectFloat& r2) const
685  {
686  if(get_xmax(r1) < get_xmin(r2))
687  return true;
688 
689  if(get_xmin(r1) > get_xmax(r2))
690  return false;
691 
692  if(get_ymax(r1) < get_ymin(r2))
693  return true;
694 
695  if(get_ymin(r1) > get_ymax(r2))
696  return false;
